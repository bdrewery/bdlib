  56 /**
  57   * @brief Encode a plaintext string into base64 (returns a buffer)
  58   * @param src A c-style string to encode
  59   * @param len Reference to length of string (to be updated on return)
  60   * @param charset The charset to use.
  61   * @return An encoded NULL-terminated c-style string (must be free()d later)
  62   */
  63 char *b64enc(const unsigned char *src, size_t *len, const char* charset = NULL);
  64 
  65 /**
  66  * @brief Base64 encode a string
  67  * @param string The string to encode
  68  * @param charset Optional charset to use. None will use standard base64
  69  * @return A new, encoded string
  70  */
  71 String base64Encode(const String& string, const char* charset = NULL);
  72 
  73 /**
  74   * @brief Encode a plaintext string into base64 (using a given buffer)
  75   * @param data The c-style string to encode
  76   * @param len Reference to length of string (to be updated on return)
  77   * @param dest Reference to the buffer to encode into
  78   * @param charset The charset to use.
  79   */
  80 void b64enc_buf(const unsigned char *data, size_t *len, char *dest, const char* charset = NULL);
  81 
  82 /**
  83   * @brief Decode a base64 encoded string into plaintext (returns a buffer)
  84   * @param data A c-style string to decode
  85   * @param len Reference to length of string (to be updated on return)
  86   * @param charset The charset to use.
  87   * @return A decoded NULL-terminated c-style string (must be free()d later)
  88   */
  89 char *b64dec(const unsigned char *data, size_t *len, const char* charset = NULL);
  90 
  91 /**
  92  * @brief Base64 decode a string
  93  * @param string The string to decode
  94  * @param charset Optional charset to use. None will use standard base64
  95  * @return A new, decoded string
  96  */
  97 String base64Decode(const String& string, const char* charset = NULL);
  98 
  99 /**
 100   * @brief Decode a base64 encoded string into plaintext (using a given buffer)
 101   * @param data The c-style string to decode
 102   * @param len Reference to length of string (to be updated on return)
 103   * @param dest Reference to the buffer to decode into
 104   * @param charset_index The character set to use
 105   */
 106 void b64dec_buf(const unsigned char *data, size_t *len, char *dest, const char* charset_index = NULL);
