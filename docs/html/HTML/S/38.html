  27 #ifndef _BD_ATOMICFILE_H
  28 #define _BD_ATOMICFILE_H 1
  29 
  30 #include "bdlib.h"
  31 #include "String.h"
  32 
  33 BDLIB_NS_BEGIN
  34 
  35 /**
  36  * @class AtomicFile
  37  * @brief Write out files atomically via tmpfile/rename.
  38  */
  39 class AtomicFile {
  40   private:
  41     String _fname;
  42     String _tmpname;
  43     int _fd;
  44     mode_t _mode;
  45   public:
  46     AtomicFile() : _fname(), _tmpname(), _fd(-1), _mode(-1) {};
  47     AtomicFile(AtomicFile& f) : _fname(f._fname), _tmpname(), _fd(f._fd),
  48                                 _mode(f._mode) {};
  49     /**
  50      * @brief Calling the destructor will abort the file if it has not already been written.
  51      * @sa abort
  52      */
  53     virtual ~AtomicFile() {
  54       if (is_open()) {
  55         abort();
  56       }
  57     }
  58 
  59     /**
  60      * @brief Open a file for writing.
  61      */
  62     bool open(const String& fname, mode_t mode = -1);
  63 
  64     /**
  65      * @brief Abort creating a file and cleanup.
  66      */
  67     bool abort();
  68 
  69     /**
  70      * @brief Check if a file was successfully opened.
  71      */
  72     inline bool is_open() const {
  73       return _fd != -1;
  74     }
  75 
  76     /**
  77      * @brief Return an fd for the open file.
  78      * @return Returns -1 if not open.
  79      */
  80     inline int fd() {
  81       return (is_open() ? _fd : -1);
  82     }
  83 
  84     /**
  85      * @brief Write out the file and close the descriptor.
  86      */
  87     bool commit();
  88 };
  89 BDLIB_NS_END
  90 #endif /* _BD_ATOMICFILE_H */
