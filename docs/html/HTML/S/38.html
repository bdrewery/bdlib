  34 #define NUM_ASCII_BYTES (3)
  35 #define NUM_ENCODED_BYTES (4)
  36 #define PADDING_CHAR '='

  38 void b64enc_buf(const unsigned char *data, size_t *len, char *dest, const char* charset)
  39 {
  40   char *buf = dest;
  41 
  42   if (!charset) {
  43     charset = b64_charset;
  44   }
  45 
  46   /* Encode 3 bytes at a time. */
  47 
  48   /*
  49    *
  50    * |       0       |       1       |       2       |
  51    *
  52    * |               |               |               |
  53    * |       |       |       |       |       |       |
  54    * |   |   |   |   |   |   |   |   |   |   |   |   |
  55    * | | | | | | | | | | | | | | | | | | | | | | | | |
  56    *  7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0
  57    * -------------------------------------------------
  58    *  5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 
  59    * | | | | | | | | | | | | | | | | | | | | | | | | |
  60    * |   | | |   |   | | |   |   | | |   |   | | |   |
  61    * |     |     |     |     |     |     |     |     |
  62    * |           |           |           |           |
  63    *
  64    * |     0     |     1     |     2     |     3     |
  65    *
  66    */
  67 
  68 
  69   while (*len >= NUM_ASCII_BYTES) {
  70     /* buf[0] is the 6 left-most bits of data[0] */
  71     buf[0] = charset[(data[0] & 0xfc) >> 2];
  72     /* buf[1] is the right-most 2 bits of data[0] and the left-most 4 bits of data[1] */
  73     buf[1] = charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
  74     /* buf[2] is the right-most 4 bits of data[1] and the 2 left-most bits of data[2] */
  75     buf[2] = charset[((data[1] & 0x0f) << 2) | ((data[2] & 0xc0) >> 6)];
  76     /* buf[3] is the right-most 6 bits of data[2] */
  77     buf[3] = charset[data[2] & 0x3f];
  78 
  79     data += NUM_ASCII_BYTES;
  80     buf += NUM_ENCODED_BYTES;
  81     *len -= NUM_ASCII_BYTES;
  82   }
  83 
  84   /* There is either 1 or 2 bytes left to encode (and possibly some padding to add) */
  85   if (*len > 0) {
  86     buf[0] = charset[(data[0] & 0xfc) >> 2];
  87     if (*len == 1) {
  88       buf[1] = charset[((data[0] & 0x03) << 4)];
  89       buf[2] = PADDING_CHAR;
  90     } else if (*len > 1) { /* == 2 .. left as >1 to help optimize with last if */
  91       /* Tack on to buf[1] the left-most 4 bits of data[1] */
  92       buf[1] = charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
  93       buf[2] = charset[(data[1] & 0x0f) << 2];
  94     }
  95     buf[3] = PADDING_CHAR;
  96     buf += NUM_ENCODED_BYTES;
  97   }
  98 
  99   *len = buf - dest;
 100 }

 102 String base64Encode(const String& string, const char* charset) {
 103   size_t len = string.length();
 104   char *p = b64enc(reinterpret_cast<const unsigned char*>(string.data()), &len, charset);
 105   bd::String encoded(p, len);
 106   free(p);
 107   return encoded;

 110 char *b64enc(const unsigned char *src, size_t *len, const char* charset)
 111 {
 112   /* Take the length and round up the next 4-byte boundary */
 113   size_t dlen = (((*len + (NUM_ASCII_BYTES - 1)) / NUM_ASCII_BYTES) * NUM_ENCODED_BYTES);
 114   char *dest = static_cast<char*>(malloc(dlen + 1));
 115 
 116   b64enc_buf(src, len, dest, charset);
 117   dest[*len] = '\0';
 118   return dest;
 119 }

 121 void b64dec_buf(const unsigned char *data, size_t *len, char *dest, const char* charset_index)
 122 {
 123   char *buf = dest;
 124   size_t blocks = (*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0);
 125 
 126   if (!charset_index) {
 127     charset_index = b64_indexes;
 128   }
 129 
 130   /* Convert the encoded base64 character back into our base255 ASCII character */
 131   while (blocks > 0) {
 132     /* buf[0] is the 6 bits from data[0] + left-most 2 bits of data[1] */
 133     *buf++ = (charset_index[data[0]] << 2) + ((charset_index[data[1]] & 0x30) >> 4);  /* 0x30 not required but to be clear. */
 134     if (data[2] != PADDING_CHAR) {
 135       /* buf[1] is the right-most 4 bits of data[1] + left-most 4 bits of data[2] */
 136       *buf++ = ((charset_index[data[1]] & 0x0f) << 4) + ((charset_index[data[2]] & 0x3c) >> 2);
 137       if (data[3] != PADDING_CHAR) {
 138         /* buf[2] is the right-most 2 bits of data[2] + data[3] */
 139         *buf++ = ((charset_index[data[2]] & 0x03) << 6) + charset_index[data[3]];
 140       }
 141     }
 142     data += NUM_ENCODED_BYTES;
 143     --blocks;
 144   }
 145 
 146   *len = (buf - dest);
 147 }

 149 char *b64dec(const unsigned char *data, size_t *len, const char* charset)
 150 {
 151   size_t dlen = ((*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0)) * NUM_ASCII_BYTES;
 152   char *dest = static_cast<char*>(malloc(dlen + 1));
 153 
 154   if (charset) {
 155     char charset_index[256];
 156     memset(charset_index, -1, sizeof(charset_index));
 157     for (size_t i = 0; i < 64; ++i)
 158       charset_index[int(charset[i])] = i;
 159     b64dec_buf(data, len, dest, charset_index);
 160   }  else
 161     b64dec_buf(data, len, dest);
 162   dest[*len] = '\0';
 163   return dest;
 164 }
