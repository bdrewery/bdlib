  60 /**
  61   * @brief Encode a plaintext string into base64 (returns a buffer)
  62   * @param src A c-style string to encode
  63   * @param len Reference to length of string (to be updated on return)
  64   * @param charset The charset to use.
  65   * @return An encoded NULL-terminated c-style string (must be free()d later)
  66   */
  67 char *b64enc(const unsigned char *src, size_t *len, const char* charset = NULL);
  68 
  69 /**
  70  * @brief Base64 encode a string
  71  * @param string The string to encode
  72  * @param charset Optional charset to use. None will use standard base64
  73  * @return A new, encoded string
  74  */
  75 String base64Encode(const String& string, const char* charset = NULL);
  76 
  77 /**
  78   * @brief Encode a plaintext string into base64 (using a given buffer)
  79   * @param data The c-style string to encode
  80   * @param len Reference to length of string (to be updated on return)
  81   * @param dest Reference to the buffer to encode into
  82   * @param charset The charset to use.
  83   */
  84 void b64enc_buf(const unsigned char *data, size_t *len, char *dest, const char* charset = NULL);
  85 
  86 /**
  87   * @brief Decode a base64 encoded string into plaintext (returns a buffer)
  88   * @param data A c-style string to decode
  89   * @param len Reference to length of string (to be updated on return)
  90   * @param charset The charset to use.
  91   * @return A decoded NULL-terminated c-style string (must be free()d later)
  92   */
  93 char *b64dec(const unsigned char *data, size_t *len, const char* charset = NULL);
  94 
  95 /**
  96  * @brief Base64 decode a string
  97  * @param string The string to decode
  98  * @param charset Optional charset to use. None will use standard base64
  99  * @return A new, decoded string
 100  */
 101 String base64Decode(const String& string, const char* charset = NULL);
 102 
 103 /**
 104   * @brief Decode a base64 encoded string into plaintext (using a given buffer)
 105   * @param data The c-style string to decode
 106   * @param len Reference to length of string (to be updated on return)
 107   * @param dest Reference to the buffer to decode into
 108   * @param charset_index The character set to use
 109   */
 110 void b64dec_buf(const unsigned char *data, size_t *len, char *dest, const char* charset_index = NULL);
