  28 #ifndef _BD_SORTEDLIST_H
  29 #define _BD_SORTEDLIST_H 1
  30 
  31 #include "bdlib.h"
  32 #include "List.h"
  33 
  34 BDLIB_NS_BEGIN
  35 
  36 template <class T>
  37 /**
  38  * @class SortedList
  39  * @brief A List which sorts on insert
  40  */
  41 class SortedList : public List<T> {
  42   typedef T iterator_type;
  43   typedef typename List<iterator_type>::Node Node;
  44 
  45   public:
  46     SortedList() : List<iterator_type>() {};
  47     virtual ~SortedList() {};
  48 
  49     /**
  50      * @brief Insert an element into the list (ordered)
  51      * @param item The element to insert
  52      * This operation is O(n/2)
  53      */
  54     virtual void insert(const iterator_type& item) {
  55       /*
  56        * 1) list = NULL
  57        * 2) list = (X)
  58        * 3) list = (X)->(Y)
  59        * 4) list = (W)->(X)->(Y)
  60        */
  61       Node* node = new Node(item), *prev = nullptr, *current = this->head;
  62       while (current != nullptr && item < current->item) {
  63         prev = current;
  64         current = current->next;
  65       }
  66 
  67       if (current == nullptr && this->head == nullptr) {
  68         this->head = this->tail = node;
  69       } else if (current == nullptr) { //Ran off the end, insert AS tail
  70         this->tail = prev->next = node;
  71         node->prev = prev; //prev=tail
  72       } else if (current == this->head) {
  73         node->next = this->head;
  74         current->prev = node;
  75         this->head = node;
  76       } else {
  77         node->next = current;
  78         node->prev = prev;
  79         current->prev = node;
  80         prev->next = node;
  81       }
  82 
  83       ++(this->my_size);
  84     }
  85 
  86 };
  87 
  88 BDLIB_NS_END
  89 #endif /* _BD_SORTEDLIST_H */
