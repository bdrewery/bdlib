  28 #ifndef _BD_THREAD_H
  29 #define _BD_THREAD_H 1
  30 
  31 #include "bdlib.h"
  32 
  33 #if !defined(USE_PTHREAD) && !defined(WIN32)
  34 # define NO_THREADS 1
  35 #endif
  36 
  37 #if defined(USE_PTHREAD)
  38 # include <pthread.h>
  39 typedef void* threadMainRet_t;
  40 typedef pthread_t threadHandle_t;
  41 #elif defined(WIN32)
  42 # include <windows.h>
  43 typedef unsigned int __stdcall threadMainRet_t;
  44 typedef HANDLE threadHandle_t;
  45 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  46 typedef void* threadMainRet_t;
  47 typedef short threadHandle_t;
  48 #endif
  49 
  50 typedef unsigned int threadId_t;
  51 
  52 BDLIB_NS_BEGIN
  53 
  54 #define THREAD_STARTED  0x1
  55 #define THREAD_DETACHED 0x2
  56 
  57 static threadMainRet_t threadMain(void*);
  58 
  59 /**
  60   * @class Thread
  61   * @brief Extendable thread class
  62   */
  63 class Thread {
  64   private:
  65     threadHandle_t handle;
  66     threadId_t id;
  67     short status;
  68     void* param;
  69 
  70   friend threadMainRet_t threadMain(void*);
  71   public:
  72     Thread(void) : handle(0), id(0), status(0), param(nullptr) {};
  73     virtual ~Thread() { stop(); };
  74     int start(void* = nullptr);
  75     void detach(void);
  76     void* wait(void);
  77     void stop(void);
  78     threadId_t getThreadId(void);
  79     static threadId_t getCurrentThreadId(void);
  80     static void sleep(int);
  81   protected: 
  82     virtual void* run(void*)=0;
  83   private:
  84     inline bool isStarted() { return status & THREAD_STARTED; };
  85     inline void setStarted() { status |= THREAD_STARTED; };
  86     inline bool isDetached() { return status & THREAD_DETACHED; };
  87     inline void setDetached() { status |= THREAD_DETACHED; };
  88     /**
  89      * @brief Is the thread started and not detached?
  90      * If status == THREAD_STARTED (1), then the THREAD_DETACHED bit is not set, 
  91      * so this is equivalent to: isStarted() && !isDetached()
  92      */
  93     inline bool shouldDetach() { return status == THREAD_STARTED; };
  94 };
  95 
  96 BDLIB_NS_END
  97 
  98 #endif /* _BD_THREAD_H */
