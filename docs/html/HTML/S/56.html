  41 static threadMainRet_t threadMain(void* object) {
  42   Thread* thread = (Thread*) object;
  43   return (threadMainRet_t) thread->run(thread->param);
  44 }

  46 int Thread::start(void* _param) {
  47   if (!isStarted()) {
  48     param = _param;
  49 #if defined(USE_PTHREAD)
  50     /* Initialize the pthread_attr needed for pthread .. */
  51     pthread_attr_t attr;
  52 
  53     pthread_attr_init(&attr);
  54     /* ... and set the detached state if we are to be detached */
  55     if (isDetached())
  56       pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);  
  57 
  58     /* Attempt to create the thread, and set to started if successfull */
  59     if (pthread_create(&(this->handle), &attr, threadMain, this) == 0)
  60       setStarted();
  61     pthread_attr_destroy(&attr);
  62 /* endif USE_PTHREAD */
  63 #elif defined(WIN32)  
  64     /* Attempt to create the thread */
  65     if (this->handle = (HANDLE)_beginthreadex(NULL, 0, threadMain, this, 0, &(this->id)) {
  66       if (isDetached())
  67         CloseHandle(this->handle);
  68       setStarted();
  69     }
  70 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  71     /* No thread support, pretend the thread started */
  72     setStarted();
  73 #endif 
  74   }
  75   return isStarted();
  76 }

  78 void Thread::detach(void) {
  79   if (shouldDetach()) {
  80 #if defined(USE_PTHREAD)
  81     pthread_detach(this->handle);
  82 #elif defined(WIN32)
  83     CloseHandle(this->handle);
  84 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  85     ; /* nop */
  86 #endif
  87   }
  88   setDetached();
  89 }

  91 void* Thread::wait(void) {
  92   void* _status = NULL;
  93 
  94   if (shouldDetach()) {
  95 #if defined(USE_PTHREAD)
  96     pthread_join(this->handle, &_status);
  97 #elif defined(WIN32)
  98     WaitForSingleObject(this->handle, INFINITE);
  99     GetExitCodeThread(this->handle, (DWORD*) &_status);
 100     CloseHandle(this->handle);
 101 #endif /* WIN32 */
 102     setDetached();
 103   }
 104   return _status;
 105 }

 107 void Thread::stop(void) {
 108   if (shouldDetach()) {
 109 #if defined(USE_PTHREAD)
 110     pthread_cancel(this->handle);
 111     pthread_detach(this->handle);
 112 #elif defined(WIN32)
 113     TerminateThread(this->handle, 0);
 114     CloseHandle(this->handle);
 115 #endif
 116     setDetached();
 117   }
 118 }

 132 void Thread::sleep(int delay) {
 133 #if defined(USE_PTHREAD) || defined(NO_THREADS)
 134   timeval timeout;
 135 
 136   timeout.tv_sec = delay / 1000;
 137   timeout.tv_usec = (delay * 1000) % 1000000;
 138   select(0, (fd_set*) NULL, (fd_set*) NULL, (fd_set*) NULL, &timeout);
 139 #elif defined(WIN32)
 140   ::Sleep(delay);
 141 #endif
 142 }
