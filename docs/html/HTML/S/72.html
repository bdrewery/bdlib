  43 int String::compare(const String& str, size_t n, size_t start) const
  44 {
  45   const size_t my_len = length();
  46   /* Same string? */
  47   if (data() == str.data() && my_len == str.length())
  48     return 0;
  49 
  50   const size_t slen = n ? std::min(str.length(), n) : str.length();
  51   const size_t len = std::min(my_len - start, slen);
  52   const int diff = std::memcmp(begin() + start, str.begin(), len);
  53   if (diff)
  54     return diff;
  55   else if (n)
  56     return std::min(my_len - start, n) - slen;
  57   else
  58     return (my_len - start) - slen;
  59 }

  63 /**
  64  * @brief Insert a cstring at the given index.
  65  * @param string The cstring to be inserted.
  66  * @param pos The index to insert at.
  67  * @param n Up to this many characters will be used from the string.
  68  * @post A buffer is allocated.
  69  * @post If the old buffer was too small, it is enlarged.
  70  * @post The string is inserted at the given index.
  71  */
  72 void String::insert(size_t pos, const char *string, size_t n)
  73 {
  74   if (n == 0) return;
  75   if (pos && !hasIndex(pos-1)) return;
  76   
  77   size_t slen = (n == npos) ? std::strlen(string) : n;
  78 
  79   AboutToModify(length() + slen);
  80   std::memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
  81   std::copy(string, string + slen, Buf() + pos);
  82   addLength(slen);
  83 }

  85 /**
  86  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  87  * @param pos The index to replace at.
  88  * @param string The cstring to replace with.
  89  * @param n How many characters to use from string.
  90  */
  91 void String::replace(size_t pos, const char *string, size_t n)
  92 {
  93   if (n == 0) return;
  94   if (pos && !hasIndex(pos-1)) return;
  95 
  96   size_t slen = (n == npos) ? std::strlen(string) : n;
  97   size_t newlen = pos + slen;
  98   
  99   if (newlen >= length()) {
 100     AboutToModify(newlen);
 101   } else {
 102     newlen = length();
 103     getOwnCopy();
 104   }
 105   std::copy(string, string + slen, Buf() + pos);
 106   setLength(newlen);
 107 }

 109 std::istream& operator>>(std::istream& is, String& string) {
 110   char ch;
 111   string = "";    // empty string, will build one char at-a-time
 112   is >> ch;    // whitespace skipped, first non-white char in ch
 113 
 114   if (!is.fail())
 115   {
 116     do
 117     {
 118       string += ch;
 119       is.get(ch);
 120     } while (!is.fail() && !isspace(ch));
 121 
 122     if (isspace(ch))    // put whitespace back on the stream
 123     {
 124       is.putback(ch);
 125     }
 126   }
 127   return is;
 128 }

 141 /*
 142  * @brief Split a string by a delimiter
 143  * @param delim The delimiter to use
 144  * @param limit Optional param which specified the max number of elements to return
 145  * @note Excess whitespace is trimmed similar to Ruby's split.
 146  */
 147 Array<String> String::split(const String& delim, size_t limit) const {
 148   if (!length()) return Array<String>();
 149   if (limit == 1) {
 150     String list[] = {*this};
 151     return Array<String>(list, 1);
 152   }
 153 
 154   const String space(' ');
 155   Array<String> array;
 156   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 157   String str(*this);
 158 
 159   while (array.size() < limit - 1) {
 160     size_t pos;
 161 
 162     // Trim out left whitespace
 163     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 164 
 165     // All done when there's nothing left
 166     if (!str) break;
 167 
 168     if ((pos = str.find(delim)) == npos)
 169       pos = str.length();
 170     array << str(0, pos);
 171     str += pos + delim.length();
 172   }
 173 
 174   // Add on extra
 175   if (limit != npos) {
 176     // Trim out left whitespace
 177     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 178     array << str;
 179   }
 180 
 181   return array;
 182 }

 202 String& String::chomp() {
 203   if (length() && (*this)[length() - 1] == '\n') --(*this);
 204   if (length() && (*this)[length() - 1] == '\r') --(*this);
 205   return *this;
 206 }

 208 String& String::trim() {
 209   // ltrim
 210   while (length() && isspace((*this)[0])) { ++offset; subLength(1); }
 211   // rtrim
 212   while (length() && isspace((*this)[length() - 1])) subLength(1);

 216 size_t String::find(const String& str) const {
 217   if (length() >= str.length()) {
 218     const size_t last_pos = length() - str.length();
 219     for (size_t pos = 0; pos <= last_pos; ++pos)
 220       if (str[0] == (*this)[pos] && !std::memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 221         return pos;
 222   }
 223   return npos;
 224 }

 226 size_t String::ifind(const String& str) const {
 227   String str_upper(str), this_upper(*this);
 228   std::transform(str.begin(), str.end(), str_upper.begin(), ::toupper);
 229   std::transform(this->begin(), this->end(), this_upper.begin(), ::toupper);
 230   return this_upper.find(str_upper);
 231 }

 240 size_t String::rfind(const String& str, const size_t lpos) const {
 241   if (length() >= str.length()) {
 242     const size_t last_pos = length() - str.length();
 243     for (size_t pos = last_pos; pos + 1 > lpos; --pos)
 244       if (str[0] == (*this)[pos] && !std::memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 245         return pos;
 246   }
 247   return npos;

 250 String String::sub(const String& search, const String& replacement, int limit) const {
 251   String newStr(length()), search_str(*this);
 252   size_t pos = 0;
 253   int cnt = 0;
 254   // Search in our const string so that stuff like \->\\ doesnt become \\\, \\\\ etc.
 255   while ((pos = search_str.find(search)) != npos) {
 256     newStr += search_str(0, pos);
 257     newStr += replacement;
 258     search_str += pos + search.length();
 259     if (limit != -1 && ++cnt == limit) break;
 260   }
 261   // Left over
 262   if (search_str.length())
 263     newStr += search_str;
 264   return newStr;
 265 }

 267 String String::subst(HashTable<String, String> hashes) const {
 268   String newStr(*this);
 269   Array<String> keys(hashes.keys());
 270   Array<String> values(hashes.values());
 271   for (size_t i = 0; i < keys.length(); ++i) {
 272     const String key(keys[i]), value(values[i]);
 273     //::printf("%s -> %s\n", key.c_str(), value.c_str());
 274     newStr = newStr.sub(key, value);
 275   }
 276   return newStr;
 277 }

 279 /**
 280  * @brief Splits off tokens
 281  * @param str The string to split a token off of
 282  * @param delim The delimiter to tokenize with
 283  * This will modify the given string to remove the first token, which is returned.
 284  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 285  */
 286 String newsplit(String& str, char delim)
 287 {
 288   if (!str.length()) return "";
 289   size_t pos = str.find(delim);
 290   if (pos == String::npos)
 291     pos = str.length();
 292 
 293   String first(str(0, pos));
 294 
 295   /* Trim out runs of whitespaces */
 296   if (delim == ' ') {
 297     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 298       ++pos;
 299   }
 300 
 301 //  str = str(pos + 1, str.length() - pos + 1);
 302   str += pos + 1;
 303   return first;
 304 }
