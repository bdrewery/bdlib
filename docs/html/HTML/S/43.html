  35 template <class T>
  36 /**
  37  * @class ArrayRef
  38  * @brief Helps the String and Array classes with reference counting
  39  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  40  */
  41 class ArrayRef {
  42   public:
  43     ~ArrayRef() { FreeBuf(buf); };
  44     mutable size_t size; //Capacity of buffer
  45     mutable T* buf;
  46     mutable int n; //References
  47 
  48     ArrayRef() : size(0), buf(NULL), n(1) {};
  49     /**
  50      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  51      * @param newSize A size that we need to Allocate the buffer to.
  52      * @param offset The offset of the old buffer so we know where to start
  53      * @param sublen The length of the subarray in use
  54      * @pre newSize is > 0 (assumed as size_t is unsigned)
  55      * @post The buffer is at least nsize bytes long.
  56      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  57      */
  58     void Reserve(const size_t newSize, size_t& offset, size_t sublen) const
  59     {
  60       /* Don't new if we already have enough room! */
  61       if (size < newSize) {
  62         size = std::max(size_t(size * 1.5), newSize);
  63 
  64         T *newbuf = AllocBuf(size);
  65 
  66         if (newbuf != buf) {
  67           /* Copy old buffer into new - only copy the subarray */
  68           std::copy(buf + offset, buf + offset + sublen, newbuf);
  69           FreeBuf(buf);
  70           buf = newbuf;
  71           offset = 0;
  72         }
  73       } else if ((size - offset) < newSize) {
  74         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  75         // Shift everything to the beginning and reset the offset
  76         /* Only copy the subarray */
  77         memmove(buf, buf + offset, sublen);
  78         offset = 0;
  79       }
  80     }
  81 
  82     /**
  83      * @brief Allocates a buffer and returns it's address.
  84      * @param bytes The number of bytes to allocate.
  85      * @post A new block of memory is allocated.
  86      * @todo Implement mempool here.
  87      */
  88     inline T* AllocBuf(const size_t bytes) const {
  89       return new T[bytes];
  90     }
  91 
  92     /**
  93      * @brief Free's up the allocated buffer.
  94      * @param p The buffer to be free'd
  95      * @post The buffer is deleted.
  96      * @todo Implement mempool here.
  97      */
  98     inline void FreeBuf(T* p) const {
  99       delete[] p;
 100     }
 101 
 102     /**
 103      * @brief Is this ReferenceCountedArray shared?
 104      */
 105     inline bool isShared() const { return n > 1; };
 106   private:
 107     // No copying allowed
 108     ArrayRef(const ArrayRef&); ///<Block implicit copy constructor
 109     ArrayRef& operator=(const ArrayRef&); ///<Block implicit copy constructor
 110 };

 179     /**
 180      * @brief Detach from the shared reference.
 181      * This is only called when losing the old buffer or when modifying the buffer (and copy-on-write is used)
 182      * @note This does not free the old reference, as it is still in use
 183      */
 184     void doDetach() const {
 185       decRef();

 189       my_hash = 0;
 190     }
 191     /**
 192      * @brief Increment our reference counter.
 193      */
 194     inline int incRef() const { return ++Ref->n; };

 196     /**
 197      * @brief Decrement our reference counter.
 198      */
 199     inline int decRef() const { return --Ref->n; };

 201   protected:
 202     /**
 203      * @brief Set the lengths to the specified length
 204      * @param newLen the new length to set to
 205      */
 206     inline void setLength(size_t newLen) const { sublen = newLen; my_hash = 0; };
 207 
 208     /**
 209      * @sa setLength()
 210      */
 211     inline void addLength(size_t diff) const { sublen += diff; my_hash = 0; };

 214      * @sa setLength()

 216     inline void subLength(size_t diff) const { sublen -= diff; my_hash = 0; };

 233     /**
 234      * @brief The array reference for reference counting

 238   protected:
 239     /**
 240      * Return the real buffer's start point, without accounting for offset. This is used for cleaning the buffer when needed.
 241      */
 242     inline const_pointer real_begin() const { return Ref->buf; };

 249      * This is for subarrays: so we know where the subarray ends.

 254      * Cache of current hash() result. 0 if stale
 255      */
 256     mutable size_t my_hash;

 258   private:
 259     /**
 260      * @brief Detach from the reference
 261      * This is called when the old buffer is no longer needed for this Array.
 262      * ie, operator=() was called.
 263      */
 264     void Detach() {
 265       if (isShared()) {
 266         doDetach();
 267       } else {
 268         setLength(0);
 269         offset = 0;
 270         my_hash = 0;
 271       }
 272     }
 273 
 274     /**
 275      * @brief Free up our reference if we have the last one.
 276      * @post The reference counter is decremented.
 277      * @post If this was the last Reference, it is free'd
 278      * This is only called in ~Array() and operator=(Array&).
 279      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 280      */
 281     inline void CheckDeallocRef() {
 282       if (decRef() < 1)
 283         delete Ref;
 284     }

 286     //    void COW(size_t) const;
 287     /**
 288      * @brief Ensure that our internal buffer is unshared.
 289      * @param n Create/Grow the buffer to this size.
 290      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 291      * @post The internal buffer/data is unshared
 292      * @post The buffer is at least size n.
 293      * @post The buffer is deep copied to a new buffer.
 294      *
 295      * Ensure that our internal buffer is unshared.
 296      * If needed, performs a deep copy into a new buffer (COW).
 297      * Also take a hint size n of the new ReferenceCountedArray's size as to avoid needless copying/allocing.
 298      */
 299     void COW(size_t n) const {
 300       const_pointer oldBuf = constBuf();
 301       size_t oldLength = length();
 302 
 303       doDetach(); //Detach from the shared reference
 304       Reserve( std::max(oldLength, n) ); //Will set capacity()/size
 305       std::copy(oldBuf, oldBuf + oldLength, Buf());
 306       setLength(oldLength);
 307     }
 308 
 309   protected:
 310     /**
 311      * @brief Force COW if needed
 312      * @post The array is no longer shared, if it was.
 313      */
 314     inline void getOwnCopy() const { AboutToModify(length()); };
 315 
 316     /**
 317      * @brief Warn the reference counting that it may need to COW
 318      * @post The buffer is detached/COW, and possibly larger
 319      * @todo If the buffer is shared and needs to shrink, the sublen should just be decreased.
 320      */
 321     inline void AboutToModify(size_t n) const {
 322       my_hash = 0;
 323       if (isShared())
 324         COW(n); // Clears the offset
 325       else {
 326         Reserve(n);
 327         /* Shift the offset away */
 328       }
 329     }
 330   public:
 331     ReferenceCountedArray() : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {};
 332     ReferenceCountedArray(const ReferenceCountedArray& rca) : Ref(rca.Ref), offset(rca.offset), sublen(rca.sublen), my_hash(rca.my_hash) { incRef(); };
 333     /**
 334      * @brief Create an empty container with at least the specified bytes in size.
 335      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 336      * @post This ReferenceCountedArray's memory will also never be shrunk.
 337      * @post A buffer has been created.
 338      *
 339      * The idea behind this is that if a specific size was asked for, the buffer is like
 340      * a char buf[N];
 341      */
 342     explicit ReferenceCountedArray(const size_t newSize) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {
 343       if (newSize <= 0) return;
 344       Reserve(newSize);
 345     };
 346     /**
 347      * @brief Create a container filled with n copies of the given value.
 348      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 349      * @param value The value to populate the array with
 350      * @post This ReferenceCountedArray's memory will also never be shrunk.
 351      * @post A buffer has been created.
 352      *
 353      */
 354     ReferenceCountedArray(const size_t newSize, const value_type value) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {
 355       if (newSize <= 0) return;
 356       Reserve(newSize);
 357 
 358       for (size_t i = 0; i < newSize; ++i)
 359         *(Buf(i)) = value;
 360       this->setLength(newSize);
 361     }

 429     /**
 430      * @brief Resize the array to the given length.
 431      * @param len The length to resize to.
 432      * @param value The optional parameter to fill the space with if the array is expanded
 433      */
 434     void resize(size_t len, value_type value = value_type()) {
 435       if (len < this->length()) {
 436         this->subLength(this->length() - len);
 437       } else {
 438         this->AboutToModify(len);
 439         for (size_t i = 0; i < (len - this->length()); ++i)
 440           *(Buf(this->length() + i)) = value;
 441         this->addLength(len - this->length());
 442       }
 443     }

 483     typedef Hash<value_type> HashType;

 485    /**
 486      * @brief Return a hash of every element in the array. Cache result as well.
 487      * @note DJB's hash function
 488      */
 489     virtual size_t hash() const {
 490       if (my_hash != 0) return my_hash;
 491       HashType hasher;
 492       size_t _hash = 5381;
 493 
 494       for(size_t i = 0; i < this->length(); ++i)
 495         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 496       return (my_hash = (_hash & 0x7FFFFFFF));
 497     }

 618     /**
 619      * @brief Return a new array from a subarray
 620      * @return a new ReferenceCountedArray
 621      * @param start The offset to begin the subarray from (indexed from 0)
 622      * @param len The length of the subarray to return
 623      * The returned slice is a reference to the original array until modified.
 624      */
 625     void slice(int start, int len = -1) {
 626       if (len == -1) len = int(length()) - start;
 627       // Start is after the end, set us to an empty array
 628       if (start >= static_cast<signed>(length())) {
 629         offset = length();
 630         setLength(0);
 631         return;
 632       }
 633 
 634       //Count backwards from the end
 635       if (start < 0)
 636         start = length() + start;
 637       // Start was before the beginning, just reset to the beginning
 638       if (start < 0)
 639         start = 0;
 640 
 641       offset += start;
 642       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 643       if (start + len >= static_cast<signed>(length()))
 644         len = length() - start;
 645       // If the length is negative, stop from counting backwards from the end
 646       else if (len < 0)
 647         len = length() - start + len;
 648       setLength(len);
 649     }

 669     /**
 670      * @brief Inserts a ReferenceCountedArray object into our buffer
 671      * @param pos The index to insert at.
 672      * @param rca The rca to insert.
 673      * @param n The length to insert.
 674      * @post The buffer contains n items from rca inserted at index pos.
 675      */
 676     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 677       if (n == 0) return;
 678       if (pos && !hasIndex(pos-1)) return;
 679 
 680       size_t slen = rca.length();
 681 
 682       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 683       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 684         *this = rca;
 685         return;
 686       }
 687 
 688       if (n == npos || n > slen)
 689         n = slen;
 690       slen -= slen - n;
 691       AboutToModify(length() + slen);
 692       memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
 693       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 694       addLength(slen);
 695     }
 696 
 697     /**
 698      * @brief Insert an item at the given index.
 699      * @param pos The index to insert at.
 700      * @param item The item to be inserted.
 701      * @post A buffer is allocated.
 702      * @post If the old buffer was too small, it is enlarged.
 703      * @post The item is inserted at the given index.
 704      */
 705     void insert(size_t pos, const_reference item)
 706     {
 707       if (pos && !hasIndex(pos-1)) return;
 708 
 709       AboutToModify(length() + 1);
 710       memmove(Buf() + pos + 1, Buf() + pos, length() - pos);
 711       *(Buf(pos)) = item;
 712       addLength(1);
 713     }
 714 
 715     /**
 716      * @brief Replace the given index with the given item.
 717      * @param pos The index to replace.
 718      * @param item The item to replace with.
 719      * @post The given index has been replaced.
 720      * @post COW is done if needed.
 721      */
 722     void replace(size_t pos, const_reference item) {
 723       if (pos && !hasIndex(pos-1)) return;
 724 
 725       getOwnCopy();
 726       *(Buf(pos)) = item;
 727     }
 728 
 729     /**
 730      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 731      * @param pos The index to replace at.
 732      * @param rca The ReferenceCountedArray object to replace with.
 733      * @param n The number of characters to use for the replace.
 734      */
 735     void replace(size_t pos, const ReferenceCountedArray &rca, size_t n = npos) {
 736       if (n == 0) return;
 737       if (pos && !hasIndex(pos-1)) return;
 738 
 739       size_t slen = rca.length();
 740 
 741       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 742       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 743         *this = rca;
 744         return;
 745       }
 746 
 747       if (n == npos || n > slen)
 748         n = slen;
 749       slen -= slen - n;
 750 
 751       size_t newlen = pos + slen;
 752 
 753       if (newlen >= length()) {
 754         AboutToModify(newlen);
 755       } else {
 756         newlen = length();
 757         getOwnCopy();
 758       }
 759       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 760       setLength(newlen);
 761     }
