  43 template <class T, class Allocator = std::allocator<T> >
  44 /**
  45  * @class ArrayRef
  46  * @brief Helps the String and Array classes with reference counting
  47  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  48  */
  49 class ArrayRef {
  50   public:
  51     typedef typename Allocator::pointer iterator;
  52     mutable Allocator alloc;
  53     mutable size_t size; //Capacity of buffer
  54     mutable iterator buf;
  55     mutable int n; //References
  56 
  57     ArrayRef(const Allocator& allocator = Allocator()) : alloc(allocator), size(0), buf(NULL), n(1) {};
  58     ~ArrayRef() {
  59       if (buf) {
  60         FreeBuf(buf);
  61       }
  62     };
  63     /**
  64      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  65      * @param newSize A size that we need to Allocate the buffer to.
  66      * @param scaling_factor How much to multiply the size by to help avoid later resizing
  67      * @param offset The offset of the old buffer so we know where to start
  68      * @param sublen The length of the subarray in use
  69      * @pre newSize is > 0 (assumed as size_t is unsigned)
  70      * @post The buffer is at least nsize bytes long.
  71      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  72      */
  73     void Reserve(size_t newSize, double scaling_factor, size_t& offset, size_t sublen) const
  74     {
  75       /* Don't new if we already have enough room! */
  76       if (size < newSize) {
  77         newSize = std::max(size_t(size * scaling_factor), newSize);
  78 
  79         iterator newbuf = alloc.allocate(newSize, buf);
  80 
  81         if (newbuf != buf) {
  82           // Initialize new memory
  83           std::uninitialized_fill(newbuf, newbuf + newSize, T());
  84           if (buf) {
  85             /* Copy old buffer into new - only copy the subarray */
  86             std::copy(buf + offset, buf + offset + sublen, newbuf);
  87             FreeBuf(buf);
  88           }
  89           buf = newbuf;
  90           size = newSize;
  91           offset = 0;
  92         }
  93       } else if ((size - offset) < newSize) {
  94         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  95         // Shift everything to the beginning and reset the offset
  96         /* Only copy the subarray */
  97         memmove(static_cast<void*>(buf), static_cast<void*>(buf + offset), sublen);
  98         offset = 0;
  99       }
 100     }
 101 
 102     /**
 103      * @brief Free's up the allocated buffer.
 104      * @param p The buffer to be free'd
 105      * @post The buffer is deleted.
 106      * @todo Implement mempool here.
 107      */
 108     inline void FreeBuf(iterator p) const {
 109       for (iterator i = p; i != p + size; ++i) {
 110         alloc.destroy(i);
 111       }
 112       alloc.deallocate(p, size);
 113     }
 114 
 115     /**
 116      * @brief Is this ReferenceCountedArray shared?
 117      */
 118     inline bool isShared() const { return n > 1; };
 119   private:
 120     // No copying allowed
 121 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 122     ArrayRef(const ArrayRef&) = delete;
 123     ArrayRef& operator=(const ArrayRef&) = delete;
 124 #else
 125     ArrayRef(const ArrayRef&);
 126     ArrayRef& operator=(const ArrayRef&);
 127 #endif
 128 };

 131 /**
 132  * @class Slice
 133  * @brief Safe subarray reading and writing.
 134  * This class should be optimized away and fully inlined such that:
 135  * ReferenceCountedArray s("look over there");
 136  * s(0, 4) = "blah"';
 137  * Should be rewritten as:
 138  * s.replace(0, "blah", 4);

 140 class Slice {
 141   private:
 142     T& rca;
 143     int start;
 144     int len;
 145 
 146 #ifdef __GXX_EXPERIMENTAL_CXX0X__

 152   public:

 157      * @brief return a new (const) slice
 158      */

 180     }

 201 template <class T, class Allocator = std::allocator<T> >

 203  * @class ReferenceCountedArray

 206 class ReferenceCountedArray : public ReferenceCountedArrayBase {
 207   public:
 208     typedef T value_type;
 209 
 210     typedef typename Allocator::size_type             size_type;
 211     typedef typename Allocator::difference_type       difference_type;
 212     typedef typename Allocator::pointer               pointer;
 213     typedef typename Allocator::const_pointer         const_pointer;
 214     typedef typename Allocator::reference             reference;
 215     typedef typename Allocator::const_reference       const_reference;
 216     typedef typename Allocator::pointer               iterator;
 217     typedef typename Allocator::const_pointer         const_iterator;
 218     typedef std::random_access_iterator_tag           iterator_category;
 219     typedef std::reverse_iterator<const_iterator>     const_reverse_iterator;
 220     typedef std::reverse_iterator<iterator>           reverse_iterator;
 221 
 222   private:
 223     /**
 224      * @brief Detach from the shared reference.
 225      * This is only called when losing the old buffer or when modifying the buffer (and copy-on-write is used)
 226      * @note This does not free the old reference, as it is still in use
 227      */
 228     void doDetach() const {
 229       decRef();
 230       Ref = NULL;
 231       sublen = 0;
 232       offset = 0;
 233       my_hash = 0;
 234     }
 235     /**
 236      * @brief Increment our reference counter.
 237      */
 238     inline int incRef() const { return Ref ? ++Ref->n : 0; };
 239 
 240     /**
 241      * @brief Decrement our reference counter.
 242      */
 243     inline int decRef() const { return Ref ? --Ref->n : 0; };
 244 
 245   protected:
 246     /**
 247      * @brief Set the lengths to the specified length
 248      * @param newLen the new length to set to
 249      */
 250     inline void setLength(size_t newLen) const { sublen = newLen; my_hash = 0; };
 251 
 252     /**
 253      * @sa setLength()
 254      */
 255     inline void addLength(size_t diff) const { sublen += diff; my_hash = 0; };
 256 
 257     /**
 258      * @sa setLength()
 259      */
 260     inline void subLength(size_t diff) const { sublen -= diff; my_hash = 0; };
 261 
 262     /**
 263      * @brief Mutable Ref->buf+offset reference for use internally
 264      */
 265     //pointer mdata() const { return Buf() + offset; };

 267     /**
 268      * @brief Mutable Ref->buf reference for use internally
 269      */
 270     inline pointer Buf(size_t pos = 0) const { return Ref ? &Ref->buf[offset + pos] : NULL; };
 271 
 272     /**
 273      * @brief Ref->buf reference for use internally
 274      */
 275     inline const_pointer constBuf(size_t pos = 0) const { return Buf(pos); };
 276   private:
 277     Allocator alloc;
 278     /**
 279      * @brief The array reference for reference counting
 280      * This is mutable so that Ref->n can be modified, which really is mutable
 281      */
 282     mutable ArrayRef<value_type, Allocator> *Ref;
 283   protected:
 284     /**
 285      * Return the real buffer's start point, without accounting for offset. This is used for cleaning the buffer when needed.
 286      */
 287     inline const_pointer real_begin() const { return Ref ? Ref->buf : NULL; };
 288 
 289     /**
 290      * This is for subarrays: so we know where the subarray starts.
 291      */
 292     mutable size_t offset;
 293     /**
 294      * This is for subarrays: so we know where the subarray ends.
 295      */
 296     mutable size_t sublen;
 297 
 298     /**
 299      * Cache of current hash() result. 0 if stale
 300      */
 301     mutable size_t my_hash;
 302 
 303   private:
 304     /**
 305      * @brief Detach from the reference
 306      * This is called when the old buffer is no longer needed.
 307      * ie, operator=() was called.
 308      */
 309     void Detach() {
 310       if (isShared()) {
 311         doDetach();
 312       } else {
 313         setLength(0);
 314         offset = 0;
 315         my_hash = 0;
 316       }
 317     }
 318 
 319     /**
 320      * @brief Free up our reference if we have the last one.
 321      * @post The reference counter is decremented.
 322      * @post If this was the last Reference, it is free'd
 323      * This is only called in ~Array() and operator=(Array&).
 324      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 325      */
 326     inline void CheckDeallocRef() {
 327       if (Ref && decRef() < 1) {
 328         delete Ref;
 329         Ref = NULL;
 330       }
 331     }

 333     //    void COW(size_t) const;
 334     /**
 335      * @brief Ensure that our internal buffer is unshared.
 336      * @param n Create/Grow the buffer to this size.
 337      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 338      * @post The internal buffer/data is unshared
 339      * @post The buffer is at least size n.
 340      * @post The buffer is deep copied to a new buffer.
 341      *
 342      * Ensure that our internal buffer is unshared.
 343      * If needed, performs a deep copy into a new buffer (COW).
 344      * Also take a hint size n of the new ReferenceCountedArray's size as to avoid needless copying/allocing.
 345      */
 346     void COW(size_t n) const {
 347       const_pointer oldBuf = constBuf();
 348       const size_t oldLength = length();
 349 
 350       doDetach(); //Detach from the shared reference
 351       Reserve(std::max(oldLength, n), _rca_cow_scaling_factor); //Will set capacity()/size
 352       std::copy(oldBuf, oldBuf + oldLength, Buf());
 353       setLength(oldLength);
 354     }
 355 
 356   protected:
 357     /**
 358      * @brief Force COW if needed
 359      * @post The array is no longer shared, if it was.
 360      */
 361     inline void getOwnCopy() const { AboutToModify(length()); };
 362 
 363     /**
 364      * @brief Warn the reference counting that it may need to COW
 365      * @post The buffer is detached/COW, and possibly larger
 366      */
 367     inline void AboutToModify(size_t n) const {
 368       my_hash = 0;
 369       if (isShared())
 370         COW(n); // Clears the offset
 371       else {
 372         Reserve(n, _rca_cow_scaling_factor);
 373         /* Shift the offset away */
 374       }
 375     }
 376   public:
 377     ReferenceCountedArray(const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(NULL), offset(0), sublen(0), my_hash(0) {
 378     };
 379     ReferenceCountedArray(const ReferenceCountedArray& rca) : ReferenceCountedArrayBase(), alloc(rca.alloc), Ref(rca.Ref), offset(rca.offset), sublen(rca.sublen), my_hash(rca.my_hash) { incRef(); };
 380 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 381     ReferenceCountedArray(ReferenceCountedArray&& rca) : ReferenceCountedArrayBase(), alloc(std::move(rca.alloc)), Ref(std::move(rca.Ref)), offset(std::move(rca.offset)), sublen(std::move(rca.sublen)), my_hash(std::move(rca.my_hash)) {
 382       rca.alloc = Allocator();
 383       rca.Ref = NULL;
 384       rca.offset = 0;
 385       rca.sublen = 0;
 386       rca.my_hash = 0;
 387     };

 389     /**
 390      * @brief Create an empty container with at least the specified bytes in size.
 391      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 392      * @post This ReferenceCountedArray's memory will also never be shrunk.
 393      * @post A buffer has been created.
 394      *
 395      * The idea behind this is that if a specific size was asked for, the buffer is like

 399       if (newSize) {

 403     /**
 404      * @brief Create a container filled with n copies of the given value.
 405      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 406      * @param value The value to populate the array with
 407      * @post This ReferenceCountedArray's memory will also never be shrunk.
 408      * @post A buffer has been created.
 409      *
 410      */
 411     ReferenceCountedArray(const size_t newSize, const value_type value, const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(NULL), offset(0), sublen(0), my_hash(0) {
 412       if (newSize) {
 413         Reserve(newSize);
 414 
 415         for (size_t i = 0; i < newSize; ++i) {

 419       }

 633    /**
 634      * @brief Return a hash of every element in the array. Cache result as well.
 635      * @note DJB's hash function
 636      */
 637     virtual size_t hash() const {
 638       if (my_hash != 0) return my_hash;
 639       HashType hasher;
 640       size_t _hash = 5381;
 641 
 642       for(size_t i = 0; i < this->length(); ++i)
 643         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 644       return (my_hash = (_hash & 0x7FFFFFFF));
 645     }

 672      * @brief Checks if the buffer has the given index or not.
 673      * @return Boolean true/false as to whether or not index exists.
 674      * @param pos Index to check.
 675      */
 676     inline bool hasIndex(size_t pos) const {
 677       if (pos >= (offset + length())) {
 678 #ifdef DEBUG
 679         ::printf("ATTEMPT TO ACCESS INDEX %zu/%zu\n", pos, offset + length());
 680 #endif
 681         return 0;
 682       }
 683       return (pos < length());
 684     };

 706     /**
 707      * @class Cref
 708      * @brief Safe element reading and writing.
 709      * This class should be optimized away and fully inlined such that:
 710      * ReferenceCountedArray s;
 711      * s[0] = 'a';
 712      * Should be rewritten as:
 713      * s.write(0, 'a');
 714      */
 715     class Cref {
 716       private:
 717         friend class ReferenceCountedArray;
 718         ReferenceCountedArray& rca;
 719         size_t k;
 720 
 721         /**
 722          * @brief Used by Cref operator[]
 723          */
 724         Cref(ReferenceCountedArray& _rca, size_t pos) : rca(_rca), k(pos) {};
 725 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 726         Cref() = delete;
 727 #else
 728         Cref();
 729 #endif
 730 
 731       public:
 732         Cref(const Cref& cref) : rca(cref.rca), k(cref.k) {};
 733         inline Cref& operator=(const Cref& cref) {
 734           (*this) = value_type(cref);
 735           return (*this);
 736         }
 737 
 738         public:
 739         /**
 740          * @sa ReferenceCountedArray::operator[]
 741          */
 742         inline operator value_type() const { return rca.read(k); };
 743 
 744         /**
 745          * Stroustrup shows using this as void with no return value, but that breaks chaining a[n] = b[n] = 'b';
 746          */
 747         inline Cref& operator=(value_type c) {
 748           rca.write(k, c);
 749           return (*this);
 750         };
 751     };

 754      * @brief Returns 'Cref' class for safe (cow) writing.

 759     /**
 760      * @brief Returns the character at the given index.
 761      * @return The character at the given index.
 762      * @param pos Index to return.
 763      * @pre The index must exist.
 764      * @sa operator[]()
 765      * @todo Perhaps this should throw an exception if out of range?
 766      */
 767     inline value_type at(size_t pos) const { return hasIndex(pos) ? (*this)[pos] : 0; };
 768 
 769     /**
 770      * @brief Return a new array from a subarray
 771      * @return a new ReferenceCountedArray
 772      * @param start The offset to begin the subarray from (indexed from 0)
 773      * @param len The length of the subarray to return
 774      * The returned slice is a reference to the original array until modified.
 775      */
 776     void slice(int start, int len = -1) {
 777       if (len == -1) len = int(length()) - start;
 778       // Start is after the end, set us to an empty array
 779       if (start >= static_cast<signed>(length())) {
 780         offset = length();
 781         setLength(0);
 782         return;
 783       }
 784 
 785       //Count backwards from the end
 786       if (start < 0)
 787         start = length() + start;
 788       // Start was before the beginning, just reset to the beginning
 789       if (start < 0)
 790         start = 0;
 791 
 792       offset += start;
 793       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 794       if (start + len >= static_cast<signed>(length()))
 795         len = length() - start;
 796       // If the length is negative, stop from counting backwards from the end
 797       else if (len < 0)
 798         len = length() - start + len;
 799       setLength(len);
 800     }

 802     /**

 812      * @param rca The rca to be appended.
 813      * @param n How many characters to copy from the ReferenceCountedArray object.
 814      * @post The buffer is allocated.
 815      * This is the same as inserting the rca at the end of the buffer.

 820     /**
 821      * @brief Inserts a ReferenceCountedArray object into our buffer
 822      * @param pos The index to insert at.
 823      * @param rca The rca to insert.
 824      * @param n The length to insert.
 825      * @post The buffer contains n items from rca inserted at index pos.
 826      */
 827     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 828       if (n == 0) return;
 829       if (pos && !hasIndex(pos-1)) return;
 830 
 831       size_t slen = rca.length();
 832 
 833       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 834       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 835         *this = rca;
 836         return;
 837       }
 838 
 839       if (n == npos || n > slen)
 840         n = slen;
 841       slen -= slen - n;
 842       AboutToModify(length() + slen);
 843       memmove(static_cast<void*>(Buf() + pos + slen), static_cast<void*>(Buf() + pos), length() - pos);
 844       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 845       addLength(slen);
 846     }

 848     /**
 849      * @brief Insert an item at the given index.
 850      * @param pos The index to insert at.
 851      * @param item The item to be inserted.
 852      * @post A buffer is allocated.
 853      * @post If the old buffer was too small, it is enlarged.
 854      * @post The item is inserted at the given index.
 855      */
 856     void insert(size_t pos, const_reference item)
 857     {
 858       if (pos && !hasIndex(pos-1)) return;
 859 
 860       AboutToModify(length() + 1);
 861       memmove(static_cast<void*>(Buf() + pos + 1), static_cast<void*>(Buf() + pos), length() - pos);
 862       *(Buf(pos)) = item;
 863       addLength(1);
 864     }

 866     /**
 867      * @brief Replace the given index with the given item.
 868      * @param pos The index to replace.
 869      * @param item The item to replace with.
 870      * @post The given index has been replaced.
 871      * @post COW is done if needed.
 872      */
 873     void replace(size_t pos, const_reference item) {
 874       if (pos && !hasIndex(pos-1)) return;
 875 
 876       getOwnCopy();
 877       *(Buf(pos)) = item;
 878     }

 880     /**
 881      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 882      * @param pos The index to replace at.
 883      * @param rca The ReferenceCountedArray object to replace with.
 884      * @param n The number of characters to use for the replace.
 885      */
 886     void replace(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 887       if (n == 0) return;
 888       if (pos && !hasIndex(pos-1)) return;
 889 
 890       size_t slen = rca.length();
 891 
 892       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 893       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 894         *this = rca;
 895         return;
 896       }
 897 
 898       if (n == npos || n > slen)
 899         n = slen;
 900       slen -= slen - n;
 901 
 902       size_t newlen = pos + slen;
 903 
 904       if (newlen >= length()) {
 905         AboutToModify(newlen);
 906       } else {
 907         newlen = length();
 908         getOwnCopy();
 909       }
 910       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 911       setLength(newlen);
 912     }
