  36 template <class T>
  37 /**
  38  * @class ArrayRef
  39  * @brief Helps the String and Array classes with reference counting
  40  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  41  */
  42 class ArrayRef {
  43   public:
  44     ~ArrayRef() { FreeBuf(buf); };
  45     mutable size_t size; //Capacity of buffer
  46     mutable T* buf;
  47     mutable int n; //References
  48 
  49     ArrayRef() : size(0), buf(NULL), n(1) {};
  50     /**
  51      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  52      * @param newSize A size that we need to Allocate the buffer to.
  53      * @param offset The offset of the old buffer so we know where to start
  54      * @param sublen The length of the subarray in use
  55      * @pre newSize is > 0 (assumed as size_t is unsigned)
  56      * @post The buffer is at least nsize bytes long.
  57      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  58      */
  59     void Reserve(const size_t newSize, size_t& offset, size_t sublen) const
  60     {
  61       /* Don't new if we already have enough room! */
  62       if (size < newSize) {
  63         size = std::max(size_t(size * 1.5), newSize);
  64 
  65         T *newbuf = AllocBuf(size);
  66 
  67         if (newbuf != buf) {
  68           /* Copy old buffer into new - only copy the subarray */
  69           std::copy(buf + offset, buf + offset + sublen, newbuf);
  70           FreeBuf(buf);
  71           buf = newbuf;
  72           offset = 0;
  73         }
  74       } else if ((size - offset) < newSize) {
  75         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  76         // Shift everything to the beginning and reset the offset
  77         /* Only copy the subarray */
  78         memmove(buf, buf + offset, sublen);
  79         offset = 0;
  80       }
  81     }
  82 
  83     /**
  84      * @brief Allocates a buffer and returns it's address.
  85      * @param bytes The number of bytes to allocate.
  86      * @post A new block of memory is allocated.
  87      * @todo Implement mempool here.
  88      */
  89     inline T* AllocBuf(const size_t bytes) const {
  90       return new T[bytes];
  91     }
  92 
  93     /**
  94      * @brief Free's up the allocated buffer.
  95      * @param p The buffer to be free'd
  96      * @post The buffer is deleted.
  97      * @todo Implement mempool here.
  98      */
  99     inline void FreeBuf(T* p) const {
 100       delete[] p;
 101     }
 102 
 103     /**
 104      * @brief Is this ReferenceCountedArray shared?
 105      */
 106     inline bool isShared() const { return n > 1; };
 107   private:
 108     // No copying allowed
 109     ArrayRef(const ArrayRef&); ///<Block implicit copy constructor
 110     ArrayRef& operator=(const ArrayRef&); ///<Block implicit copy constructor
 111 };

 270   private:
 271     /**
 272      * @brief Detach from the reference
 273      * This is called when the old buffer is no longer needed.
 274      * ie, operator=() was called.
 275      */
 276     void Detach() {
 277       if (isShared()) {
 278         doDetach();
 279       } else {
 280         setLength(0);
 281         offset = 0;
 282         my_hash = 0;
 283       }
 284     }

 286     /**
 287      * @brief Free up our reference if we have the last one.
 288      * @post The reference counter is decremented.
 289      * @post If this was the last Reference, it is free'd
 290      * This is only called in ~Array() and operator=(Array&).
 291      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 292      */
 293     inline void CheckDeallocRef() {
 294       if (decRef() < 1)
 295         delete Ref;
 296     }

 298     //    void COW(size_t) const;
 299     /**
 300      * @brief Ensure that our internal buffer is unshared.
 301      * @param n Create/Grow the buffer to this size.
 302      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 303      * @post The internal buffer/data is unshared
 304      * @post The buffer is at least size n.
 305      * @post The buffer is deep copied to a new buffer.
 306      *
 307      * Ensure that our internal buffer is unshared.
 308      * If needed, performs a deep copy into a new buffer (COW).
 309      * Also take a hint size n of the new ReferenceCountedArray's size as to avoid needless copying/allocing.
 310      */
 311     void COW(size_t n) const {
 312       const_pointer oldBuf = constBuf();
 313       size_t oldLength = length();
 314 
 315       doDetach(); //Detach from the shared reference
 316       Reserve( std::max(oldLength, n) ); //Will set capacity()/size
 317       std::copy(oldBuf, oldBuf + oldLength, Buf());
 318       setLength(oldLength);
 319     }
 320 
 321   protected:
 322     /**
 323      * @brief Force COW if needed
 324      * @post The array is no longer shared, if it was.
 325      */
 326     inline void getOwnCopy() const { AboutToModify(length()); };
 327 
 328     /**
 329      * @brief Warn the reference counting that it may need to COW
 330      * @post The buffer is detached/COW, and possibly larger
 331      * @todo If the buffer is shared and needs to shrink, the sublen should just be decreased.
 332      */
 333     inline void AboutToModify(size_t n) const {
 334       my_hash = 0;
 335       if (isShared())
 336         COW(n); // Clears the offset
 337       else {
 338         Reserve(n);
 339         /* Shift the offset away */
 340       }
 341     }

 343     ReferenceCountedArray() : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {};
 344     ReferenceCountedArray(const ReferenceCountedArray& rca) : Ref(rca.Ref), offset(rca.offset), sublen(rca.sublen), my_hash(rca.my_hash) { incRef(); };
 345     /**
 346      * @brief Create an empty container with at least the specified bytes in size.
 347      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 348      * @post This ReferenceCountedArray's memory will also never be shrunk.
 349      * @post A buffer has been created.
 350      *
 351      * The idea behind this is that if a specific size was asked for, the buffer is like
 352      * a char buf[N];
 353      */
 354     explicit ReferenceCountedArray(const size_t newSize) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {
 355       if (newSize <= 0) return;

 357     };
 358     /**
 359      * @brief Create a container filled with n copies of the given value.
 360      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 361      * @param value The value to populate the array with
 362      * @post This ReferenceCountedArray's memory will also never be shrunk.
 363      * @post A buffer has been created.
 364      *
 365      */
 366     ReferenceCountedArray(const size_t newSize, const value_type value) : Ref(new ArrayRef<value_type>()), offset(0), sublen(0), my_hash(0) {
 367       if (newSize <= 0) return;
 368       Reserve(newSize);

 371         *(Buf(i)) = value;
 372       this->setLength(newSize);
 373     }

 447       if (len < this->length()) {

 536     typedef Hash<value_type> HashType;

 538    /**
 539      * @brief Return a hash of every element in the array. Cache result as well.
 540      * @note DJB's hash function
 541      */
 542     virtual size_t hash() const {
 543       if (my_hash != 0) return my_hash;
 544       HashType hasher;
 545       size_t _hash = 5381;
 546 
 547       for(size_t i = 0; i < this->length(); ++i)
 548         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 549       return (my_hash = (_hash & 0x7FFFFFFF));
 550     }

 671     /**
 672      * @brief Return a new array from a subarray
 673      * @return a new ReferenceCountedArray
 674      * @param start The offset to begin the subarray from (indexed from 0)
 675      * @param len The length of the subarray to return
 676      * The returned slice is a reference to the original array until modified.
 677      */
 678     void slice(int start, int len = -1) {
 679       if (len == -1) len = int(length()) - start;
 680       // Start is after the end, set us to an empty array
 681       if (start >= static_cast<signed>(length())) {
 682         offset = length();
 683         setLength(0);
 684         return;
 685       }
 686 
 687       //Count backwards from the end
 688       if (start < 0)
 689         start = length() + start;
 690       // Start was before the beginning, just reset to the beginning
 691       if (start < 0)
 692         start = 0;
 693 
 694       offset += start;
 695       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 696       if (start + len >= static_cast<signed>(length()))
 697         len = length() - start;
 698       // If the length is negative, stop from counting backwards from the end
 699       else if (len < 0)
 700         len = length() - start + len;
 701       setLength(len);
 702     }
 703 
 704     /**
 705      * @brief Appends one item to end of buffer.
 706      * @param item The item to be appended.
 707      * @post The buffer is allocated.
 708      * @post The item is appended at the end of the buffer.
 709      * This is the same as inserting the item at the end of the buffer.
 710      */
 711     inline void append(const_reference item) { insert(length(), item); };
 712     /**
 713      * @brief Appends given rca to the end of buffer
 714      * @param rca The rca to be appended.
 715      * @param n How many characters to copy from the ReferenceCountedArray object.
 716      * @post The buffer is allocated.
 717      * This is the same as inserting the rca at the end of the buffer.
 718      */
 719     inline void append(const ReferenceCountedArray& rca, size_t n = npos) { insert(length(), rca, n); };

 722     /**
 723      * @brief Inserts a ReferenceCountedArray object into our buffer
 724      * @param pos The index to insert at.
 725      * @param rca The rca to insert.
 726      * @param n The length to insert.
 727      * @post The buffer contains n items from rca inserted at index pos.
 728      */
 729     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 730       if (n == 0) return;
 731       if (pos && !hasIndex(pos-1)) return;
 732 
 733       size_t slen = rca.length();
 734 
 735       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 736       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 737         *this = rca;
 738         return;
 739       }
 740 
 741       if (n == npos || n > slen)
 742         n = slen;
 743       slen -= slen - n;
 744       AboutToModify(length() + slen);
 745       memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
 746       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 747       addLength(slen);
 748     }

 750     /**
 751      * @brief Insert an item at the given index.
 752      * @param pos The index to insert at.
 753      * @param item The item to be inserted.
 754      * @post A buffer is allocated.
 755      * @post If the old buffer was too small, it is enlarged.
 756      * @post The item is inserted at the given index.
 757      */
 758     void insert(size_t pos, const_reference item)
 759     {
 760       if (pos && !hasIndex(pos-1)) return;
 761 
 762       AboutToModify(length() + 1);
 763       memmove(Buf() + pos + 1, Buf() + pos, length() - pos);
 764       *(Buf(pos)) = item;
 765       addLength(1);
 766     }

 768     /**
 769      * @brief Replace the given index with the given item.
 770      * @param pos The index to replace.
 771      * @param item The item to replace with.
 772      * @post The given index has been replaced.
 773      * @post COW is done if needed.
 774      */
 775     void replace(size_t pos, const_reference item) {
 776       if (pos && !hasIndex(pos-1)) return;
 777 
 778       getOwnCopy();
 779       *(Buf(pos)) = item;
 780     }
 781 
 782     /**
 783      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 784      * @param pos The index to replace at.
 785      * @param rca The ReferenceCountedArray object to replace with.
 786      * @param n The number of characters to use for the replace.
 787      */
 788     void replace(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 789       if (n == 0) return;
 790       if (pos && !hasIndex(pos-1)) return;
 791 
 792       size_t slen = rca.length();
 793 
 794       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 795       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 796         *this = rca;
 797         return;
 798       }
 799 
 800       if (n == npos || n > slen)
 801         n = slen;
 802       slen -= slen - n;
 803 
 804       size_t newlen = pos + slen;
 805 
 806       if (newlen >= length()) {
 807         AboutToModify(newlen);
 808       } else {
 809         newlen = length();
 810         getOwnCopy();
 811       }
 812       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 813       setLength(newlen);
 814     }
