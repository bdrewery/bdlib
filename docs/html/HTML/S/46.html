  43 int String::compare(const String& str, size_t n, size_t start) const
  44 {
  45   /* Same string? */
  46   if (cbegin() == str.cbegin() && length() == str.length())
  47     return 0;
  48   if (start != 0)
  49     validateIndex(start);
  50 
  51   const size_t slen = n ? std::min(str.length(), n) : str.length();
  52   const size_t len = std::min(length() - start, slen);
  53   const int diff = std::memcmp(begin() + start, str.begin(), len);
  54   if (diff)
  55     return diff;
  56   else if (n)
  57     return std::min(length() - start, n) - slen;
  58   else
  59     return (length() - start) - slen;
  60 }

  75 /**
  76  * @brief Insert a cstring at the given index.
  77  * @param string The cstring to be inserted.
  78  * @param pos The index to insert at.
  79  * @param n Up to this many characters will be used from the string.
  80  * @post A buffer is allocated.
  81  * @post If the old buffer was too small, it is enlarged.
  82  * @post The string is inserted at the given index.
  83  */
  84 void String::insert(size_t pos, const char *string, size_t n)
  85 {
  86   if (n == 0) return;
  87   if (pos != 0)
  88     validateIndex(pos - 1);
  89   
  90   size_t slen = (n == npos) ? std::strlen(string) : n;
  91 
  92   AboutToModify(length() + slen);
  93   std::memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
  94   std::copy(string, string + slen, Buf() + pos);
  95   addLength(slen);
  96 }

  98 /**
  99  * @brief Replaces the given index with the given cstring up to the given amount of characters.
 100  * @param pos The index to replace at.
 101  * @param string The cstring to replace with.
 102  * @param n How many characters to use from string.
 103  */
 104 void String::replace(size_t pos, const char *string, size_t n)
 105 {
 106   if (n == 0) return;
 107   if (pos != 0)
 108     validateIndex(pos - 1);
 109 
 110   size_t slen = (n == npos) ? std::strlen(string) : n;
 111   size_t newlen = pos + slen;
 112   
 113   if (newlen >= length()) {
 114     AboutToModify(newlen);
 115   } else {
 116     newlen = length();
 117     getOwnCopy();
 118   }
 119   std::copy(string, string + slen, Buf() + pos);
 120   setLength(newlen);
 121 }

 123 std::istream& operator>>(std::istream& is, String& string) {
 124   char ch;
 125   string = "";    // empty string, will build one char at-a-time
 126   is >> ch;    // whitespace skipped, first non-white char in ch
 127 
 128   if (!is.fail())
 129   {
 130     do
 131     {
 132       string += ch;
 133       is.get(ch);
 134     } while (!is.fail() && !isspace(ch));
 135 
 136     if (isspace(ch))    // put whitespace back on the stream
 137     {
 138       is.putback(ch);
 139     }
 140   }
 141   return is;
 142 }

 155 /*
 156  * @brief Split a string by a delimiter
 157  * @param delim The delimiter to use
 158  * @param limit Optional param which specified the max number of elements to return
 159  * @note Excess whitespace is trimmed similar to Ruby's split.
 160  */
 161 Array<String> String::split(const String& delim, size_t limit) const {
 162   if (!length()) return Array<String>();
 163   if (limit == 1) {
 164     String list[] = {*this};
 165     return Array<String>(list, 1);
 166   }
 167 
 168   const String space(' ');
 169   Array<String> array;
 170   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 171   String str(*this);
 172 
 173   while (array.size() < limit - 1) {
 174     size_t pos;
 175 
 176     // Trim out left whitespace
 177     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 178 
 179     // All done when there's nothing left
 180     if (!str) break;
 181 
 182     if ((pos = str.find(delim)) == npos)
 183       pos = str.length();
 184     array << str(0, pos);
 185     str += pos + delim.length();
 186   }
 187 
 188   // Add on extra
 189   if (limit != npos) {
 190     // Trim out left whitespace
 191     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 192     array << str;
 193   }
 194 
 195   return array;
 196 }

 216 String& String::chomp() {
 217   if (length() && (*this)[length() - 1] == '\n') --(*this);
 218   if (length() && (*this)[length() - 1] == '\r') --(*this);
 219   return *this;

 222 String& String::trim() {
 223   // ltrim
 224   while (length() && isspace((*this)[0])) { ++offset; subLength(1); }
 225   // rtrim
 226   while (length() && isspace((*this)[length() - 1])) subLength(1);
 227   return *this;

 230 size_t String::find(const String& str) const {
 231   if (length() >= str.length()) {
 232     const size_t last_pos = length() - str.length();
 233     for (size_t pos = 0; pos <= last_pos; ++pos)
 234       if (str[0] == (*this)[pos] && !std::memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 235         return pos;
 236   }
 237   return npos;
 238 }

 240 size_t String::ifind(const String& str) const {
 241   String str_upper(str), this_upper(*this);
 242   std::transform(str.begin(), str.end(), str_upper.begin(), ::toupper);
 243   std::transform(this->begin(), this->end(), this_upper.begin(), ::toupper);
 244   return this_upper.find(str_upper);

 254 size_t String::rfind(const String& str, const size_t lpos) const {
 255   if (length() >= str.length()) {
 256     const size_t last_pos = length() - str.length();
 257     for (size_t pos = last_pos; pos + 1 > lpos; --pos)
 258       if (str[0] == (*this)[pos] && !std::memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 259         return pos;
 260   }
 261   return npos;
 262 }

 264 String String::sub(const String& search, const String& replacement, int limit) const {
 265   String newStr(length()), search_str(*this);
 266   size_t pos = 0;
 267   int cnt = 0;
 268   // Search in our const string so that stuff like \->\\ doesnt become \\\, \\\\ etc.
 269   while ((pos = search_str.find(search)) != npos) {
 270     newStr += search_str(0, pos);
 271     newStr += replacement;
 272     search_str += pos + search.length();
 273     if (limit != -1 && ++cnt == limit) break;
 274   }
 275   // Left over
 276   if (search_str.length())
 277     newStr += search_str;
 278   return newStr;
 279 }

 281 String String::subst(HashTable<String, String> hashes) const {
 282   String newStr(*this);
 283   Array<String> keys(hashes.keys());
 284   Array<String> values(hashes.values());
 285   for (size_t i = 0; i < keys.length(); ++i) {
 286     const String key(keys[i]), value(values[i]);
 287     //::printf("%s -> %s\n", key.c_str(), value.c_str());
 288     newStr = newStr.sub(key, value);
 289   }
 290   return newStr;
 291 }

 293 /**
 294  * @brief Splits off tokens
 295  * @param str The string to split a token off of
 296  * @param delim The delimiter to tokenize with
 297  * This will modify the given string to remove the first token, which is returned.
 298  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 299  */
 300 String newsplit(String& str, char delim)
 301 {
 302   if (!str.length()) return "";
 303   size_t pos = str.find(delim);
 304   if (pos == String::npos)
 305     pos = str.length();
 306 
 307   String first(str(0, pos));
 308 
 309   /* Trim out runs of whitespaces */
 310   if (delim == ' ') {
 311     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 312       ++pos;
 313   }
 314 
 315 //  str = str(pos + 1, str.length() - pos + 1);
 316   str += pos + 1;
 317   return first;
 318 }
