  36 template <class Key, class Value>
  37 /**
  38   * @class BinaryTree
  39   * @brief Binary tree data structure
  40   */
  41 class BinaryTree {
  42   class BinaryTreeIterator;
  43 
  44   public:
  45     typedef BinaryTreeIterator iterator;
  46     typedef KeyValue<Key, Value> iterator_type;
  47 
  48   private:
  49     class Node {
  50       public:
  51         Node *left;
  52         Node *right;
  53         iterator_type kv;
  54 
  55         Node(const Key k, const Value v) : left(NULL), right(NULL), kv(k, v) { };
  56         Node(const Node& n) : left(n.left ? new Node(*(n.left)) : NULL),
  57                               right(n.right ? new Node(*(n.right)) : NULL),
  58                               kv(n.kv) {};
  59     };
  60 
  61     size_t my_size;
  62     Node *root;
  63 
  64     /** 
  65       * @brief Returns a reference to the node with a matching key
  66       * @param search Where to start the search in the tree
  67       * @param key The key to search for
  68       * @return The node searched for, if found, otherwise NULL
  69       */
  70     Node*& fetchNode(Node* const* search, const Key& key) const {
  71       while ((*search) != NULL) {
  72         if (key < (*search)->kv.key())
  73           search = &(*search)->left;
  74         else if (key > (*search)->kv.key())
  75           search = &(*search)->right;
  76         else {
  77           return const_cast<Node*&>(*search);
  78         }
  79       }
  80       return const_cast<Node*&>(*search);
  81     }
  82 
  83     /** 
  84       * @brief Insert a node into the given tree
  85       * @param search The tree to insert the node into
  86       * @param node The node to insert
  87       * @sa fetchNode
  88       * This first looks for the where the node SHOULD be in the tree.
  89       * It will of course fail, but this is the location to place the node.
  90       */
  91     inline void insertNode(Node** search, Node* const node) {
  92       Node*& insertAt = fetchNode(search, node->kv.key());
  93       insertNode(insertAt);
  94     }
  95 
  96     inline void insertNode(Node*& insertAt, Node* const node) {
  97       insertAt = node;
  98       ++my_size;
  99     }
 100 
 101     /**
 102       * @brief Delete a node from the tree. (possibly recursive)
 103       * @param node A reference to the node to delete.
 104       */
 105     void deleteNode(Node*& node) {
 106       if (node == NULL) 
 107         return;
 108       if (node->left == NULL) {
 109         Node* temp = node->right;
 110         delete node;
 111         node = temp;
 112         --my_size;
 113       } else if (node->right == NULL) {
 114         Node* temp = node->left;
 115         delete node;
 116         node = temp;
 117         --my_size;
 118       } else {
 119         //Two children, find max of left subtree and swap
 120         Node** temp = &(node->left);
 121 
 122         while ((*temp)->right != NULL)
 123           temp = &(*temp)->right;
 124         node->kv = (*temp)->kv;
 125         deleteNode(*temp);
 126       }
 127     }
 128   public:
 129     /** 
 130       * @brief Checks whether the given key is in the tree
 131       * @param key The key to search for
 132       * @return true/false depending on found/not-found
 133       */
 134     bool contains(const Key& key) const {
 135       if (isEmpty())
 136         return false;
 137       if (fetchNode(&root, key))
 138         return true;
 139       return false;
 140     }
 141 
 142     /**
 143       * @brief Returns the current size of the tree
 144       * @return The current size of the tree
 145       */
 146     inline size_t size() const { return my_size; };
 147     inline bool isEmpty() const { return size() == 0; };
 148 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 149     inline explicit operator bool() const { return !isEmpty(); };
 150 #endif

 152   public:
 153     BinaryTree() : my_size(0), root(NULL) {};
 154     BinaryTree(const BinaryTree& tree) : my_size(tree.my_size), root(new Node(*(tree.root))) {};
 155 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 156     BinaryTree(BinaryTree&& tree) : my_size(std::move(tree.my_size)), root(std::move(tree.root)) {
 157       root = NULL;
 158       my_size = 0;
 159     };
 160 #endif

 162     friend void swap(BinaryTree& a, BinaryTree& b) {
 163       using std::swap;
 164 
 165       swap(a.my_size, b.my_size);
 166       swap(a.root, b.root);
 167     }
 168 
 169     BinaryTree& operator=(BinaryTree tree) {
 170       swap(*this, tree);
 171       return *this;
 172     }
 173 
 174     /**
 175       * @brief Destructor
 176       * @sa deleteNode
 177       * This simply deletes the root node over and over until the root is NULL.
 178       * There may possibly be a quicker way that does not use recursion (deleteNode does) but this seems the simplest.
 179       */
 180     virtual ~BinaryTree() { clear(); };
 181 
 182     /**
 183       * @brief insert Key/Value pair into tree
 184       * @param key The key to be inserted (or replaced)
 185       * @param value The value to be inserted (or replaced)
 186       * @post The tree's size is increased by 1 if the element was not already in the tree
 187       */
 188     bool insert(const Key& key, const Value& value) {
 189       Node*& rNode = fetchNode(&root, key);
 190       if (rNode)
 191         rNode->kv = iterator_type(key, value);
 192       else {
 193         Node *newNode = new Node(key, value);
 194         insertNode(rNode, newNode);
 195       }
 196       return true;
 197     }
 198 
 199     /**
 200       * @brief Associate array type accessor (rvalue)
 201       * @sa getValue
 202       * @param key The key to search for
 203       */
 204     inline const Value operator[](const Key& key) const { return getValue(key); }
 205 
 206     /**
 207       * @brief Associate array type accessor (lvalue)
 208       * @param key The key to search for
 209       * @sa insert
 210       * If the key is not in the tree, it is inserted, and the value set to the rvalue given.
 211       */
 212     Value& operator[](const Key& key) {
 213       Node*& node = fetchNode(&root, key);
 214       if (!node)
 215         insertNode(node, new Node(key, Value()));
 216       //The key was inserted at the return node's address! node is now guranteed to be NON-NULL ! */
 217       return node->kv.v;
 218     }
 219 
 220     /** 
 221       * @brief Remove the given key (and its value) from the tree
 222       * @param key The key to be searched/removed
 223       * @return Whether or not the key was removed
 224       */
 225     bool remove(const Key& key) {
 226       Node*& node = fetchNode(&root, key);
 227 
 228       if (node != NULL) {
 229         deleteNode(node);
 230         return true;
 231       }
 232       return false;
 233     }
 234 
 235     inline void clear() {
 236       while (root != NULL)
 237         deleteNode(root);
 238     }
 239 
 240     /**
 241       * @brief Find a value in the tree from the key given
 242       * @param key The key to search for
 243       * @return The value of the key searched for, or NULL if not found
 244       */
 245     Value getValue(const Key& key) const {
 246       const Value empty;
 247       if (isEmpty()) return empty;
 248 
 249       Node* node = fetchNode(&root, key);
 250       if (node)
 251         return node->kv.value();
 252       return empty;
 253     }

 262       const Key empty;
 263       if (isEmpty()) return empty;

 265       for (iterator iter = begin(); iter; ++iter) {

 273   private:
 274     class BinaryTreeIterator : public Iterator<iterator_type> {
 275       friend class BinaryTree;
 276       typedef BinaryTree<Key, Value> Tree;
 277       private:
 278         Tree* tree;
 279         size_t index;
 280         size_t my_size;
 281         bool reverse;
 282         iterator_type* storage;
 283 
 284         void fillArray(int& i, const Node* node) {
 285           if (node == NULL) 
 286             return;
 287           fillArray(i, node->left);
 288           storage[i++] = iterator_type(node->kv);
 289           fillArray(i, node->right);
 290         }
 291 
 292         BinaryTreeIterator(Tree* t, Node* node, size_t size, bool _reverse = 0) : Iterator<iterator_type>(),
 293                                                                           tree(t),
 294                                                                           index(_reverse ? size - 1 : 0),
 295                                                                           my_size(size), 
 296                                                                           reverse(_reverse),
 297                                                                           storage(new iterator_type[size]) {
 298           int i = 0;
 299           fillArray(i, node);
 300         }
 301         //BinaryTreeIterator& operator=(const BinaryTreeIterator&); ///<Block implicit copy constructor
 302       public:
 303         BinaryTreeIterator(const iterator& iter) : Iterator<iterator_type>(), 
 304                                                    tree(iter.tree),
 305                                                    index(iter.index), 
 306                                                    my_size(iter.my_size), 
 307                                                    reverse(iter.reverse),
 308                                                    storage(new iterator_type[iter.my_size]) {
 309           for (size_t i = 0; i < my_size; ++i)
 310             storage[i] = iter.storage[i];
 311         }
 312 
 313         BinaryTreeIterator() : Iterator<iterator_type>(),
 314                                tree(NULL),
 315                                index(0),
 316                                my_size(0),
 317                                reverse(0),
 318                                storage(NULL) {
 319         };
 320 
 321         iterator& operator=(const iterator& iter) {
 322           if (&iter != this) {
 323             if (my_size != iter.my_size) {
 324               delete[] storage;
 325               storage = new iterator_type[iter.my_size];
 326             }
 327 
 328             tree = iter.tree;
 329             index = iter.index;
 330             my_size = iter.my_size;
 331 
 332             for (size_t i = 0; i < my_size; ++i)
 333               storage[i] = iter.storage[i];
 334           }
 335           return *this;
 336         }
 337 
 338         virtual ~BinaryTreeIterator() { delete[] storage; }
 339 
 340         virtual void remove() {
 341           tree->remove(storage[index].key());
 342 /*
 343           --my_size;
 344           for (int i = index; i < my_size; ++i)
 345             storage[i] = storage[i + 1];
 346           //The iterator is 'invalid' now, but the index is shifter down in case iter++ is called
 347           --index;
 348 */
 349         }
 350 
 351         virtual
 352 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 353           explicit
 354 #endif
 355         operator bool() const {
 356           return ((reverse && index > 0) || (!reverse && index < (my_size)));
 357         };
 358 
 359         virtual operator iterator_type() { return operator*(); };
 360         virtual iterator_type& operator*(){ return storage[index]; }
 361 
 362         /**
 363          * @brief Postfix increment
 364          */
 365         virtual iterator operator++(int) {
 366           iterator tmp(*this);
 367           ++index;
 368           return tmp;
 369         }
 370 
 371         /**
 372          * @brief Prefix increment
 373          */
 374         virtual iterator& operator++() {
 375           ++index;
 376           return *this;
 377         }
 378 
 379         /**
 380          * @brief Postfix decrement
 381          */
 382         virtual iterator operator--(int) {
 383           iterator tmp(*this);
 384           --index;
 385           return tmp;
 386         }
 387 
 388         /**
 389          * @brief Prefix decrement
 390          */
 391         virtual iterator& operator--() {
 392           --index;
 393           return *this;
 394         }
 395     };
 396   public:
 397     inline iterator begin() { return iterator(this, root, my_size); };
 398     inline iterator end() { return iterator(this, root, my_size, 1); };
