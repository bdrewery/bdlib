  48 typedef struct {
  49         int len;
  50         int family;
  51         union {
  52                 struct sockaddr addr;
  53                 struct sockaddr_in ipv4;
  54 #ifdef DO_IPV6
  55                 struct sockaddr_in6 ipv6;
  56 #endif
  57         } u;
  58 } sockname_t;

  61 /**
  62  * @class Socket
  63  * @brief Socket class
  64  */
  65 class Socket {
  66   private:
  67     int sock;
  68     sockname_t addr;
  69     int flags;
  70   public:
  71     Socket() : sock(-1), addr(0), pfamily(0) {};
  72     Socket(int f) : sock(-1), addr(0), flags(f) {};
  73     Socket(Socket& s) : sock(s.sock), addr(s.addr), flags(s.flags) {};
  74     virtual ~Socket();
  75     
  76     bool create();
  77     bool connect(const String host, const in_port_t port);
  78     bool bind(const in_port_t port);
  79     bool listen() const;
  80     bool accept (Socket&) const;
  81 
  82     bool send(const String) const;
  83     bool recv(String&) const;
  84 
  85     int setNonBlocking(bool = 1);
  86     inline bool isValid() const { return sock != -1; };
  87 };
