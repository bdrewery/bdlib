  32 template <class Key, class Value>
  33 /**
  34   * @class BinaryTree
  35   * @brief Binary tree data structure
  36   */
  37 class BinaryTree {
  38   class BinaryTreeIterator;
  39 
  40   public:
  41     typedef BinaryTreeIterator iterator;
  42     typedef KeyValue<Key, Value> iterator_type;
  43 
  44   private:
  45     class Node {
  46       public:
  47         Node *left;
  48         Node *right;
  49         iterator_type kv;
  50 
  51         Node(const Key k, const Value v) : left(NULL), right(NULL), kv(k, v) { };
  52         Node(const Node& n) : left(n.left ? new Node(*(n.left)) : NULL),
  53                               right(n.right ? new Node(*(n.right)) : NULL),
  54                               kv(n.kv) {};
  55     };
  56 
  57     size_t my_size;
  58     Node *root;
  59 
  60     /** 
  61       * @brief Returns a reference to the node with a matching key
  62       * @param search Where to start the search in the tree
  63       * @param key The key to search for
  64       * @return The node searched for, if found, otherwise NULL
  65       */
  66     Node*& fetchNode(Node* const* search, const Key& key) const {
  67       while ((*search) != NULL) {
  68         if (key < (*search)->kv.key())
  69           search = &(*search)->left;
  70         else if (key > (*search)->kv.key())
  71           search = &(*search)->right;
  72         else {
  73           return const_cast<Node*&>(*search);
  74         }
  75       }
  76       return const_cast<Node*&>(*search);
  77     }
  78 
  79     /** 
  80       * @brief Insert a node into the given tree
  81       * @param search The tree to insert the node into
  82       * @param node The node to insert
  83       * @sa fetchNode
  84       * This first looks for the where the node SHOULD be in the tree.
  85       * It will of course fail, but this is the location to place the node.
  86       */
  87     inline void insertNode(Node** search, Node* const node) {
  88       Node*& insertAt = fetchNode(search, node->kv.key());
  89       insertNode(insertAt);
  90     }
  91 
  92     inline void insertNode(Node*& insertAt, Node* const node) {
  93       insertAt = node;
  94       ++my_size;
  95     }
  96 
  97     /**
  98       * @brief Delete a node from the tree. (possibly recursive)
  99       * @param node A reference to the node to delete.
 100       */
 101     void deleteNode(Node*& node) {
 102       if (node == NULL) 
 103         return;
 104       if (node->left == NULL) {
 105         Node* temp = node->right;
 106         delete node;
 107         node = temp;
 108         --my_size;
 109       } else if (node->right == NULL) {
 110         Node* temp = node->left;
 111         delete node;
 112         node = temp;
 113         --my_size;
 114       } else {
 115         //Two children, find max of left subtree and swap
 116         Node** temp = &(node->left);
 117 
 118         while ((*temp)->right != NULL)
 119           temp = &(*temp)->right;
 120         node->kv = (*temp)->kv;
 121         deleteNode(*temp);
 122       }
 123     }
 124   public:
 125     /** 
 126       * @brief Checks whether the given key is in the tree
 127       * @param key The key to search for
 128       * @return true/false depending on found/not-found
 129       */
 130     bool contains(const Key& key) const {
 131       if (isEmpty())
 132         return false;
 133       if (fetchNode(&root, key))
 134         return true;
 135       return false;
 136     }
 137 
 138     /**
 139       * @brief Returns the current size of the tree
 140       * @return The current size of the tree
 141       */
 142     inline size_t size() const { return my_size; };
 143     inline bool isEmpty() const { return size() == 0; };
 144     inline explicit operator bool() const { return !isEmpty(); };
 145 
 146   public:
 147     BinaryTree() : my_size(0), root(NULL) {};
 148     BinaryTree(const BinaryTree& tree) : my_size(tree.my_size), root(new Node(*(tree.root))) {};
 149 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 150     BinaryTree(BinaryTree&& tree) : my_size(0), root(NULL) {
 151       swap(*this, tree);
 152     };
 153 #endif

 155     friend void swap(BinaryTree& a, BinaryTree& b) {
 156       using std::swap;
 157 
 158       swap(a.my_size, b.my_size);
 159       swap(a.root, b.root);
 160     }
 161 
 162     BinaryTree& operator=(BinaryTree tree) {
 163       swap(*this, tree);
 164       return *this;
 165     }
 166 
 167     /**
 168       * @brief Destructor
 169       * @sa deleteNode
 170       * This simply deletes the root node over and over until the root is NULL.
 171       * There may possibly be a quicker way that does not use recursion (deleteNode does) but this seems the simplest.
 172       */
 173     virtual ~BinaryTree() { clear(); };
 174 
 175     /**
 176       * @brief insert Key/Value pair into tree
 177       * @param key The key to be inserted (or replaced)
 178       * @param value The value to be inserted (or replaced)
 179       * @post The tree's size is increased by 1 if the element was not already in the tree
 180       */
 181     bool insert(const Key& key, const Value& value) {
 182       Node*& rNode = fetchNode(&root, key);
 183       if (rNode)
 184         rNode->kv = iterator_type(key, value);
 185       else {
 186         Node *newNode = new Node(key, value);
 187         insertNode(rNode, newNode);
 188       }
 189       return true;
 190     }
 191 
 192     /**
 193       * @brief Associate array type accessor (rvalue)
 194       * @sa getValue
 195       * @param key The key to search for
 196       */
 197     inline const Value operator[](const Key& key) const { return getValue(key); }
 198 
 199     /**
 200       * @brief Associate array type accessor (lvalue)
 201       * @param key The key to search for
 202       * @sa insert
 203       * If the key is not in the tree, it is inserted, and the value set to the rvalue given.
 204       */
 205     Value& operator[](const Key& key) {
 206       Node*& node = fetchNode(&root, key);
 207       if (!node)
 208         insertNode(node, new Node(key, Value()));
 209       //The key was inserted at the return node's address! node is now guranteed to be NON-NULL ! */
 210       return node->kv.v;
 211     }
 212 
 213     /** 
 214       * @brief Remove the given key (and its value) from the tree
 215       * @param key The key to be searched/removed
 216       * @return Whether or not the key was removed
 217       */
 218     bool remove(const Key& key) {
 219       Node*& node = fetchNode(&root, key);
 220 
 221       if (node != NULL) {
 222         deleteNode(node);
 223         return true;
 224       }
 225       return false;
 226     }
 227 
 228     inline void clear() {
 229       while (root != NULL)
 230         deleteNode(root);
 231     }
 232 
 233     /**
 234       * @brief Find a value in the tree from the key given
 235       * @param key The key to search for
 236       * @return The value of the key searched for, or NULL if not found
 237       */
 238     Value getValue(const Key& key) const {
 239       const Value empty;
 240       if (isEmpty()) return empty;
 241 
 242       Node* node = fetchNode(&root, key);
 243       if (node)
 244         return node->kv.value();
 245       return empty;
 246     }

 266   private:
 267     class BinaryTreeIterator : public Iterator<iterator_type> {
 268       friend class BinaryTree;
 269       typedef BinaryTree<Key, Value> Tree;
 270       private:
 271         Tree* tree;
 272         size_t index;
 273         size_t my_size;
 274         iterator_type* storage;
 275 
 276         void fillArray(int& i, const Node* node) {
 277           if (node == NULL) 
 278             return;
 279           fillArray(i, node->left);
 280           storage[i++] = iterator_type(node->kv);
 281           fillArray(i, node->right);
 282         }
 283 
 284         BinaryTreeIterator(Tree* t, Node* node, size_t size, bool end = 0) : Iterator<iterator_type>(), 
 285                                                                           tree(t),
 286                                                                           index(end ? size - 1: 0), 
 287                                                                           my_size(size), 
 288                                                                           storage(new iterator_type[size]) {
 289           int i = 0;
 290           fillArray(i, node);
 291         }
 292         //BinaryTreeIterator& operator=(const BinaryTreeIterator&); ///<Block implicit copy constructor
 293       public:
 294         BinaryTreeIterator(const iterator& iter) : Iterator<iterator_type>(), 
 295                                                    tree(iter.tree),
 296                                                    index(iter.index), 
 297                                                    my_size(iter.my_size), 
 298                                                    storage(new iterator_type[iter.my_size]) {
 299           for (size_t i = 0; i < my_size; ++i)
 300             storage[i] = iter.storage[i];
 301         }
 302 
 303         BinaryTreeIterator() : Iterator<iterator_type>(),
 304                                tree(NULL),
 305                                index(0),
 306                                my_size(0),
 307                                storage(NULL) {
 308         };
 309 
 310         iterator& operator=(const iterator& iter) {
 311           if (&iter != this) {
 312             if (my_size != iter.my_size) {
 313               delete[] storage;
 314               storage = new iterator_type[iter.my_size];
 315             }
 316 
 317             tree = iter.tree;
 318             index = iter.index;
 319             my_size = iter.my_size;
 320 
 321             for (size_t i = 0; i < my_size; ++i)
 322               storage[i] = iter.storage[i];
 323           }
 324           return *this;
 325         }
 326 
 327         virtual ~BinaryTreeIterator() { delete[] storage; }

 329         virtual void remove() {
 330           tree->remove(storage[index].key());
 331 /*
 332           --my_size;
 333           for (int i = index; i < my_size; ++i)
 334             storage[i] = storage[i + 1];
 335           //The iterator is 'invalid' now, but the index is shifter down in case iter++ is called
 336           --index;
 337 */
 338         }
 339 
 340         virtual explicit operator bool() const { return (index > 0 && index < my_size); };
 341 
 342         virtual operator iterator_type() { return operator*(); };
 343         virtual iterator_type& operator*(){ return storage[index]; }
 344 
 345         /**
 346          * @brief Postfix increment
 347          */
 348         virtual iterator operator++(int) {
 349           iterator tmp(*this);
 350           ++index;
 351           return tmp;
 352         }
 353 
 354         /**
 355          * @brief Prefix increment
 356          */
 357         virtual iterator& operator++() {
 358           ++index;
 359           return *this;
 360         }

 362         /**
 363          * @brief Postfix decrement
 364          */
 365         virtual iterator operator--(int) {
 366           iterator tmp(*this);
 367           --index;
 368           return tmp;
 369         }

 372          * @brief Prefix decrement
 373          */
 374         virtual iterator& operator--() {
 375           --index;
 376           return *this;
 377         }
 378     };
 379   public:
 380     inline iterator begin() { return iterator(this, root, my_size); };
