  32 template <class Key, class Value>
  33 /**
  34   * @class BinaryTree
  35   * @brief Binary tree data structure
  36   */
  37 class BinaryTree {
  38   class BinaryTreeIterator;
  39 
  40   public:
  41     typedef BinaryTreeIterator iterator;
  42     typedef KeyValue<Key, Value> iterator_type;
  43 
  44   private:
  45     class Node {
  46       public:
  47         Node *left;
  48         Node *right;
  49         iterator_type kv;
  50 
  51         Node(const Key k, const Value v) : left(NULL), right(NULL), kv(k, v) { };
  52         Node(const Node& n) : left(n.left ? new Node(*(n.left)) : NULL),
  53                               right(n.right ? new Node(*(n.right)) : NULL),
  54                               kv(n.kv) {};
  55     };
  56 
  57     size_t my_size;
  58     Node *root;
  59 
  60     /** 
  61       * @brief Returns a reference to the node with a matching key
  62       * @param search Where to start the search in the tree
  63       * @param key The key to search for
  64       * @return The node searched for, if found, otherwise NULL
  65       */
  66     Node* &fetchNode(Node* const* search, const Key &key) const {
  67       while ((*search) != NULL) {
  68         if (key < (*search)->kv.key())
  69           search = &(*search)->left;
  70         else if (key > (*search)->kv.key())
  71           search = &(*search)->right;
  72         else {
  73           return const_cast<Node*&>(*search);
  74         }
  75       }
  76       return const_cast<Node*&>(*search);
  77     }
  78 
  79     /** 
  80       * @brief Insert a node into the given tree
  81       * @param search The tree to insert the node into
  82       * @param node The node to insert
  83       * @sa fetchNode
  84       * This first looks for the where the node SHOULD be in the tree.
  85       * It will of course fail, but this is the location to place the node.
  86       */
  87     inline void insertNode(Node** search, Node* const node) {
  88       Node* &insertAt = fetchNode(search, node->kv.key());
  89       insertNode(insertAt);
  90     }
  91 
  92     inline void insertNode(Node*& insertAt, Node* const node) {
  93       insertAt = node;
  94       ++my_size;
  95     }
  96 
  97     /**
  98       * @brief Delete a node from the tree. (possibly recursive)
  99       * @param node A reference to the node to delete.
 100       */
 101     void deleteNode(Node*& node) {
 102       if (node == NULL) 
 103         return;
 104       if (node->left == NULL) {
 105         Node* temp = node->right;
 106         delete node;
 107         node = temp;
 108         --my_size;
 109       } else if (node->right == NULL) {
 110         Node* temp = node->left;
 111         delete node;
 112         node = temp;
 113         --my_size;
 114       } else {
 115         //Two children, find max of left subtree and swap
 116         Node** temp = &(node->left);
 117 
 118         while ((*temp)->right != NULL)
 119           temp = &(*temp)->right;
 120         node->kv = (*temp)->kv;
 121         deleteNode(*temp);
 122       }
 123     }
 124   public:
 125     /** 
 126       * @brief Checks whether the given key is in the tree
 127       * @param key The key to search for
 128       * @return true/false depending on found/not-found
 129       */
 130     bool contains(const Key &key) const {
 131       if (isEmpty())
 132         return false;
 133       if (fetchNode(&root, key))
 134         return true;
 135       return false;
 136     }
 137 
 138     /**
 139       * @brief Returns the current size of the tree
 140       * @return The current size of the tree
 141       */
 142     inline size_t size() const { return my_size; };
 143     inline bool isEmpty() const { return size() == 0; };
 144     //inline operator bool() const { return !isEmpty(); };
 145 
 146   public:
 147     BinaryTree() : my_size(0), root(NULL) {};
 148     BinaryTree(const BinaryTree& tree) : my_size(tree.my_size), root(new Node(*(tree.root))) {};
 149 
 150     BinaryTree& operator = (const BinaryTree& tree) {
 151       if (&tree != this) {
 152         clear();
 153         my_size = tree.my_size;
 154         root = new Node(*(tree.root));
 155       }
 156       return *this;
 157     }
 158 
 159     /**
 160       * @brief Destructor
 161       * @sa deleteNode
 162       * This simply deletes the root node over and over until the root is NULL.
 163       * There may possibly be a quicker way that does not use recursion (deleteNode does) but this seems the simplest.
 164       */
 165     virtual ~BinaryTree() { clear(); };
 166 
 167     /**
 168       * @brief insert Key/Value pair into tree
 169       * @param key The key to be inserted (or replaced)
 170       * @param value The value to be inserted (or replaced)
 171       * @post The tree's size is increased by 1 if the element was not already in the tree
 172       */
 173     bool insert(const Key &key, const Value &value) {
 174       Node*& rNode = fetchNode(&root, key);
 175       if (rNode)
 176         rNode->kv = iterator_type(key, value);
 177       else {
 178         Node *newNode = new Node(key, value);
 179         insertNode(rNode, newNode);
 180       }
 181       return true;
 182     }
 183 
 184     /**
 185       * @brief Associate array type accessor (rvalue)
 186       * @sa getValue
 187       * @param key The key to search for
 188       */
 189     inline const Value operator [](const Key& key) const { return getValue(key); }
 190 
 191     /**
 192       * @brief Associate array type accessor (lvalue)
 193       * @param key The key to search for
 194       * @sa insert
 195       * If the key is not in the tree, it is inserted, and the value set to the rvalue given.
 196       */
 197     Value& operator [](const Key& key) {
 198       Node*& node = fetchNode(&root, key);
 199       if (!node)
 200         insertNode(node, new Node(key, Value()));
 201       //The key was inserted at the return node's address! node is now guranteed to be NON-NULL ! */
 202       return node->kv.v;
 203     }
 204 
 205     /** 
 206       * @brief Remove the given key (and its value) from the tree
 207       * @param key The key to be searched/removed
 208       * @return Whether or not the key was removed
 209       */
 210     bool remove(const Key &key) {
 211       Node* &node = fetchNode(&root, key);
 212 
 213       if (node != NULL) {
 214         deleteNode(node);
 215         return true;
 216       }
 217       return false;
 218     }
 219 
 220     inline void clear() {
 221       while (root != NULL)
 222         deleteNode(root);
 223     }
 224 
 225     /**
 226       * @brief Find a value in the tree from the key given
 227       * @param key The key to search for
 228       * @return The value of the key searched for, or NULL if not found
 229       */
 230     Value getValue(const Key &key) const {
 231       const Value empty;
 232       if (isEmpty()) return empty;
 233 
 234       Node* node = fetchNode(&root, key);
 235       if (node)
 236         return node->kv.value();
 237       return empty;
 238     }

 246     Value getKey(const Value &value) const {
 247       const Key empty;

 258   private:
 259     class BinaryTreeIterator : public Iterator<iterator_type> {
 260       friend class BinaryTree;
 261       typedef BinaryTree<Key, Value> Tree;
 262       private:
 263         Tree* tree;
 264         size_t index;
 265         size_t my_size;
 266         iterator_type* storage;
 267 
 268         void fillArray(int& i, const Node* node) {
 269           if (node == NULL) 
 270             return;
 271           fillArray(i, node->left);
 272           storage[i++] = iterator_type(node->kv);
 273           fillArray(i, node->right);
 274         }
 275 
 276         BinaryTreeIterator(Tree* t, Node* node, size_t size, bool end = 0) : Iterator<iterator_type>(), 
 277                                                                           tree(t),
 278                                                                           index(end ? size - 1: 0), 
 279                                                                           my_size(size), 
 280                                                                           storage(new iterator_type[size]) {
 281           int i = 0;
 282           fillArray(i, node);
 283         }
 284         //BinaryTreeIterator& operator=(const BinaryTreeIterator&); ///<Block implicit copy constructor
 285       public:
 286         BinaryTreeIterator(const iterator& iter) : Iterator<iterator_type>(), 
 287                                                    tree(iter.tree),
 288                                                    index(iter.index), 
 289                                                    my_size(iter.my_size), 
 290                                                    storage(new iterator_type[iter.my_size]) {
 291           for (size_t i = 0; i < my_size; ++i)
 292             storage[i] = iter.storage[i];
 293         }
 294 
 295         BinaryTreeIterator() : Iterator<iterator_type>(),
 296                                tree(NULL),
 297                                index(0),
 298                                my_size(0),
 299                                storage(NULL) {
 300         };
 301 
 302         iterator& operator =(const iterator& iter) {
 303           if (&iter != this) {
 304             if (my_size != iter.my_size) {
 305               delete[] storage;
 306               storage = new iterator_type[iter.my_size];
 307             }
 308 
 309             tree = iter.tree;
 310             index = iter.index;
 311             my_size = iter.my_size;
 312 
 313             for (size_t i = 0; i < my_size; ++i)
 314               storage[i] = iter.storage[i];
 315           }
 316           return *this;
 317         }
 318 
 319         virtual ~BinaryTreeIterator() { delete[] storage; }
 320 
 321         virtual void remove() {
 322           tree->remove(storage[index].key());
 323 /*
 324           --my_size;
 325           for (int i = index; i < my_size; ++i)
 326             storage[i] = storage[i + 1];
 327           //The iterator is 'invalid' now, but the index is shifter down in case iter++ is called
 328           --index;
 329 */
 330         }
 331 
 332         virtual operator bool() { return (index > 0 && index < my_size); };
 333 
 334         virtual operator iterator_type () { return operator*(); };
 335         virtual iterator_type& operator *() { return storage[index]; }
 336 
 337         //Postfix
 338         virtual iterator operator ++(int) {
 339           iterator tmp(*this);
 340           ++index;
 341           return tmp;
 342         }
 343 
 344         virtual iterator& operator ++() {
 345           ++index;
 346           return *this;
 347         }
 348 
 349         //Postfix
 350         virtual iterator operator --(int) {
 351           iterator tmp(*this);
 352           --index;
 353           return tmp;
 354         }
 355 
 356         virtual iterator& operator --() {
 357           --index;
 358           return *this;
 359         }
 360     };
 361   public:
 362     inline iterator begin() { return iterator(this, root, my_size); };
 363     inline iterator end() { return iterator(this, root, my_size, 1); };
