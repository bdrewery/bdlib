  38 bool Socket::create() {
  39   int ptype = 0;
  40 
  41   if (flags & SOCKET_UDP)
  42     ptype = SOCK_DGRAM;
  43   else if (flags & SOCKET_TCP)
  44     ptype = SOCK_STREAM;
  45 
  46   sock = socket(pfamily, ptype, 0);
  47   if (sock == -1)
  48     return false;
  49 }
