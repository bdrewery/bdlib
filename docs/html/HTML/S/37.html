  44 template <class T>
  45 /**
  46  * @class Array
  47  * @brief Dynamic Array
  48  */
  49 class Array : public ReferenceCountedArray<T> {
  50   public:
  51     typedef T value_type;
  52 
  53     typedef size_t             size_type;
  54     typedef std::ptrdiff_t     difference_type;
  55     typedef value_type*        pointer;
  56     typedef const value_type*  const_pointer;
  57     typedef value_type&        reference;
  58     typedef const value_type&  const_reference;
  59 
  60     /* Constructors */
  61     Array() : ReferenceCountedArray<value_type>() {};
  62     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  63     Array(Array<value_type>&& array) : ReferenceCountedArray<value_type>(std::move(array)) {};
  64 
  65     /**
  66      * @brief Create an array from an initializer list
  67      * @param list An initializer_list
  68      */
  69     Array(std::initializer_list<value_type> list) : ReferenceCountedArray<value_type>() {
  70       *this = list;
  71     }
  72 
  73     /**
  74      * @brief Create a Array from a given carray.
  75      * @param carray The null-terminated array to create the object from.
  76      * @param len How big is the carray?
  77      * @post A ArrayBuf has been initialized.
  78      * @post The buffer has been filled with the array.
  79      * @test Array test("Some array");
  80      */
  81     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  82       this->Reserve(len);
  83       for (size_t i = 0; i < len; ++i)
  84         push(carray[i]);
  85     };
  86 
  87     /**
  88      * @brief Create an empty Array container with at least the specified elements in size.
  89      * @param newSize Reserve at least this many buckets for this Array.
  90      * @post This array's memory will also never be shrunk.
  91      * @post A buffer has been created.
  92      */
  93     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
  94     Array(const size_type newSize, const value_type value) : ReferenceCountedArray<value_type>(newSize, value) {};
  95 
  96     virtual ~Array() {};
  97 
  98     /**
  99      * @brief Create an array from an initializer list
 100      * @param list An initializer_list
 101      */
 102     Array& operator=(std::initializer_list<value_type> list) {
 103       this->clear();
 104       this->Reserve(list.size());
 105       for (value_type item : list) {
 106         push(std::move(item));
 107       }
 108       return *this;
 109     }
 110 
 111     Array& operator=(const Array<value_type>& array) {
 112       ReferenceCountedArray<value_type>::operator=(array);
 113       return *this;
 114     }
 115 
 116     Array& operator=(Array<value_type>&& array) {
 117       ReferenceCountedArray<value_type>::operator=(std::move(array));
 118       return *this;
 119     }
 120 
 121     /**
 122      * @brief Add an item to the end of the array
 123      */
 124     inline void push(const value_type item) {
 125       this->AboutToModify(this->length() + 1);
 126       *(this->Buf(this->length())) = item;
 127       this->addLength(1);
 128     }
 129 
 130     /**
 131      * @sa push
 132      */
 133     inline friend Array<value_type>& operator<<(Array<value_type>& array, const_reference item) {
 134       array.push(item);
 135       return array;
 136     }
 137 
 138     /**
 139      * @brief Shift the array left, removing the first element.
 140      * @return The first element.
 141      */
 142     inline value_type shift() {
 143       if (this->isEmpty()) return value_type();
 144 
 145       value_type temp(*(this->Buf(0)));
 146       ++(this->offset);
 147       this->subLength(1);
 148       return temp;
 149     }
 150 
 151     /**
 152      * @brief Pop a value off the end of the array
 153      * @return The last element.
 154      */
 155     inline value_type pop() {
 156       if (this->isEmpty()) return value_type();
 157 
 158       value_type temp(*(this->Buf(this->length() - 1)));
 159       this->subLength(1);
 160       return temp;
 161     }
 162 
 163     /**
 164      * @sa pop
 165      */
 166     inline friend Array<value_type>& operator>>(Array<value_type>& array, reference item) {
 167       item = array.pop();
 168       return array;
 169     }
 170 
 171     /**
 172      * @brief Join an array by a delimiter into a string
 173      */
 174     String join(const String& delim, bool quoted = false) const {
 175       if (!this->length()) return String();
 176       size_t str_size = 0, i;
 177 
 178       // Preallocate a fully sized String to avoid frequent resizing
 179       str_size += (this->length() * delim.length()) - 1;
 180       if (quoted)
 181         str_size += this->length() * 2;
 182       for (i = 0; i < this->length(); ++i)
 183         str_size += this->Buf(i)->length();
 184 
 185       String str(str_size);
 186 
 187       for (i = 0; i < this->length(); ++i) {
 188         if (i)
 189           str += delim;
 190         if (quoted) {
 191           str += '"' + *(this->Buf(i)) + '"';
 192         } else {
 193           str += *(this->Buf(i));
 194         }
 195       }
 196       return str;
 197     }
 198 
 199     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 200     /**
 201      * @brief Compare our Array object with another Array object, but only n elements
 202      * @param array The Array object to equals to.
 203      * @param n The number of items to equals.
 204      * @return True if the number of elements are the same, and they all are equal.
 205      */
 206     bool equals(const Array& array, size_t n) const
 207     {
 208       size_t my_len = this->length();
 209       bool same_length = (my_len == array.length());
 210 
 211       /* Same array? */
 212       if (this->data() == array.data() && same_length)
 213         return true;
 214 
 215       if (!same_length)
 216         return false;
 217 
 218       size_t slen = std::min(array.length(), n);
 219       size_t len = std::min(my_len, slen);
 220 
 221       for (size_t i = 0; i < len; ++i) {
 222         if ((*this)[i] != array[i])
 223           return false;
 224       }
 225       return true;
 226     }
 227 
 228     inline friend bool operator==(const Array& lhs, const Array& rhs) {
 229       return lhs.size() == rhs.size() && lhs.equals(rhs);
 230     };
 231     inline friend bool operator!=(const Array& lhs, const Array& rhs) { return !(lhs == rhs); };
 232 
 233     // Subarrays
 234     /**
 235      * @sa ReferenceCountedArray::slice()
 236      */
 237     inline Array subarray(int start, int len = -1) const {
 238       Array newArray(*this);
 239       newArray.slice(start, len);
 240       return newArray;
 241     };
 242 
 243     /**
 244      * @sa subarray
 245      */
 246     inline Array operator()(int start, int len = -1) const { return subarray(start, len); };
 247 
 248     /**
 249      * @brief Returns a 'Slice' class for safe (cow) writing into the array
 250      * @sa Slice
 251      * @param start Starting position
 252      * @param len How many items to use
 253      */
 254     inline Slice<Array> operator()(int start, int len = -1) { return Slice<Array>(*this, start, len); };
 255 
 256 #ifdef CPPUNIT_VERSION
 257     void CPPUNIT_checkArrayEqual(Array actual, CPPUNIT_NS::SourceLine sourceLine) {
 258       if ((*this) == actual) return;
 259       ::CPPUNIT_NS::Asserter::failNotEqual(this->join("|").c_str(), actual.join("|").c_str(), sourceLine);
 260     }
 261 #endif /* CPPUNIT_VERSION */

 263     /* Operators */
 264     /**
 265      * @relates Array
 266      * @brief Concatenates two array objects together.
 267      * @param array1 The LHS array.
 268      * @param array2 The RHS array.
 269      * @post A new array is allocated, reference copied and returned.
 270      * @return Returns a new array that can be reference copied by the lvalue.
 271      */
 272     inline friend Array operator+(Array array1, const Array& array2) {
 273       array1 += array2;
 274       return array1;
 275     }
 276 
 277     /**
 278      * @brief Prefix increment
 279      */
 280     inline const Array& operator++() {
 281       return (*this) += 1;
 282     }
 283 
 284     /**
 285      * @brief Postfix increment
 286      */
 287     inline const Array operator++(int) {
 288       Array tmp((*this)(0, 1));
 289       ++(this->offset);
 290       this->subLength(1);
 291       return tmp;
 292     }
 293 
 294     /**
 295      * @brief Prefix decrement
 296      */
 297     inline const Array& operator--() {
 298       return (*this) -= 1;
 299     }
 300 
 301     /**
 302      * @brief Postfix decrement
 303      */
 304     inline const Array operator--(int) {
 305       Array tmp((*this)(this->length() - 1, 1));
 306       this->subLength(1);
 307       return tmp;
 308     }
 309 
 310     /**
 311      * \sa append(const char)
 312      */
 313     inline Array& operator+=(const_reference item) {
 314       this->append(item);
 315       return *this;
 316     }
 317 
 318     /**
 319      * \sa append(const Array&)
 320      */
 321     inline Array& operator+=(const Array& array) {
 322       this->append(array);
 323       return *this;
 324     }
 325 
 326     inline Array& operator+=(const int n) {
 327       if (!this->length())
 328         return *this;
 329       if (int(this->length()) - n < 0) {
 330         this->offset = this->length();
 331         this->setLength(0);
 332       } else {
 333         this->offset += n;
 334         this->subLength(n);
 335       }
 336       return *this;
 337     }
 338 
 339     inline Array& operator-=(const int n) {
 340       if (!this->length())
 341         return *this;
 342       if (int(this->length()) - n < 0) {
 343         this->offset = this->length();
 344         this->setLength(0);
 345       } else
 346         this->subLength(n);
 347       return *this;
 348     }
 349 
 350 };
