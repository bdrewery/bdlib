  27 #ifndef _BD_LIST_H
  28 #define _BD_LIST_H 1
  29 
  30 #include "bdlib.h"
  31 #include "Iterator.h"

  34 template <class T>
  35 /**
  36   * @class List
  37   * @brief List data structure
  38   */
  39 class List {
  40   class ListIterator;
  41 
  42   public:
  43     typedef ListIterator iterator;
  44     typedef T value_type;
  45 
  46     typedef size_t             size_type;
  47     typedef std::ptrdiff_t     difference_type;
  48     typedef value_type*        pointer;
  49     typedef const value_type*  const_pointer;
  50     typedef value_type&        reference;
  51     typedef const value_type&  const_reference;
  52 
  53   protected:
  54     struct Node {
  55       value_type item;
  56       Node* next;
  57       Node* prev;
  58       
  59       Node(const_reference p) : item(p), next(NULL), prev(NULL) {};
  60       /* To avoid -Weffc++ warnings */
  61       Node(const Node& n) : item(n.item), next(n.next), prev(n.prev) {};
  62 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  63       Node(const value_type&& p) : item(std::move(p.item)), next(std::move(p.next)), prev(std::move(p.prev)) {
  64         item = NULL;
  65         next = NULL;
  66         prev = NULL;
  67       };
  68 #endif
  69 
  70       friend void swap(Node& a, Node& b) {
  71         using std::swap;
  72 
  73         swap(a.item, b.item);
  74         swap(a.next, b.next);
  75         swap(a.prev, b.prev);
  76       }
  77 
  78       Node& operator=(Node n) {
  79         swap(*this, n);
  80         return *this;
  81       }
  82     };
  83 
  84     void deleteNode(Node* node) {
  85       Node* next = node->next; //NULL if node = tail
  86       Node* prev = node->prev; //NULL if node = head
  87 
  88       delete node;
  89 
  90       if (next) next->prev = prev;
  91       if (prev) prev->next = next;
  92 
  93       if (node == head) head = next;
  94       if (node == tail) tail = prev;
  95 
  96       --my_size;
  97     }
  98 
  99     Node* findNode(const_reference item) const {
 100       for (Node* current = head; current; current = current->next) {
 101         if (current->item == item) {
 102           return current;
 103         }
 104       }
 105       return NULL;
 106     }
 107 
 108     Node* head;
 109     Node* tail;
 110     size_type my_size;
 111 
 112   public:
 113     List() : head(NULL), tail(NULL), my_size(0) {};
 114     virtual ~List() { clear(); };
 115 
 116     List(const List& list) : head(NULL), tail(NULL), my_size(0) {
 117       for (Node* search = list.head; search; search = search->next)
 118         insert(search->item);
 119     }
 120 
 121 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 122     List(List&& list) : head(NULL), tail(NULL), my_size(0) {
 123       swap(*this, list);
 124     }
 125 #endif
 126 
 127     friend void swap(List& a, List& b) {
 128       using std::swap;
 129 
 130       swap(a.head, b.head);
 131       swap(a.tail, b.tail);
 132       swap(a.my_size, b.my_size);
 133     }
 134 
 135     List& operator=(List list) {
 136       swap(*this, list);
 137       return *this;
 138     }
 139 
 140     void clear() {
 141       Node* node = head, *node_n = NULL;
 142       while (node) {
 143         node_n = node->next;
 144         delete node;
 145         node = node_n;
 146       }
 147       head = tail = NULL;
 148       my_size = 0;
 149     };
 150 
 151     inline size_type size() const { return my_size; };
 152     inline bool isEmpty() const { return size() == 0; };
 153 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 154     inline explicit operator bool() const { return !isEmpty(); };
 155 #endif
 156 
 157     /**
 158      * @brief Insert into the list at the head
 159      * @param item The item to insert
 160      */
 161     void insert(const_reference item) {
 162       Node* node = new Node(item);
 163       if (!head) {
 164         head = tail = node;
 165       } else {
 166         head->prev = node;
 167         node->next = head;
 168         head = node;
 169       }
 170       ++my_size;
 171     }
 172 
 173     /**
 174      * @sa insert
 175      */
 176     inline friend List<value_type>& operator<< (List<value_type>& list, const_reference item) {
 177       list.insert(item);
 178       return list;
 179     }
 180 
 181     bool contains(const_reference item) const {
 182       return findNode(item) ? true : false;
 183     }
 184 
 185    const value_type find(const_reference item) const {
 186       Node* node = findNode(item);
 187       return node ? node->item : value_type();
 188     }
 189 
 190     /**
 191      * @brief Return a reference to the found element
 192      */
 193     reference findRef(const_reference item) {
 194       Node* node = findNode(item);
 195       return node ? node->item : head->item;
 196     }
 197 
 198     bool remove(const_reference item) {
 199       /* cases (removing X):
 200        * 1) list = (X)
 201        * 2) list = (X)->(Y)
 202        * 3) list = (W)->(X)
 203        * 4) list = (W)->(X)->(Y)
 204        */
 205       Node* node = findNode(item);
 206       if (node) {
 207         deleteNode(node);
 208         return true;
 209       }
 210       return false;
 211     }
 212 
 213     private:
 214     class ListIterator : public Iterator<value_type> {
 215       friend class List;
 216       private:
 217         List *list;
 218         Node *current;
 219         bool end;
 220 
 221         ListIterator(List &_list, bool _end = 0) : Iterator<value_type>(),
 222                                                               list(&_list),
 223                                                               current(_end ? _list.head : _list.tail), /* tail is beginning */
 224                                                               end(_end) {
 225         };
 226       public:
 227         ListIterator(const iterator& iter) : Iterator<value_type>(),
 228                                              list(iter.list),
 229                                              current(iter.current),
 230                                              end(iter.end) {
 231         };
 232         ListIterator() : Iterator<value_type>(),
 233                          list(NULL),
 234                          current(NULL),
 235                          end(0) {
 236         };
 237 
 238         iterator& operator =(const iterator& iter) {
 239           list = iter.list;
 240           current = iter.current;
 241           end = iter.end;
 242           return *this;
 243         }
 244 
 245         virtual ~ListIterator() { }
 246 
 247         virtual void remove() {
 248           Node *nextValue = end == 0 ? current->next : current->prev;
 249 
 250           list->deleteNode(current);
 251 
 252           current = nextValue;
 253         }
 254 
 255         virtual
 256 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 257           explicit
 258 #endif
 259         operator bool() const { return (current != NULL); };
 260         virtual operator value_type () { return operator*(); };
 261           
 262         virtual value_type& operator *() { return static_cast<value_type&>(current->item); }
 263 
 264         //Postfix
 265         virtual iterator operator ++(int) {
 266           iterator tmp(*this);
 267           current = current->prev;
 268           return tmp;
 269         }
 270 
 271         virtual iterator& operator ++() {
 272           current = current->prev;
 273           return *this;
 274         }
 275 
 276         //Postfix
 277         virtual iterator operator --(int) {
 278           iterator tmp(*this);
 279           current = current->next;
 280           return tmp;
 281         }
 282 
 283         virtual iterator& operator --() {
 284           current = current->next;
 285           return *this;
 286         }
 287     };
 288   public:
 289     inline iterator begin() { return iterator(*this); };
 290     inline iterator end() { return iterator(*this, 1); };
 291 };

 295 #endif /* _BD_LIST_H */
