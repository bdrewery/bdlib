  30 template <class T>
  31 /**
  32   * @class List
  33   * @brief List data structure
  34   */
  35 class List {
  36   class ListIterator;
  37 
  38   public:
  39     typedef ListIterator iterator;
  40     typedef T iterator_type;
  41 
  42 
  43   protected:
  44     struct Node {
  45       iterator_type item;
  46       Node* next;
  47       Node* prev;
  48       
  49       Node(const iterator_type& p) : item(p), next(NULL), prev(NULL) {};
  50       /* To avoid -Weffc++ warnings */
  51       Node(const Node& n) : item(n.item), next(n.next), prev(n.prev) {};
  52 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  53       Node(iterator_type&& p) : item(NULL), next(NULL), prev(NULL) {
  54         swap(*this, p);
  55       };
  56 #endif
  57 
  58       friend void swap(Node& a, Node& b) {
  59         using std::swap;
  60 
  61         swap(a.item, b.item);
  62         swap(a.next, b.next);
  63         swap(a.prev, b.prev);
  64       }
  65 
  66       Node& operator=(Node n) {
  67         swap(*this, n);
  68         return *this;
  69       }
  70     };
  71 
  72     void deleteNode(Node* node) {
  73       Node* next = node->next; //NULL if node = tail
  74       Node* prev = node->prev; //NULL if node = head
  75 
  76       delete node;
  77 
  78       if (next) next->prev = prev;
  79       if (prev) prev->next = next;
  80 
  81       if (node == head) head = next;
  82       if (node == tail) tail = prev;
  83 
  84       --my_size;
  85     }
  86 
  87     Node* findNode(const iterator_type& item) const {
  88       for (Node* current = head; current; current = current->next) {
  89         if (current->item == item) {
  90           return current;
  91         }
  92       }
  93       return NULL;
  94     }
  95 
  96     Node* head;
  97     Node* tail;
  98     size_t my_size;
  99 
 100   public:
 101     List() : head(NULL), tail(NULL), my_size(0) {};
 102     virtual ~List() { clear(); };
 103 
 104     List(const List& list) : head(NULL), tail(NULL), my_size(0) {
 105       for (Node* search = list.head; search; search = search->next)
 106         insert(search->item);
 107     }
 108 
 109 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 110     List(List&& list) : head(NULL), tail(NULL), my_size(0) {
 111       swap(*this, list);
 112     }
 113 #endif
 114 
 115     friend void swap(List& a, List& b) {
 116       using std::swap;
 117 
 118       swap(a.head, b.head);
 119       swap(a.tail, b.tail);
 120       swap(a.my_size, b.my_size);
 121     }
 122 
 123     List& operator=(List list) {
 124       swap(*this, list);
 125       return *this;
 126     }
 127 
 128     void clear() {
 129       Node* node = head, *node_n = NULL;
 130       while (node) {
 131         node_n = node->next;
 132         delete node;
 133         node = node_n;
 134       }
 135       head = tail = NULL;
 136       my_size = 0;
 137     };
 138 
 139     inline size_t size() const { return my_size; };
 140     inline bool isEmpty() const { return size() == 0; };
 141     inline explicit operator bool() const { return !isEmpty(); };
 142 
 143     /**
 144      * @brief Insert into the list at the head
 145      * @param item The item to insert
 146      */
 147     void insert(const iterator_type& item) {
 148       Node* node = new Node(item);
 149       if (!head) {
 150         head = tail = node;
 151       } else {
 152         head->prev = node;
 153         node->next = head;
 154         head = node;
 155       }
 156       ++my_size;
 157     }
 158 
 159     /**
 160      * @sa insert
 161      */
 162     inline friend List<iterator_type>& operator<< (List<iterator_type>& list, const iterator_type& item) {
 163       list.insert(item);
 164       return list;
 165     }
 166 
 167     bool contains(const iterator_type& item) const {
 168       return findNode(item) ? true : false;
 169     }
 170 
 171    const iterator_type find(const iterator_type& item) const {
 172       Node* node = findNode(item);
 173       return node ? node->item : iterator_type();
 174     }
 175 
 176     /**
 177      * @brief Return a reference to the found element
 178      */
 179     iterator_type& findRef(const iterator_type& item) {
 180       Node* node = findNode(item);
 181       return node ? node->item : head->item;
 182     }
 183 
 184     bool remove(const iterator_type& item) {
 185       /* cases (removing X):
 186        * 1) list = (X)
 187        * 2) list = (X)->(Y)
 188        * 3) list = (W)->(X)
 189        * 4) list = (W)->(X)->(Y)
 190        */
 191       Node* node = findNode(item);
 192       if (node) {
 193         deleteNode(node);
 194         return true;
 195       }
 196       return false;
 197     }
 198 
 199     private:
 200     class ListIterator : public Iterator<iterator_type> {
 201       friend class List;
 202       private:
 203         List *list;
 204         Node *current;
 205         bool end;
 206 
 207         ListIterator(List &_list, bool _end = 0) : Iterator<iterator_type>(),
 208                                                               list(&_list),
 209                                                               current(_end ? _list.head : _list.tail), /* tail is beginning */
 210                                                               end(_end) {
 211         };
 212       public:
 213         ListIterator(const iterator& iter) : Iterator<iterator_type>(),
 214                                              list(iter.list),
 215                                              current(iter.current),
 216                                              end(iter.end) {
 217         };
 218         ListIterator() : Iterator<iterator_type>(),
 219                          list(NULL),
 220                          current(NULL),
 221                          end(0) {
 222         };
 223 
 224         iterator& operator =(const iterator& iter) {
 225           list = iter.list;
 226           current = iter.current;
 227           end = iter.end;
 228           return *this;
 229         }
 230 
 231         virtual ~ListIterator() { }
 232 
 233         virtual void remove() {
 234           Node *nextValue = end == 0 ? current->next : current->prev;
 235 
 236           list->deleteNode(current);
 237 
 238           current = nextValue;
 239         }
 240 
 241         virtual explicit operator bool() const { return (current != NULL); };
 242         virtual operator iterator_type () { return operator*(); };
 243           
 244         virtual iterator_type& operator *() { return static_cast<iterator_type&>(current->item); }
 245 
 246         //Postfix
 247         virtual iterator operator ++(int) {
 248           iterator tmp(*this);
 249           current = current->prev;
 250           return tmp;
 251         }
 252 
 253         virtual iterator& operator ++() {
 254           current = current->prev;
 255           return *this;
 256         }
 257 
 258         //Postfix
 259         virtual iterator operator --(int) {
 260           iterator tmp(*this);
 261           current = current->next;
 262           return tmp;
 263         }
 264 
 265         virtual iterator& operator --() {
 266           current = current->next;
 267           return *this;
 268         }
 269     };
 270   public:
 271     inline iterator begin() { return iterator(*this); };
 272     inline iterator end() { return iterator(*this, 1); };
 273 };
