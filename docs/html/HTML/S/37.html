  30 template <class T>
  31 /**
  32   * @class List
  33   * @brief List data structure
  34   */
  35 class List {
  36   class ListIterator;
  37 
  38   public:
  39     typedef ListIterator iterator;
  40     typedef T iterator_type;
  41 
  42 
  43   protected:
  44     struct Node {
  45       iterator_type ptr;
  46       Node* next;
  47       Node* prev;
  48       
  49       Node(const iterator_type& p) : ptr(p), next(NULL), prev(NULL) {};
  50       /* To avoid -Weffc++ warnings */
  51       Node(const Node& n) : ptr(n.ptr), next(n.next), prev(n.prev) {};
  52       Node& operator=(const Node& n) {
  53         ptr = n.ptr;
  54         next = n.next;
  55         prev = n.prev;
  56       }
  57     };
  58 
  59     void deleteNode(Node* node) {
  60       Node* next = node->next; //NULL if node = tail
  61       Node* prev = node->prev; //NULL if node = head
  62 
  63       delete node;
  64 
  65       if (next) next->prev = prev;
  66       if (prev) prev->next = next;
  67 
  68       if (node == head) head = next;
  69       if (node == tail) tail = prev;
  70 
  71       --my_size;
  72     }
  73 
  74     Node* head;
  75     Node* tail;
  76     size_t my_size;
  77 
  78   public:
  79     List() : head(NULL), tail(NULL), my_size(0) {};
  80     virtual ~List() { clear(); };
  81 
  82     List(const List& list) : head(NULL), tail(NULL), my_size(0) {
  83       for (Node* search = list.head; search; search = search->next)
  84         insert(search->ptr);
  85     }
  86 
  87     List& operator =(const List& list) {
  88       if (&list != this) {
  89         clear();
  90         for (Node* search = list.head; search; search = search->next)
  91           insert(search->ptr);
  92       }
  93       return *this;
  94     }
  95 
  96     void clear() {
  97       Node* node = head, *node_n = NULL;
  98       while (node) {
  99         node_n = node->next;
 100         delete node;
 101         node = node_n;
 102       }
 103       head = tail = NULL;
 104       my_size = 0;
 105     };
 106 
 107     inline size_t size() const { return my_size; };
 108     inline bool isEmpty() const { return size() == 0; };
 109     //inline operator bool() const { return !isEmpty(); };
 110 
 111     /**
 112      * @brief Insert into the list at the head
 113      * @param ptr The ptr to insert
 114      */
 115     void insert(const iterator_type& ptr) {
 116       Node* node = new Node(ptr);
 117       if (!head) {
 118         head = tail = node;
 119       } else {
 120         head->prev = node;
 121         node->next = head;
 122         head = node;
 123       }
 124       ++my_size;
 125     }
 126 
 127     /**
 128      * @sa insert
 129      */
 130     inline friend List<iterator_type>& operator<< (List<iterator_type>& list, const iterator_type& item) {
 131       list.insert(item);
 132       return list;
 133     }
 134 
 135     bool contains(const iterator_type& ptr) const {
 136       if (head) {
 137         for (Node* node = head; node; node = node->next) {
 138           if (node->ptr == ptr)
 139             return 1;
 140         }
 141       }
 142       return 0;
 143     }
 144 
 145    const iterator_type find(const iterator_type& ptr) const {
 146       const iterator_type empty;
 147       if (isEmpty()) return empty;
 148 
 149       for (Node* current = head; current; current = current->next) {
 150         if (current->ptr == ptr)
 151           return current->ptr;
 152       }
 153       return empty;
 154     }
 155 
 156     /**
 157      * @brief Return a reference to the found element
 158      */
 159     iterator_type& findRef(const iterator_type& ptr) {
 160       Node* current = head;
 161       for (; current; current = current->next) {
 162         if (current->ptr == ptr)
 163           break;
 164       }
 165       return current->ptr;
 166     }
 167 
 168     bool remove(const iterator_type& ptr) {
 169       if (head) {
 170       /* cases (removing X):
 171        * 1) list = (X)
 172        * 2) list = (X)->(Y)
 173        * 3) list = (W)->(X)
 174        * 4) list = (W)->(X)->(Y)
 175        */
 176         for (Node* node = head; node; node = node->next) {
 177           if (node->ptr == ptr) {
 178             deleteNode(node);
 179             return true;
 180           }
 181         }
 182       }
 183       return false;
 184     }
 185 
 186     private:
 187     class ListIterator : public Iterator<iterator_type> {
 188       friend class List;
 189       private:
 190         List *list;
 191         Node *current;
 192         bool end;
 193 
 194         ListIterator(List &_list, bool _end = 0) : Iterator<iterator_type>(),
 195                                                               list(&_list),
 196                                                               current(_end ? _list.head : _list.tail), /* tail is beginning */
 197                                                               end(_end) {
 198         };
 199       public:
 200         ListIterator(const iterator& iter) : Iterator<iterator_type>(),
 201                                              list(iter.list),
 202                                              current(iter.current),
 203                                              end(iter.end) {
 204         };
 205         ListIterator() : Iterator<iterator_type>(),
 206                          list(NULL),
 207                          current(NULL),
 208                          end(0) {
 209         };
 210 
 211         iterator& operator =(const iterator& iter) {
 212           list = iter.list;
 213           current = iter.current;
 214           end = iter.end;
 215           return *this;
 216         }
 217 
 218         virtual ~ListIterator() { }
 219 
 220         virtual void remove() {
 221           Node *nextValue = end == 0 ? current->next : current->prev;
 222 
 223           list->deleteNode(current);
 224 
 225           current = nextValue;
 226         }
 227 
 228         virtual operator bool() { return (current != NULL); };
 229         virtual operator iterator_type () { return operator*(); };
 230           
 231         virtual iterator_type& operator *() { return static_cast<iterator_type&>(current->ptr); }
 232 
 233         //Postfix
 234         virtual iterator operator ++(int) {
 235           iterator tmp(*this);
 236           current = current->prev;
 237           return tmp;
 238         }
 239 
 240         virtual iterator& operator ++() {
 241           current = current->prev;
 242           return *this;
 243         }
 244 
 245         //Postfix
 246         virtual iterator operator --(int) {
 247           iterator tmp(*this);
 248           current = current->next;
 249           return tmp;
 250         }
 251 
 252         virtual iterator& operator --() {
 253           current = current->next;
 254           return *this;
 255         }
 256     };
 257   public:
 258     inline iterator begin() { return iterator(*this); };
 259     inline iterator end() { return iterator(*this, 1); };
 260 };
