  42 template <class T, class Allocator = std::allocator<T> >
  43 /**
  44  * @class ArrayRef
  45  * @brief Helps the String and Array classes with reference counting
  46  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  47  */
  48 class ArrayRef {
  49   public:
  50     typedef typename Allocator::pointer iterator;
  51     mutable Allocator alloc;
  52     mutable size_t size; //Capacity of buffer
  53     union {
  54       mutable iterator buf;
  55       mutable T buf_local[16];
  56     };
  57     mutable std::atomic<int> refs; //References
  58 
  59     ArrayRef(const Allocator& allocator = Allocator()) : alloc(allocator), size(0), buf(nullptr), refs(1) {};
  60     ~ArrayRef() {
  61       if (buf) {
  62         FreeBuf(buf);
  63       }
  64     };
  65     /**
  66      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  67      * @param newSize A size that we need to Allocate the buffer to.
  68      * @param scaling_factor How much to multiply the size by to help avoid later resizing
  69      * @param offset The offset of the old buffer so we know where to start
  70      * @param sublen The length of the subarray in use
  71      * @pre newSize is > 0 (assumed as size_t is unsigned)
  72      * @post The buffer is at least nsize bytes long.
  73      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  74      */
  75     void Reserve(size_t newSize, double scaling_factor, size_t& offset, size_t sublen) const
  76     {
  77       /* Don't new if we already have enough room! */
  78       if (size < newSize) {
  79         newSize = std::max(size_t(size * scaling_factor), newSize);
  80 
  81         iterator newbuf = alloc.allocate(newSize, buf);
  82 
  83         if (newbuf != buf) {
  84           // Initialize new memory
  85           std::uninitialized_fill(newbuf, newbuf + newSize, T());
  86           if (buf) {
  87             /* Copy old buffer into new - only copy the subarray */
  88             std::copy(buf + offset, buf + offset + sublen, newbuf);
  89             FreeBuf(buf);
  90           }
  91           buf = newbuf;
  92           size = newSize;
  93           offset = 0;
  94         }
  95       } else if ((size - offset) < newSize) {
  96         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  97         // Shift everything to the beginning and reset the offset
  98         /* Only copy the subarray */
  99         std::memmove(static_cast<void*>(buf), static_cast<void*>(buf + offset), sublen);
 100         offset = 0;
 101       }
 102     }
 103 
 104     /**
 105      * @brief Free's up the allocated buffer.
 106      * @param p The buffer to be free'd
 107      * @post The buffer is deleted.
 108      * @todo Implement mempool here.
 109      */
 110     inline void FreeBuf(iterator p) const {
 111       for (iterator i = p; i != p + size; ++i) {
 112         alloc.destroy(i);
 113       }
 114       alloc.deallocate(p, size);
 115     }
 116 
 117     /**
 118      * @brief Is this ReferenceCountedArray shared?
 119      */
 120     inline bool isShared() const { return refs > 1; };
 121   private:
 122     // No copying allowed
 123     ArrayRef(const ArrayRef&) = delete;
 124     ArrayRef& operator=(const ArrayRef&) = delete;
 125 };

 128 /**
 129  * @class Slice
 130  * @brief Safe subarray reading and writing.
 131  * This class should be optimized away and fully inlined such that:
 132  * ReferenceCountedArray s("look over there");
 133  * s(0, 4) = "blah"';
 134  * Should be rewritten as:
 135  * s.replace(0, "blah", 4);
 136  */
 137 class Slice {
 138   private:
 139     T& rca;
 140     int start;
 141     int len;
 142 
 143     Slice() = delete;
 144 
 145   public:
 146     Slice(T& _rca, int _start, int _len) : rca(_rca), start(_start), len(_len) {};
 147     Slice(const Slice& slice) : rca(slice.rca), start(slice.start), len(slice.len) {};
 148 
 149     /**
 150      * @brief return a new (const) slice
 151      */
 152     inline operator T() const {
 153       T newArray(rca);
 154       newArray.slice(start, len);
 155       return newArray;
 156     };
 157 
 158     friend void swap(Slice& a, Slice& b) {
 159       using std::swap;
 160 
 161       swap(a.alloc, b.alloc);
 162       swap(a.rca, b.rca);
 163       swap(a.start, b.start);
 164       swap(a.len, b.len);
 165     }
 166 
 167     /**
 168      * @brief Assign a Slice to a Slice
 169      */
 170     inline Slice& operator=(const Slice& slice) {
 171       (*this) = T(slice);
 172       return (*this);
 173     }
 174 
 175     /**
 176      * @brief Assign to a Slice
 177      * @todo This needs to account for negative start/len
 178      */
 179     inline Slice& operator=(const T& array) {
 180       rca.replace(start, array, len);
 181       return (*this);
 182     }

 194 template <class T, class Allocator = std::allocator<T> >
 195 /**
 196  * @class ReferenceCountedArray
 197  * @brief Common template base class for String and Array
 198  */
 199 class ReferenceCountedArray : public ReferenceCountedArrayBase {
 200   public:
 201     typedef T value_type;
 202 
 203     typedef typename Allocator::size_type             size_type;
 204     typedef typename Allocator::difference_type       difference_type;
 205     typedef typename Allocator::pointer               pointer;
 206     typedef typename Allocator::const_pointer         const_pointer;
 207     typedef typename Allocator::reference             reference;
 208     typedef typename Allocator::const_reference       const_reference;
 209     typedef typename Allocator::pointer               iterator;
 210     typedef typename Allocator::const_pointer         const_iterator;
 211     typedef std::random_access_iterator_tag           iterator_category;
 212     typedef std::reverse_iterator<const_iterator>     const_reverse_iterator;
 213     typedef std::reverse_iterator<iterator>           reverse_iterator;
 214 
 215   private:
 216     /**
 217      * @brief Detach from the shared reference.
 218      * This is only called when losing the old buffer or when modifying the buffer (and copy-on-write is used)
 219      * @note This does not free the old reference, as it is still in use
 220      */
 221     void doDetach() const {
 222       decRef();
 223       Ref = nullptr;
 224       sublen = 0;
 225       offset = 0;
 226       my_hash = 0;
 227     }
 228     /**
 229      * @brief Increment our reference counter.
 230      */
 231     inline int incRef() const { return Ref ? ++Ref->refs : 0; };
 232 
 233     /**
 234      * @brief Decrement our reference counter.
 235      */
 236     inline int decRef() const { return Ref ? --Ref->refs : 0; };
 237 
 238   protected:
 239     /**
 240      * @brief Set the lengths to the specified length
 241      * @param newLen the new length to set to
 242      */
 243     inline void setLength(size_t newLen) const { sublen = newLen; my_hash = 0; };
 244 
 245     /**
 246      * @sa setLength()
 247      */
 248     inline void addLength(size_t diff) const { sublen += diff; my_hash = 0; };
 249 
 250     /**
 251      * @sa setLength()
 252      */
 253     inline void subLength(size_t diff) const { sublen -= diff; my_hash = 0; };
 254 
 255     /**
 256      * @brief Mutable Ref->buf+offset reference for use internally
 257      */
 258     //pointer mdata() const { return Buf() + offset; };

 260     /**
 261      * @brief Mutable Ref->buf reference for use internally
 262      */
 263     inline pointer Buf(size_t pos = 0) const { return Ref ? &Ref->buf[offset + pos] : nullptr; };
 264 
 265     /**
 266      * @brief Ref->buf reference for use internally
 267      */
 268     inline const_pointer constBuf(size_t pos = 0) const { return Buf(pos); };
 269   private:
 270     Allocator alloc;
 271     /**
 272      * @brief The array reference for reference counting
 273      * This is mutable so that Ref->refs can be modified, which really is mutable
 274      */
 275     mutable ArrayRef<value_type, Allocator> *Ref;
 276   protected:
 277     /**
 278      * Return the real buffer's start point, without accounting for offset. This is used for cleaning the buffer when needed.
 279      */
 280     inline const_pointer real_begin() const { return Ref ? Ref->buf : nullptr; };
 281 
 282     /**
 283      * This is for subarrays: so we know where the subarray starts.
 284      */
 285     mutable size_t offset;
 286     /**
 287      * This is for subarrays: so we know where the subarray ends.
 288      */
 289     mutable size_t sublen;
 290 
 291     /**
 292      * Cache of current hash() result. 0 if stale
 293      */
 294     mutable size_t my_hash;
 295 
 296   private:
 297     /**
 298      * @brief Detach from the reference
 299      * This is called when the old buffer is no longer needed.
 300      * ie, operator=() was called.
 301      */
 302     void Detach() {
 303       if (isShared()) {
 304         doDetach();
 305       } else {
 306         setLength(0);
 307         offset = 0;
 308         my_hash = 0;
 309       }
 310     }
 311 
 312     /**
 313      * @brief Free up our reference if we have the last one.
 314      * @post The reference counter is decremented.
 315      * @post If this was the last Reference, it is free'd
 316      * This is only called in ~Array() and operator=(Array&).
 317      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 318      */
 319     inline void CheckDeallocRef() {
 320       if (Ref && decRef() < 1) {
 321         delete Ref;
 322         Ref = nullptr;
 323       }
 324     }
 325 
 326     //    void COW(size_t) const;
 327     /**
 328      * @brief Ensure that our internal buffer is unshared.
 329      * @param n Create/Grow the buffer to this size.
 330      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 331      * @post The internal buffer/data is unshared
 332      * @post The buffer is at least size n.
 333      * @post The buffer is deep copied to a new buffer.
 334      *
 335      * Ensure that our internal buffer is unshared.
 336      * If needed, performs a deep copy into a new buffer (COW).
 337      * Also take a hint size n of the new ReferenceCountedArray's size as to avoid needless copying/allocing.
 338      */
 339     void COW(size_t n) const {
 340       const_pointer oldBuf = constBuf();
 341       const size_t oldLength = length();
 342 
 343       doDetach(); //Detach from the shared reference
 344       Reserve(std::max(oldLength, n), _rca_cow_scaling_factor); //Will set capacity()/size
 345       std::copy(oldBuf, oldBuf + oldLength, Buf());
 346       setLength(oldLength);
 347     }
 348 
 349   protected:
 350     /**
 351      * @brief Force COW if needed
 352      * @post The array is no longer shared, if it was.
 353      */
 354     inline void getOwnCopy() const { AboutToModify(length()); };
 355 
 356     /**
 357      * @brief Warn the reference counting that it may need to COW
 358      * @post The buffer is detached/COW, and possibly larger
 359      */
 360     inline void AboutToModify(size_t n) const {
 361       my_hash = 0;
 362       if (isShared())
 363         COW(n); // Clears the offset
 364       else {
 365         Reserve(n, _rca_cow_scaling_factor);
 366         /* Shift the offset away */
 367       }
 368     }

 370     /**
 371      * @brief Checks if the buffer has the given index or not.
 372      * @param pos Index to check.
 373      */
 374     inline void validateIndex(size_t pos) const {
 375       if (pos >= length())
 376         throw std::out_of_range("ReferenceCountedArray::validateIndex");
 377     };

 383     ReferenceCountedArray(ReferenceCountedArray&& rca) : ReferenceCountedArrayBase(), alloc(std::move(rca.alloc)), Ref(std::move(rca.Ref)), offset(std::move(rca.offset)), sublen(std::move(rca.sublen)), my_hash(std::move(rca.my_hash)) {

 391     /**
 392      * @brief Create an empty container with at least the specified bytes in size.
 393      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 394      * @post This ReferenceCountedArray's memory will also never be shrunk.
 395      * @post A buffer has been created.
 396      *
 397      * The idea behind this is that if a specific size was asked for, the buffer is like
 398      * a char buf[N];

 400     explicit ReferenceCountedArray(const size_t newSize, const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(nullptr), offset(0), sublen(0), my_hash(0) {
 401       if (newSize) {

 403       }
 404     };
 405     /**
 406      * @brief Create a container filled with n copies of the given value.
 407      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 408      * @param value The value to populate the array with
 409      * @post This ReferenceCountedArray's memory will also never be shrunk.
 410      * @post A buffer has been created.
 411      *
 412      */
 413     ReferenceCountedArray(const size_t newSize, const value_type value, const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(nullptr), offset(0), sublen(0), my_hash(0) {
 414       if (newSize) {
 415         Reserve(newSize);
 416 
 417         for (size_t i = 0; i < newSize; ++i) {
 418           *(Buf(i)) = value;
 419         }
 420         this->setLength(newSize);
 421       }
 422     }

 641    /**
 642      * @brief Return a hash of every element in the array. Cache result as well.
 643      * @note DJB's hash function
 644      */
 645     virtual size_t hash() const {
 646       if (my_hash != 0) return my_hash;
 647       std::hash<value_type> hasher;
 648       size_t _hash = 5381;
 649 
 650       for(size_t i = 0; i < this->length(); ++i)
 651         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 652       return (my_hash = (_hash & 0x7FFFFFFF));
 653     }

 667     /**
 668      * @brief Find an item in the array, starting from the end
 669      * @return The position of the item if found, or npos if not found
 670      **/
 671     size_t rfind(const_reference item) const {
 672       for (size_t i = length() - 1; i + 1 > 0; --i)
 673         if (*(Buf(i)) == item)
 674           return i;
 675       return npos;
 676     }

 697      * @brief Safe element access operator
 698      * @todo This is only called on a (const) ReferenceCountedArray, but should for a ReferenceCountedArray as well.

 704     /**
 705      * @class Cref
 706      * @brief Safe element reading and writing.
 707      * This class should be optimized away and fully inlined such that:
 708      * ReferenceCountedArray s;
 709      * s[0] = 'a';
 710      * Should be rewritten as:
 711      * s.write(0, 'a');
 712      */
 713     class Cref {
 714       private:
 715         friend class ReferenceCountedArray;
 716         ReferenceCountedArray& rca;
 717         size_t k;
 718 
 719         /**
 720          * @brief Used by Cref operator[]
 721          */
 722         Cref(ReferenceCountedArray& _rca, size_t pos) : rca(_rca), k(pos) {};
 723         Cref() = delete;
 724 
 725       public:
 726         Cref(const Cref& cref) : rca(cref.rca), k(cref.k) {};
 727         inline Cref& operator=(const Cref& cref) {
 728           (*this) = value_type(cref);
 729           return (*this);
 730         }
 731 
 732         public:
 733         /**
 734          * @sa ReferenceCountedArray::operator[]
 735          */
 736         inline operator value_type() const noexcept { return rca.read(k); };
 737 
 738         /**
 739          * Stroustrup shows using this as void with no return value, but that breaks chaining a[n] = b[n] = 'b';
 740          */
 741         inline Cref& operator=(value_type c) {
 742           rca.write(k, c);
 743           return (*this);
 744         };
 745     };
 746 
 747     /**
 748      * @brief Returns 'Cref' class for safe (cow) writing.
 749      * @sa Cref
 750      */
 751     inline Cref operator[](size_t pos) { return Cref(*this, pos); };
 752 
 753     /**
 754      * @brief Returns the character at the given index.
 755      * @return The character at the given index.
 756      * @param pos Index to return.
 757      * @pre The index must exist.
 758      * @sa operator[]()
 759      * @todo Perhaps this should throw an exception if out of range?
 760      */
 761     inline value_type at(size_t pos) const {
 762       validateIndex(pos);
 763       return (*this)[pos];
 764     };
 765 
 766     /**
 767      * @param start The offset to begin the subarray from (indexed from 0)
 768      * @param len The length of the subarray to return
 769      * The returned slice is a reference to the original array until modified.
 770      */
 771     void slice(int start, int len = -1) {
 772       if (len == -1) len = int(length()) - start;
 773       // Start is after the end, set us to an empty array
 774       if (start >= static_cast<signed>(length())) {
 775         offset = length();
 776         setLength(0);
 777         return;
 778       }
 779 
 780       //Count backwards from the end
 781       if (start < 0)
 782         start = length() + start;
 783       // Start was before the beginning, just reset to the beginning
 784       if (start < 0)
 785         start = 0;
 786 
 787       offset += start;
 788       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 789       if (start + len >= static_cast<signed>(length()))
 790         len = length() - start;
 791       // If the length is negative, stop from counting backwards from the end
 792       else if (len < 0)
 793         len = length() - start + len;
 794       setLength(len);
 795     }

 797     /**
 798      * @brief Appends one item to end of buffer.
 799      * @param item The item to be appended.
 800      * @post The buffer is allocated.
 801      * @post The item is appended at the end of the buffer.
 802      * This is the same as inserting the item at the end of the buffer.
 803      */
 804     inline void append(const_reference item) { insert(length(), item); };
 805     /**
 806      * @brief Appends given rca to the end of buffer
 807      * @param rca The rca to be appended.
 808      * @param n How many characters to copy from the ReferenceCountedArray object.
 809      * @post The buffer is allocated.
 810      * This is the same as inserting the rca at the end of the buffer.
 811      */
 812     inline void append(const ReferenceCountedArray& rca, size_t n = npos) { insert(length(), rca, n); };

 815     /**
 816      * @brief Inserts a ReferenceCountedArray object into our buffer
 817      * @param pos The index to insert at.
 818      * @param rca The rca to insert.
 819      * @param n The length to insert.
 820      * @post The buffer contains n items from rca inserted at index pos.
 821      */
 822     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 823       if (n == 0) return;
 824       if (pos != 0)
 825         validateIndex(pos - 1);
 826 
 827       size_t slen = rca.length();
 828 
 829       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 830       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 831         *this = rca;
 832         return;
 833       }
 834 
 835       if (n == npos || n > slen)
 836         n = slen;
 837       slen -= slen - n;
 838       AboutToModify(length() + slen);
 839       std::memmove(static_cast<void*>(Buf() + pos + slen), static_cast<void*>(Buf() + pos), length() - pos);
 840       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 841       addLength(slen);
 842     }

 844     /**
 845      * @brief Insert an item at the given index.
 846      * @param pos The index to insert at.
 847      * @param item The item to be inserted.
 848      * @post A buffer is allocated.
 849      * @post If the old buffer was too small, it is enlarged.
 850      * @post The item is inserted at the given index.
 851      */
 852     void insert(size_t pos, const_reference item)
 853     {
 854       if (pos != 0)
 855         validateIndex(pos - 1);
 856 
 857       AboutToModify(length() + 1);
 858       std::memmove(static_cast<void*>(Buf() + pos + 1), static_cast<void*>(Buf() + pos), length() - pos);
 859       *(Buf(pos)) = item;
 860       addLength(1);
 861     }

 863     /**
 864      * @brief Replace the given index with the given item.
 865      * @param pos The index to replace.
 866      * @param item The item to replace with.
 867      * @post The given index has been replaced.
 868      * @post COW is done if needed.
 869      */
 870     void replace(size_t pos, const_reference item) {
 871       if (pos != 0)
 872         validateIndex(pos - 1);
 873 
 874       getOwnCopy();
 875       *(Buf(pos)) = item;
 876     }
 877 
 878     /**
 879      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 880      * @param pos The index to replace at.
 881      * @param rca The ReferenceCountedArray object to replace with.
 882      * @param n The number of characters to use for the replace.
 883      */
 884     void replace(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 885       if (n == 0) return;
 886       if (pos != 0)
 887         validateIndex(pos - 1);
 888 
 889       size_t slen = rca.length();
 890 
 891       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 892       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 893         *this = rca;
 894         return;
 895       }
 896 
 897       if (n == npos || n > slen)
 898         n = slen;
 899       slen -= slen - n;
 900 
 901       size_t newlen = pos + slen;
 902 
 903       if (newlen >= length()) {
 904         AboutToModify(newlen);
 905       } else {
 906         newlen = length();
 907         getOwnCopy();
 908       }
 909       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 910       setLength(newlen);
 911     }
