 153 void StreamTest :: readTest (void)
 154 {
 155   String sbuf, gbuf;
 156 
 157   a->puts("This is line 1\n");
 158   a->puts("This is line 2\n");
 159   a->puts("This is line 3\n");
 160   a->puts("This is line 4\n");
 161   a->puts("This is line 5\n");
 162   a->puts("This is line 6\n");
 163   a->puts("This is line 7\n");
 164   a->puts("This is line 8\n");
 165   a->puts("This is line 9\n");
 166   a->puts("This is line10\n");
 167   a->puts("This is line11\n");
 168   a->puts("This is line12\n");
 169   a->puts("This is line13\n");
 170   a->puts("This is line14\n");
 171   a->seek(0, SEEK_SET);
 172   int x = 0;
 173   size_t pos = 0;
 174   while (a->tell() < a->length()) {
 175     sbuf = String::printf("This is line%2d\nThis is line%2d\n", x + 1, x + 2);
 176     gbuf = a->read(30);
 177     pos += gbuf.length();
 178     CPPUNIT_ASSERT_EQUAL(pos, a->tell());
 179     CPPUNIT_ASSERT_EQUAL((size_t) 30, gbuf.length());
 180     CPPUNIT_ASSERT_STRING_EQUAL(sbuf, gbuf);
 181     x += 2;
 182   }
 183   CPPUNIT_ASSERT_EQUAL(14, x);
 184 
 185   size_t savelen = a->tell();
 186   a->seek(a->tell(), SEEK_SET);
 187   a->puts("This is line15\n");
 188   a->seek(savelen, SEEK_SET);
 189   sbuf = "This is line15\n";
 190   gbuf = a->read(30);
 191   pos += gbuf.length();
 192   CPPUNIT_ASSERT_EQUAL(pos, a->tell());
 193   CPPUNIT_ASSERT_EQUAL((size_t) 15, gbuf.length());
 194   CPPUNIT_ASSERT_STRING_EQUAL(gbuf, sbuf);
 195 }

 229 void StreamTest :: loadFileTest (void)
 230 {
 231   const char *file = "/etc/services";
 232 
 233   FILE *f = nullptr;
 234   f = fopen(file, "rb");
 235   if (f == nullptr)
 236     return;
 237 
 238   fseek(f, 0, SEEK_END);
 239   size_t size = ftell(f);
 240   fseek(f, 0, SEEK_SET);
 241 
 242   char *buffer = (char*) malloc(size + 1);
 243   fread(buffer, 1, size, f);
 244   buffer[size] = 0;
 245 
 246   a->loadFile(file);
 247 
 248   CPPUNIT_ASSERT_EQUAL(size, a->length());
 249   CPPUNIT_ASSERT(*a == buffer);
 250   free(buffer);
 251 
 252   char buf[1024] = "";
 253   String gbuf;
 254 
 255   fseek(f, 0, SEEK_SET);
 256   while (fgets(buf, sizeof(buf), f) != nullptr) {
 257     gbuf = a->getline(sizeof(buf) - 1);
 258     CPPUNIT_ASSERT_STRING_EQUAL(String(buf), gbuf);
 259   }
 260   fclose(f);
 261 
 262   struct stat s;
 263   stat(file, &s);
 264   CPPUNIT_ASSERT_EQUAL(a->length(), size_t(s.st_size));
 265 }
