 454 void ScriptInterpTCLTest :: createCommandEventTest (void)
 455 {
 456   ScriptInterpTCL tcl_script;
 457   ScriptCallbackerPtr scb;
 458   String result, input;
 459   Array<String> params;
 460 
 461   tcl_script.createCommand("param_test", param_test);
 462 
 463   /* Try a TCL Callback from C++ (event binding from C)
 464    * Execute script_bind from TCL, which will record the bound event callback in Events
 465    * Then call the Event and verify that it returns our expected result
 466    */
 467   // Create script_bind which will add the passed proc to our event handling procedures (Just an Array for now)
 468   tcl_script.createCommand("bind", script_bind);
 469   // Bind an event
 470   tcl_script.eval("proc echo {args} { return $args }");
 471   tcl_script.eval("bind test echo");
 472 
 473   // Call the event from C++ and then verify it echos back the input
 474   CPPUNIT_ASSERT_NO_THROW(scb = Events.at("test"));
 475   input = "my params";
 476   params << input;
 477   // FIXME: This call line needs help to accept arbitrary params
 478   result = scb->call(params);
 479   CPPUNIT_ASSERT_STRING_EQUAL(BDLIB_NS::String::printf("{%s}", input.c_str()), result);
 480   params.clear();
 481 
 482   tcl_script.eval("bind \"complex test\" param_test");
 483   CPPUNIT_ASSERT_NO_THROW(scb = Events.at("complex test"));
 484 
 485   // Now trigger the callback and verify that it calls the passed event
 486   params << "test";
 487   params << "42";
 488   result = scb->call(params);
 489   CPPUNIT_ASSERT_STRING_EQUAL("I got 2 args, arg1: test arg2: 42", result);
 490   params.clear();
 491 
 492   params << "some argument";
 493   params << "45";
 494   result = scb->call(params);
 495   CPPUNIT_ASSERT_STRING_EQUAL("I got 2 args, arg1: some argument arg2: 45", result);
