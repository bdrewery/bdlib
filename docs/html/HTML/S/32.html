  83 template <class Key, class Value>
  84 struct KeyValue {
  85   public:
  86     Key k;
  87     Value v;
  88 
  89     KeyValue() : k(), v() {};
  90     KeyValue(const Key& _key, const Value& _value) : k(_key), v(_value) {};
  91     KeyValue(const KeyValue& kv) : k(kv.k), v(kv.v) {};
  92 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  93     KeyValue(KeyValue&& kv) : k(), v() {
  94       swap(*this, kv);
  95     }
  96 #endif
  97       friend void swap(KeyValue& a, KeyValue& b) {
  98         using std::swap;
  99 
 100         swap(a.k, b.k);
 101         swap(a.v, b.v);
 102       }
 103       KeyValue& operator=(KeyValue kv) {
 104         swap(*this, kv);
 105         return *this;
 106       }
 107 
 108     inline const Key& key() const { return k; };
 109     inline const Value& value() const { return v; };
