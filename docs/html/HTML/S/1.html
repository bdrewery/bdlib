  43 #define SOCKET_CLIENT   (1UL << 0)
  44 #define SOCKET_SERVER   (1UL << 1)
  45 #define SOCKET_BIND     (1UL << 2)
  46 #define SOCKET_ASYNC    (1UL << 3)
  47 #define SOCKET_TCP      (1UL << 4)
  48 #define SOCKET_UDP      (1UL << 5)
  49 
  50 typedef struct {
  51         int len;
  52         int family;
  53         union {
  54                 struct sockaddr addr;
  55                 struct sockaddr_in ipv4;
  56 #ifdef DO_IPV6
  57                 struct sockaddr_in6 ipv6;
  58 #endif
  59         } u;
  60 } sockname_t;
  61 
  62 
  63 /**
  64  * @class Socket
  65  * @brief Socket class
  66  */
  67 class Socket {
  68   private:
  69     int sock;
  70     sockname_t addr;
  71     int flags;
  72   public:
  73     Socket() : sock(-1), addr(0), flags(0) {};
  74     Socket(int f) : sock(-1), addr(0), flags(f) {};
  75     Socket(Socket& s) : sock(s.sock), addr(s.addr), flags(s.flags) {};
  76     virtual ~Socket();
  77     
  78     bool create();
  79     bool connect(const String& host, const in_port_t port);
  80     bool bind(const in_port_t port);
  81     bool listen() const;
  82     bool accept (Socket&) const;
  83 
  84     bool send(const String) const;
  85     bool recv(String&) const;
  86 
  87     int setNonBlocking(bool = 1);
  88     inline bool isValid() const { return sock != -1; };
  89 };
  90 
  91 class ClientSocket : public Socket {
  92   private:
  93   public:
  94     ClientSocket(const String& dest, const in_port_t port);
  95     ClientSocket(const String& dest, const in_port_t port, const String& src, const in_port_t sport);
  96     ClientSocket(const ClientSocket& cs);
  97     virtual ~ClientSocket();
  98 };

 105 //    ServerSocket(const ServerSocket& ss);
 106     virtual ~ServerSocket();
 107 };
