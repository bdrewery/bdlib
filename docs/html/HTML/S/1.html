  47 typedef struct {
  48         int len;
  49         int family;
  50         union {
  51                 struct sockaddr addr;
  52                 struct sockaddr_in ipv4;
  53 #ifdef DO_IPV6
  54                 struct sockaddr_in6 ipv6;
  55 #endif
  56         } u;
  57 } sockname_t;

  60 /**
  61  * @class Socket
  62  * @brief Socket class
  63  */
  64 class Socket {
  65   private:
  66     int sock;
  67     sockname_t addr;
  68     int flags;
  69   public:
  70     Socket() : sock(-1), addr(0), pfamily(0) {};
  71     Socket(int f) : sock(-1), addr(0), flags(f) {};
  72     Socket(Socket& s) : sock(s.sock), addr(s.addr), flags(s.flags) {};
  73     virtual ~Socket();
  74     
  75     bool create();
  76     bool connect(const String host, const in_port_t port);
  77     bool bind(const in_port_t port);
  78     bool listen() const;
  79     bool accept (Socket&) const;
  80 
  81     bool send(const String) const;
  82     bool recv(String&) const;
  83 
  84     int setNonBlocking(bool = 1);
  85     inline bool isValid() const { return sock != -1; };
  86 };
