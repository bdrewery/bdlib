  31 /*
  32  * Copy src to string dst of size siz.  At most siz-1 characters
  33  * will be copied.  Always NUL terminates (unless siz == 0).
  34  * Returns strlen(src); if retval >= siz, truncation occurred.
  35  */
  36 size_t
  37 strlcpy(char *dst, const char *src, size_t siz)
  38 {
  39   register char *d = dst;
  40   register const char *s = src;
  41   register size_t n = siz;
  42 
  43   /* Copy as many bytes as will fit */
  44   if (n != 0 && --n != 0) {
  45     do {
  46       if ((*d++ = *s++) == 0)
  47         break;
  48     } while (--n != 0);
  49   }
  50 
  51   /* Not enough room in dst, add NUL and traverse rest of src */
  52   if (n == 0) {
  53     if (siz != 0)
  54       *d = '\0';                /* NUL-terminate dst */
  55     while (*s++) ;
  56   }
  57 
  58   return (s - src - 1);         /* count does not include NUL */
  59 }

  61 /*
  62  * Appends src to string dst of size siz (unlike strncat, siz is the
  63  * full size of dst, not space left).  At most siz-1 characters
  64  * will be copied.  Always NUL terminates (unless siz == 0).
  65  * Returns strlen(src); if retval >= siz, truncation occurred.
  66  */
  67 size_t
  68 strlcat(char *dst, const char *src, size_t siz)
  69 {
  70   register char *d = dst;
  71   register const char *s = src;
  72   register size_t n = siz;
  73   size_t dlen;
  74 
  75   /* Find the end of dst and adjust bytes left but don't go past end */
  76   while (*d != '\0' && n-- != 0)
  77     d++;
  78   dlen = d - dst;
  79   n = siz - dlen;
  80 
  81   if (n == 0)
  82     return (dlen + strlen(s));
  83   while (*s != '\0') {
  84     if (n != 1) {
  85       *d++ = *s;
  86       n--;
  87     }
  88     s++;
  89   }
  90   *d = '\0';
  91 
  92   return (dlen + (s - src));    /* count does not include NUL */
  93 }

  95 void
  96 str_redup(char **str, const char *newstr)
  97 {
  98         size_t len;
  99 
 100         if (!newstr) {
 101                 if (*str) free(*str);
 102                 *str = NULL;
 103                 return;
 104         }
 105         len = strlen(newstr) + 1;
 106         *str = static_cast<char*>(realloc(*str, len));
 107         memcpy(*str, newstr, len);
 108 }

 110 char *
 111 strdup(const char *entry)
 112 {
 113   size_t len = strlen(entry);
 114   char *target = static_cast<char*>(calloc(1, len + 1));
 115   if (target == NULL) return NULL;
 116   target[len] = 0;
 117   return static_cast<char*>(memcpy(target, entry, len));
 118 }

 120 char *
 121 strldup(const char *entry, size_t maxlen)
 122 {
 123   size_t slen = strlen(entry);
 124   size_t len = slen < maxlen ? slen : maxlen;
 125   char *target = static_cast<char*>(calloc(1, len + 1));
 126   if (target == NULL) return NULL;
 127   target[len] = 0;
 128   return static_cast<char*>(memcpy(target, entry, len));
 129 }
