  27 #ifndef _BD_HASHTABLE_H
  28 #define _BD_HASHTABLE_H 1
  29 
  30 #include <unordered_map>
  31 #include <vector>
  32 #include "bdlib.h"
  33 #include "Array.h"
  34 BDLIB_NS_BEGIN
  35 
  36 
  37 /**
  38   * @class HashTable
  39   * @brief HashTable data structure
  40   * @todo replace()
  41   * @todo iterators
  42   * @todo resizing/growing
  43   *
  44   */
  45 template <class Key, class Value>
  46 class HashTable {
  47   private:
  48     static const size_t default_list_size = 100;
  49     typedef typename std::unordered_map<Key, Value>::value_type iterator_type;
  50     typedef void (*hash_table_block)(const Key, Value, void *param);
  51 
  52     std::unordered_map<Key, Value> map;
  53 
  54   public:
  55     HashTable() : map(default_list_size) {} ;
  56     explicit HashTable(size_t capacity_in) : map(capacity_in) {};
  57     HashTable(const HashTable<Key, Value>& table) : map(table.map) {}
  58     HashTable(HashTable<Key, Value>&& table) : map(std::move(table.map)) {
  59       table.map.clear();
  60     }
  61     HashTable(std::initializer_list<iterator_type> list) : map(list) {}
  62 
  63     virtual ~HashTable() {}
  64 
  65     void clear() {
  66       map.clear();
  67     }
  68 
  69     /**
  70      * @brief A ruby style block which will yield to the passed callback for each Key/Value pair.
  71      * @param block The block to execute for each element
  72      * @param param An optional parameter to pass to the block.
  73      * @return How many iterations were called
  74      */
  75     int each(hash_table_block block, void* param = nullptr) {
  76       int n = 0;
  77       if (!size()) return n;
  78 
  79       // Make a list of KeyValues to yield from.
  80       // Don't yield in this loop as the block may actually modify (this), thus making this iterator stale
  81       std::vector<iterator_type> items;
  82       for (const auto& item : map) {
  83         items.push_back(item);
  84       }
  85 
  86       // Now yield on our temporary, so (this) isn't a factor.
  87       for (const auto& item : items) {
  88         ++n;
  89         block(item.first, item.second, param);
  90       }
  91       return n;
  92     }
  93 
  94     friend void swap(HashTable<Key, Value>& a, HashTable<Key, Value>& b) {
  95       using std::swap;
  96 
  97       swap(a.map, b.map);
  98     }
  99 
 100     HashTable& operator=(HashTable<Key, Value> table) {
 101       swap(*this, table);
 102       return *this; 
 103     }
 104 
 105     /**
 106      * @brief Create an array from an initializer list
 107      * @param list An initializer_list
 108      */
 109     HashTable& operator=(std::initializer_list<iterator_type> list) {
 110       for (iterator_type item : list) {
 111         (*this)[item.key()] = item.value();
 112       }
 113       return *this;
 114     }
 115 
 116     inline size_t size() const { return map.size(); };
 117     inline size_t capacity() const { return map.max_size(); };
 118     inline bool isEmpty() const { return map.empty(); };
 119     inline explicit operator bool() const { return !isEmpty(); };
 120 
 121     bool insert(const Key& key, const Value& value) {
 122       if (contains(key)) return false;
 123       map[key] = std::move(value);
 124       return true;
 125     }
 126 
 127     inline bool contains(const Key& key) const {
 128       if (isEmpty()) return false;
 129       return map.find(key) != std::end(map);
 130     };
 131 
 132     bool remove(const Key& key) {
 133       if (isEmpty()) return false;
 134       return map.erase(key) > 0 ? true : false;
 135     };
 136 
 137     inline Value getValue(const Key& key) const {
 138       auto result = map.find(key);
 139       if (result == std::end(map))
 140         return Value();
 141       return result->second;
 142     };
 143 
 144     /**
 145       * @brief Associate array type accessor (rvalue)
 146       * @param key The key to search for
 147       */
 148     inline const Value operator[](const Key& key) const { return getValue(key); }
 149 
 150     /**
 151      * @brief Return an array of all the keys
 152      */
 153     Array<Key> keys() const {
 154       Array<Key> tmp(size());
 155 
 156       for (const auto& item : map) {
 157         tmp << item.first;
 158       }
 159       return tmp;
 160     }
 161 
 162     /**
 163      * @brief Return an array of all the values
 164      */
 165     Array<Value> values() const {
 166       Array<Value> tmp(size());
 167 
 168       for (const auto& item : map) {
 169         tmp << item.second;
 170       }
 171       return tmp;
 172     }
 173     
 174     /**
 175       * @brief Associate array type accessor (lvalue)
 176       * @param key The key to search for
 177       * @sa find_or_insert_key 
 178       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 179       */
 180     inline Value& operator[](const Key& key) {
 181       return map[key];
 182     }
 183 };
