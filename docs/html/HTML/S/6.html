  38 template <class Key, class Value>
  39 /**
  40   * @class HashTable
  41   * @brief HashTable data structure
  42   * @todo replace()
  43   * @todo iterators
  44   * @todo resizing/growing
  45   *
  46   */
  47 class HashTable {
  48   private:
  49     static const size_t default_list_size = 100;
  50     typedef KeyValue<Key, Value> iterator_type;
  51     typedef Hash<Key> hasher;
  52     typedef void (*hash_table_block)(const Key, Value, void *param);
  53 
  54     List<iterator_type> *_list;
  55     size_t _size;
  56     size_t _capacity;
  57     hasher _hash;
  58 
  59     inline size_t getIndex(const Key& key) const {
  60       return _hash(key) % _capacity;
  61     }
  62   public:
  63     HashTable() : _list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {};
  64     explicit HashTable(size_t capacity_in) : _list(new List<iterator_type>[capacity_in]), _size(0), _capacity(capacity_in), _hash() {};
  65     HashTable(const HashTable<Key, Value>& table) : _list(new List<iterator_type>[table._capacity]), _size(table._size), _capacity(table._capacity), _hash(table._hash) {
  66       for (size_t i = 0; i < _capacity; ++i)
  67           _list[i] = table._list[i];
  68     };
  69     HashTable(HashTable<Key, Value>&& table) : _list(std::move(table._list)), _size(std::move(table._Size)), _capacity(std::move(table._capacity)), _hash(std::move(table._hash)) {
  70       table._list = nullptr;
  71       table._size = 0;
  72       table._capacity = default_list_size;
  73       table._hash = _hash();
  74     }
  75     HashTable(std::initializer_list<iterator_type> list) : _list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {
  76       *this = list;
  77     }
  78 
  79     virtual ~HashTable() {
  80       delete[] _list;
  81     }
  82 
  83     void clear() {
  84       for (size_t i = 0; i < _capacity; ++i)
  85         _list[i].clear();
  86       _size = 0;
  87     }
  88 
  89     /**
  90      * @brief A ruby style block which will yield to the passed callback for each Key/Value pair.
  91      * @param block The block to execute for each element
  92      * @param param An optional parameter to pass to the block.
  93      * @return How many iterations were called
  94      */
  95     int each(hash_table_block block, void* param = nullptr) {
  96       int n = 0;
  97       if (!size()) return n;
  98 
  99       // Make a list of KeyValues to yield from.
 100       // Don't yield in this loop as the block may actually modify (this), thus making this iterator stale
 101       typename List<iterator_type>::iterator iter;
 102       List<iterator_type> items;
 103       for (size_t i = 0; i < _capacity; ++i) {
 104         if (_list[i].size()) {
 105           for (iter = _list[i].begin(); iter; (++iter)) {
 106             items << *iter;
 107           }
 108         }
 109       }
 110 
 111       // Now yield on our temporary, so (this) isn't a factor.
 112       for (iter = items.begin(); iter; (++iter)) {
 113         iterator_type kv = *iter;
 114         ++n;
 115         block(kv.key(), kv.value(), param);
 116       }
 117       return n;
 118     }
 119 
 120     friend void swap(HashTable<Key, Value>& a, HashTable<Key, Value>& b) {
 121       using std::swap;
 122 
 123       swap(a._list, b._list);
 124       swap(a._size, b._size);
 125       swap(a._capacity, b._capacity);
 126       swap(a._hash, b._hash);
 127     }
 128 
 129     HashTable& operator=(HashTable<Key, Value> table) {
 130       swap(*this, table);
 131       return *this; 
 132     }
 133 
 134     /**
 135      * @brief Create an array from an initializer list
 136      * @param list An initializer_list
 137      */
 138     HashTable& operator=(std::initializer_list<iterator_type> list) {
 139       for (iterator_type item : list) {
 140         (*this)[item.key()] = item.value();
 141       }
 142       return *this;
 143     }
 144 
 145     inline size_t size() const { return _size; };
 146     inline size_t capacity() const { return _capacity; };
 147     inline bool isEmpty() const { return size() == 0; };
 148     inline explicit operator bool() const { return !isEmpty(); };
 149 
 150     bool insert(const Key& key, const Value& value) {
 151       if (contains(key)) return false;
 152       _list[getIndex(key)] << iterator_type(key, value);
 153       ++_size;
 154       return true;
 155     }
 156 
 157     inline bool contains(const Key& key) const {
 158       if (isEmpty()) return false;
 159       return _list[getIndex(key)].contains(iterator_type(key, Value()));
 160     };
 161 
 162     bool remove(const Key& key) {
 163       if (isEmpty()) return false;
 164       if (_list[getIndex(key)].remove(iterator_type(key, Value()))) {
 165         --_size;
 166         return true;
 167       }
 168       return false;
 169     };
 170 
 171     inline Value getValue(const Key& key) const {
 172       Value empty;
 173       if (isEmpty()) return empty;
 174       return _list[getIndex(key)].find(iterator_type(key, Value())).value();
 175     };
 176 
 177     /**
 178       * @brief Associate array type accessor (rvalue)
 179       * @param key The key to search for
 180       */
 181     inline const Value operator[](const Key& key) const { return getValue(key); }
 182 
 183     /**
 184      * @brief Return an array of all the keys
 185      */
 186     Array<Key> keys() const {
 187       Array<Key> tmp(size());
 188 
 189       typename List<iterator_type>::iterator iter;
 190       for (size_t i = 0; i < capacity(); ++i) {
 191         if (_list[i].size()) {
 192           for (iter = _list[i].begin(); iter; (++iter)) {
 193             iterator_type kv = *iter;
 194             tmp << kv.key();
 195           }
 196         }
 197       }
 198       return tmp;
 199     }
 200 
 201     /**
 202      * @brief Return an array of all the values
 203      */
 204     Array<Value> values() const {
 205       Array<Value> tmp(size());
 206 
 207       typename List<iterator_type>::iterator iter;
 208       for (size_t i = 0; i < capacity(); ++i) {
 209         if (_list[i].size()) {
 210           for (iter = _list[i].begin(); iter; (++iter)) {
 211             iterator_type kv = *iter;
 212             tmp << kv.value();
 213           }
 214         }
 215       }
 216       return tmp;
 217     }
 218     
 219     /**
 220       * @brief Associate array type accessor (lvalue)
 221       * @param key The key to search for
 222       * @sa find_or_insert_key 
 223       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 224       */
 225     inline Value& operator[](const Key& key) {
 226       return find_or_insert_key(key);
 227     }
 228  
 229     /**
 230      * @brief Find a key in the list or insert it.
 231      */
 232     inline Value& find_or_insert_key(const Key& key) {
 233       if (!contains(key))
 234         insert(key, Value());
 235       return _list[getIndex(key)].findRef(iterator_type(key, Value())).v;
 236     }
 237 };
