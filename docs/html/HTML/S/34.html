  40 inline String ScriptInterpTCL::eval(const String& script) {
  41   if (Tcl_EvalEx(interp, script.c_str(), script.length(), TCL_EVAL_GLOBAL) == TCL_OK) {

  48 ScriptInterp::LoadError ScriptInterpTCL::loadScript(const String& fileName, String& resultStr) {
  49   if (fileName.rfind(".tcl", fileName.length() - 4) == String::npos)
  50     return SCRIPT_LOAD_WRONG_INTERP;
  51   if (Tcl_EvalFile(interp, *fileName) != TCL_OK) {
  52     Tcl_Obj* value = Tcl_GetObjResult(interp);
  53     resultStr = tcl_to_c_cast<String>::from(value, this);
  54     return SCRIPT_LOAD_ERROR;
  55   }
  56   return SCRIPT_LOAD_OK;
  57 }

  62 int ScriptInterpTCL::ScriptCmdTCL::_createCommand_callback(
  63     ClientData clientData, Tcl_Interp *interp, int objc,
  64     Tcl_Obj *CONST objv[]) {
  65   String cmdName(tcl_to_c_cast<String>::from(objv[0], nullptr));
  66   ScriptCmdPtr ccd = CmdHandlerData[cmdName];
  67 
  68   if ((size_t(objc) - 1) < ccd->callbackParamMin || size_t(objc) - 1 > ccd->callbackParamMax) {
  69     String errorResult(String::printf("Wrong # args. Expected %zu, got %d.", ccd->callbackParamMin, objc - 1));
  70     if (ccd->usage)
  71       errorResult += String::printf(" Should be \"%s %s\"", cmdName.c_str(), ccd->usage);
  72     Tcl_SetObjResult(interp,
  73         c_to_tcl_cast<const String&>::from(errorResult, interp));
  74     return TCL_ERROR;
  75   }
  76 
  77   try {
  78     ccd->callback_proxy->call(objc, reinterpret_cast<void* CONST*>(objv), ccd->si, interp);
  79   } catch (BDLIB_NS::String& e) {
  80     Tcl_SetObjResult(interp,
  81         c_to_tcl_cast<const String&>::from(e, interp));
  82     return TCL_ERROR;
  83   } catch (...) {
  84     Tcl_SetObjResult(interp, c_to_tcl_cast<const char*>::from("Unhandled exception.", interp));
  85     return TCL_ERROR;
  86   }
  87   return TCL_OK;
  88 }

  90 void ScriptInterpTCL::setupTraces(const String& varName, ClientData var,
  91     Tcl_VarTraceProc* get, Tcl_VarTraceProc* set, link_var_hook hook_func) {
  92   Tcl_SetVar(interp, *varName, "", TCL_GLOBAL_ONLY);
  93   Tcl_TraceVar(interp, *varName, TCL_TRACE_READS | TCL_GLOBAL_ONLY, get, var);
  94   Tcl_TraceVar(interp, *varName, TCL_TRACE_WRITES | TCL_GLOBAL_ONLY, set, var);
  95   if (hook_func != nullptr)
  96     link_var_hooks[varName] = hook_func;
  97 }
