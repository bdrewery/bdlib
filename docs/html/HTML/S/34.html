  37 #define STREAM_BLOCKSIZE        1024
  38 /* The possibilities for the third argument to `fseek'.
  39    These values should not be changed.  */
  40 #define SEEK_SET        0       /* Seek from beginning of file.  */
  41 #define SEEK_CUR        1       /* Seek from current position.  */
  42 #define SEEK_END        2       /* Seek from end of file.  */
  43 
  44 class Stream {
  45   protected:
  46         String str;
  47         unsigned int pos;
  48         bool loading;
  49 
  50   public:
  51         Stream() : str(), pos(0), loading(0) {};
  52         Stream(const Stream& stream) : str(stream.str), pos(stream.pos), loading(0) {};
  53 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  54         Stream(Stream&& stream) : str(), pos(0), loading(0) {
  55           swap(*this, stream);
  56         }
  57 #endif
  58         Stream(const String& string) : str(string), pos(0), loading(0) {};
  59         Stream(const int newSize) : str(), pos(0), loading(0) { if (newSize > 0) Reserve(newSize); };
  60         virtual ~Stream() {};
  61 
  62         friend void swap(Stream& a, Stream& b) {
  63           using std::swap;
  64 
  65           swap(a.str, b.str);
  66           swap(a.pos, b.pos);
  67           swap(a.loading, b.loading);
  68         }
  69 
  70         Stream& operator=(Stream stream) {
  71           swap(*this, stream);
  72           return *this;
  73         }
  74 
  75         void Reserve(const size_t) const;
  76 
  77         /**
  78          * @brief Returns the position of the Stream.
  79          * @return Position of the Stream.
  80         */
  81         size_t tell() const { return pos; };
  82 
  83         /**
  84          * @brief Truncates the stream at the current position.
  85         */
  86         void truncate() { str = str(0, pos); };
  87 
  88         String peek() { return str(pos); };
  89 
  90 //        operator void*() { return tell() <= length(); };
  91 
  92         int seek(int, int);
  93         void clear() { str.clear(); pos = 0; }
  94 
  95         /**
  96          * @brief Insert a string into the stream.
  97          * @note The stream pointer is advanced as well
  98          */
  99         virtual void puts (const String& string) {
 100           str.replace(tell(), string);
 101           pos += string.length();
 102         }
 103 
 104         /**
 105          * @brief Reads 1 line from the stream
 106          * @note The stream pointer is advanced as well
 107          * @param maxSize Optional param which specifies max data to pull
 108          * @sa read
 109          */
 110         virtual String getline (size_t maxSize = 99999999) { return read(maxSize == 99999999 ? (length() - pos) : maxSize, '\n'); }
 111 
 112         /**
 113          * @brief Reads specified number of bytes from the stream
 114          * @param maxSize How many bytes to read
 115          * @param delim What to split the read on. For example: '\n' will return 1 line.
 116          * @note The stream pointer is advanced as well
 117          */
 118         virtual String read (size_t maxSize, char delim = 0);
 119 
 120         /**
 121          * @brief Load a file into the stream
 122          * @returns 1 on error, 0 on success.
 123          * @param fd File descriptor to read from
 124          * @note this.loading is set to 1 during this process.
 125          * @note Uses puts() for populating the stream.
 126          * @note The fd is not closed.
 127          * @sa puts
 128          */
 129         virtual int loadFile(const int);
 130 
 131         /**
 132          * @brief Load a file into the stream
 133          * @returns 1 on error, 0 on success.
 134          * @param fname File to read from
 135          * @note this.loading is set to 1 during this process.
 136          * @note Uses puts() for populating the stream.
 137          * @sa puts
 138          */
 139         virtual int loadFile(const String& fname);
 140 
 141         /**
 142          * @brief Write stream out to a file
 143          * @returns 1 on error, 0 on success
 144          * @param fd File descriptor to write to
 145          * @note The fd is not closed.
 146          */
 147         virtual int writeFile(const int fd) const;
 148 
 149         /**
 150          * @brief Write stream out to a file
 151          * @returns 1 on error, 0 on success
 152          * @param fname Filename to write to
 153          * @param mode Optional param to specify mode for new file
 154          */
 155         virtual int writeFile(const String& fname, mode_t mode = (S_IRUSR|S_IWUSR)) const;
 156 
 157 
 158         inline operator String() const { return str; };
 159         inline size_t length() const { return str.length(); };
 160         inline size_t capacity() const { return str.capacity(); };
 161         inline bool operator!() const { return str.isEmpty(); };
 162 
 163         friend Stream& operator<<(Stream&, const String&);
 164 };
 165 
 166 inline Stream& operator<<(Stream& stream, const String& string) { stream.puts(string); return stream; }
