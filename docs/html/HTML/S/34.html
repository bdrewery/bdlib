  42 #define STREAM_BLOCKSIZE        1024
  43 /* The possibilities for the third argument to `fseek'.
  44    These values should not be changed.  */
  45 #define SEEK_SET        0       /* Seek from beginning of file.  */
  46 #define SEEK_CUR        1       /* Seek from current position.  */
  47 #define SEEK_END        2       /* Seek from end of file.  */
  48 
  49 class Stream {
  50   protected:
  51         String str;
  52         unsigned int pos;
  53         bool loading;
  54 
  55   public:
  56         Stream() : str(), pos(0), loading(0) {};
  57         Stream(const Stream& stream) : str(stream.str), pos(stream.pos), loading(0) {};
  58 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  59         Stream(Stream&& stream) : str(stream.str), pos(stream.pos), loading(stream.loading) {
  60           str = String();
  61           pos = 0;
  62           loading = 0;
  63         }
  64 #endif
  65         Stream(const String& string) : str(string), pos(0), loading(0) {};
  66         Stream(const int newSize) : str(), pos(0), loading(0) { if (newSize > 0) Reserve(newSize); };
  67         virtual ~Stream() {};
  68 
  69         friend void swap(Stream& a, Stream& b) {
  70           using std::swap;
  71 
  72           swap(a.str, b.str);
  73           swap(a.pos, b.pos);
  74           swap(a.loading, b.loading);
  75         }
  76 
  77         Stream& operator=(Stream stream) {
  78           swap(*this, stream);
  79           return *this;
  80         }
  81 
  82         void Reserve(const size_t) const;
  83 
  84         /**
  85          * @brief Returns the position of the Stream.
  86          * @return Position of the Stream.
  87         */
  88         size_t tell() const { return pos; };
  89 
  90         /**
  91          * @brief Truncates the stream at the current position.
  92         */
  93         void truncate() { str = str(0, pos); };
  94 
  95         String peek() { return str(pos); };
  96 
  97 //        operator void*() { return tell() <= length(); };
  98 
  99         int seek(int, int);
 100         void clear() { str.clear(); pos = 0; }
 101 
 102         /**
 103          * @brief Insert a string into the stream.
 104          * @note The stream pointer is advanced as well
 105          */
 106         virtual void puts (const String& string) {
 107           str.replace(tell(), string);
 108           pos += string.length();
 109         }
 110 
 111         /**
 112          * @brief Reads 1 line from the stream
 113          * @note The stream pointer is advanced as well
 114          * @param maxSize Optional param which specifies max data to pull
 115          * @sa read
 116          */
 117         virtual String getline (size_t maxSize = 99999999) { return read(maxSize == 99999999 ? (length() - pos) : maxSize, '\n'); }
 118 
 119         /**
 120          * @brief Reads specified number of bytes from the stream
 121          * @param maxSize How many bytes to read
 122          * @param delim What to split the read on. For example: '\n' will return 1 line.
 123          * @note The stream pointer is advanced as well
 124          */
 125         virtual String read (size_t maxSize, char delim = 0);
 126 
 127         /**
 128          * @brief Load a file into the stream
 129          * @returns 1 on error, 0 on success.
 130          * @param fd File descriptor to read from
 131          * @note this.loading is set to 1 during this process.
 132          * @note Uses puts() for populating the stream.
 133          * @note The fd is not closed.
 134          * @sa puts
 135          */
 136         virtual int loadFile(const int);
 137 
 138         /**
 139          * @brief Load a file into the stream
 140          * @returns 1 on error, 0 on success.
 141          * @param fname File to read from
 142          * @note this.loading is set to 1 during this process.
 143          * @note Uses puts() for populating the stream.
 144          * @sa puts
 145          */
 146         virtual int loadFile(const String& fname);
 147 
 148         /**
 149          * @brief Write stream out to a file
 150          * @returns 1 on error, 0 on success
 151          * @param fd File descriptor to write to
 152          * @note The fd is not closed.
 153          */
 154         virtual int writeFile(const int fd) const;
 155 
 156         /**
 157          * @brief Write stream out to a file
 158          * @returns 1 on error, 0 on success
 159          * @param fname Filename to write to
 160          * @param mode Optional param to specify mode for new file
 161          */
 162         virtual int writeFile(const String& fname, mode_t mode = (S_IRUSR|S_IWUSR)) const;
 163 
 164 
 165         inline operator String() const { return str; };
 166         inline size_t length() const { return str.length(); };
 167         inline size_t capacity() const { return str.capacity(); };
 168         inline bool operator!() const { return str.isEmpty(); };
 169 
 170         friend Stream& operator<<(Stream&, const String&);
 171 };
 172 
 173 inline Stream& operator<<(Stream& stream, const String& string) { stream.puts(string); return stream; }
