  37 #define STREAM_BLOCKSIZE        1024
  38 /* The possibilities for the third argument to `fseek'.
  39    These values should not be changed.  */
  40 #define SEEK_SET        0       /* Seek from beginning of file.  */
  41 #define SEEK_CUR        1       /* Seek from current position.  */
  42 #define SEEK_END        2       /* Seek from end of file.  */
  43 
  44 class Stream {
  45   protected:
  46         String str;
  47         unsigned int pos;
  48         bool loading;
  49 
  50   public:
  51         Stream() : str(), pos(0), loading(0) {};
  52         Stream(const Stream& stream) : str(stream.str), pos(stream.pos), loading(0) {};
  53         Stream(const String& string) : str(string), pos(0), loading(0) {};
  54         Stream(const int newSize) : str(), pos(0), loading(0) { if (newSize > 0) Reserve(newSize); };
  55         virtual ~Stream() {};
  56 
  57         void Reserve(const size_t) const;
  58 
  59         /**
  60          * @brief Returns the position of the Stream.
  61          * @return Position of the Stream.
  62         */
  63         size_t tell() const { return pos; };
  64 
  65         /**
  66          * @brief Truncates the stream at the current position.
  67         */
  68         void truncate() { str = str(0, pos); };
  69 
  70         String peek() { return str(pos); };
  71 
  72 //        operator void*() { return tell() <= length(); };
  73 
  74         int seek(int, int);
  75         void clear() { str.clear(); pos = 0; }
  76 
  77         /**
  78          * @brief Insert a string into the stream.
  79          * @note The stream pointer is advanced as well
  80          */
  81         virtual void puts (const String& string) {
  82           str.replace(tell(), string);
  83           pos += string.length();
  84         }
  85 
  86         /**
  87          * @brief Reads 1 line from the stream
  88          * @note The stream pointer is advanced as well
  89          * @param maxSize Optional param which specifies max data to pull
  90          * @sa read
  91          */
  92         virtual String getline (size_t maxSize = 99999999) { return read(maxSize == 99999999 ? (length() - pos) : maxSize, '\n'); }
  93 
  94         /**
  95          * @brief Reads specified number of bytes from the stream
  96          * @param maxSize How many bytes to read
  97          * @param delim What to split the read on. For example: '\n' will return 1 line.
  98          * @note The stream pointer is advanced as well
  99          */
 100         virtual String read (size_t maxSize, char delim = 0);
 101 
 102         /**
 103          * @brief Load a file into the stream
 104          * @returns 1 on error, 0 on success.
 105          * @param fd File descriptor to read from
 106          * @note this.loading is set to 1 during this process.
 107          * @note Uses puts() for populating the stream.
 108          * @note The fd is not closed.
 109          * @sa puts
 110          */
 111         virtual int loadFile(const int);
 112 
 113         /**
 114          * @brief Load a file into the stream
 115          * @returns 1 on error, 0 on success.
 116          * @param fname File to read from
 117          * @note this.loading is set to 1 during this process.
 118          * @note Uses puts() for populating the stream.
 119          * @sa puts
 120          */
 121         virtual int loadFile(const String& fname);
 122 
 123         /**
 124          * @brief Write stream out to a file
 125          * @returns 1 on error, 0 on success
 126          * @param fd File descriptor to write to
 127          * @note The fd is not closed.
 128          */
 129         virtual int writeFile(const int fd) const;
 130 
 131         /**
 132          * @brief Write stream out to a file
 133          * @returns 1 on error, 0 on success
 134          * @param fname Filename to write to
 135          * @param mode Optional param to specify mode for new file
 136          */
 137         virtual int writeFile(const String& fname, mode_t mode = (S_IRUSR|S_IWUSR)) const;
 138 
 139 
 140         inline operator String() const { return str; };
 141         inline size_t length() const { return str.length(); };
 142         inline size_t capacity() const { return str.capacity(); };
 143         inline bool operator ! () const { return str.isEmpty(); };
 144 
 145         friend Stream& operator << (Stream&, const String&);
 146 };
 147 
 148 inline Stream& operator << (Stream& stream, const String& string) { stream.puts(string); return stream; }
