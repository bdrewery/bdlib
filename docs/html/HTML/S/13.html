  42 int main (int argc, char* argv[])
  43 {
  44     // informs test-listener about testresults
  45     CPPUNIT_NS :: TestResult testresult;
  46 
  47     // register listener for collecting the test-results
  48     CPPUNIT_NS :: TestResultCollector collectedresults;
  49     testresult.addListener (&collectedresults);
  50 
  51     CPPUNIT_NS :: BriefTestProgressListener listener;
  52     if (argc > 1 || getenv("V")) {
  53       // Shows a message as each test starts
  54       testresult.addListener( &listener );
  55     }
  56 
  57     // insert test-suite at test-runner by registry
  58     CPPUNIT_NS :: TestRunner testrunner;
  59     testrunner.addTest (CPPUNIT_NS :: TestFactoryRegistry :: getRegistry ().makeTest ());
  60     testrunner.run (testresult);
  61 
  62     // output results in compiler-format
  63     CPPUNIT_NS :: CompilerOutputter compileroutputter (&collectedresults, std::cerr);
  64     compileroutputter.write ();
  65 
  66     // return 0 if tests were successful
  67     int ret = collectedresults.wasSuccessful() ? 0 : 1;
  68 
  69 #ifdef USE_SCRIPT_TCL
  70     /* Avoid valgrind memory leaks. */
  71     Tcl_Finalize();
  72 #endif
  73 
  74     return ret;
  75 }
