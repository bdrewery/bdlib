 116 void HashTableTest :: containsTest (void)
 117 {
 118   a->insert(1, "Blah");
 119   a->insert(1, "Bleck");
 120   a->insert(2, "Blah");
 121   a->insert(8, "Blah");
 122   a->insert(4, "Blah");
 123   
 124   CPPUNIT_ASSERT_EQUAL(true, a->contains(1));
 125   CPPUNIT_ASSERT_EQUAL(false, a->contains(3));
 126   CPPUNIT_ASSERT_EQUAL(true, a->contains(4));
 127 
 128   (*b) = (*a);
 129 
 130   CPPUNIT_ASSERT_EQUAL(true, b->contains(1));
 131   CPPUNIT_ASSERT_EQUAL(false, b->contains(3));
 132   CPPUNIT_ASSERT_EQUAL(true, b->contains(4));
 133 
 134   b->insert(5, "test");
 135   CPPUNIT_ASSERT_EQUAL(true, b->contains(5));
 136   CPPUNIT_ASSERT_EQUAL(false, a->contains(5));
 137 
 138   (*b) = (*a);
 139   CPPUNIT_ASSERT_EQUAL(true, b->contains(1));
 140   CPPUNIT_ASSERT_EQUAL(false, b->contains(3));
 141   CPPUNIT_ASSERT_EQUAL(true, b->contains(4));
 142   CPPUNIT_ASSERT_EQUAL(false, b->contains(5));
 143   CPPUNIT_ASSERT_EQUAL(false, a->contains(5));
 144 
 145   (*a)[42] = "42";
 146   CPPUNIT_ASSERT_EQUAL((size_t)5, a->size());
 147   CPPUNIT_ASSERT_EQUAL(true, a->contains(42));
 148 
 149 
 150   sa->insert("1", "Blah");
 151   sa->insert("1", "Bleck");
 152   sa->insert("2", "Blah");
 153   sa->insert("8", "Blah");
 154   sa->insert("4", "Blah");
 155   
 156   CPPUNIT_ASSERT_EQUAL(true, sa->contains("1"));
 157   CPPUNIT_ASSERT_EQUAL(false, sa->contains("3"));
 158   CPPUNIT_ASSERT_EQUAL(true, sa->contains("4"));
 159 
 160   (*sb) = (*sa);
 161 
 162   CPPUNIT_ASSERT_EQUAL(true, sb->contains("1"));
 163   CPPUNIT_ASSERT_EQUAL(false, sb->contains("3"));
 164   CPPUNIT_ASSERT_EQUAL(true, sb->contains("4"));
 165 
 166   sb->insert("5", "test");
 167   CPPUNIT_ASSERT_EQUAL(true, sb->contains("5"));
 168   CPPUNIT_ASSERT_EQUAL(false, sa->contains("5"));
 169 
 170   (*sb) = (*sa);
 171   CPPUNIT_ASSERT_EQUAL(true, sb->contains("1"));
 172   CPPUNIT_ASSERT_EQUAL(false, sb->contains("3"));
 173   CPPUNIT_ASSERT_EQUAL(true, sb->contains("4"));
 174   CPPUNIT_ASSERT_EQUAL(false, sb->contains("5"));
 175   CPPUNIT_ASSERT_EQUAL(false, sa->contains("5"));
 176 
 177   (*sa)["42"] = "42";
 178   CPPUNIT_ASSERT_EQUAL((size_t)5, sa->size());
 179   CPPUNIT_ASSERT_EQUAL(true, sa->contains("42"));
 180 
 181 
 182   //Test .keys and .values
 183   Array<String> keys = sa->keys();
 184   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("1") != size_t(-1));
 185   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("2") != size_t(-1));
 186   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("8") != size_t(-1));
 187   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("4") != size_t(-1));
 188   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("42") != size_t(-1));
 189   CPPUNIT_ASSERT_EQUAL(size_t(5), keys.size());
 190 
 191   Array<String> values = sa->values();
 192   CPPUNIT_ASSERT_EQUAL(bool(1), values.find("Blah") != size_t(-1));
 193   CPPUNIT_ASSERT_EQUAL(bool(1), values.find("42") != size_t(-1));
 194   CPPUNIT_ASSERT_STRING_EQUAL("Blah Blah Blah Blah 42", values.join(' '));
 195   CPPUNIT_ASSERT_EQUAL(size_t(5), values.size());
 196 }

 411 void HashTableTest :: eachTest(void) {
 412   int x = 5;

 414
