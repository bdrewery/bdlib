 152 void StreamTest :: readTest (void)
 153 {
 154   String sbuf, gbuf;
 155 
 156   a->puts("This is line 1\n");
 157   a->puts("This is line 2\n");
 158   a->puts("This is line 3\n");
 159   a->puts("This is line 4\n");
 160   a->puts("This is line 5\n");
 161   a->puts("This is line 6\n");
 162   a->puts("This is line 7\n");
 163   a->puts("This is line 8\n");
 164   a->puts("This is line 9\n");
 165   a->puts("This is line10\n");
 166   a->puts("This is line11\n");
 167   a->puts("This is line12\n");
 168   a->puts("This is line13\n");
 169   a->puts("This is line14\n");
 170   a->seek(0, SEEK_SET);
 171   int x = 0;
 172   size_t pos = 0;
 173   while (a->tell() < a->length()) {
 174     sbuf = String::printf("This is line%2d\nThis is line%2d\n", x + 1, x + 2);
 175     gbuf = a->read(30);
 176     pos += gbuf.length();
 177     CPPUNIT_ASSERT_EQUAL(pos, a->tell());
 178     CPPUNIT_ASSERT_EQUAL((size_t) 30, gbuf.length());
 179     CPPUNIT_ASSERT_STRING_EQUAL(sbuf, gbuf);
 180     x += 2;
 181   }
 182   CPPUNIT_ASSERT_EQUAL(14, x);
 183 
 184   size_t savelen = a->tell();
 185   a->seek(a->tell(), SEEK_SET);
 186   a->puts("This is line15\n");
 187   a->seek(savelen, SEEK_SET);
 188   sbuf = "This is line15\n";
 189   gbuf = a->read(30);
 190   pos += gbuf.length();
 191   CPPUNIT_ASSERT_EQUAL(pos, a->tell());
 192   CPPUNIT_ASSERT_EQUAL((size_t) 15, gbuf.length());
 193   CPPUNIT_ASSERT_STRING_EQUAL(gbuf, sbuf);
 194 }

 228 void StreamTest :: loadFileTest (void)
 229 {
 230   const char *file = "/etc/passwd";
 231 
 232   FILE *f = NULL;
 233   f = fopen(file, "rb");
 234   if (f == NULL)
 235     return;
 236 
 237   fseek(f, 0, SEEK_END);
 238   size_t size = ftell(f);
 239   fseek(f, 0, SEEK_SET);
 240 
 241   char *buffer = (char*) malloc(size + 1);
 242   fread(buffer, 1, size, f);
 243   buffer[size] = 0;
 244 
 245   a->loadFile(file);
 246 
 247   CPPUNIT_ASSERT_EQUAL(size, a->length());
 248   CPPUNIT_ASSERT(*a == buffer);
 249   free(buffer);
 250 
 251   char buf[1024] = "";
 252   String gbuf;
 253 
 254   fseek(f, 0, SEEK_SET);
 255   while (fgets(buf, sizeof(buf), f) != NULL) {
 256     gbuf = a->getline(sizeof(buf) - 1);
 257     CPPUNIT_ASSERT_STRING_EQUAL(String(buf), gbuf);
 258   }
 259   fclose(f);
 260 
 261   struct stat s;
 262   stat(file, &s);
 263   CPPUNIT_ASSERT_EQUAL(a->length(), size_t(s.st_size));
 264 }
