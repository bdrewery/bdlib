  23 #ifndef _BD_THREAD_H
  24 #define _BD_THREAD_H 1
  25 
  26 #include "bdlib.h"
  27 
  28 #if !defined(USE_PTHREAD) && !defined(WIN32)
  29 # define NO_THREADS 1
  30 #endif
  31 
  32 #if defined(USE_PTHREAD)
  33 # include <pthread.h>
  34 typedef void* threadMainRet_t;
  35 typedef pthread_t threadHandle_t;
  36 #elif defined(WIN32)
  37 # include <windows.h>
  38 typedef unsigned int __stdcall threadMainRet_t;
  39 typedef HANDLE threadHandle_t;
  40 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  41 typedef void* threadMainRet_t;
  42 typedef short threadHandle_t;
  43 #endif
  44 
  45 typedef unsigned int threadId_t;
  46 
  47 BDLIB_NS_BEGIN
  48 
  49 #define THREAD_STARTED  0x1
  50 #define THREAD_DETACHED 0x2
  51 
  52 static threadMainRet_t threadMain(void*);
  53 
  54 /**
  55   * @class Thread
  56   * @brief Extendable thread class
  57   */
  58 class Thread {
  59   private:
  60     threadHandle_t handle;
  61     threadId_t id;
  62     short status;
  63     void* param;
  64 
  65   friend threadMainRet_t threadMain(void*);
  66   public:
  67     Thread(void) : handle(0), id(0), status(0), param(NULL) {};
  68     virtual ~Thread() { stop(); };
  69     int start(void* = NULL);
  70     void detach(void);
  71     void* wait(void);
  72     void stop(void);
  73     threadId_t getThreadId(void);
  74     static threadId_t getCurrentThreadId(void);
  75     static void sleep(int);
  76   protected: 
  77     virtual void* run(void*)=0;
  78   private:
  79     inline bool isStarted() { return status & THREAD_STARTED; };
  80     inline void setStarted() { status |= THREAD_STARTED; };
  81     inline bool isDetached() { return status & THREAD_DETACHED; };
  82     inline void setDetached() { status |= THREAD_DETACHED; };
  83     /**
  84      * @brief Is the thread started and not detached?
  85      * If status == THREAD_STARTED (1), then the THREAD_DETACHED bit is not set, 
  86      * so this is equivalent to: isStarted() && !isDetached()
  87      */
  88     inline bool shouldDetach() { return status == THREAD_STARTED; };
  89 };
  90 
  91 BDLIB_NS_END
  92 
  93 #endif /* _BD_THREAD_H */
