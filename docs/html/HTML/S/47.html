  56 /**
  57  * @class String
  58  * @brief Provides custom string class for easy and optimized string manipulation.
  59  * @todo compute hash on insert, then use to compare instead of strcmp
  60  * @todo an updating hash as the copy is done.
  61  */
  62 class String : public ReferenceCountedArray<String_Array_Type> {
  63   private:
  64         typedef std::allocator<String_Array_Type> Allocator;
  65         static unsigned char cleanse_ctr;
  66 
  67         /* Cleanse our buffer using OPENSSL_cleanse() */
  68         void cleanse() {
  69           const size_t len = capacity();
  70           char* ptr = const_cast<char*>(real_begin()); //real_begin() will ignore offset.
  71 
  72           unsigned char *p = reinterpret_cast<unsigned char *>(ptr);
  73           size_t loop = len;
  74           while(loop--)
  75           {
  76             *(p++) = static_cast<unsigned char>(cleanse_ctr);
  77             cleanse_ctr += (17 + (reinterpret_cast<size_t>(p) & 0xF));
  78           }
  79           if(memchr(ptr, cleanse_ctr, len))
  80             cleanse_ctr += (63 + reinterpret_cast<size_t>(p));
  81         }
  82   public:

 447 inline String& String::operator+=(const size_t n) {
 448   if (!length())
 449     return *this;
 450   if (n > length()) {
 451     offset = length();
 452     setLength(0);
 453   } else {
 454     offset += n;
 455     subLength(n);
 456   }
 457   return *this;
 458 }
 459 
 460 inline String& String::operator-=(const size_t n) {
 461   if (!length())
 462     return *this;
 463   if (n > length()) {
 464     offset = length();
 465     setLength(0);
 466   } else
 467     subLength(n);
 468   return *this;
 469 }
