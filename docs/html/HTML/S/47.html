  40 template <class T>
  41 /**
  42  * @class Queue
  43  * @brief A FIFO Queue, what else!
  44  */
  45 class Queue : public List<T> {
  46   typedef typename List<T>::Node Node;
  47 
  48   public:
  49     friend class QueueIterator<T>;
  50     friend class QueueConstIterator<T>;
  51 
  52     typedef QueueIterator<T> iterator;
  53     typedef QueueConstIterator<T> const_iterator;
  54     typedef T value_type;
  55 
  56     typedef size_t             size_type;
  57     typedef std::ptrdiff_t     difference_type;
  58     typedef value_type*        pointer;
  59     typedef const value_type*  const_pointer;
  60     typedef value_type&        reference;
  61     typedef const value_type&  const_reference;
  62 
  63 
  64     Queue() : List<value_type>() {};
  65     virtual ~Queue() {};
  66 
  67     /**
  68      * @brief Dequeue an element from the tail
  69      */
  70     value_type dequeue() {
  71       value_type item = this->tail->item;
  72       this->deleteNode(this->tail);
  73       return item;
  74     };
  75 
  76     /**
  77      * @sa dequeue
  78      */
  79     inline friend Queue<value_type>& operator>> (Queue<value_type>& queue, reference item) {
  80       item = queue.dequeue();
  81       return queue;
  82     }
  83 
  84     /**
  85      * @brief Return the first element, but don't dequeue it
  86      */
  87     inline const_iterator front() {
  88       return this->isEmpty() ? this->end() : const_iterator(this, this->tail);
  89     }
  90 
  91 
  92     /**
  93      * @brief Enqueue an element into the queue
  94      * @param item The element to enqueue
  95      * @todo Make this take an iterator as a position to insert at
  96      */
  97     inline void enqueue(const_reference item) {
  98       this->insert(item);
  99     };
 100 
 101     /**
 102      * @sa push
 103      */
 104     inline friend Queue<value_type>& operator<< (Queue<value_type>& queue, const_reference item) {
 105       queue.enqueue(item);
 106       return queue;
 107     }
 108 
 109     inline iterator begin() { return iterator(this, this->tail); };
 110     inline iterator end() { return iterator(this, NULL); };
 111     inline const_iterator begin() const { return const_iterator(this, this->tail); };
 112     inline const_iterator end() const { return const_iterator(this, NULL, 1); };
 113 };

 115 template <class T>
 116 class QueueIterator {
 117   friend class Queue<T>;
 118   friend class QueueConstIterator<T>;
 119 
 120   public:
 121     typedef QueueIterator<T>            iterator;
 122     typedef QueueConstIterator<T>       const_iterator;
 123     typedef typename Queue<T>::Node     Node;
 124     typedef std::forward_iterator_tag   iterator_category;
 125     typedef T                           value_type;
 126     typedef std::ptrdiff_t              difference_type;
 127     typedef size_t                      size_type;
 128     typedef value_type&                 reference;
 129     typedef value_type*                 pointer;
 130 
 131   private:
 132     Queue<T> *queue;
 133     Node *current;
 134 
 135     QueueIterator(Queue<T>* _queue, Node* start) : queue(_queue), current(start) {};
 136   public:
 137     QueueIterator() : queue(NULL), current(NULL) {};
 138     ~QueueIterator() { }
 139 
 140     reference operator *() const { return current->item; }
 141     pointer operator ->() const { return &(operator*()); }
 142 
 143 
 144     bool operator==(const iterator& iter) const { return iter.current == current; }
 145     bool operator!=(const iterator& iter) const { return iter.current != current; }
 146 
 147     //Postfix
 148     iterator operator ++(int) {
 149       iterator tmp(*this);
 150       current = current->next;
 151       return tmp;
 152     }
 153 
 154     iterator& operator ++() {
 155       current = current->next;
 156       return *this;
 157     }
 158 };
