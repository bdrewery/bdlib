  35 template <class T>
  36 /**
  37  * @class Queue
  38  * @brief A FIFO Queue, what else!
  39  */
  40 class Queue : public List<T> {
  41   typedef typename List<T>::Node Node;
  42 
  43   public:
  44     friend class QueueIterator<T>;
  45     friend class QueueConstIterator<T>;
  46 
  47     typedef QueueIterator<T> iterator;
  48     typedef QueueConstIterator<T> const_iterator;
  49     typedef T value_type;
  50 
  51     typedef size_t             size_type;
  52     typedef std::ptrdiff_t     difference_type;
  53     typedef value_type*        pointer;
  54     typedef const value_type*  const_pointer;
  55     typedef value_type&        reference;
  56     typedef const value_type&  const_reference;
  57 
  58 
  59     Queue() : List<value_type>() {};
  60     virtual ~Queue() {};
  61 
  62     /**
  63      * @brief Dequeue an element from the tail
  64      */
  65     value_type dequeue() {
  66       value_type item = this->tail->ptr;
  67       this->deleteNode(this->tail);
  68       return item;
  69     };
  70 
  71     /**
  72      * @sa dequeue
  73      */
  74     inline friend Queue<value_type>& operator>> (Queue<value_type>& queue, reference item) {
  75       item = queue.dequeue();
  76       return queue;
  77     }
  78 
  79     /**
  80      * @brief Return the first element, but don't dequeue it
  81      */
  82     inline const_iterator front() {
  83       return this->isEmpty() ? this->end() : const_iterator(this, this->tail);
  84     }
  85 
  86 
  87     /**
  88      * @brief Enqueue an element into the queue
  89      * @param item The element to enqueue
  90      * @todo Make this take an iterator as a position to insert at
  91      */
  92     inline void enqueue(const_reference item) {
  93       this->insert(item);
  94     };
  95 
  96     /**
  97      * @sa push
  98      */
  99     inline friend Queue<value_type>& operator<< (Queue<value_type>& queue, const_reference item) {
 100       queue.enqueue(item);
 101       return queue;
 102     }
 103 
 104     inline iterator begin() { return iterator(this, this->tail); };
 105     inline iterator end() { return iterator(this, NULL); };
 106     inline const_iterator begin() const { return const_iterator(this, this->tail); };
 107     inline const_iterator end() const { return const_iterator(this, NULL, 1); };
 108 };

 110 template <class T>
 111 class QueueIterator {
 112   friend class Queue<T>;
 113   friend class QueueConstIterator<T>;
 114 
 115   public:
 116     typedef Queue<T> Queue_t;
 117     typedef QueueIterator<T> iterator;
 118     typedef QueueConstIterator<T> const_iterator;
 119     typedef typename Queue_t::Node Node;
 120     typedef std::forward_iterator_tag iterator_category;
 121     typedef T value_type;
 122     typedef std::ptrdiff_t difference_type;
 123     typedef size_t size_type;
 124     typedef T& reference;
 125     typedef T* pointer;
 126 
 127   private:
 128     Queue_t *queue;
 129     Node *current;
 130 
 131     QueueIterator(Queue_t* _queue, Node* start) : queue(_queue), current(start) {};
 132   public:
 133     QueueIterator() : queue(NULL), current(NULL) {};
 134     ~QueueIterator() { }
 135 
 136     reference operator *() const { return current->ptr; }
 137     pointer operator ->() const { return &(operator*()); }
 138 
 139 
 140     bool operator==(const iterator& iter) const { return iter.current == current; }
 141     bool operator!=(const iterator& iter) const { return iter.current != current; }
 142 
 143     //Postfix
 144     iterator operator ++(int) {
 145       iterator tmp(*this);
 146       current = current->next;
 147       return tmp;
 148     }
 149 
 150     iterator& operator ++() {
 151       current = current->next;
 152       return *this;
 153     }
 154 };
