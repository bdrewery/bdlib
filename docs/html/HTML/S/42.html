  57 /**
  58  * @class ScriptInterp
  59  * @brief Handles generalized script interpreter access
  60  */
  61 class ScriptInterp {
  62   private:
  63         ScriptInterp(const ScriptInterp&) = delete;
  64         ScriptInterp& operator=(const ScriptInterp&) = delete;
  65 
  66   protected:
  67         virtual int init() = 0;
  68         virtual int destroy() = 0;
  69 
  70         struct ScriptCmd {
  71           ScriptCmd(const ScriptCmd&) = delete;
  72           ScriptCmd& operator=(const ScriptCmd&) = delete;
  73           ScriptInterp* si;
  74           const String cmdName;
  75           std::unique_ptr<ScriptCommandHandlerBase> callback_proxy;
  76           const char* usage;
  77           size_t callbackParamMin;
  78           size_t callbackParamMax;
  79           bool registered;
  80           ScriptCmd(ScriptInterp* _si, const String &_cmdName,
  81               std::unique_ptr<ScriptCommandHandlerBase> _callback_proxy,
  82               const char* _usage, size_t _callbackParamMin,
  83               size_t _callbackParamMax) :
  84             si(_si),
  85             cmdName(_cmdName),
  86             callback_proxy(std::move(_callback_proxy)),
  87             usage(_usage),
  88             callbackParamMin(_callbackParamMin),
  89             callbackParamMax(_callbackParamMax),
  90             registered(false)
  91           {}
  92           virtual ~ScriptCmd() {};
  93           virtual void registerCmd() {
  94             registered = true;
  95           }
  96           virtual void unregisterCmd() {
  97             registered = false;
  98           }
  99         };
 100         typedef std::shared_ptr<ScriptCmd>
 101           ScriptCmdPtr;
 102   public:
 103         enum script_type {
 104           SCRIPT_TYPE_TCL,
 105         };
 106         typedef void (*link_var_hook)(const void*, const void*);
 107 
 108         ScriptInterp() {};
 109         virtual ~ScriptInterp() {};
 110 
 111         /**
 112          * @param script The script to evaluate
 113          */
 114         virtual String eval(const String& script) = 0;
 115 
 116         enum LoadError {
 117           SCRIPT_LOAD_OK,
 118           SCRIPT_LOAD_ERROR,
 119           SCRIPT_LOAD_WRONG_INTERP
 120         };
 121         /**
 122          * @brief Load script file
 123          * @param fileName The file to load
 124          * @param resultStr String to hold error output from interp
 125          */
 126         virtual LoadError loadScript(const String& fileName, String& resultStr) = 0;
 127 
 128         template<typename InterpType, typename ReturnType, typename... Params>
 129         static void createCommand(InterpType& si, const String& cmdName, ReturnType(*callback)(Params...), const char* usage = nullptr, size_t min_params = size_t(-1)) {
 130           si.createCommand(cmdName, callback, usage, min_params);
 131         }
 132 
 133         /**
 134          * @brief Remove a command from the interp
 135          * @param name The name of the command to remove
 136          */
 137         virtual void deleteCommand(const String& name) = 0;
 138 
 139         /**
 140          * @brief Link a C variable to the interp
 141          * @param si The ScriptInterp to link the variable to
 142          * @param varName The name to create the variable as
 143          * @param var The variable to link to
 144          * @note This is dispatching to the derived class due to being unable to do specialized template methods
 145          */
 146         template <typename Interp, typename T>
 147           static void linkVar(Interp& si, const String& varName, T& var) {
 148             si.linkVar(varName, var);
 149           };
 150 
 151         /**
 152          * @brief Remove a variable from the interp
 153          * @param varName The name of the variable to remove
 154          */
 155         virtual void unlinkVar(const String& varName) = 0;
 156 
 157         // Used for downcasting hack due to no virtual templates
 158         virtual script_type type() const = 0;
 159 };
