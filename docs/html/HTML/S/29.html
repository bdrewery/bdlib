  57 /**
  58  * @class String
  59  * @brief Provides custom string class for easy and optimized string manipulation.
  60  */
  61 class String : public ReferenceCountedArray<String_Array_Type> {
  62   private:
  63         typedef std::allocator<String_Array_Type> Allocator;
  64         static unsigned char cleanse_ctr;
  65 
  66         /* Cleanse our buffer using OPENSSL_cleanse() */
  67         void cleanse() {
  68           const size_t len = capacity();
  69           char* ptr = const_cast<char*>(real_begin()); //real_begin() will ignore offset.
  70 
  71           unsigned char *p = reinterpret_cast<unsigned char *>(ptr);
  72           size_t loop = len;
  73           while(loop--)
  74           {
  75             *(p++) = static_cast<unsigned char>(cleanse_ctr);
  76             cleanse_ctr += (17 + (reinterpret_cast<size_t>(p) & 0xF));
  77           }
  78           if(memchr(ptr, cleanse_ctr, len))
  79             cleanse_ctr += (63 + reinterpret_cast<size_t>(p));
  80         }
  81   public:
  82 
  83         /* Constructors */
  84         String(const Allocator& allocator = Allocator()) : ReferenceCountedArray<String_Array_Type, Allocator>(allocator) {};
  85         String(const String& string) : ReferenceCountedArray<String_Array_Type, Allocator>(string) {};
  86         String(String&& string) : ReferenceCountedArray<String_Array_Type, Allocator>(std::move(string)) {};
  87         /**
  88          * @brief Create a String from a given cstring.
  89          * @param cstring The null-terminated character array to create the object from.
  90          * @post The buffer has been filled with the string.
  91          * @test String test("Some string");

 446 inline String& String::operator+=(const size_t n) {
 447   if (!length())
 448     return *this;
 449   if (n > length()) {
 450     offset = length();
 451     setLength(0);
 452   } else {
 453     offset += n;
 454     subLength(n);
 455   }
 456   return *this;
 457 }
 458 
 459 inline String& String::operator-=(const size_t n) {
 460   if (!length())
 461     return *this;
 462   if (n > length()) {
 463     offset = length();
 464     setLength(0);
 465   } else
 466     subLength(n);
 467   return *this;
 468 }

 472 // comparison operators:
 473 inline bool operator==(const String& lhs, const String& rhs) {
 474   return (lhs.length() == rhs.length() &&
 475       lhs.compare(rhs) == 0);
 476 }
 477 
 478 inline bool operator!=(const String& lhs, const String& rhs) {
 479   return ! (lhs == rhs);
 480 }
 481 
 482 inline bool operator<(const String& lhs, const String& rhs) {
 483   return (lhs.compare(rhs) < 0);
 484 }
 485 
 486 inline bool operator<=(const String& lhs, const String& rhs) {
 487   return ! (rhs < lhs);
 488 }
 489 
 490 inline bool operator>(const String& lhs, const String& rhs) {
 491   return (rhs < lhs);
 492 }
 493 
 494 inline bool operator>=(const String& lhs, const String& rhs) {
 495   return ! (lhs < rhs);
 496 }
