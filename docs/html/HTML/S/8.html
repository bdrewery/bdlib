  23 #ifndef _BD_SORTEDLIST_H
  24 #define _BD_SORTEDLIST_H 1
  25 
  26 #include "bdlib.h"
  27 #include "List.h"
  28 
  29 BDLIB_NS_BEGIN
  30 
  31 template <class T>
  32 /**
  33  * @class SortedList
  34  * @brief A List which sorts on insert
  35  */
  36 class SortedList : public List<T> {
  37   typedef T iterator_type;
  38   typedef typename List<iterator_type>::Node Node;
  39 
  40   public:
  41     SortedList() : List<iterator_type>() {};
  42     virtual ~SortedList() {};
  43 
  44     /**
  45      * @brief Insert an element into the list (ordered)
  46      * @param item The element to insert
  47      * This operation is O(n/2)
  48      */
  49     virtual void insert(const iterator_type& item) {
  50       /*
  51        * 1) list = NULL
  52        * 2) list = (X)
  53        * 3) list = (X)->(Y)
  54        * 4) list = (W)->(X)->(Y)
  55        */
  56       Node* node = new Node(item), *prev = NULL, *current = this->head;
  57       while (current != NULL && item < current->item) {
  58         prev = current;
  59         current = current->next;
  60       }
  61 
  62       if (current == NULL && this->head == NULL) {
  63         this->head = this->tail = node;
  64       } else if (current == NULL) { //Ran off the end, insert AS tail
  65         this->tail = prev->next = node;
  66         node->prev = prev; //prev=tail
  67       } else if (current == this->head) {
  68         node->next = this->head;
  69         current->prev = node;
  70         this->head = node;
  71       } else {
  72         node->next = current;
  73         node->prev = prev;
  74         current->prev = node;
  75         prev->next = node;
  76       }
  77 
  78       ++(this->my_size);
  79     }
  80 
  81 };
  82 
  83 BDLIB_NS_END
  84 #endif /* _BD_SORTEDLIST_H */
