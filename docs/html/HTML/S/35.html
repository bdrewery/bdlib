  53 /**
  54  * @class String
  55  * @brief Provides custom string class for easy and optimized string manipulation.
  56  * @todo compute hash on insert, then use to compare instead of strcmp
  57  * @todo an updating hash as the copy is done.
  58  */
  59 class String : public ReferenceCountedArray<String_Array_Type> {
  60   private:
  61         typedef std::allocator<String_Array_Type> Allocator;
  62         static unsigned char cleanse_ctr;
  63 
  64         /* Cleanse our buffer using OPENSSL_cleanse() */
  65         void cleanse() {
  66           const size_t len = capacity();
  67           char* ptr = const_cast<char*>(real_begin()); //real_begin() will ignore offset.
  68 
  69           unsigned char *p = reinterpret_cast<unsigned char *>(ptr);
  70           size_t loop = len;
  71           while(loop--)
  72           {
  73             *(p++) = static_cast<unsigned char>(cleanse_ctr);
  74             cleanse_ctr += (17 + (reinterpret_cast<size_t>(p) & 0xF));
  75           }
  76           if(memchr(ptr, cleanse_ctr, len))
  77             cleanse_ctr += (63 + reinterpret_cast<size_t>(p));
  78         }
  79   public:

 420 inline String& String::operator+=(const size_t n) {
 421   if (!length())
 422     return *this;
 423   if (n > length()) {
 424     offset = length();
 425     setLength(0);
 426   } else {
 427     offset += n;
 428     subLength(n);
 429   }
 430   return *this;
 431 }
 432 
 433 inline String& String::operator-=(const size_t n) {
 434   if (!length())
 435     return *this;
 436   if (n > length()) {
 437     offset = length();
 438     setLength(0);
 439   } else
 440     subLength(n);
 441   return *this;
 442 }
