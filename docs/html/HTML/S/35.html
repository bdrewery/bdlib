  54 /**
  55  * @class String
  56  * @brief Provides custom string class for easy and optimized string manipulation.
  57  * @todo compute hash on insert, then use to compare instead of strcmp
  58  * @todo an updating hash as the copy is done.
  59  */
  60 class String : public ReferenceCountedArray<String_Array_Type> {
  61   private:
  62         static unsigned char cleanse_ctr;
  63 
  64         /* Cleanse our buffer using OPENSSL_cleanse() */
  65         void cleanse() {
  66           const size_t len = capacity();
  67           char* ptr = const_cast<char*>(real_begin()); //real_begin() will ignore offset.
  68 
  69           unsigned char *p = reinterpret_cast<unsigned char *>(ptr);
  70           size_t loop = len;
  71           while(loop--)
  72           {
  73             *(p++) = static_cast<unsigned char>(cleanse_ctr);
  74             cleanse_ctr += (17 + (reinterpret_cast<size_t>(p) & 0xF));
  75           }
  76           if(memchr(ptr, cleanse_ctr, len))
  77             cleanse_ctr += (63 + reinterpret_cast<size_t>(p));
  78         }
  79   public:
  80         static const size_t npos = static_cast<size_t>(-1);

 415 inline String& String::operator += (const size_t n) {
 416   if (!length())
 417     return *this;
 418   if (n > length()) {
 419     offset = length();
 420     setLength(0);
 421   } else {
 422     offset += n;
 423     subLength(n);
 424   }
 425   return *this;
 426 }
 427 
 428 inline String& String::operator -= (const size_t n) {
 429   if (!length())
 430     return *this;
 431   if (n > length()) {
 432     offset = length();
 433     setLength(0);
 434   } else
 435     subLength(n);
 436   return *this;
 437 }
