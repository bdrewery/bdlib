  58 /**
  59  * @class String
  60  * @brief Provides custom string class for easy and optimized string manipulation.
  61  * @todo compute hash on insert, then use to compare instead of strcmp
  62  * @todo an updating hash as the copy is done.
  63  */
  64 class String : public ReferenceCountedArray<String_Array_Type> {
  65   private:
  66         typedef std::allocator<String_Array_Type> Allocator;
  67         static unsigned char cleanse_ctr;
  68 
  69         /* Cleanse our buffer using OPENSSL_cleanse() */
  70         void cleanse() {
  71           const size_t len = capacity();
  72           char* ptr = const_cast<char*>(real_begin()); //real_begin() will ignore offset.
  73 
  74           unsigned char *p = reinterpret_cast<unsigned char *>(ptr);
  75           size_t loop = len;
  76           while(loop--)
  77           {
  78             *(p++) = static_cast<unsigned char>(cleanse_ctr);
  79             cleanse_ctr += (17 + (reinterpret_cast<size_t>(p) & 0xF));
  80           }
  81           if(memchr(ptr, cleanse_ctr, len))
  82             cleanse_ctr += (63 + reinterpret_cast<size_t>(p));
  83         }
  84   public:

  87         String(const Allocator& allocator = Allocator()) : ReferenceCountedArray<String_Array_Type, Allocator>(allocator) {};

  89 #ifdef __GXX_EXPERIMENTAL_CXX0X__

 453 inline String& String::operator+=(const size_t n) {
 454   if (!length())
 455     return *this;
 456   if (n > length()) {
 457     offset = length();
 458     setLength(0);
 459   } else {
 460     offset += n;
 461     subLength(n);
 462   }
 463   return *this;
 464 }
 465 
 466 inline String& String::operator-=(const size_t n) {
 467   if (!length())
 468     return *this;
 469   if (n > length()) {
 470     offset = length();
 471     setLength(0);
 472   } else
 473     subLength(n);
 474   return *this;
 475 }
