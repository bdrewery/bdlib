 294 void BinaryTreeTest :: iterateTest (void)
 295 {
 296   a->insert(1, "Blah");
 297   a->insert(1, "Bleck");
 298   a->insert(2, "Blah2");
 299   a->insert(8, "Blah8");
 300   a->insert(4, "Blah4");
 301   CPPUNIT_ASSERT_EQUAL(size_t(4), a->size());
 302 
 303   BinaryTree<int, String>::iterator iter = a->begin();
 304 
 305   CPPUNIT_ASSERT_EQUAL(true, (bool) iter);
 306   while (iter) {
 307 //    int key = (int) iter.next();
 308 //    printf("%d\n", key);
 309     String value = (iter++)->value();
 310 //    printf("%d: %s\n", key, a->getValue(key));
 311   }
 312 
 313   iter = a->begin();
 314   CPPUNIT_ASSERT_EQUAL(true, (bool) iter);
 315   CPPUNIT_ASSERT_STRING_EQUAL("Bleck", (iter++)->value());
 316   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", (iter++)->value());
 317   CPPUNIT_ASSERT_STRING_EQUAL("Blah4", (iter++)->value());
 318   CPPUNIT_ASSERT_STRING_EQUAL("Blah8", (iter++)->value());
 319   CPPUNIT_ASSERT_EQUAL(size_t(4), a->size());
 320 
 321   a->remove(4);
 322   iter = a->begin();
 323   CPPUNIT_ASSERT_STRING_EQUAL("Bleck", (iter++)->value());
 324   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", (iter++)->value());
 325   CPPUNIT_ASSERT_STRING_EQUAL("Blah8", (iter++)->value());
 326   CPPUNIT_ASSERT_EQUAL(size_t(3), a->size());
 327 
 328   a->remove(2);
 329   iter = a->begin();
 330   CPPUNIT_ASSERT_STRING_EQUAL("Bleck", (iter++)->value());
 331   CPPUNIT_ASSERT_STRING_EQUAL("Blah8", (iter++)->value());
 332   CPPUNIT_ASSERT_EQUAL(size_t(2), a->size());
 333 
 334   a->remove(8);
 335   iter = a->begin();
 336   CPPUNIT_ASSERT_STRING_EQUAL("Bleck", (iter++)->value());
 337   CPPUNIT_ASSERT_EQUAL(size_t(1), a->size());
 338 
 339   a->remove(1);
 340   iter = a->begin();
 341   CPPUNIT_ASSERT_EQUAL(size_t(0), a->size());
 342   CPPUNIT_ASSERT_EQUAL(false, (bool) iter);
 343 
 344   /* Now reinsert some data N:testN .. */
 345   for (int i = 0; i < 5; ++i) {
 346     char buf[10] = "";
 347     sprintf(buf, "test%d", i);
 348     (*a).insert(i, String(buf));
 349   }
 350 
 351   int n = 0;
 352   /* And ensure it comes out in key order in tact */
 353   CPPUNIT_ASSERT_EQUAL(true, (bool) (*a).begin());
 354   for (iter = (*a).begin(); iter; (++iter)) {
 355     char buf[10] = "";
 356     sprintf(buf, "test%d", n);
 357     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 358     CPPUNIT_ASSERT_STRING_EQUAL( String(buf), iter->value());
 359     ++n;
 360   }
 361 
 362   n = (*a).size() - 1;
 363   /* Test reverse */
 364   for (iter = (*a).end(); iter; (--iter)) {
 365     char buf[10] = "";
 366     sprintf(buf, "test%d", n);
 367     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 368     CPPUNIT_ASSERT_STRING_EQUAL( String(buf), iter->value());
 369     --n;
 370   }
 371 
 372   /* make sure the reverse did not trample the tree */
 373   n = 0;
 374   /* And ensure it comes out in key order in tact */
 375   for (iter = (*a).begin(); iter; (++iter)) {
 376     char buf[10] = "";
 377     sprintf(buf, "test%d", n);
 378     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 379     CPPUNIT_ASSERT_STRING_EQUAL( String(buf), iter->value());
 380     ++n;
 381   }
 382 
 383   //Remove evens .. check after
 384   n = 0;
 385   for (iter = (*a).begin(); iter; (++iter)) {
 386     //While removing, this next condition should hold true
 387     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 388     if (n % 2) {
 389       iter.remove();
 390     }
 391 //   The iterator is INVALID now
 392 //   CPPUNIT_ASSERT_EQUAL(n, iter->key());
 393     ++n;
 394   }
 395 
 396   //Ensure odds are still there
 397   n = 0;
 398   for (iter = (*a).begin(); iter; (++iter)) {
 399     char buf[10] = "";
 400     sprintf(buf, "test%d", n);
 401     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 402     CPPUNIT_ASSERT_STRING_EQUAL( String(buf), iter->value());
 403     n += 2;
 404   }
 405 
 406   (*a).clear();
 407 
 408   /* Now reinsert some data N:testN .. */
 409   for (int i = 0; i < 5; ++i) {
 410     char buf[10] = "";
 411     sprintf(buf, "test%d", i);
 412     (*a).insert(i, String(buf));
 413   }
 414 
 415   /* Now do the same test in reverse (removing from the end) */
 416   //Remove evens .. check after
 417   n = (*a).size() - 1;
 418   int keep_n = n;
 419   for (iter = (*a).end(); iter; (--iter)) {
 420     //While removing, this next condition should hold true
 421     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 422     if (n % 2) {
 423       iter.remove();
 424     }
 425 //   The iterator is INVALID now
 426 //   CPPUNIT_ASSERT_EQUAL(n, iter->key());
 427     --n;
 428   }
 429 
 430   CPPUNIT_ASSERT_EQUAL(false, a->isEmpty());
 431 
 432   //Ensure odds are still there
 433   n = keep_n;
 434   for (iter = (*a).end(); iter; (--iter)) {
 435     char buf[10] = "";
 436     sprintf(buf, "test%d", n);
 437     CPPUNIT_ASSERT_EQUAL(n, iter->key());
 438     CPPUNIT_ASSERT_STRING_EQUAL( String(buf), iter->value());
 439     n -= 2;
 440   }
