  88 template <class Key, class Value>
  89 struct KeyValue {
  90   public:
  91     Key k;
  92     Value v;
  93 
  94     KeyValue() : k(), v() {};
  95     KeyValue(const Key& _key, const Value& _value) : k(_key), v(_value) {};
  96     KeyValue(const KeyValue& kv) : k(kv.k), v(kv.v) {};
  97     KeyValue(KeyValue&& kv) : k(), v() {
  98       swap(*this, kv);
  99     }
 100       friend void swap(KeyValue& a, KeyValue& b) {
 101         using std::swap;
 102 
 103         swap(a.k, b.k);
 104         swap(a.v, b.v);
 105       }
 106       KeyValue& operator=(KeyValue kv) {
 107         swap(*this, kv);
 108         return *this;
 109       }
 110 
 111     inline const Key& key() const { return k; };
 112     inline const Value& value() const { return v; };
