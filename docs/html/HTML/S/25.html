 137         struct ScriptCmdTCL : public ScriptCmd {
 138           static int _createCommand_callback(ClientData clientData,
 139               Tcl_Interp *interp, int objc, Tcl_Obj *CONST objv[]);
 140 
 141           ScriptCmdTCL(const ScriptCmdTCL&) = delete;
 142           ScriptCmdTCL& operator=(const ScriptCmdTCL&) = delete;
 143           ScriptCmdTCL(ScriptInterpTCL* _si, const String &_cmdName,
 144               std::unique_ptr<ScriptCommandHandlerBase> _callback_proxy,
 145               const char* _usage, size_t _callbackParamMin,
 146               size_t _callbackParamMax) :
 147               ScriptCmd(_si, std::move(_cmdName), std::move(_callback_proxy),
 148                   _usage, _callbackParamMin, _callbackParamMax) {};
 149           virtual ~ScriptCmdTCL() {
 150             this->unregisterCmd();
 151           }
 152           virtual void registerCmd() {
 153             if (registered)
 154               return;
 155             Tcl_CreateObjCommand(static_cast<ScriptInterpTCL*>(si)->interp,
 156                 *cmdName, _createCommand_callback, nullptr, nullptr);
 157             ScriptCmd::registerCmd();
 158           }
 159           virtual void unregisterCmd() {
 160             if (!registered)
 161               return;
 162             Tcl_DeleteCommand(static_cast<ScriptInterpTCL*>(si)->interp,
 163                 *cmdName);
 164             ScriptCmd::unregisterCmd();
 165           }
 166         };

 168         void _createCommand(const String& cmdName,
 169             std::unique_ptr<ScriptCommandHandlerBase> callback_proxy,
 170             const char* usage,
 171             size_t callbackParamMin, size_t callbackParamMax) {
 172           auto ccd = std::make_shared<ScriptCmdTCL>(this, cmdName,
 173               std::move(callback_proxy), usage, callbackParamMin,
 174               callbackParamMax);
 175           CmdHandlerData[cmdName] = ccd;
 176           ccd->registerCmd();
 177         }

 219         template<typename T>
 220           static const char* tcl_traceSet(ClientData clientData,
 221               Tcl_Interp* interp, char* name1, char* name2, int flags) {
 222             Tcl_Obj *obj = ScriptInterpTCL::TraceSet(interp, name1, name2,
 223                 flags);
 224             if (!obj)
 225               return name1;
 226             const auto oldval(std::move(*static_cast<T*>(clientData)));
 227 
 228             *static_cast<T*>(clientData) =
 229               std::move(tcl_to_c_cast<T>::from(obj, nullptr));
 230             const auto link_var_hook = link_var_hooks.find(name1);
 231             if (link_var_hook != std::end(link_var_hooks))
 232               link_var_hook->second(
 233                   (const void*)&oldval, (const void*)(clientData));
 234             return nullptr;
 235           }
 236 
 237         template<typename T>
 238           static const char* tcl_traceSetPtr(ClientData clientData,
 239               Tcl_Interp* interp, char* name1, char* name2, int flags) {
 240             Tcl_Obj *obj = ScriptInterpTCL::TraceSet(interp, name1, name2,
 241                 flags);
 242             if (!obj)
 243               return name1;
 244             const auto link_var_hook = link_var_hooks.find(name1);
 245             const trace_ptr_data *data =
 246               static_cast<trace_ptr_data*>(clientData);
 247             /* Save oldval */
 248             typedef typename std::pointer_traits<T>::element_type T_name;
 249             std::unique_ptr<T_name[]> oldval;
 250             if (link_var_hook != std::end(link_var_hooks)) {
 251               oldval = std::move(std::make_unique<T_name[]>(data->size));
 252               memcpy(oldval.get(), static_cast<T>(data->ptr), data->size);
 253             }
 254 
 255             memmove(data->ptr, tcl_to_c_cast<T>::from(obj, nullptr),
 256                 data->size);
 257             if (std::is_same<T, char*>::value ||
 258                 std::is_same<T, const char*>::value)
 259               static_cast<T>(data->ptr)[data->size - 1] = '\0';
 260             if (link_var_hook != std::end(link_var_hooks))
 261               link_var_hook->second(
 262                   (const void*)(oldval.get()), (const void*)(data->ptr));
 263             return nullptr;
 264           }

 306         /**
 307          * @brief Link a C variable to the interp
 308          * @param varName The name to create the variable as
 309          * @param var The variable to link to
 310          */
 311         template <typename T>
 312           inline void linkVar(const String& varName,
 313               T& var,
 314               link_var_hook hook_func = nullptr) {
 315             setupTraces(varName, (ClientData) &var,
 316                 (Tcl_VarTraceProc*) tcl_traceGet<const T>,
 317                 (Tcl_VarTraceProc*) tcl_traceSet<T>, hook_func);
 318           };
 319 
 320         /**
 321          * @brief Link a C variable to the interp
 322          * @param varName The name to create the variable as
 323          * @param var The variable to link to
 324          */
 325         template <typename T>
 326           inline void linkVar(const String& varName,
 327               T* var, size_t size,
 328               link_var_hook hook_func = nullptr) {
 329             auto data = std::make_unique<trace_ptr_data>(var, size);
 330             setupTraces(varName, (ClientData) data.get(),
 331                 (Tcl_VarTraceProc*) tcl_traceGetPtrData<const T*>,
 332                 (Tcl_VarTraceProc*) tcl_traceSetPtr<T*>, hook_func);
 333             trace_ptrs[varName] = std::move(data);
 334           };
 335

 337          * @brief Link a const C variable to the interp
 338          * @param varName The name to create the variable as
