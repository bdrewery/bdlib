  36 class ScriptCallbackBase {
  37   public:
  38     virtual ~ScriptCallbackBase() {};
  39     virtual void call(size_t argc, void* const argv[], ScriptInterp* si, void *proxy_data = NULL) = 0;
  40 };
  41 
  42 class ScriptCallbacker {
  43   private:
  44     ScriptCallbacker(const ScriptCallbacker&) = delete;
  45     ScriptCallbacker& operator=(const ScriptCallbacker&) = delete;
  46   public:
  47     ScriptInterp* si;
  48     const String cmd;
  49 
  50     ScriptCallbacker(ScriptInterp* _si, const String& _cmd) : si(_si), cmd(_cmd) {};
  51     virtual ~ScriptCallbacker() {};
  52     virtual String call(const Array<String>& params = Array<String>()) = 0;
  53 };

  55 /**
  56  * @class ScriptInterp
  57  * @brief Handles generalized script interpreter access
  58  */
  59 class ScriptInterp {
  60   private:
  61         ScriptInterp(const ScriptInterp&) = delete;
  62         ScriptInterp& operator=(const ScriptInterp&) = delete;
  63 
  64   protected:
  65         virtual int init() = 0;
  66         virtual int destroy() = 0;
  67 
  68   public:
  69         struct script_cmd_handler_clientdata {
  70           ScriptInterp* si;
  71           ScriptCallbackBase* callback_proxy;
  72           const char* usage;
  73           size_t callbackParamMin;
  74           size_t callbackParamMax;
  75           script_cmd_handler_clientdata (ScriptInterp* _si, ScriptCallbackBase* _callback_proxy, const char* _usage, size_t _callbackParamMin, size_t _callbackParamMax) :
  76             si(_si),
  77             callback_proxy(_callback_proxy),
  78             usage(_usage),
  79             callbackParamMin(_callbackParamMin),
  80             callbackParamMax(_callbackParamMax) {};
  81         };
  82 
  83         enum script_type {
  84           SCRIPT_TYPE_TCL,
  85         };
  86         typedef void (*link_var_hook)(const void*, const void*);
  87 
  88         ScriptInterp() {};
  89         virtual ~ScriptInterp() {};
  90 
  91         /**
  92          * @param script The script to evaluate
  93          */
  94         virtual String eval(const String& script) = 0;
  95 
  96         enum LoadError {
  97           SCRIPT_LOAD_OK,
  98           SCRIPT_LOAD_ERROR,
  99           SCRIPT_LOAD_WRONG_INTERP
 100         };
 101         /**
 102          * @brief Load script file
 103          * @param fileName The file to load
 104          * @param resultStr String to hold error output from interp
 105          */
 106         virtual LoadError loadScript(const String& fileName, String& resultStr) = 0;
 107 
 108         template<typename InterpType, typename ReturnType, typename... Params>
 109         static void createCommand(InterpType& si, const String& cmdName, ReturnType(*callback)(Params...), const char* usage = nullptr, size_t min_params = size_t(-1)) {
 110           si.createCommand(cmdName, callback, usage, min_params);
 111         }
 112 
 113         /**
 114          * @brief Remove a command from the interp
 115          * @param name The name of the command to remove
 116          */
 117         virtual void deleteCommand(const String& name) = 0;
 118 
 119         /**
 120          * @brief Link a C variable to the interp
 121          * @param si The ScriptInterp to link the variable to
 122          * @param varName The name to create the variable as
 123          * @param var The variable to link to
 124          * @note This is dispatching to the derived class due to being unable to do specialized template methods
 125          */
 126         template <typename Interp, typename T>
 127           static void linkVar(Interp& si, const String& varName, T& var) {
 128             si.linkVar(varName, var);
 129           };
 130 
 131         /**
 132          * @brief Remove a variable from the interp
 133          * @param varName The name of the variable to remove
 134          */
 135         virtual void unlinkVar(const String& varName) = 0;
 136 
 137         // Used for downcasting hack due to no virtual templates
 138         virtual script_type type() const = 0;
 139 };
