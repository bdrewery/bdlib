  42 template <class T>
  43 /**
  44  * @class Array
  45  * @brief Dynamic Array
  46  */
  47 class Array : public ReferenceCountedArray<T> {
  48   public:
  49     typedef T value_type;
  50 
  51     typedef size_t             size_type;
  52     typedef std::ptrdiff_t     difference_type;
  53     typedef value_type*        pointer;
  54     typedef const value_type*  const_pointer;
  55     typedef value_type&        reference;
  56     typedef const value_type&  const_reference;
  57 
  58     /* Constructors */
  59     Array() : ReferenceCountedArray<value_type>() {};
  60     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  61     Array(Array<value_type>&& array) : ReferenceCountedArray<value_type>(std::move(array)) {};
  62     /**
  63      * @brief Create a Array from a given carray.
  64      * @param carray The null-terminated array to create the object from.
  65      * @post A ArrayBuf has been initialized.
  66      * @post The buffer has been filled with the array.
  67      * @test Array test("Some array");
  68      */
  69     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  70       size_t i = 0;
  71       while (1) {
  72         if (!carray[i]) break;
  73         push(carray[i]);
  74         ++i;
  75       }
  76     };
  77 
  78     /**
  79      * @brief Create an array from an initializer list
  80      * @param list An initializer_list
  81      */
  82     Array(std::initializer_list<value_type> list) : ReferenceCountedArray<value_type>() {
  83       *this = list;
  84     }
  85 
  86     /**
  87      * @brief Create a Array from a given carray.
  88      * @param carray The null-terminated array to create the object from.
  89      * @param len How big is the carray?
  90      * @post A ArrayBuf has been initialized.
  91      * @post The buffer has been filled with the array.
  92      * @test Array test("Some array");
  93      */
  94     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  95       this->Reserve(len);
  96       for (size_t i = 0; i < len; ++i)
  97         push(carray[i]);
  98     };
  99 
 100     /**
 101      * @brief Create an empty Array container with at least the specified elements in size.
 102      * @param newSize Reserve at least this many buckets for this Array.
 103      * @post This array's memory will also never be shrunk.
 104      * @post A buffer has been created.
 105      */
 106     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
 107     Array(const size_type newSize, const value_type value) : ReferenceCountedArray<value_type>(newSize, value) {};
 108 
 109     virtual ~Array() {};
 110 
 111     /**
 112      * @brief Create an array from an initializer list
 113      * @param list An initializer_list
 114      */
 115     Array& operator=(std::initializer_list<value_type> list) {
 116       this->clear();
 117       this->Reserve(list.size());
 118       for (value_type item : list) {
 119         push(std::move(item));
 120       }
 121       return *this;
 122     }
 123 
 124     Array& operator=(const Array<value_type>& array) {
 125       ReferenceCountedArray<value_type>::operator=(array);
 126       return *this;
 127     }
 128 
 129     Array& operator=(Array<value_type>&& array) {
 130       ReferenceCountedArray<value_type>::operator=(std::move(array));
 131       return *this;
 132     }
 133 
 134     /**
 135      * @brief Add an item to the end of the array
 136      */
 137     inline void push(const value_type item) {
 138       this->AboutToModify(this->length() + 1);
 139       *(this->Buf(this->length())) = item;
 140       this->addLength(1);
 141     }
 142 
 143     /**
 144      * @sa push
 145      */
 146     inline friend Array<value_type>& operator<<(Array<value_type>& array, const_reference item) {
 147       array.push(item);
 148       return array;
 149     }
 150 
 151     /**
 152      * @brief Shift the array left, removing the first element.
 153      * @return The first element.
 154      */
 155     inline value_type shift() {
 156       if (this->isEmpty()) return value_type();
 157 
 158       value_type temp(*(this->Buf(0)));
 159       ++(this->offset);
 160       this->subLength(1);
 161       return temp;
 162     }
 163 
 164     /**
 165      * @brief Pop a value off the end of the array
 166      * @return The last element.
 167      */
 168     inline value_type pop() {
 169       if (this->isEmpty()) return value_type();
 170 
 171       value_type temp(*(this->Buf(this->length() - 1)));
 172       this->subLength(1);
 173       return temp;
 174     }
 175 
 176     /**
 177      * @sa pop
 178      */
 179     inline friend Array<value_type>& operator>>(Array<value_type>& array, reference item) {
 180       item = array.pop();
 181       return array;
 182     }
 183 
 184     /**
 185      * @brief Join an array by a delimiter into a string
 186      */
 187     String join(const String& delim, bool quoted = false) const {
 188       if (!this->length()) return String();
 189       size_t str_size = 0, i;
 190 
 191       // Preallocate a fully sized String to avoid frequent resizing
 192       str_size += (this->length() * delim.length()) - 1;
 193       if (quoted)
 194         str_size += this->length() * 2;
 195       for (i = 0; i < this->length(); ++i)
 196         str_size += this->Buf(i)->length();
 197 
 198       String str(str_size);
 199 
 200       for (i = 0; i < this->length(); ++i) {
 201         if (i)
 202           str += delim;
 203         if (quoted) {
 204           str += '"' + *(this->Buf(i)) + '"';
 205         } else {
 206           str += *(this->Buf(i));
 207         }
 208       }
 209       return str;
 210     }

 212     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 213     /**
 214      * @brief Compare our Array object with another Array object, but only n elements
 215      * @param array The Array object to equals to.
 216      * @param n The number of items to equals.
 217      * @return True if the number of elements are the same, and they all are equal.
 218      */
 219     bool equals(const Array& array, size_t n) const
 220     {
 221       size_t my_len = this->length();
 222       bool same_length = (my_len == array.length());
 223 
 224       /* Same array? */
 225       if (this->data() == array.data() && same_length)
 226         return true;
 227 
 228       if (!same_length)
 229         return false;
 230 
 231       size_t slen = std::min(array.length(), n);
 232       size_t len = std::min(my_len, slen);
 233 
 234       for (size_t i = 0; i < len; ++i) {
 235         if ((*this)[i] != array[i])
 236           return false;
 237       }
 238       return true;
 239     }
 240 
 241     inline friend bool operator==(const Array& lhs, const Array& rhs) { return lhs.equals(rhs); };
 242     inline friend bool operator!=(const Array& lhs, const Array& rhs) { return !(lhs == rhs); };

 245     /**
 246      * @sa ReferenceCountedArray::slice()
 247      */
 248     inline Array subarray(int start, int len = -1) const {
 249       Array newArray(*this);
 250       newArray.slice(start, len);
 251       return newArray;
 252     };

 254     /**
 255      * @sa subarray
 256      */
 257     inline Array operator()(int start, int len = -1) const { return subarray(start, len); };

 259     /**
 260      * @brief Returns a 'Slice' class for safe (cow) writing into the array

 262      * @param start Starting position
 263      * @param len How many items to use
 264      */
 265     inline Slice<Array> operator()(int start, int len = -1) { return Slice<Array>(*this, start, len); };

 267 #ifdef CPPUNIT_VERSION
 268     void CPPUNIT_checkArrayEqual(Array actual, CPPUNIT_NS::SourceLine sourceLine) {
 269       if ((*this) == actual) return;
 270       ::CPPUNIT_NS::Asserter::failNotEqual(this->join("|").c_str(), actual.join("|").c_str(), sourceLine);
 271     }

 274     /* Operators */
 275     /**
 276      * @relates Array
 277      * @brief Concatenates two array objects together.
 278      * @param array1 The LHS array.
 279      * @param array2 The RHS array.
 280      * @post A new array is allocated, reference copied and returned.
 281      * @return Returns a new array that can be reference copied by the lvalue.
 282      */
 283     inline friend Array operator+(Array array1, const Array& array2) {
 284       array1 += array2;
 285       return array1;
 286     }
 287 
 288     /**
 289      * @brief Prefix increment
 290      */
 291     inline const Array& operator++() {
 292       return (*this) += 1;
 293     }
 294 
 295     /**
 296      * @brief Postfix increment
 297      */
 298     inline const Array operator++(int) {
 299       Array tmp((*this)(0, 1));
 300       ++(this->offset);
 301       this->subLength(1);
 302       return tmp;
 303     }
 304 
 305     /**
 306      * @brief Prefix decrement
 307      */
 308     inline const Array& operator--() {
 309       return (*this) -= 1;
 310     }
 311 
 312     /**
 313      * @brief Postfix decrement
 314      */
 315     inline const Array operator--(int) {
 316       Array tmp((*this)(this->length() - 1, 1));
 317       this->subLength(1);
 318       return tmp;
 319     }
 320 
 321     /**
 322      * \sa append(const char)
 323      */
 324     inline Array& operator+=(const_reference item) {
 325       this->append(item);
 326       return *this;
 327     }
 328 
 329     /**
 330      * \sa append(const Array&)
 331      */
 332     inline Array& operator+=(const Array& array) {
 333       this->append(array);
 334       return *this;
 335     }
 336 
 337     inline Array& operator+=(const int n) {
 338       if (!this->length())
 339         return *this;
 340       if (int(this->length()) - n < 0) {
 341         this->offset = this->length();
 342         this->setLength(0);
 343       } else {
 344         this->offset += n;
 345         this->subLength(n);
 346       }
 347       return *this;
 348     }
 349 
 350     inline Array& operator-=(const int n) {
 351       if (!this->length())
 352         return *this;
 353       if (int(this->length()) - n < 0) {
 354         this->offset = this->length();
 355         this->setLength(0);
 356       } else
 357         this->subLength(n);
 358       return *this;
 359     }

 361 };
