  66 #define tr1_hashtable_define_trivial_hash(T)                                      \
  67 template<>                                                                        \
  68   struct Hash<T>                                                                  \
  69   {                                                                               \
  70     inline size_t operator()(T val) const { return static_cast<size_t>(val); }    \
  71   }
