  37 bool AtomicFile::open(const String& fname, mode_t mode) {
  38   size_t slash_pos;
  39   String dir;
  40 
  41   _fname = fname;
  42   _mode = mode;
  43 
  44   /*
  45    * Get a tempfile in the same directory as the target so that rename(2)
  46    * is not cross-device.
  47    */
  48   if ((slash_pos = _fname.rfind("/")) == String::npos) {
  49     slash_pos = 0;
  50     dir = "./";
  51   } else {
  52     dir = _fname(0, slash_pos + 1);     /* include the '/' */
  53   }
  54 
  55   _tmpname = String::printf("%s.%s.XXXXXX", dir.c_str(),
  56       static_cast<String>(_fname(slash_pos)).c_str());
  57 
  58   /* XXX: Move nul-writing to String per C++11 */
  59   _tmpname.resize(_tmpname.length() + 1);
  60   _fd = mkstemp(_tmpname.begin());
  61 
  62   return is_open();
  63 }
  64 
  65 bool AtomicFile::abort() {
  66   if (!is_open()) {
  67     return false;
  68   }
  69   ::unlink(_tmpname.c_str());
  70   ::close(_fd);
  71   _fd = -1;
  72   return true;
  73 }
  74 
  75 bool AtomicFile::commit() {
  76   bool result, remove_temp;
  77 
  78   result = false;
  79   remove_temp = true;
  80   if (_mode != mode_t(-1) && fchmod(_fd, _mode) != 0) {
  81     goto cleanup;
  82   }
  83   if (::fsync(_fd) != 0) {
  84     goto cleanup;
  85   }
  86   if (::close(_fd) != 0) {
  87     goto cleanup;
  88   }
  89   if (std::rename(_tmpname.c_str(), _fname.c_str()) != 0) {
  90     goto cleanup;
  91   }
  92 
  93   remove_temp = false;
  94   result = true;
  95 
  96 cleanup:
  97   if (remove_temp) {
  98     ::unlink(_tmpname.c_str());
  99   }
 100   _fd = -1;
 101   return result;
 102 }
