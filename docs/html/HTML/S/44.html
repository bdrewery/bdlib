  88 #define define_tcl_traceGet(T)                                                                                                     \
  89 template<>                                                                                                                         \
  90 const char* tcl_traceGet<T> (ClientData clientData, Tcl_Interp* interp, char* name1, char* name2, int flags) {                     \
  91   return ScriptInterpTCL::TraceGet(c_to_tcl_cast<T>::from(*static_cast<T*>(clientData), interp), interp, name1, name2, flags);     \
  92 }
  93 
  94 #define define_tcl_traceSet(T)                                                                                                     \
  95 template<>                                                                                                                         \
  96 const char* tcl_traceSet<T> (ClientData clientData, Tcl_Interp* interp, char* name1, char* name2, int flags) {                     \
  97   Tcl_Obj *obj = ScriptInterpTCL::TraceSet(interp, name1, name2, flags);                                                           \
  98   if (!obj) return name1;                                                                                                          \
  99   const T oldval(*static_cast<T*>(clientData));                                                                                    \
 100                                                                                                                                    \
 101   *static_cast<T*>(clientData) = std::move(tcl_to_c_cast<T>::from(obj, nullptr));                                                  \
 102   if (ScriptInterpTCL::link_var_hooks[name1])                                                                                      \
 103     (ScriptInterpTCL::link_var_hooks[name1])((const void*)&oldval, (const void*)(clientData));                                     \
 104   return nullptr;                                                                                                                  \
 105 }

 130 class ScriptInterpTCL : public ScriptInterp {
 131   friend Array<String> tcl_to_c_cast<Array<String>>::from(Tcl_Obj* obj, ScriptInterp* si);
 132   friend Array<Array<String>> tcl_to_c_cast<Array<Array<String>>>::from(Tcl_Obj* obj, ScriptInterp* si);
 133   private:
 134         Tcl_Interp *interp;
 135         static HashTable<String, script_cmd_handler_clientdata*> CmdHandlerData;
 136 
 137         ScriptInterpTCL(const ScriptInterpTCL&) = delete;
 138         ScriptInterpTCL& operator=(const ScriptInterpTCL&) = delete;
 139 
 140         void setupTraces(const String& name, ClientData var, Tcl_VarTraceProc* get, Tcl_VarTraceProc* set, link_var_hook hook_func);
 141         static int _createCommand_callback(ClientData clientData, Tcl_Interp *interp, int objc, Tcl_Obj *CONST objv[]);
 142 
 143         void _createCommand(const String& cmdName, ScriptCallbackBase* callback_proxy, const char* usage, size_t callbackParamMin, size_t callbackParamMax) {
 144           script_cmd_handler_clientdata* ccd = new script_cmd_handler_clientdata(this, callback_proxy, usage, callbackParamMin, callbackParamMax);
 145           CmdHandlerData[cmdName] = ccd;
 146           Tcl_CreateObjCommand(interp, *cmdName, _createCommand_callback, nullptr, nullptr);
 147         }
 148 
 149   protected:
 150         virtual int init();
 151         virtual int destroy();
 152 
 153   public:
 154         static HashTable<String, link_var_hook> link_var_hooks;
 155         ScriptInterpTCL() : ScriptInterp(), interp(nullptr) {init();};
 156         virtual ~ScriptInterpTCL() {
 157 
 158           // Delete all of my ccd
 159           for (auto ccd : CmdHandlerData.values()) {
 160               delete ccd->callback_proxy;
 161               delete ccd;
 162           }
 163           CmdHandlerData.clear();
 164           destroy();
 165         };
 166 
 167         virtual String eval(const String& script);
 168         virtual LoadError loadScript(const String& fileName, String& resultStr);
 169 
 170         template<typename ReturnType, typename... Params>
 171         inline void createCommand(const String& cmdName, ReturnType(*callback)(Params...), const char* usage = nullptr, size_t min_params = size_t(-1)) {
 172           _createCommand(cmdName, new ScriptCallbackTCL<ReturnType, Params...>(callback), usage, min_params == size_t(-1) ? sizeof...(Params) : min_params, sizeof...(Params));
 173         }
 174 
 175         virtual void deleteCommand(const String& cmdName) {
 176           script_cmd_handler_clientdata* ccd = CmdHandlerData[cmdName];
 177           delete ccd->callback_proxy;
 178           delete ccd;
 179           CmdHandlerData.remove(cmdName);
 180           Tcl_DeleteCommand(interp, *cmdName);
 181         }
 182 
 183         /* Variable linking */
 184 
 185         /**
 186          * @brief Link a C variable to the interp
 187          * @param varName The name to create the variable as
 188          * @param var The variable to link to
 189          */
 190         template <typename T>
 191           inline void linkVar(const String& varName, T& var, link_var_hook hook_func = nullptr) {
 192             setupTraces(varName, (ClientData) &var, (Tcl_VarTraceProc*) tcl_traceGet<T>, (Tcl_VarTraceProc*) tcl_traceSet<T>, hook_func);
 193           };
 194 
 195         /**
 196          * @brief Link a const C variable to the interp
 197          * @param varName The name to create the variable as
 198          * @param var The variable to link to
 199          * @note The variable will be created as read-only
 200          */
 201         template <typename T>
 202           inline void linkVar(const String& varName, const T& var, link_var_hook hook_func = nullptr) {
 203             setupTraces(varName, (ClientData) &var, (Tcl_VarTraceProc*) tcl_traceGet<T>, (Tcl_VarTraceProc*) TraceSetRO, hook_func);
 204           };
 205 
 206         /**
 207          * @brief Remove a variable from the interp
 208          * @param varName The name of the variable to remove
 209          */
 210         virtual void unlinkVar(const String& varName) {
 211           Tcl_UnsetVar(interp, *varName, TCL_GLOBAL_ONLY);
 212           link_var_hooks.remove(varName);
 213         }
 214 
 215         virtual script_type type() const { return SCRIPT_TYPE_TCL; }
 216   private:
 217         static const char* TraceSetRO (ClientData clientData, Tcl_Interp *interp, char *name1, char *name2, int flags);
 218 
 219   public:
 220         static Tcl_Obj* TraceSet (Tcl_Interp *interp, char *name1, char *name2, int flags);
 221         static const char* TraceGet (Tcl_Obj* value, Tcl_Interp *interp, char *name1, char *name2, int flags);
 222 };
