  40 /**
  41  * @class Array

  44 class Array : public ReferenceCountedArray<T> {

  51     typedef T value_type;

  53     typedef size_t             size_type;
  54     typedef std::ptrdiff_t     difference_type;
  55     typedef value_type*        pointer;
  56     typedef const value_type*  const_pointer;
  57     typedef value_type&        reference;
  58     typedef const value_type&  const_reference;

  60     /* Constructors */
  61     Array() : ReferenceCountedArray<value_type>() {};
  62     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  63     /**
  64      * @brief Create a Array from a given carray.
  65      * @param carray The null-terminated array to create the object from.
  66      * @post A ArrayBuf has been initialized.
  67      * @post The buffer has been filled with the array.
  68      * @test Array test("Some array");
  69      */
  70     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  71       size_t i = 0;
  72       while (1) {
  73         if (!carray[i]) break;
  74         push(carray[i]);
  75         ++i;
  76       }
  77     };
  78 
  79     /**
  80      * @brief Create a Array from a given carray.
  81      * @param carray The null-terminated array to create the object from.
  82      * @param len How big is the carray?
  83      * @post A ArrayBuf has been initialized.
  84      * @post The buffer has been filled with the array.
  85      * @test Array test("Some array");
  86      */
  87     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  88       this->Reserve(len);
  89       for (size_t i = 0; i < len; ++i)
  90         push(carray[i]);
  91     };
  92 
  93     /**
  94      * @brief Create an empty Array container with at least the specified elements in size.
  95      * @param newSize Reserve at least this many buckets for this Array.
  96      * @post This array's memory will also never be shrunk.
  97      * @post A buffer has been created.
  98      */
  99     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
 100     Array(const size_type newSize, const value_type value) : ReferenceCountedArray<value_type>(newSize, value) {};
 101 
 102     virtual ~Array() {};
 103 
 104     /**
 105      * @brief Add an item to the end of the array
 106      */
 107     inline void push(const value_type item) {
 108       AboutToModify(this->length() + 1);
 109       *(Buf(this->length())) = item;
 110       this->addLength(1);
 111     }
 112 
 113     /**
 114      * @sa push
 115      */
 116     inline friend Array<value_type>& operator<< (Array<value_type>& array, const_reference item) {
 117       array.push(item);
 118       return array;
 119     }
 120 
 121     /**
 122      * @brief Shift the array left, removing the first element.
 123      * @return The first element.
 124      */
 125     inline value_type shift() {
 126       if (this->isEmpty()) return value_type();
 127 
 128       value_type temp(*(this->Buf(0)));
 129       ++(this->offset);
 130       this->subLength(1);
 131       return temp;
 132     }
 133 
 134     /**
 135      * @brief Pop a value off the end of the array
 136      * @return The last element.
 137      */
 138     inline value_type pop() {
 139       if (this->isEmpty()) return value_type();
 140 
 141       value_type temp(*(this->Buf(this->length() - 1)));
 142       this->subLength(1);
 143       return temp;
 144     }
 145 
 146     /**
 147      * @sa pop
 148      */
 149     inline friend Array<value_type>& operator>> (Array<value_type>& array, reference item) {
 150       item = array.pop();
 151       return array;
 152     }
 153 
 154     /**
 155      * @brief Join an array by a delimiter into a string
 156      */
 157     String join(const String& delim, bool quoted = false) const {
 158       if (!this->length()) return String();
 159       String str;
 160       for (size_t i = 0; i < this->length(); ++i) {
 161         if (i)
 162           str += delim;
 163         if (quoted) {
 164           str += '"' + *(this->Buf(i)) + '"';
 165         } else {
 166           str += *(this->Buf(i));
 167         }
 168       }
 169       return str;
 170     }
 171 
 172     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 173     /**
 174      * @brief Compare our Array object with another Array object, but only n elements
 175      * @param array The Array object to equals to.
 176      * @param n The number of items to equals.
 177      * @return True if the number of elements are the same, and they all are equal.
 178      */
 179     bool equals(const Array &array, size_t n) const
 180     {
 181       size_t my_len = this->length();
 182       bool same_length = (my_len == array.length());
 183 
 184       /* Same array? */
 185       if (this->data() == array.data() && same_length)
 186         return true;
 187 
 188       if (!same_length)
 189         return false;
 190 
 191       size_t slen = std::min(array.length(), n);
 192       size_t len = std::min(my_len, slen);
 193 
 194       for (size_t i = 0; i < len; ++i) {
 195         if ((*this)[i] != array[i])
 196           return false;
 197       }
 198       return true;
 199     }

 288     inline Array& operator+= (const int n) {
 289       if (!this->length())
 290         return *this;
 291       if (int(this->length()) - n < 0) {
 292         this->offset = this->length();
 293         this->setLength(0);
 294       } else {
 295         this->offset += n;
 296         this->subLength(n);
 297       }
 298       return *this;
 299     }
 300 
 301     inline Array& operator-= (const int n) {
 302       if (!this->length())
 303         return *this;
 304       if (int(this->length()) - n < 0) {
 305         this->offset = this->length();
 306         this->setLength(0);
 307       } else
 308         this->subLength(n);
 309       return *this;
 310     }
