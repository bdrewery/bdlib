  37 template <class T, class Allocator = std::allocator<T>>
  38 /**
  39  * @class ArrayRef
  40  * @brief Helps the String and Array classes with reference counting
  41  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  42  */
  43 class ArrayRef {
  44   public:
  45     typedef typename Allocator::pointer iterator;
  46     mutable Allocator alloc;
  47     mutable size_t size; //Capacity of buffer
  48     mutable iterator buf;
  49     mutable int n; //References
  50 
  51     ArrayRef(const Allocator& allocator = Allocator()) : alloc(allocator), size(0), buf(NULL), n(1) {};
  52     ~ArrayRef() { FreeBuf(buf); };
  53     /**
  54      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  55      * @param newSize A size that we need to Allocate the buffer to.
  56      * @param offset The offset of the old buffer so we know where to start
  57      * @param sublen The length of the subarray in use
  58      * @pre newSize is > 0 (assumed as size_t is unsigned)
  59      * @post The buffer is at least nsize bytes long.
  60      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  61      */
  62     void Reserve(size_t newSize, size_t& offset, size_t sublen) const
  63     {
  64       /* Don't new if we already have enough room! */
  65       if (size < newSize) {
  66         newSize = std::max(size_t(size * 1.5), newSize);
  67 
  68         iterator newbuf = alloc.allocate(newSize, buf);
  69 
  70         if (newbuf != buf) {
  71           // Initialize new memory
  72           std::uninitialized_fill(newbuf, newbuf + newSize, T());
  73           /* Copy old buffer into new - only copy the subarray */
  74           std::copy(buf + offset, buf + offset + sublen, newbuf);
  75           FreeBuf(buf);
  76           buf = newbuf;
  77           size = newSize;
  78           offset = 0;
  79         }
  80       } else if ((size - offset) < newSize) {
  81         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
  82         // Shift everything to the beginning and reset the offset
  83         /* Only copy the subarray */
  84         memmove(buf, buf + offset, sublen);
  85         offset = 0;
  86       }
  87     }
  88 
  89     /**
  90      * @brief Free's up the allocated buffer.
  91      * @param p The buffer to be free'd
  92      * @post The buffer is deleted.
  93      * @todo Implement mempool here.
  94      */
  95     inline void FreeBuf(iterator p) const {
  96       for (iterator i = p; i != p + size; ++i) {
  97         alloc.destroy(i);
  98       }
  99       alloc.deallocate(p, size);
 100     }
 101 
 102     /**
 103      * @brief Is this ReferenceCountedArray shared?
 104      */
 105     inline bool isShared() const { return n > 1; };
 106   private:
 107     // No copying allowed
 108     ArrayRef(const ArrayRef&) = delete;
 109     ArrayRef& operator=(const ArrayRef&) = delete;
 110 };

 296      * @post The reference counter is decremented.

 299      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.

 302       if (Ref && decRef() < 1)

 306     //    void COW(size_t) const;
 307     /**
 308      * @brief Ensure that our internal buffer is unshared.
 309      * @param n Create/Grow the buffer to this size.
 310      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 311      * @post The internal buffer/data is unshared
 312      * @post The buffer is at least size n.
 313      * @post The buffer is deep copied to a new buffer.
 314      *
 315      * Ensure that our internal buffer is unshared.
 316      * If needed, performs a deep copy into a new buffer (COW).
 317      * Also take a hint size n of the new ReferenceCountedArray's size as to avoid needless copying/allocing.
 318      */
 319     void COW(size_t n) const {
 320       const_pointer oldBuf = constBuf();
 321       const size_t oldLength = length();
 322 
 323       doDetach(); //Detach from the shared reference
 324       Reserve( std::max(oldLength, n) ); //Will set capacity()/size
 325       std::copy(oldBuf, oldBuf + oldLength, Buf());
 326       setLength(oldLength);
 327     }

 336     /**
 337      * @brief Warn the reference counting that it may need to COW
 338      * @post The buffer is detached/COW, and possibly larger
 339      */
 340     inline void AboutToModify(size_t n) const {
 341       my_hash = 0;
 342       if (isShared())
 343         COW(n); // Clears the offset
 344       else {
 345         Reserve(n);
 346         /* Shift the offset away */
 347       }
 348     }

 575    /**
 576      * @brief Return a hash of every element in the array. Cache result as well.
 577      * @note DJB's hash function
 578      */
 579     virtual size_t hash() const {
 580       if (my_hash != 0) return my_hash;
 581       HashType hasher;
 582       size_t _hash = 5381;
 583 
 584       for(size_t i = 0; i < this->length(); ++i)
 585         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 586       return (my_hash = (_hash & 0x7FFFFFFF));
 587     }

 681 
 682         /**
 683          * Stroustrup shows using this as void with no return value, but that breaks chaining a[n] = b[n] = 'b';
 684          */
 685         inline Cref& operator=(value_type c) {
 686           rca.write(k, c);
 687           return (*this);
 688         };
 689     };

 691     /**
 692      * @brief Returns 'Cref' class for safe (cow) writing.
 693      * @sa Cref
 694      */
 695     inline Cref operator[](size_t pos) { return Cref(*this, pos); };
 696 
 697     /**
 698      * @brief Returns the character at the given index.
 699      * @return The character at the given index.
 700      * @param pos Index to return.
 701      * @pre The index must exist.
 702      * @sa operator[]()
 703      * @todo Perhaps this should throw an exception if out of range?
 704      */
 705     inline value_type at(size_t pos) const { return hasIndex(pos) ? (*this)[pos] : 0; };
 706 
 707     /**
 708      * @brief Return a new array from a subarray
 709      * @return a new ReferenceCountedArray
 710      * @param start The offset to begin the subarray from (indexed from 0)
 711      * @param len The length of the subarray to return
 712      * The returned slice is a reference to the original array until modified.
 713      */
 714     void slice(int start, int len = -1) {
 715       if (len == -1) len = int(length()) - start;
 716       // Start is after the end, set us to an empty array
 717       if (start >= static_cast<signed>(length())) {
 718         offset = length();
 719         setLength(0);
 720         return;
 721       }
 722 
 723       //Count backwards from the end
 724       if (start < 0)
 725         start = length() + start;
 726       // Start was before the beginning, just reset to the beginning
 727       if (start < 0)
 728         start = 0;
 729 
 730       offset += start;
 731       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 732       if (start + len >= static_cast<signed>(length()))
 733         len = length() - start;
 734       // If the length is negative, stop from counting backwards from the end
 735       else if (len < 0)
 736         len = length() - start + len;
 737       setLength(len);
 738     }

 740     /**
 741      * @brief Appends one item to end of buffer.
 742      * @param item The item to be appended.
 743      * @post The buffer is allocated.
 744      * @post The item is appended at the end of the buffer.
 745      * This is the same as inserting the item at the end of the buffer.
 746      */
 747     inline void append(const_reference item) { insert(length(), item); };
 748     /**
 749      * @brief Appends given rca to the end of buffer
 750      * @param rca The rca to be appended.
 751      * @param n How many characters to copy from the ReferenceCountedArray object.
 752      * @post The buffer is allocated.
 753      * This is the same as inserting the rca at the end of the buffer.
 754      */
 755     inline void append(const ReferenceCountedArray& rca, size_t n = npos) { insert(length(), rca, n); };

 758     /**
 759      * @brief Inserts a ReferenceCountedArray object into our buffer
 760      * @param pos The index to insert at.
 761      * @param rca The rca to insert.
 762      * @param n The length to insert.
 763      * @post The buffer contains n items from rca inserted at index pos.
 764      */
 765     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 766       if (n == 0) return;
 767       if (pos && !hasIndex(pos-1)) return;
 768 
 769       size_t slen = rca.length();
 770 
 771       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 772       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 773         *this = rca;
 774         return;
 775       }
 776 
 777       if (n == npos || n > slen)
 778         n = slen;
 779       slen -= slen - n;
 780       AboutToModify(length() + slen);
 781       memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
 782       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 783       addLength(slen);
 784     }

 786     /**
 787      * @brief Insert an item at the given index.
 788      * @param pos The index to insert at.
 789      * @param item The item to be inserted.
 790      * @post A buffer is allocated.
 791      * @post If the old buffer was too small, it is enlarged.
 792      * @post The item is inserted at the given index.
 793      */
 794     void insert(size_t pos, const_reference item)
 795     {
 796       if (pos && !hasIndex(pos-1)) return;
 797 
 798       AboutToModify(length() + 1);
 799       memmove(Buf() + pos + 1, Buf() + pos, length() - pos);
 800       *(Buf(pos)) = item;
 801       addLength(1);
 802     }

 804     /**
 805      * @brief Replace the given index with the given item.
 806      * @param pos The index to replace.
 807      * @param item The item to replace with.
 808      * @post The given index has been replaced.
 809      * @post COW is done if needed.
 810      */
 811     void replace(size_t pos, const_reference item) {
 812       if (pos && !hasIndex(pos-1)) return;
 813 
 814       getOwnCopy();
 815       *(Buf(pos)) = item;
 816     }
 817 
 818     /**
 819      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 820      * @param pos The index to replace at.
 821      * @param rca The ReferenceCountedArray object to replace with.
 822      * @param n The number of characters to use for the replace.
 823      */
 824     void replace(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 825       if (n == 0) return;
 826       if (pos && !hasIndex(pos-1)) return;
 827 
 828       size_t slen = rca.length();
 829 
 830       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 831       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 832         *this = rca;
 833         return;
 834       }
 835 
 836       if (n == npos || n > slen)
 837         n = slen;
 838       slen -= slen - n;
 839 
 840       size_t newlen = pos + slen;
 841 
 842       if (newlen >= length()) {
 843         AboutToModify(newlen);
 844       } else {
 845         newlen = length();
 846         getOwnCopy();
 847       }
 848       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 849       setLength(newlen);
 850     }
