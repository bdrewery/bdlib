  43 template <class T>
  44 /**
  45  * @class Array
  46  * @brief Dynamic Array
  47  */
  48 class Array : public ReferenceCountedArray<T> {
  49   public:
  50     typedef T value_type;
  51 
  52     typedef size_t             size_type;
  53     typedef std::ptrdiff_t     difference_type;
  54     typedef value_type*        pointer;
  55     typedef const value_type*  const_pointer;
  56     typedef value_type&        reference;
  57     typedef const value_type&  const_reference;
  58 
  59     /* Constructors */
  60     Array() : ReferenceCountedArray<value_type>() {};
  61     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  62 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  63     Array(Array<value_type>&& array) : ReferenceCountedArray<value_type>(std::move(array)) {};
  64 #endif
  65     /**
  66      * @brief Create a Array from a given carray.
  67      * @param carray The null-terminated array to create the object from.
  68      * @post A ArrayBuf has been initialized.
  69      * @post The buffer has been filled with the array.
  70      * @test Array test("Some array");
  71      */
  72     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  73       size_t i = 0;
  74       while (1) {
  75         if (!carray[i]) break;
  76         push(carray[i]);
  77         ++i;
  78       }
  79     };
  80 
  81 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  82     /**
  83      * @brief Create an array from an initializer list
  84      * @param list An initializer_list
  85      */
  86     Array(std::initializer_list<value_type> list) : ReferenceCountedArray<value_type>() {
  87       *this = list;
  88     }
  89 #endif
  90 
  91     /**
  92      * @brief Create a Array from a given carray.
  93      * @param carray The null-terminated array to create the object from.
  94      * @param len How big is the carray?
  95      * @post A ArrayBuf has been initialized.
  96      * @post The buffer has been filled with the array.
  97      * @test Array test("Some array");
  98      */
  99     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
 100       this->Reserve(len);
 101       for (size_t i = 0; i < len; ++i)
 102         push(carray[i]);
 103     };
 104 
 105     /**
 106      * @brief Create an empty Array container with at least the specified elements in size.
 107      * @param newSize Reserve at least this many buckets for this Array.
 108      * @post This array's memory will also never be shrunk.
 109      * @post A buffer has been created.
 110      */
 111     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
 112     Array(const size_type newSize, const value_type value) : ReferenceCountedArray<value_type>(newSize, value) {};
 113 
 114     virtual ~Array() {};
 115 
 116 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 117     /**
 118      * @brief Create an array from an initializer list
 119      * @param list An initializer_list
 120      */
 121     Array& operator=(std::initializer_list<value_type> list) {
 122       this->clear();
 123       this->Reserve(list.size());
 124       for (value_type item : list) {
 125         push(std::move(item));
 126       }
 127       return *this;
 128     }
 129 #endif
 130 
 131     Array& operator=(const Array<value_type>& array) {
 132       ReferenceCountedArray<value_type>::operator=(array);
 133       return *this;
 134     }
 135 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 136     Array& operator=(Array<value_type>&& array) {
 137       ReferenceCountedArray<value_type>::operator=(std::move(array));
 138       return *this;
 139     }
 140 #endif
 141 
 142     /**
 143      * @brief Add an item to the end of the array
 144      */
 145     inline void push(const value_type item) {
 146       this->AboutToModify(this->length() + 1);
 147       *(this->Buf(this->length())) = item;
 148       this->addLength(1);
 149     }
 150 
 151     /**
 152      * @sa push
 153      */
 154     inline friend Array<value_type>& operator<<(Array<value_type>& array, const_reference item) {
 155       array.push(item);
 156       return array;
 157     }
 158 
 159     /**
 160      * @brief Shift the array left, removing the first element.
 161      * @return The first element.
 162      */
 163     inline value_type shift() {
 164       if (this->isEmpty()) return value_type();
 165 
 166       value_type temp(*(this->Buf(0)));
 167       ++(this->offset);
 168       this->subLength(1);
 169       return temp;
 170     }
 171 
 172     /**
 173      * @brief Pop a value off the end of the array
 174      * @return The last element.
 175      */
 176     inline value_type pop() {
 177       if (this->isEmpty()) return value_type();
 178 
 179       value_type temp(*(this->Buf(this->length() - 1)));
 180       this->subLength(1);
 181       return temp;
 182     }
 183 
 184     /**
 185      * @sa pop
 186      */
 187     inline friend Array<value_type>& operator>>(Array<value_type>& array, reference item) {
 188       item = array.pop();
 189       return array;
 190     }
 191 
 192     /**
 193      * @brief Join an array by a delimiter into a string
 194      */
 195     String join(const String& delim, bool quoted = false) const {
 196       if (!this->length()) return String();
 197       size_t str_size = 0, i;
 198 
 199       // Preallocate a fully sized String to avoid frequent resizing
 200       str_size += (this->length() * delim.length()) - 1;
 201       if (quoted)
 202         str_size += this->length() * 2;
 203       for (i = 0; i < this->length(); ++i)
 204         str_size += this->Buf(i)->length();
 205 
 206       String str(str_size);
 207 
 208       for (i = 0; i < this->length(); ++i) {
 209         if (i)
 210           str += delim;
 211         if (quoted) {
 212           str += '"' + *(this->Buf(i)) + '"';
 213         } else {
 214           str += *(this->Buf(i));
 215         }
 216       }
 217       return str;
 218     }

 220     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 221     /**
 222      * @brief Compare our Array object with another Array object, but only n elements
 223      * @param array The Array object to equals to.
 224      * @param n The number of items to equals.
 225      * @return True if the number of elements are the same, and they all are equal.
 226      */
 227     bool equals(const Array& array, size_t n) const
 228     {
 229       size_t my_len = this->length();
 230       bool same_length = (my_len == array.length());
 231 
 232       /* Same array? */
 233       if (this->data() == array.data() && same_length)
 234         return true;
 235 
 236       if (!same_length)
 237         return false;
 238 
 239       size_t slen = std::min(array.length(), n);
 240       size_t len = std::min(my_len, slen);
 241 
 242       for (size_t i = 0; i < len; ++i) {
 243         if ((*this)[i] != array[i])
 244           return false;
 245       }
 246       return true;
 247     }
 248 
 249     inline friend bool operator==(const Array& lhs, const Array& rhs) { return lhs.equals(rhs); };
 250     inline friend bool operator!=(const Array& lhs, const Array& rhs) { return !(lhs == rhs); };

 253     /**
 254      * @sa ReferenceCountedArray::slice()
 255      */
 256     inline Array subarray(int start, int len = -1) const {
 257       Array newArray(*this);
 258       newArray.slice(start, len);
 259       return newArray;
 260     };

 262     /**
 263      * @sa subarray
 264      */
 265     inline Array operator()(int start, int len = -1) const { return subarray(start, len); };
 266 
 267     /**
 268      * @brief Returns a 'Slice' class for safe (cow) writing into the array
 269      * @sa Slice
 270      * @param start Starting position
 271      * @param len How many items to use
 272      */
 273     inline Slice<Array> operator()(int start, int len = -1) { return Slice<Array>(*this, start, len); };

 275 #ifdef CPPUNIT_VERSION
 276     void CPPUNIT_checkArrayEqual(Array actual, CPPUNIT_NS::SourceLine sourceLine) {
 277       if ((*this) == actual) return;
 278       ::CPPUNIT_NS::Asserter::failNotEqual(this->join("|").c_str(), actual.join("|").c_str(), sourceLine);
 279     }

 282     /* Operators */
 283     /**
 284      * @relates Array
 285      * @brief Concatenates two array objects together.
 286      * @param array1 The LHS array.
 287      * @param array2 The RHS array.
 288      * @post A new array is allocated, reference copied and returned.
 289      * @return Returns a new array that can be reference copied by the lvalue.
 290      */
 291     inline friend Array operator+(Array array1, const Array& array2) {
 292       array1 += array2;
 293       return array1;
 294     }
 295 
 296     /**
 297      * @brief Prefix increment
 298      */
 299     inline const Array& operator++() {
 300       return (*this) += 1;
 301     }
 302 
 303     /**
 304      * @brief Postfix increment
 305      */
 306     inline const Array operator++(int) {
 307       Array tmp((*this)(0, 1));
 308       ++(this->offset);
 309       this->subLength(1);
 310       return tmp;
 311     }
 312 
 313     /**
 314      * @brief Prefix decrement
 315      */
 316     inline const Array& operator--() {
 317       return (*this) -= 1;
 318     }
 319 
 320     /**
 321      * @brief Postfix decrement
 322      */
 323     inline const Array operator--(int) {
 324       Array tmp((*this)(this->length() - 1, 1));
 325       this->subLength(1);
 326       return tmp;
 327     }
 328 
 329     /**
 330      * \sa append(const char)
 331      */
 332     inline Array& operator+=(const_reference item) {
 333       this->append(item);
 334       return *this;
 335     }
 336 
 337     /**
 338      * \sa append(const Array&)
 339      */
 340     inline Array& operator+=(const Array& array) {
 341       this->append(array);
 342       return *this;
 343     }
 344 
 345     inline Array& operator+=(const int n) {
 346       if (!this->length())
 347         return *this;
 348       if (int(this->length()) - n < 0) {
 349         this->offset = this->length();
 350         this->setLength(0);
 351       } else {
 352         this->offset += n;
 353         this->subLength(n);
 354       }
 355       return *this;
 356     }
 357 
 358     inline Array& operator-=(const int n) {
 359       if (!this->length())
 360         return *this;
 361       if (int(this->length()) - n < 0) {
 362         this->offset = this->length();
 363         this->setLength(0);
 364       } else
 365         this->subLength(n);
 366       return *this;
 367     }

 369 };
