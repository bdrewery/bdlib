  39 template <class T>
  40 /**
  41  * @class Array
  42  * @brief Dynamic Array
  43  */
  44 class Array : public ReferenceCountedArray<T> {
  45   public:
  46     typedef T value_type;
  47 
  48     typedef size_t             size_type;
  49     typedef std::ptrdiff_t     difference_type;
  50     typedef value_type*        pointer;
  51     typedef const value_type*  const_pointer;
  52     typedef value_type&        reference;
  53     typedef const value_type&  const_reference;
  54 
  55     /* Constructors */
  56     Array() : ReferenceCountedArray<value_type>() {};
  57     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  58     /**
  59      * @brief Create a Array from a given carray.
  60      * @param carray The null-terminated array to create the object from.
  61      * @post A ArrayBuf has been initialized.
  62      * @post The buffer has been filled with the array.
  63      * @test Array test("Some array");
  64      */
  65     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  66       size_t i = 0;
  67       while (1) {
  68         if (!carray[i]) break;
  69         push(carray[i]);
  70         ++i;
  71       }
  72     };
  73 
  74     /**
  75      * @brief Create a Array from a given carray.
  76      * @param carray The null-terminated array to create the object from.
  77      * @param len How big is the carray?
  78      * @post A ArrayBuf has been initialized.
  79      * @post The buffer has been filled with the array.
  80      * @test Array test("Some array");
  81      */
  82     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  83       this->Reserve(len);
  84       for (size_t i = 0; i < len; ++i)
  85         push(carray[i]);
  86     };
  87 
  88     /**
  89      * @brief Create an empty Array container with at least the specified elements in size.
  90      * @param newSize Reserve at least this many buckets for this Array.
  91      * @post This array's memory will also never be shrunk.
  92      * @post A buffer has been created.
  93      */
  94     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
  95     Array(const size_type newSize, const value_type value) : ReferenceCountedArray<value_type>(newSize, value) {};
  96 
  97     virtual ~Array() {};
  98 
  99     /**
 100      * @brief Add an item to the end of the array
 101      */
 102     inline void push(const value_type item) {
 103       this->AboutToModify(this->length() + 1);
 104       *(this->Buf(this->length())) = item;
 105       this->addLength(1);
 106     }
 107 
 108     /**
 109      * @sa push
 110      */
 111     inline friend Array<value_type>& operator<<(Array<value_type>& array, const_reference item) {
 112       array.push(item);
 113       return array;
 114     }
 115 
 116     /**
 117      * @brief Shift the array left, removing the first element.
 118      * @return The first element.
 119      */
 120     inline value_type shift() {
 121       if (this->isEmpty()) return value_type();
 122 
 123       value_type temp(*(this->Buf(0)));
 124       ++(this->offset);
 125       this->subLength(1);
 126       return temp;
 127     }
 128 
 129     /**
 130      * @brief Pop a value off the end of the array
 131      * @return The last element.
 132      */
 133     inline value_type pop() {
 134       if (this->isEmpty()) return value_type();
 135 
 136       value_type temp(*(this->Buf(this->length() - 1)));
 137       this->subLength(1);
 138       return temp;
 139     }
 140 
 141     /**
 142      * @sa pop
 143      */
 144     inline friend Array<value_type>& operator>>(Array<value_type>& array, reference item) {
 145       item = array.pop();
 146       return array;
 147     }
 148 
 149     /**
 150      * @brief Join an array by a delimiter into a string
 151      */
 152     String join(const String& delim, bool quoted = false) const {
 153       if (!this->length()) return String();
 154       String str;
 155       for (size_t i = 0; i < this->length(); ++i) {
 156         if (i)
 157           str += delim;
 158         if (quoted) {
 159           str += '"' + *(this->Buf(i)) + '"';
 160         } else {
 161           str += *(this->Buf(i));
 162         }
 163       }
 164       return str;
 165     }
 166 
 167     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 168     /**
 169      * @brief Compare our Array object with another Array object, but only n elements
 170      * @param array The Array object to equals to.
 171      * @param n The number of items to equals.
 172      * @return True if the number of elements are the same, and they all are equal.
 173      */
 174     bool equals(const Array& array, size_t n) const
 175     {
 176       size_t my_len = this->length();
 177       bool same_length = (my_len == array.length());
 178 
 179       /* Same array? */
 180       if (this->data() == array.data() && same_length)
 181         return true;
 182 
 183       if (!same_length)
 184         return false;
 185 
 186       size_t slen = std::min(array.length(), n);
 187       size_t len = std::min(my_len, slen);
 188 
 189       for (size_t i = 0; i < len; ++i) {
 190         if ((*this)[i] != array[i])
 191           return false;
 192       }
 193       return true;
 194     }
 195 
 196     inline friend bool operator==(const Array& lhs, const Array& rhs) { return lhs.equals(rhs); };
 197     inline friend bool operator!=(const Array& lhs, const Array& rhs) { return !(lhs == rhs); };
 198 
 199     // Subarrays
 200     /**
 201      * @sa ReferenceCountedArray::slice()
 202      */
 203     inline Array subarray(int start, int len = -1) const {
 204       Array newArray(*this);
 205       newArray.slice(start, len);
 206       return newArray;
 207     };
 208 
 209     /**
 210      * @sa subarray
 211      */
 212     inline Array operator()(int start, int len = -1) const { return subarray(start, len); };
 213 
 214     /**
 215      * @brief Returns a 'Slice' class for safe (cow) writing into the array
 216      * @sa Slice
 217      * @param start Starting position
 218      * @param len How many items to use
 219      */
 220     inline Slice<Array> operator()(int start, int len = -1) { return Slice<Array>(*this, start, len); };

 222 #ifdef CPPUNIT_VERSION
 223     void CPPUNIT_checkArrayEqual(Array actual, CPPUNIT_NS::SourceLine sourceLine) {
 224       if ((*this) == actual) return;
 225       ::CPPUNIT_NS::Asserter::failNotEqual(this->join("|").c_str(), actual.join("|").c_str(), sourceLine);
 226     }

 229     /* Operators */
 230     /**
 231      * @relates Array
 232      * @brief Concatenates two array objects together.
 233      * @param array1 The LHS array.
 234      * @param array2 The RHS array.
 235      * @post A new array is allocated, reference copied and returned.
 236      * @return Returns a new array that can be reference copied by the lvalue.
 237      */
 238     inline friend Array operator+(const Array& array1, const Array& array2) {
 239       Array temp(array1);
 240       temp += array2;
 241       return temp;
 242     }
 243 
 244     /**
 245      * @brief Prefix increment
 246      */
 247     inline const Array& operator++() {
 248       return (*this) += 1;
 249     }
 250 
 251     /**
 252      * @brief Postfix increment
 253      */
 254     inline const Array operator++(int) {
 255       Array tmp((*this)(0, 1));
 256       ++(this->offset);
 257       this->subLength(1);
 258       return tmp;
 259     }
 260 
 261     /**
 262      * @brief Prefix decrement
 263      */
 264     inline const Array& operator--() {
 265       return (*this) -= 1;
 266     }
 267 
 268     /**
 269      * @brief Postfix decrement
 270      */
 271     inline const Array operator--(int) {
 272       Array tmp((*this)(this->length() - 1, 1));
 273       this->subLength(1);
 274       return tmp;
 275     }
 276 
 277     /**
 278      * \sa append(const char)
 279      */
 280     inline Array& operator+=(const_reference item) {
 281       this->append(item);
 282       return *this;
 283     }
 284 
 285     /**
 286      * \sa append(const Array&)
 287      */
 288     inline Array& operator+=(const Array& array) {
 289       this->append(array);
 290       return *this;
 291     }
 292 
 293     inline Array& operator+=(const int n) {
 294       if (!this->length())
 295         return *this;
 296       if (int(this->length()) - n < 0) {
 297         this->offset = this->length();
 298         this->setLength(0);
 299       } else {
 300         this->offset += n;
 301         this->subLength(n);
 302       }
 303       return *this;
 304     }
 305 
 306     inline Array& operator-=(const int n) {
 307       if (!this->length())
 308         return *this;
 309       if (int(this->length()) - n < 0) {
 310         this->offset = this->length();
 311         this->setLength(0);
 312       } else
 313         this->subLength(n);
 314       return *this;
 315     }
 316 
 317 };
