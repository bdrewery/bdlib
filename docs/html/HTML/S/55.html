  42 int String::compare(const String& str, size_t n, size_t start) const
  43 {
  44   const size_t my_len = length();
  45   /* Same string? */
  46   if (data() == str.data() && my_len == str.length())
  47     return 0;
  48 
  49   const size_t slen = n ? std::min(str.length(), n) : str.length();
  50   const size_t len = std::min(my_len - start, slen);
  51   const int diff = memcmp(begin() + start, str.begin(), len);
  52   if (diff)
  53     return diff;
  54   else if (n)
  55     return std::min(my_len - start, n) - slen;
  56   else
  57     return (my_len - start) - slen;
  58 }

  62 /**
  63  * @brief Insert a cstring at the given index.
  64  * @param string The cstring to be inserted.
  65  * @param pos The index to insert at.
  66  * @param n Up to this many characters will be used from the string.
  67  * @post A buffer is allocated.
  68  * @post If the old buffer was too small, it is enlarged.
  69  * @post The string is inserted at the given index.
  70  */
  71 void String::insert(size_t pos, const char *string, size_t n)
  72 {
  73   if (n == 0) return;
  74   if (pos && !hasIndex(pos-1)) return;
  75   
  76   size_t slen = (n == npos) ? strlen(string) : n;
  77 
  78   AboutToModify(length() + slen);
  79   memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
  80   std::copy(string, string + slen, Buf() + pos);
  81   addLength(slen);
  82 }

  84 /**
  85  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  86  * @param pos The index to replace at.
  87  * @param string The cstring to replace with.
  88  * @param n How many characters to use from string.
  89  */
  90 void String::replace(size_t pos, const char *string, size_t n)
  91 {
  92   if (n == 0) return;
  93   if (pos && !hasIndex(pos-1)) return;
  94 
  95   size_t slen = (n == npos) ? strlen(string) : n;
  96   size_t newlen = pos + slen;
  97   
  98   if (newlen >= length()) {
  99     AboutToModify(newlen);
 100   } else {
 101     newlen = length();
 102     getOwnCopy();
 103   }
 104   std::copy(string, string + slen, Buf() + pos);
 105   setLength(newlen);
 106 }

 108 std::istream& operator>>(std::istream& is, String& string) {
 109   char ch;
 110   string = "";    // empty string, will build one char at-a-time
 111   is >> ch;    // whitespace skipped, first non-white char in ch
 112 
 113   if (!is.fail())
 114   {
 115     do
 116     {
 117       string += ch;
 118       is.get(ch);
 119     } while (!is.fail() && !isspace(ch));
 120 
 121     if (isspace(ch))    // put whitespace back on the stream
 122     {
 123       is.putback(ch);
 124     }
 125   }
 126   return is;
 127 }

 140 /*
 141  * @brief Split a string by a delimiter
 142  * @param delim The delimiter to use
 143  * @param limit Optional param which specified the max number of elements to return
 144  * @note Excess whitespace is trimmed similar to Ruby's split.
 145  */
 146 Array<String> String::split(const String& delim, size_t limit) const {
 147   if (!length()) return Array<String>();
 148   if (limit == 1) {
 149     String list[] = {*this};
 150     return Array<String>(list, 1);
 151   }
 152 
 153   const String space(' ');
 154   Array<String> array;
 155   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 156   String str(*this);
 157   size_t pos = 0;
 158 
 159   while (array.size() < limit - 1) {
 160     // Trim out left whitespace
 161     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 162 
 163     // All done when there's nothing left
 164     if (!str) break;
 165 
 166     if ((pos = str.find(delim)) == npos)
 167       pos = str.length();
 168     array << str(0, pos);
 169     str += pos + delim.length();
 170   }
 171 
 172   // Add on extra
 173   if (limit != npos) {
 174     // Trim out left whitespace
 175     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 176     array << str;
 177   }
 178 
 179   return array;
 180 }

 200 String& String::chomp() {
 201   if (length() && (*this)[length() - 1] == '\n') --(*this);
 202   if (length() && (*this)[length() - 1] == '\r') --(*this);
 203   return *this;

 214 size_t String::find(const String& str) const {
 215   if (length() >= str.length()) {
 216     const size_t last_pos = length() - str.length();
 217     for (size_t pos = 0; pos <= last_pos; ++pos)
 218       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 219         return pos;
 220   }
 221   return npos;
 222 }

 224 size_t String::rfind(const String& str, const size_t lpos) const {
 225   if (length() >= str.length()) {
 226     const size_t last_pos = length() - str.length();
 227     for (size_t pos = last_pos; pos + 1 > lpos; --pos)
 228       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 229         return pos;
 230   }
 231   return npos;
 232 }

 234 String String::sub(const String& search, const String& replacement, int limit) const {
 235   String newStr(length()), search_str(*this);
 236   size_t pos = 0;
 237   int cnt = 0;
 238   // Search in our const string so that stuff like \->\\ doesnt become \\\, \\\\ etc.
 239   while ((pos = search_str.find(search)) != npos) {
 240     newStr += search_str(0, pos);
 241     newStr += replacement;
 242     search_str += pos + search.length();
 243     if (limit != -1 && ++cnt == limit) break;
 244   }
 245   // Left over
 246   if (search_str.length())
 247     newStr += search_str;
 248   return newStr;
 249 }

 252   String newStr(*this);
 253   Array<String> keys(hashes.keys());
 254   Array<String> values(hashes.values());
 255   for (size_t i = 0; i < keys.length(); ++i) {

 257     //::printf("%s -> %s\n", key.c_str(), value.c_str());
 258     newStr = newStr.sub(key, value);
 259   }
 260   return newStr;

 263 /**
 264  * @brief Splits off tokens
 265  * @param str The string to split a token off of
 266  * @param delim The delimiter to tokenize with
 267  * This will modify the given string to remove the first token, which is returned.
 268  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 269  */
 270 String newsplit(String& str, char delim)
 271 {
 272   if (!str.length()) return "";
 273   size_t pos = str.find(delim);
 274   if (pos == String::npos)
 275     pos = str.length();
 276 
 277   String first(str(0, pos));
 278 
 279   /* Trim out runs of whitespaces */
 280   if (delim == ' ') {
 281     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 282       ++pos;
 283   }
 284 
 285 //  str = str(pos + 1, str.length() - pos + 1);
 286   str += pos + 1;
 287   return first;
 288 }
