 115 void HashTableTest :: containsTest (void)
 116 {
 117   a->insert(1, "Blah");
 118   a->insert(1, "Bleck");
 119   a->insert(2, "Blah");
 120   a->insert(8, "Blah");
 121   a->insert(4, "Blah");
 122   
 123   CPPUNIT_ASSERT_EQUAL(true, a->contains(1));
 124   CPPUNIT_ASSERT_EQUAL(false, a->contains(3));
 125   CPPUNIT_ASSERT_EQUAL(true, a->contains(4));
 126 
 127   (*b) = (*a);
 128 
 129   CPPUNIT_ASSERT_EQUAL(true, b->contains(1));
 130   CPPUNIT_ASSERT_EQUAL(false, b->contains(3));
 131   CPPUNIT_ASSERT_EQUAL(true, b->contains(4));
 132 
 133   b->insert(5, "test");
 134   CPPUNIT_ASSERT_EQUAL(true, b->contains(5));
 135   CPPUNIT_ASSERT_EQUAL(false, a->contains(5));
 136 
 137   (*b) = (*a);
 138   CPPUNIT_ASSERT_EQUAL(true, b->contains(1));
 139   CPPUNIT_ASSERT_EQUAL(false, b->contains(3));
 140   CPPUNIT_ASSERT_EQUAL(true, b->contains(4));
 141   CPPUNIT_ASSERT_EQUAL(false, b->contains(5));
 142   CPPUNIT_ASSERT_EQUAL(false, a->contains(5));
 143 
 144   (*a)[42] = "42";
 145   CPPUNIT_ASSERT_EQUAL((size_t)5, a->size());
 146   CPPUNIT_ASSERT_EQUAL(true, a->contains(42));
 147 
 148 
 149   sa->insert("1", "Blah");
 150   sa->insert("1", "Bleck");
 151   sa->insert("2", "Blah");
 152   sa->insert("8", "Blah");
 153   sa->insert("4", "Blah");
 154   
 155   CPPUNIT_ASSERT_EQUAL(true, sa->contains("1"));
 156   CPPUNIT_ASSERT_EQUAL(false, sa->contains("3"));
 157   CPPUNIT_ASSERT_EQUAL(true, sa->contains("4"));
 158 
 159   (*sb) = (*sa);
 160 
 161   CPPUNIT_ASSERT_EQUAL(true, sb->contains("1"));
 162   CPPUNIT_ASSERT_EQUAL(false, sb->contains("3"));
 163   CPPUNIT_ASSERT_EQUAL(true, sb->contains("4"));
 164 
 165   sb->insert("5", "test");
 166   CPPUNIT_ASSERT_EQUAL(true, sb->contains("5"));
 167   CPPUNIT_ASSERT_EQUAL(false, sa->contains("5"));
 168 
 169   (*sb) = (*sa);
 170   CPPUNIT_ASSERT_EQUAL(true, sb->contains("1"));
 171   CPPUNIT_ASSERT_EQUAL(false, sb->contains("3"));
 172   CPPUNIT_ASSERT_EQUAL(true, sb->contains("4"));
 173   CPPUNIT_ASSERT_EQUAL(false, sb->contains("5"));
 174   CPPUNIT_ASSERT_EQUAL(false, sa->contains("5"));
 175 
 176   (*sa)["42"] = "42";
 177   CPPUNIT_ASSERT_EQUAL((size_t)5, sa->size());
 178   CPPUNIT_ASSERT_EQUAL(true, sa->contains("42"));
 179 
 180 
 181   //Test .keys and .values
 182   Array<String> keys = sa->keys();
 183   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("1") != size_t(-1));
 184   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("2") != size_t(-1));
 185   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("8") != size_t(-1));
 186   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("4") != size_t(-1));
 187   CPPUNIT_ASSERT_EQUAL(bool(1), keys.find("42") != size_t(-1));
 188   CPPUNIT_ASSERT_EQUAL(size_t(5), keys.size());
 189 
 190   Array<String> values = sa->values();
 191   CPPUNIT_ASSERT_EQUAL(bool(1), values.find("Blah") != size_t(-1));
 192   CPPUNIT_ASSERT_EQUAL(bool(1), values.find("42") != size_t(-1));
 193   CPPUNIT_ASSERT_STRING_EQUAL("Blah Blah Blah Blah 42", values.join(' '));
 194   CPPUNIT_ASSERT_EQUAL(size_t(5), values.size());
 195 }

 245 void HashTableTest :: getValueTest (void)
 246 {
 247   a->insert(1, "Blah1");
 248   CPPUNIT_ASSERT_STRING_EQUAL("Blah1", a->getValue(1));
 249   CPPUNIT_ASSERT_EQUAL((size_t)1, a->size());
 250 
 251   CPPUNIT_ASSERT_EQUAL(false, a->insert(1, "Bleck"));
 252   CPPUNIT_ASSERT_STRING_EQUAL("Blah1", a->getValue(1));
 253   CPPUNIT_ASSERT_EQUAL((size_t)1, a->size());
 254 
 255 
 256   a->insert(2, "Blah2");
 257   CPPUNIT_ASSERT_STRING_EQUAL("Blah1", a->getValue(1));
 258   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", a->getValue(2));
 259   CPPUNIT_ASSERT_EQUAL((size_t)2, a->size());
 260 
 261   a->insert(8, "Blah8");
 262   CPPUNIT_ASSERT_STRING_EQUAL("Blah1", a->getValue(1));
 263   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", a->getValue(2));
 264   CPPUNIT_ASSERT_STRING_EQUAL("Blah8", a->getValue(8));
 265   CPPUNIT_ASSERT_EQUAL((size_t)3, a->size());
 266 
 267   a->insert(4, "Blah4");
 268   CPPUNIT_ASSERT_STRING_EQUAL("Blah4", (*a)[4]);
 269   CPPUNIT_ASSERT_STRING_EQUAL("Blah1", (*a)[1]);
 270   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", (*a)[2]);
 271   CPPUNIT_ASSERT_STRING_EQUAL("Blah8", a->getValue(8));
 272   CPPUNIT_ASSERT_EQUAL((size_t)4, a->size());
 273 
 274   
 275   String result = (*a)[58];
 276   CPPUNIT_ASSERT_EQUAL(true, result.isEmpty());
 277   CPPUNIT_ASSERT_EQUAL((size_t)5, a->size());
 278 
 279   (*b) = (*a);

 281   CPPUNIT_ASSERT_STRING_EQUAL("Blah1", b->getValue(1));
 282   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", b->getValue(2));
 283   CPPUNIT_ASSERT_STRING_EQUAL("Blah8", b->getValue(8));

 287   HashTable<int, String> test(*a);

 290   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", test.getValue(2));

 316 void HashTableTest :: removeTest (void)
 317 {
 318   a->insert(1, "Blah");
 319   a->insert(1, "Bleck");
 320   a->insert(2, "Blah2");
 321   a->insert(8, "Blah8");
 322   a->insert(4, "Blah4");
 323 
 324   a->remove(8);  
 325 
 326   String result;
 327 
 328   result = a->getValue(58);
 329   CPPUNIT_ASSERT_EQUAL(true, result.isEmpty());
 330 
 331  
 332   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->getValue(1));
 333   CPPUNIT_ASSERT_STRING_EQUAL("Blah4", a->getValue(4));
 334   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", a->getValue(2));
 335   result = a->getValue(8);
 336   CPPUNIT_ASSERT_EQUAL(true, result.isEmpty());
 337 
 338   CPPUNIT_ASSERT_EQUAL(false, a->remove(8));
 339 
 340   a->remove(4);
 341   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->getValue(1));
 342   CPPUNIT_ASSERT_STRING_EQUAL("Blah2", a->getValue(2));
 343   result = a->getValue(4);
 344   CPPUNIT_ASSERT_EQUAL(true, result.isEmpty());
 345 
 346   a->remove(2);
 347   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->getValue(1));
 348   result = a->getValue(2);

 350
