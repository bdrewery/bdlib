  37 void AtomicFileTest :: setUp (void)
  38 {
  39   FILE *f;
  40   size_t source_len;
  41 
  42   chdir("/tmp/");
  43 
  44   /* Copy /etc/services into source */
  45   f = fopen("/etc/services", "r");
  46   CPPUNIT_ASSERT(f != NULL);
  47   fseek(f, 0, SEEK_END);
  48   source_len = ftell(f);
  49   CPPUNIT_ASSERT(source_len > 0);
  50   source = new String();
  51   source->resize(source_len);
  52   fseek(f, 0, SEEK_SET);
  53   CPPUNIT_ASSERT_EQUAL(source_len, fread(source->begin(), 1, source_len, f));
  54   CPPUNIT_ASSERT_EQUAL(source_len, source->length());
  55   fclose(f);
  56 }

  58 void AtomicFileTest :: tearDown (void)

  60   delete source;

  63 void AtomicFileTest :: basicTest (void)
  64 {
  65   AtomicFile *a;
  66   int fd;
  67   FILE *f, *f2;
  68   const char *dst_name;
  69   char buf[4];
  70   String verification;
  71   struct stat st;
  72 
  73   dst_name = "test.file";
  74   unlink(dst_name);
  75 
  76   /* Generate a junk file to replace and keep it open */
  77   f2 = fopen(dst_name, "w");
  78   CPPUNIT_ASSERT(f2 != NULL);
  79   CPPUNIT_ASSERT_EQUAL(size_t(4), fwrite("test", 1, 4, f2));
  80   fflush(f2);
  81   /* !! Keep dst_name open so it is busy */
  82 
  83   /* Write out source into the dst */
  84   a = new AtomicFile;
  85   a->open(dst_name, S_IRUSR);
  86   CPPUNIT_ASSERT_EQUAL(true, a->is_open());
  87   /* Write out the data to the fd through a FILE stream */
  88   fd = a->fd();
  89   CPPUNIT_ASSERT(fd != -1);
  90   /* dup(2) the fd due to lack of fdclose() */
  91   fd = dup(fd);
  92   f = fdopen(fd, "w");
  93   CPPUNIT_ASSERT(f != NULL);
  94   CPPUNIT_ASSERT_EQUAL(source->length(), fwrite(
  95         static_cast<const char*>(source->begin()), 1, source->length(), f));
  96   fclose(f);
  97 
  98   /* Verify the contents of the dst have not yet changed */
  99   f = fopen(dst_name, "r");
 100   CPPUNIT_ASSERT_EQUAL(size_t(4), fread(buf, 1, 4, f));
 101   fclose(f);
 102   CPPUNIT_ASSERT_STRING_EQUAL("test", String(buf, 4));
 103 
 104   /* Write out the file atomically. */
 105   CPPUNIT_ASSERT_EQUAL(true, a->commit());
 106   /* !! a left open, it is expected to flush out / rename at close() */
 107 
 108   /* Now verify the contents of dst against source */
 109 
 110   f = fopen(dst_name, "r");
 111   CPPUNIT_ASSERT(f != NULL);
 112   fseek(f, 0, SEEK_END);
 113   CPPUNIT_ASSERT_EQUAL(source->length(), static_cast<size_t>(ftell(f)));
 114   verification.resize(source->length());
 115   fseek(f, 0, SEEK_SET);
 116   CPPUNIT_ASSERT_EQUAL(source->length(), fread(verification.begin(), 1,
 117         source->length(), f));
 118   CPPUNIT_ASSERT_STRING_EQUAL(*source, verification);
 119 
 120   /* Also verify the chmod */
 121   CPPUNIT_ASSERT_EQUAL(0, fstat(fileno(f), &st));
 122   CPPUNIT_ASSERT_EQUAL(mode_t(S_IRUSR), st.st_mode & ~S_IFMT);
 123 
 124   fclose(f);
 125 
 126   fclose(f2);           /* The junk file kept open */
 127   unlink(dst_name);
 128   delete a;
 129 }

 199 void AtomicFileTest :: abortTest (void)
 200 {
 201   AtomicFile *a;
 202   int fd;
 203   FILE *f, *f2;
 204   const char *dst_name;
 205   char buf[4];
 206   String verification;
 207 
 208   dst_name = "test.file";
 209   unlink(dst_name);
 210 
 211   /* Generate a junk file to replace and keep it open */
 212   f2 = fopen(dst_name, "w");
 213   CPPUNIT_ASSERT(f2 != NULL);
 214   CPPUNIT_ASSERT_EQUAL(size_t(4), fwrite("test", 1, 4, f2));
 215   fflush(f2);
 216   /* !! Keep dst_name open so it is busy */
 217 
 218   /* Write out source into the dst */
 219   a = new AtomicFile;
 220   a->open(dst_name);
 221   CPPUNIT_ASSERT_EQUAL(true, a->is_open());
 222   /* Write out the data to the fd through a FILE stream */
 223   fd = a->fd();
 224   CPPUNIT_ASSERT(fd != -1);
 225   /* dup(2) the fd due to lack of fdclose() */
 226   fd = dup(fd);
 227   f = fdopen(fd, "w");
 228   CPPUNIT_ASSERT(f != NULL);
 229   CPPUNIT_ASSERT_EQUAL(source->length(), fwrite(
 230         static_cast<const char*>(source->begin()), 1, source->length(), f));
 231   fsync(fd);
 232   fclose(f);
 233 
 234   /* Verify the contents of the dst have not yet changed */
 235   f = fopen(dst_name, "r");
 236   CPPUNIT_ASSERT_EQUAL(size_t(4), fread(buf, 1, 4, f));
 237   fclose(f);
 238   CPPUNIT_ASSERT_STRING_EQUAL("test", String(buf, 4));
 239 
 240   /* Abort the atomic file writing and verify the contents and file remain. */
 241   CPPUNIT_ASSERT_EQUAL(true, a->abort());
 242 
 243   f = fopen(dst_name, "r");
 244   CPPUNIT_ASSERT_EQUAL(size_t(4), fread(buf, 1, 4, f));
 245   fseek(f, 0, SEEK_END);
 246   CPPUNIT_ASSERT_EQUAL(size_t(4), static_cast<size_t>(ftell(f)));
 247   fclose(f);
 248   CPPUNIT_ASSERT_STRING_EQUAL("test", String(buf, 4));
 249 
 250   fclose(f2);           /* The junk file kept open */
 251   unlink(dst_name);
 252   delete a;
 253 }
