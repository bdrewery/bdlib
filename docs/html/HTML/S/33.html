 319 {

 321 
 322   (*str_a).push("test");
 323   (*str_a).push("test2");
 324   (*str_b).push("test");
 325   (*str_b).push("test2");

 332   str_b->clear();

 371 void ArrayTest :: operatorsTest(void)
 372 {
 373   str_a->push("1");
 374   str_a->push("2");
 375   str_b->push("3");
 376   str_b->push("4");
 377 
 378   str_c = new Array<String>();
 379   (*str_c) = (*str_a) + (*str_b);
 380 
 381   CPPUNIT_ASSERT_STRING_EQUAL("1 2 3 4", str_c->join(' '));
 382   CPPUNIT_ASSERT_EQUAL(size_t(4), str_c->size());
 383 
 384   (*str_c) += 1;
 385   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4", str_c->join(' '));
 386   CPPUNIT_ASSERT_EQUAL(size_t(3), str_c->size());
 387 
 388   (*str_b)[0] = "5";
 389   (*str_b)[1] = "6";
 390   (*str_c) += (*str_b);
 391 
 392   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4 5 6", str_c->join(' '));
 393   CPPUNIT_ASSERT_EQUAL(size_t(5), str_c->size());
 394 
 395   (*str_b) = (*str_c)--;
 396 
 397   CPPUNIT_ASSERT_STRING_EQUAL("6", str_b->join(' '));
 398   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4 5", str_c->join(' '));
 399   CPPUNIT_ASSERT_EQUAL(size_t(1), str_b->size());
 400   CPPUNIT_ASSERT_EQUAL(size_t(4), str_c->size());
 401 
 402   --(*str_c);
 403 
 404   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4", str_c->join(' '));
 405   CPPUNIT_ASSERT_EQUAL(size_t(3), str_c->size());
 406 
 407   ++(*str_c);
 408   CPPUNIT_ASSERT_STRING_EQUAL("3 4", str_c->join(' '));
 409   CPPUNIT_ASSERT_EQUAL(size_t(2), str_c->size());
 410 
 411   (*str_b) = (*str_c)++;
 412   CPPUNIT_ASSERT_EQUAL(size_t(1), str_c->size());
 413   CPPUNIT_ASSERT_STRING_EQUAL("3", str_b->join(' '));
 414   CPPUNIT_ASSERT_STRING_EQUAL("4", str_c->join(' '));
 415 }
