  42 #define STREAM_BLOCKSIZE        1024
  43 /* The possibilities for the third argument to `fseek'.
  44    These values should not be changed.  */
  45 #define SEEK_SET        0       /* Seek from beginning of file.  */
  46 #define SEEK_CUR        1       /* Seek from current position.  */
  47 #define SEEK_END        2       /* Seek from end of file.  */
  48 
  49 class Stream {
  50   protected:
  51         String str;
  52         unsigned int pos;
  53         bool loading;
  54 
  55   public:
  56         Stream() : str(), pos(0), loading(0) {};
  57         Stream(const Stream& stream) : str(stream.str), pos(stream.pos), loading(0) {};
  58         Stream(Stream&& stream) : str(std::move(stream.str)), pos(std::move(stream.pos)), loading(std::move(stream.loading)) {
  59           stream.str = String();
  60           stream.pos = 0;
  61           stream.loading = 0;
  62         }
  63         Stream(const String& string) : str(string), pos(0), loading(0) {};
  64         Stream(const int newSize) : str(), pos(0), loading(0) { if (newSize > 0) Reserve(newSize); };
  65         virtual ~Stream() {};
  66 
  67         friend void swap(Stream& a, Stream& b) {
  68           using std::swap;
  69 
  70           swap(a.str, b.str);
  71           swap(a.pos, b.pos);
  72           swap(a.loading, b.loading);
  73         }
  74 
  75         Stream& operator=(Stream stream) {
  76           swap(*this, stream);
  77           return *this;
  78         }
  79 
  80         void Reserve(const size_t) const;
  81 
  82         /**
  83          * @brief Returns the position of the Stream.
  84          * @return Position of the Stream.
  85         */
  86         size_t tell() const { return pos; };
  87 
  88         /**
  89          * @brief Truncates the stream at the current position.
  90         */
  91         void truncate() { str = str(0, pos); };
  92 
  93         String peek() { return str(pos); };
  94 
  95 //        operator void*() { return tell() <= length(); };
  96 
  97         int seek(int, int);
  98         void clear() { str.clear(); pos = 0; }
  99 
 100         /**
 101          * @brief Insert a string into the stream.
 102          * @note The stream pointer is advanced as well
 103          */
 104         virtual void puts (const String& string) {
 105           str.replace(tell(), string);
 106           pos += string.length();
 107         }
 108 
 109         /**
 110          * @brief Reads 1 line from the stream
 111          * @note The stream pointer is advanced as well
 112          * @param maxSize Optional param which specifies max data to pull
 113          * @sa read
 114          */
 115         virtual String getline (size_t maxSize = 99999999) { return read(maxSize == 99999999 ? (length() - pos) : maxSize, '\n'); }
 116 
 117         /**
 118          * @brief Reads specified number of bytes from the stream
 119          * @param maxSize How many bytes to read
 120          * @param delim What to split the read on. For example: '\n' will return 1 line.
 121          * @note The stream pointer is advanced as well
 122          */
 123         virtual String read (size_t maxSize, char delim = 0);
 124 
 125         /**
 126          * @brief Load a file into the stream
 127          * @returns 1 on error, 0 on success.
 128          * @param fd File descriptor to read from
 129          * @note this.loading is set to 1 during this process.
 130          * @note Uses puts() for populating the stream.
 131          * @note The fd is not closed.
 132          * @sa puts
 133          */
 134         virtual int loadFile(const int fd);
 135 
 136         /**
 137          * @brief Load a file into the stream
 138          * @returns 1 on error, 0 on success.
 139          * @param fname File to read from
 140          * @note this.loading is set to 1 during this process.
 141          * @note Uses puts() for populating the stream.
 142          * @sa puts
 143          */
 144         virtual int loadFile(const String& fname);
 145 
 146         /**
 147          * @brief Write stream out to a file
 148          * @returns 1 on error, 0 on success
 149          * @param fd File descriptor to write to
 150          * @note The fd is not closed.
 151          */
 152         virtual int writeFile(const int fd) const;
 153 
 154         /**
 155          * @brief Write stream out to a file
 156          * @returns 1 on error, 0 on success
 157          * @param fname Filename to write to
 158          * @param mode Optional param to specify mode for new file
 159          */
 160         virtual int writeFile(const String& fname, mode_t mode = (S_IRUSR|S_IWUSR)) const;
 161 
 162 
 163         inline operator String() const { return str; };
 164         inline size_t length() const { return str.length(); };
 165         inline size_t capacity() const { return str.capacity(); };
 166         inline bool operator!() const { return str.isEmpty(); };
 167 
 168         friend Stream& operator<<(Stream&, const String&);
 169 };
 170 
 171 inline Stream& operator<<(Stream& stream, const String& string) { stream.puts(string); return stream; }
