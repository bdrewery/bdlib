  38 template <class T>
  39 /**
  40  * @class Array
  41  * @brief Dynamic Array
  42  */
  43 class Array : public ReferenceCountedArray<T> {
  44   public:
  45     typedef T value_type;
  46 
  47     typedef size_t             size_type;
  48     typedef std::ptrdiff_t     difference_type;
  49     typedef value_type*        pointer;
  50     typedef const value_type*  const_pointer;
  51     typedef value_type&        reference;
  52     typedef const value_type&  const_reference;
  53 
  54     /* Constructors */
  55     Array() : ReferenceCountedArray<value_type>() {};
  56     Array(const Array<value_type>& array) : ReferenceCountedArray<value_type>(array) {};
  57 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  58     Array(Array<value_type>&& array) : ReferenceCountedArray<value_type>(array) {};
  59 #endif
  60     /**
  61      * @brief Create a Array from a given carray.
  62      * @param carray The null-terminated array to create the object from.
  63      * @post A ArrayBuf has been initialized.
  64      * @post The buffer has been filled with the array.
  65      * @test Array test("Some array");
  66      */
  67     Array(const_pointer carray) : ReferenceCountedArray<value_type>() {
  68       size_t i = 0;
  69       while (1) {
  70         if (!carray[i]) break;
  71         push(carray[i]);
  72         ++i;
  73       }
  74     };
  75 
  76 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  77     /**
  78      * @brief Create an array from an initializer list
  79      * @param list An initializer_list
  80      */
  81     Array(std::initializer_list<value_type> list) : ReferenceCountedArray<value_type>() {
  82       *this = list;
  83     }
  84 #endif
  85 
  86     /**
  87      * @brief Create a Array from a given carray.
  88      * @param carray The null-terminated array to create the object from.
  89      * @param len How big is the carray?
  90      * @post A ArrayBuf has been initialized.
  91      * @post The buffer has been filled with the array.
  92      * @test Array test("Some array");
  93      */
  94     Array(const_pointer carray, size_t len) : ReferenceCountedArray<value_type>() {
  95       this->Reserve(len);
  96       for (size_t i = 0; i < len; ++i)
  97         push(carray[i]);
  98     };
  99 
 100     /**
 101      * @brief Create an empty Array container with at least the specified elements in size.
 102      * @param newSize Reserve at least this many buckets for this Array.
 103      * @post This array's memory will also never be shrunk.
 104      * @post A buffer has been created.
 105      */
 106     explicit Array(const size_type newSize) : ReferenceCountedArray<value_type>(newSize) {};
 107     Array(const size_type newSize, const value_type value) : ReferenceCountedArray<value_type>(newSize, value) {};
 108 
 109     virtual ~Array() {};
 110 
 111 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 112     /**
 113      * @brief Create an array from an initializer list
 114      * @param list An initializer_list
 115      */
 116     Array& operator=(std::initializer_list<value_type> list) {
 117       this->clear();
 118       this->Reserve(list.size());
 119       for (value_type item : list) {
 120         push(std::move(item));
 121       }
 122       return *this;
 123     }
 124 #endif
 125     using ReferenceCountedArray<value_type>::operator=;
 126 
 127     /**
 128      * @brief Add an item to the end of the array
 129      */
 130     inline void push(const value_type item) {
 131       this->AboutToModify(this->length() + 1);
 132       *(this->Buf(this->length())) = item;
 133       this->addLength(1);
 134     }
 135 
 136     /**
 137      * @sa push
 138      */
 139     inline friend Array<value_type>& operator<<(Array<value_type>& array, const_reference item) {
 140       array.push(item);
 141       return array;
 142     }
 143 
 144     /**
 145      * @brief Shift the array left, removing the first element.
 146      * @return The first element.
 147      */
 148     inline value_type shift() {
 149       if (this->isEmpty()) return value_type();
 150 
 151       value_type temp(*(this->Buf(0)));
 152       ++(this->offset);
 153       this->subLength(1);
 154       return temp;
 155     }
 156 
 157     /**
 158      * @brief Pop a value off the end of the array
 159      * @return The last element.
 160      */
 161     inline value_type pop() {
 162       if (this->isEmpty()) return value_type();
 163 
 164       value_type temp(*(this->Buf(this->length() - 1)));
 165       this->subLength(1);
 166       return temp;
 167     }
 168 
 169     /**
 170      * @sa pop
 171      */
 172     inline friend Array<value_type>& operator>>(Array<value_type>& array, reference item) {
 173       item = array.pop();
 174       return array;
 175     }
 176 
 177     /**
 178      * @brief Join an array by a delimiter into a string
 179      */
 180     String join(const String& delim, bool quoted = false) const {
 181       if (!this->length()) return String();
 182       String str;
 183       for (size_t i = 0; i < this->length(); ++i) {
 184         if (i)
 185           str += delim;
 186         if (quoted) {
 187           str += '"' + *(this->Buf(i)) + '"';
 188         } else {
 189           str += *(this->Buf(i));
 190         }
 191       }
 192       return str;
 193     }
 194 
 195     inline bool equals(const Array& array) const { return equals(array, array.length()); };
 196     /**
 197      * @brief Compare our Array object with another Array object, but only n elements
 198      * @param array The Array object to equals to.
 199      * @param n The number of items to equals.
 200      * @return True if the number of elements are the same, and they all are equal.
 201      */
 202     bool equals(const Array& array, size_t n) const
 203     {
 204       size_t my_len = this->length();
 205       bool same_length = (my_len == array.length());
 206 
 207       /* Same array? */
 208       if (this->data() == array.data() && same_length)
 209         return true;
 210 
 211       if (!same_length)
 212         return false;
 213 
 214       size_t slen = std::min(array.length(), n);
 215       size_t len = std::min(my_len, slen);
 216 
 217       for (size_t i = 0; i < len; ++i) {
 218         if ((*this)[i] != array[i])
 219           return false;
 220       }
 221       return true;
 222     }
 223 
 224     inline friend bool operator==(const Array& lhs, const Array& rhs) { return lhs.equals(rhs); };
 225     inline friend bool operator!=(const Array& lhs, const Array& rhs) { return !(lhs == rhs); };

 228     /**
 229      * @sa ReferenceCountedArray::slice()
 230      */
 231     inline Array subarray(int start, int len = -1) const {
 232       Array newArray(*this);
 233       newArray.slice(start, len);
 234       return newArray;
 235     };

 238      * @sa subarray
 239      */
 240     inline Array operator()(int start, int len = -1) const { return subarray(start, len); };

 243      * @brief Returns a 'Slice' class for safe (cow) writing into the array

 245      * @param start Starting position
 246      * @param len How many items to use
 247      */
 248     inline Slice<Array> operator()(int start, int len = -1) { return Slice<Array>(*this, start, len); };

 251     void CPPUNIT_checkArrayEqual(Array actual, CPPUNIT_NS::SourceLine sourceLine) {

 258     /**
 259      * @relates Array

 261      * @param array1 The LHS array.
 262      * @param array2 The RHS array.
 263      * @post A new array is allocated, reference copied and returned.

 267       array1 += array2;
 268       return array1;
 269     }

 271     /**
 272      * @brief Prefix increment
 273      */
 274     inline const Array& operator++() {
 275       return (*this) += 1;
 276     }
 277 
 278     /**
 279      * @brief Postfix increment
 280      */
 281     inline const Array operator++(int) {
 282       Array tmp((*this)(0, 1));
 283       ++(this->offset);
 284       this->subLength(1);
 285       return tmp;
 286     }
 287 
 288     /**
 289      * @brief Prefix decrement
 290      */
 291     inline const Array& operator--() {
 292       return (*this) -= 1;
 293     }
 294 
 295     /**
 296      * @brief Postfix decrement
 297      */
 298     inline const Array operator--(int) {
 299       Array tmp((*this)(this->length() - 1, 1));
 300       this->subLength(1);
 301       return tmp;
 302     }
 303 
 304     /**
 305      * \sa append(const char)
 306      */
 307     inline Array& operator+=(const_reference item) {
 308       this->append(item);
 309       return *this;
 310     }
 311 
 312     /**
 313      * \sa append(const Array&)
 314      */
 315     inline Array& operator+=(const Array& array) {
 316       this->append(array);
 317       return *this;
 318     }
 319 
 320     inline Array& operator+=(const int n) {
 321       if (!this->length())
 322         return *this;
 323       if (int(this->length()) - n < 0) {
 324         this->offset = this->length();
 325         this->setLength(0);
 326       } else {
 327         this->offset += n;
 328         this->subLength(n);
 329       }
 330       return *this;
 331     }
 332 
 333     inline Array& operator-=(const int n) {
 334       if (!this->length())
 335         return *this;
 336       if (int(this->length()) - n < 0) {
 337         this->offset = this->length();
 338         this->setLength(0);
 339       } else
 340         this->subLength(n);
 341       return *this;
 342     }
