  43 void Stream::Reserve(const size_t newSize) const {
  44   if (newSize < capacity())
  45     return;
  46   str.Reserve(STREAM_BLOCKSIZE * ((newSize + STREAM_BLOCKSIZE -1) / STREAM_BLOCKSIZE));
  47 }

  49 int Stream::seek(int offset, int whence) {
  50   int newpos;
  51 
  52   switch (whence) {
  53     case SEEK_SET:
  54       newpos = offset;
  55       break;
  56     case SEEK_CUR:
  57       newpos = tell() + offset;
  58       break;
  59     case SEEK_END:
  60       newpos = capacity() - offset;
  61       break;
  62     default:
  63       newpos = tell();
  64   }
  65   if (newpos < 0)
  66     newpos = 0;
  67   else if (static_cast<size_t>(newpos) > capacity())
  68     newpos = capacity();
  69   pos = newpos;
  70 
  71   return newpos;
  72 }

  74 String Stream::read(size_t maxSize, char delim) {
  75   size_t toRead = std::min(maxSize, length() - pos);
  76 
  77   /* No need to split the string, return a substring */
  78   if (!delim) {
  79     String ret(str(pos, toRead));
  80     pos += toRead;
  81     return ret;
  82   }
  83 
  84   /* Must reconstruct a new string */
  85   String ret(toRead);
  86 
  87   while ((ret.length() < toRead) && (str[pos] != delim)) {
  88     ret += str[pos++];
  89   }
  90   if (str.size() > pos && str[pos] == delim) {
  91     ret += delim;
  92     ++pos;
  93   }
  94 
  95   return ret;
  96 }

 108 int Stream::loadFile(const int fd)
 109 {
 110   clear();
 111 #ifdef HAVE_MMAP
 112   ssize_t size = lseek(fd, 0, SEEK_END);
 113   if (size < 0) {
 114     return 1;
 115   }
 116   Reserve(size);
 117   void* map = mmap(0, size, PROT_READ, MAP_PRIVATE, fd, 0);
 118   if (map == MAP_FAILED) {
 119     return 1;
 120   }
 121   loading = 1;
 122 
 123   puts(String(static_cast<const char*>(map), size));
 124 
 125   munmap(static_cast<void*>(map), size);
 126 #else
 127   int my_fd = dup(fd);
 128   if (my_fd == -1) {
 129     return 1;
 130   }
 131   FILE *f = fdopen(my_fd, "rb");
 132   if (f == nullptr) {
 133     close(my_fd);
 134     return 1;
 135   }
 136 
 137   loading = 1;
 138   fseek(f, 0, SEEK_END);
 139   size_t size = ftell(f);
 140   fseek(f, 0, SEEK_SET);
 141   Reserve(size);
 142 
 143   size_t len = 0;
 144   char buf[STREAM_BLOCKSIZE + 1];
 145 
 146   while ((len = fread(buf, 1, sizeof(buf) - 1, f))) {
 147     puts(String(buf, len));
 148   }
 149   fclose(f);
 150 #endif
 151   seek(0, SEEK_SET);
 152   loading = 0;
 153   return 0;
 154 }

 156 int Stream::writeFile(const String& file, mode_t mode) const
 157 {
 158   int fd = open(file.c_str(), O_CREAT|O_RDWR|O_TRUNC, mode);
 159   if (fd == -1)
 160     return 1;
 161   int ret = writeFile(fd);
 162   close(fd);
 163   return ret;
 164 }

 166 int Stream::writeFile(const int fd) const
 167 {
 168 #ifdef HAVE_MMAP
 169   /* Write to end of file to make its size match */
 170   /* Seek past wherever it is at now plus our length */
 171   if (ftruncate(fd, length())) return 1;
 172   if (!length())
 173     return 0;
 174 
 175   void* map = mmap(0, length(), PROT_READ|PROT_WRITE, MAP_SHARED, fd, 0);
 176 
 177   if (map == MAP_FAILED) return 1;
 178 
 179   std::memcpy(map, str.data(), length());
 180 
 181   if (munmap(map, length()) == -1) return 1;
 182 #else
 183   int my_fd = dup(fd);
 184   if (my_fd == -1) {
 185     return 1;
 186   }
 187   FILE *f = fdopen(my_fd, "wb");
 188   if (f == nullptr) {
 189     close(my_fd);
 190     return 1;
 191   }
 192 
 193   if ((fwrite(str.data(), 1, length(), f) != length()) || (fflush(f))) {
 194     fclose(f);
 195     return 1;
 196   }
 197   fclose(f);
 198 #endif
 199   return 0;
 200 }
