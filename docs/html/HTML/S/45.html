  27 #ifndef _BD_LIST_H
  28 #define _BD_LIST_H 1
  29 
  30 #include "bdlib.h"
  31 #include "Iterator.h"

  34 template <class T>
  35 /**
  36   * @class List
  37   * @brief List data structure
  38   */
  39 class List {
  40   class ListIterator;
  41 
  42   public:
  43     typedef ListIterator iterator;
  44     typedef T value_type;
  45 
  46     typedef size_t             size_type;
  47     typedef std::ptrdiff_t     difference_type;
  48     typedef value_type*        pointer;
  49     typedef const value_type*  const_pointer;
  50     typedef value_type&        reference;
  51     typedef const value_type&  const_reference;
  52 
  53   protected:
  54     struct Node {
  55       value_type item;
  56       Node* next;
  57       Node* prev;
  58       
  59       Node(const_reference p) : item(p), next(nullptr), prev(nullptr) {};
  60       /* To avoid -Weffc++ warnings */
  61       Node(const Node& n) : item(n.item), next(n.next), prev(n.prev) {};
  62       Node(const value_type&& p) : item(std::move(p.item)), next(std::move(p.next)), prev(std::move(p.prev)) {
  63         p.item = nullptr;
  64         p.next = nullptr;
  65         p.prev = nullptr;
  66       };
  67 
  68       friend void swap(Node& a, Node& b) {
  69         using std::swap;
  70 
  71         swap(a.item, b.item);
  72         swap(a.next, b.next);
  73         swap(a.prev, b.prev);
  74       }
  75 
  76       Node& operator=(Node n) {
  77         swap(*this, n);
  78         return *this;
  79       }
  80     };
  81 
  82     void deleteNode(Node* node) {
  83       Node* next = node->next; //NULL if node = tail
  84       Node* prev = node->prev; //NULL if node = head
  85 
  86       delete node;
  87 
  88       if (next) next->prev = prev;
  89       if (prev) prev->next = next;
  90 
  91       if (node == head) head = next;
  92       if (node == tail) tail = prev;
  93 
  94       --my_size;
  95     }
  96 
  97     Node* findNode(const_reference item) const {
  98       for (Node* current = head; current; current = current->next) {
  99         if (current->item == item) {
 100           return current;
 101         }
 102       }
 103       return nullptr;
 104     }
 105 
 106     Node* head;
 107     Node* tail;
 108     size_type my_size;
 109 
 110   public:
 111     List() : head(nullptr), tail(nullptr), my_size(0) {};
 112     virtual ~List() { clear(); };
 113 
 114     List(const List& list) : head(nullptr), tail(nullptr), my_size(0) {
 115       for (Node* search = list.head; search; search = search->next)
 116         insert(search->item);
 117     }
 118 
 119     List(List&& list) : head(nullptr), tail(nullptr), my_size(0) {
 120       swap(*this, list);
 121     }
 122 
 123     friend void swap(List& a, List& b) {
 124       using std::swap;
 125 
 126       swap(a.head, b.head);
 127       swap(a.tail, b.tail);
 128       swap(a.my_size, b.my_size);
 129     }
 130 
 131     List& operator=(List list) {
 132       swap(*this, list);
 133       return *this;
 134     }
 135 
 136     void clear() {
 137       Node* node = head, *node_n = nullptr;
 138       while (node) {
 139         node_n = node->next;
 140         delete node;
 141         node = node_n;
 142       }
 143       head = tail = nullptr;
 144       my_size = 0;
 145     };
 146 
 147     inline size_type size() const { return my_size; };
 148     inline bool isEmpty() const { return size() == 0; };
 149     inline explicit operator bool() const { return !isEmpty(); };
 150 
 151     /**
 152      * @brief Insert into the list at the head
 153      * @param item The item to insert
 154      */
 155     void insert(const_reference item) {
 156       Node* node = new Node(item);
 157       if (!head) {
 158         head = tail = node;
 159       } else {
 160         head->prev = node;
 161         node->next = head;
 162         head = node;
 163       }
 164       ++my_size;
 165     }
 166 
 167     /**
 168      * @sa insert
 169      */
 170     inline friend List<value_type>& operator<< (List<value_type>& list, const_reference item) {
 171       list.insert(item);
 172       return list;
 173     }
 174 
 175     bool contains(const_reference item) const {
 176       return findNode(item) ? true : false;
 177     }
 178 
 179    const value_type find(const_reference item) const {
 180       Node* node = findNode(item);
 181       return node ? node->item : value_type();
 182     }
 183 
 184     /**
 185      * @brief Return a reference to the found element
 186      */
 187     reference findRef(const_reference item) {
 188       Node* node = findNode(item);
 189       return node ? node->item : head->item;
 190     }
 191 
 192     bool remove(const_reference item) {
 193       /* cases (removing X):
 194        * 1) list = (X)
 195        * 2) list = (X)->(Y)
 196        * 3) list = (W)->(X)
 197        * 4) list = (W)->(X)->(Y)
 198        */
 199       Node* node = findNode(item);
 200       if (node) {
 201         deleteNode(node);
 202         return true;
 203       }
 204       return false;
 205     }
 206 
 207     private:
 208     class ListIterator : public Iterator<value_type> {
 209       friend class List;
 210       private:
 211         List *list;
 212         Node *current;
 213         bool end;
 214 
 215         ListIterator(List &_list, bool _end = 0) : Iterator<value_type>(),
 216                                                               list(&_list),
 217                                                               current(_end ? _list.head : _list.tail), /* tail is beginning */
 218                                                               end(_end) {
 219         };
 220       public:
 221         ListIterator(const iterator& iter) : Iterator<value_type>(),
 222                                              list(iter.list),
 223                                              current(iter.current),
 224                                              end(iter.end) {
 225         };
 226         ListIterator() : Iterator<value_type>(),
 227                          list(nullptr),
 228                          current(nullptr),
 229                          end(0) {
 230         };
 231 
 232         iterator& operator =(const iterator& iter) {
 233           list = iter.list;
 234           current = iter.current;
 235           end = iter.end;
 236           return *this;
 237         }
 238 
 239         virtual ~ListIterator() { }
 240 
 241         virtual void remove() {
 242           Node *nextValue = end == 0 ? current->next : current->prev;
 243 
 244           list->deleteNode(current);
 245 
 246           current = nextValue;
 247         }
 248 
 249         virtual explicit operator bool() const { return (current != nullptr); };
 250         virtual operator value_type () { return operator*(); };
 251           
 252         virtual value_type& operator *() { return static_cast<value_type&>(current->item); }
 253 
 254         //Postfix
 255         virtual iterator operator ++(int) {
 256           iterator tmp(*this);
 257           current = current->prev;
 258           return tmp;
 259         }
 260 
 261         virtual iterator& operator ++() {
 262           current = current->prev;
 263           return *this;
 264         }
 265 
 266         //Postfix
 267         virtual iterator operator --(int) {
 268           iterator tmp(*this);
 269           current = current->next;
 270           return tmp;
 271         }
 272 
 273         virtual iterator& operator --() {
 274           current = current->next;
 275           return *this;
 276         }
 277     };
 278   public:
 279     inline iterator begin() { return iterator(*this); };
 280     inline iterator end() { return iterator(*this, 1); };
 281 };
