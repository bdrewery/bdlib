  31 #define NUM_ASCII_BYTES (3)
  32 #define NUM_ENCODED_BYTES (4)
  33 #define PADDING_CHAR '='

  35 void b64enc_buf(const unsigned char *data, size_t *len, char *dest, const char* charset)
  36 {
  37   char *buf = dest;
  38 
  39   if (!charset) {
  40     charset = b64_charset;
  41   }
  42 
  43   /* Encode 3 bytes at a time. */
  44 
  45   /*
  46    *
  47    * |       0       |       1       |       2       |
  48    *
  49    * |               |               |               |
  50    * |       |       |       |       |       |       |
  51    * |   |   |   |   |   |   |   |   |   |   |   |   |
  52    * | | | | | | | | | | | | | | | | | | | | | | | | |
  53    *  7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0 7 6 5 4 3 2 1 0
  54    * -------------------------------------------------
  55    *  5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 5 4 3 2 1 0 
  56    * | | | | | | | | | | | | | | | | | | | | | | | | |
  57    * |   | | |   |   | | |   |   | | |   |   | | |   |
  58    * |     |     |     |     |     |     |     |     |
  59    * |           |           |           |           |
  60    *
  61    * |     0     |     1     |     2     |     3     |
  62    *
  63    */
  64 
  65 
  66   while (*len >= NUM_ASCII_BYTES) {
  67     /* buf[0] is the 6 left-most bits of data[0] */
  68     buf[0] = charset[(data[0] & 0xfc) >> 2];
  69     /* buf[1] is the right-most 2 bits of data[0] and the left-most 4 bits of data[1] */
  70     buf[1] = charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
  71     /* buf[2] is the right-most 4 bits of data[1] and the 2 left-most bits of data[2] */
  72     buf[2] = charset[((data[1] & 0x0f) << 2) | ((data[2] & 0xc0) >> 6)];
  73     /* buf[3] is the right-most 6 bits of data[2] */
  74     buf[3] = charset[data[2] & 0x3f];
  75 
  76     data += NUM_ASCII_BYTES;
  77     buf += NUM_ENCODED_BYTES;
  78     *len -= NUM_ASCII_BYTES;
  79   }
  80 
  81   /* There is either 1 or 2 bytes left to encode (and possibly some padding to add) */
  82   if (*len > 0) {
  83     buf[0] = charset[(data[0] & 0xfc) >> 2];
  84     if (*len == 1) {
  85       buf[1] = charset[((data[0] & 0x03) << 4)];
  86       buf[2] = PADDING_CHAR;
  87     } else if (*len > 1) { /* == 2 .. left as >1 to help optimize with last if */
  88       /* Tack on to buf[1] the left-most 4 bits of data[1] */
  89       buf[1] = charset[((data[0] & 0x03) << 4) | ((data[1] & 0xf0) >> 4)];
  90       buf[2] = charset[(data[1] & 0x0f) << 2];
  91     }
  92     buf[3] = PADDING_CHAR;
  93     buf += NUM_ENCODED_BYTES;
  94   }
  95 
  96   *len = buf - dest;
  97 }

  99 String base64Encode(const String& string, const char* charset) {
 100   size_t len = string.length();
 101   char *p = b64enc(reinterpret_cast<const unsigned char*>(string.data()), &len, charset);
 102   bd::String encoded(p, len);
 103   free(p);
 104   return encoded;
 105 }

 107 char *b64enc(const unsigned char *src, size_t *len, const char* charset)
 108 {
 109   /* Take the length and round up the next 4-byte boundary */
 110   size_t dlen = (((*len + (NUM_ASCII_BYTES - 1)) / NUM_ASCII_BYTES) * NUM_ENCODED_BYTES);
 111   char *dest = static_cast<char*>(malloc(dlen + 1));
 112 
 113   b64enc_buf(src, len, dest, charset);
 114   dest[*len] = '\0';
 115   return dest;
 116 }

 118 void b64dec_buf(const unsigned char *data, size_t *len, char *dest, const char* charset_index)
 119 {
 120   char *buf = dest;
 121   size_t blocks = (*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0);
 122 
 123   if (!charset_index) {
 124     charset_index = b64_indexes;
 125   }
 126 
 127   /* Convert the encoded base64 character back into our base255 ASCII character */
 128   while (blocks > 0) {
 129     /* buf[0] is the 6 bits from data[0] + left-most 2 bits of data[1] */
 130     *buf++ = (charset_index[data[0]] << 2) + ((charset_index[data[1]] & 0x30) >> 4);  /* 0x30 not required but to be clear. */
 131     if (data[2] != PADDING_CHAR) {
 132       /* buf[1] is the right-most 4 bits of data[1] + left-most 4 bits of data[2] */
 133       *buf++ = ((charset_index[data[1]] & 0x0f) << 4) + ((charset_index[data[2]] & 0x3c) >> 2);
 134       if (data[3] != PADDING_CHAR) {
 135         /* buf[2] is the right-most 2 bits of data[2] + data[3] */
 136         *buf++ = ((charset_index[data[2]] & 0x03) << 6) + charset_index[data[3]];
 137       }
 138     }
 139     data += NUM_ENCODED_BYTES;
 140     --blocks;
 141   }
 142 
 143   *len = (buf - dest);
 144 }

 146 char *b64dec(const unsigned char *data, size_t *len, const char* charset)
 147 {
 148   size_t dlen = ((*len / NUM_ENCODED_BYTES) + ((*len % NUM_ENCODED_BYTES) ? 1 : 0)) * NUM_ASCII_BYTES;
 149   char *dest = static_cast<char*>(malloc(dlen + 1));
 150 
 151   if (charset) {
 152     char charset_index[256];
 153     memset(charset_index, -1, sizeof(charset_index));
 154     for (size_t i = 0; i < 64; ++i)
 155       charset_index[int(charset[i])] = i;
 156     b64dec_buf(data, len, dest, charset_index);
 157   }  else
 158     b64dec_buf(data, len, dest);
 159   dest[*len] = '\0';
 160   return dest;
 161 }
