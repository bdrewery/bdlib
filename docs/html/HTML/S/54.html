  41 int String::compare(const String& str, size_t n, size_t start) const
  42 {
  43   const size_t my_len = length();
  44   /* Same string? */
  45   if (data() == str.data() && my_len == str.length())
  46     return 0;
  47 
  48   const size_t slen = n ? std::min(str.length(), n) : str.length();
  49   const size_t len = std::min(my_len - start, slen);
  50   const int diff = memcmp(begin() + start, str.begin(), len);
  51   if (diff)
  52     return diff;
  53   else if (n)
  54     return std::min(my_len - start, n) - slen;
  55   else
  56     return (my_len - start) - slen;
  57 }

  61 /**
  62  * @brief Insert a cstring at the given index.
  63  * @param string The cstring to be inserted.
  64  * @param pos The index to insert at.
  65  * @param n Up to this many characters will be used from the string.
  66  * @post A buffer is allocated.
  67  * @post If the old buffer was too small, it is enlarged.
  68  * @post The string is inserted at the given index.
  69  */
  70 void String::insert(size_t pos, const char *string, size_t n)
  71 {
  72   if (n == 0) return;
  73   if (pos && !hasIndex(pos-1)) return;
  74   
  75   size_t slen = (n == npos) ? strlen(string) : n;
  76 
  77   AboutToModify(length() + slen);
  78   memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
  79   std::copy(string, string + slen, Buf() + pos);
  80   addLength(slen);
  81 }

  83 /**
  84  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  85  * @param pos The index to replace at.
  86  * @param string The cstring to replace with.
  87  * @param n How many characters to use from string.
  88  */
  89 void String::replace(size_t pos, const char *string, size_t n)
  90 {
  91   if (n == 0) return;
  92   if (pos && !hasIndex(pos-1)) return;
  93 
  94   size_t slen = (n == npos) ? strlen(string) : n;
  95   size_t newlen = pos + slen;
  96   
  97   if (newlen >= length()) {
  98     AboutToModify(newlen);
  99   } else {
 100     newlen = length();
 101     getOwnCopy();
 102   }
 103   std::copy(string, string + slen, Buf() + pos);
 104   setLength(newlen);
 105 }

 107 std::istream& operator>>(std::istream& is, String& string) {
 108   char ch;
 109   string = "";    // empty string, will build one char at-a-time
 110   is >> ch;    // whitespace skipped, first non-white char in ch
 111 
 112   if (!is.fail())
 113   {
 114     do
 115     {
 116       string += ch;
 117       is.get(ch);
 118     } while (!is.fail() && !isspace(ch));
 119 
 120     if (isspace(ch))    // put whitespace back on the stream
 121     {
 122       is.putback(ch);
 123     }
 124   }
 125   return is;
 126 }

 139 /*
 140  * @brief Split a string by a delimiter
 141  * @param delim The delimiter to use
 142  * @param limit Optional param which specified the max number of elements to return
 143  * @note Excess whitespace is trimmed similar to Ruby's split.
 144  */
 145 Array<String> String::split(const String& delim, size_t limit) const {
 146   if (!length()) return Array<String>();
 147   if (limit == 1) {
 148     String list[] = {*this};
 149     return Array<String>(list, 1);
 150   }
 151 
 152   const String space(' ');
 153   Array<String> array;
 154   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 155   String str(*this);
 156   size_t pos = 0;
 157 
 158   while (array.size() < limit - 1) {
 159     // Trim out left whitespace
 160     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 161 
 162     // All done when there's nothing left
 163     if (!str) break;
 164 
 165     if ((pos = str.find(delim)) == npos)
 166       pos = str.length();
 167     array << str(0, pos);
 168     str += pos + delim.length();
 169   }
 170 
 171   // Add on extra
 172   if (limit != npos) {
 173     // Trim out left whitespace
 174     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 175     array << str;
 176   }
 177 
 178   return array;
 179 }

 199 String& String::chomp() {
 200   if (length() && (*this)[length() - 1] == '\n') --(*this);
 201   if (length() && (*this)[length() - 1] == '\r') --(*this);
 202   return *this;
 203 }

 207   while (length() && isspace((*this)[0])) { ++offset; subLength(1); }

 213 size_t String::find(const String& str) const {
 214   if (length() >= str.length()) {
 215     const size_t last_pos = length() - str.length();
 216     for (size_t pos = 0; pos <= last_pos; ++pos)
 217       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 218         return pos;
 219   }
 220   return npos;
 221 }

 223 size_t String::rfind(const String& str, const size_t lpos) const {
 224   if (length() >= str.length()) {
 225     const size_t last_pos = length() - str.length();
 226     for (size_t pos = last_pos; pos + 1 > lpos; --pos)
 227       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 228         return pos;
 229   }
 230   return npos;
 231 }

 233 String String::sub(const String& search, const String& replacement, int limit) const {
 234   String newStr(length()), search_str(*this);
 235   size_t pos = 0;
 236   int cnt = 0;
 237   // Search in our const string so that stuff like \->\\ doesnt become \\\, \\\\ etc.
 238   while ((pos = search_str.find(search)) != npos) {
 239     newStr += search_str(0, pos);
 240     newStr += replacement;
 241     search_str += pos + search.length();
 242     if (limit != -1 && ++cnt == limit) break;
 243   }
 244   // Left over
 245   if (search_str.length())
 246     newStr += search_str;
 247   return newStr;
 248 }

 262 /**
 263  * @brief Splits off tokens
 264  * @param str The string to split a token off of
 265  * @param delim The delimiter to tokenize with
 266  * This will modify the given string to remove the first token, which is returned.
 267  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 268  */
 269 String newsplit(String& str, char delim)
 270 {
 271   if (!str.length()) return "";
 272   size_t pos = str.find(delim);
 273   if (pos == String::npos)
 274     pos = str.length();
 275 
 276   String first(str(0, pos));
 277 
 278   /* Trim out runs of whitespaces */
 279   if (delim == ' ') {
 280     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 281       ++pos;
 282   }
 283 
 284 //  str = str(pos + 1, str.length() - pos + 1);
 285   str += pos + 1;
 286   return first;
 287 }
