  28 int Stream::seek(int offset, int whence) {
  29   int newpos;
  30 
  31   switch (whence) {
  32     case SEEK_SET:
  33       newpos = offset;
  34       break;
  35     case SEEK_CUR:
  36       newpos = tell() + offset;
  37       break;
  38     case SEEK_END:
  39       newpos = capacity() - offset;
  40       break;
  41     default:
  42       newpos = tell();
  43   }
  44   if (newpos < 0)
  45     newpos = 0;
  46   else if (static_cast<size_t>(newpos) > capacity())
  47     newpos = capacity();
  48   pos = newpos;
  49 
  50   return newpos;
  51 }

  53 String Stream::read(size_t maxSize, char delim) {
  54   size_t toRead = std::min(maxSize, length() - pos);
  55 
  56   /* No need to split the string, return a substring */
  57   if (!delim) {
  58     String ret(str(pos, toRead));
  59     pos += toRead;
  60     return ret;
  61   }
  62 
  63   /* Must reconstruct a new string */
  64   String ret(toRead);
  65 
  66   while ((ret.length() < toRead) && (str[pos] != delim)) {
  67     ret += str[pos++];
  68   }
  69   if (str.hasIndex(pos) && str[pos] == delim) {
  70     ret += delim;
  71     ++pos;
  72   }
  73 
  74   return ret;

 137 int Stream::writeFile(const int fd) const
 138 {
 139 #ifdef HAVE_MMAP
 140   /* Write to end of file to make its size match */
 141   /* Seek past wherever it is at now plus our length */
 142   if (ftruncate(fd, length())) return 1;
 143   if (!length())
 144     return 0;
 145 
 146   void* map = mmap(0, length(), PROT_READ|PROT_WRITE, MAP_SHARED, fd, 0);
 147 
 148   if (map == MAP_FAILED) return 1;
 149 
 150   std::memcpy(map, str.data(), length());
 151 
 152   if (munmap(map, length()) == -1) return 1;
 153 #else
 154   FILE *f = fdopen(fd, "wb");
 155   if (f == NULL)
 156     return 1;
 157 
 158   if ((fwrite(str.data(), 1, length(), f) != length()) || (fflush(f))) {
 159     fclose(f);
 160     return 1;
 161   }
 162   fclose(f);
