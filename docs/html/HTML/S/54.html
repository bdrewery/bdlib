  45 int String::compare(const String& str, size_t n, size_t start) const
  46 {
  47   const size_t my_len = length();
  48   /* Same string? */
  49   if (data() == str.data() && my_len == str.length())
  50     return 0;
  51 
  52   const size_t slen = n ? std::min(str.length(), n) : str.length();
  53   const size_t len = std::min(my_len - start, slen);
  54   const int diff = memcmp(begin() + start, str.begin(), len);
  55   if (diff)
  56     return diff;
  57   else if (n)
  58     return std::min(my_len - start, n) - slen;
  59   else
  60     return (my_len - start) - slen;
  61 }

  65 /**
  66  * @brief Insert a cstring at the given index.
  67  * @param string The cstring to be inserted.
  68  * @param pos The index to insert at.
  69  * @param n Up to this many characters will be used from the string.
  70  * @post A buffer is allocated.
  71  * @post If the old buffer was too small, it is enlarged.
  72  * @post The string is inserted at the given index.
  73  */
  74 void String::insert(size_t pos, const char *string, size_t n)
  75 {
  76   if (n == 0) return;
  77   if (pos && !hasIndex(pos-1)) return;
  78   
  79   size_t slen = (n == npos) ? strlen(string) : n;
  80 
  81   AboutToModify(length() + slen);
  82   memmove(Buf() + pos + slen, Buf() + pos, length() - pos);
  83   std::copy(string, string + slen, Buf() + pos);
  84   addLength(slen);
  85 }

  87 /**
  88  * @brief Replaces the given index with the given cstring up to the given amount of characters.
  89  * @param pos The index to replace at.
  90  * @param string The cstring to replace with.
  91  * @param n How many characters to use from string.
  92  */
  93 void String::replace(size_t pos, const char *string, size_t n)
  94 {
  95   if (n == 0) return;
  96   if (pos && !hasIndex(pos-1)) return;
  97 
  98   size_t slen = (n == npos) ? strlen(string) : n;
  99   size_t newlen = pos + slen;
 100   
 101   if (newlen >= length()) {
 102     AboutToModify(newlen);
 103   } else {
 104     newlen = length();
 105     getOwnCopy();
 106   }
 107   std::copy(string, string + slen, Buf() + pos);
 108   setLength(newlen);
 109 }

 111 std::istream& operator>>(std::istream& is, String& string) {
 112   char ch;
 113   string = "";    // empty string, will build one char at-a-time
 114   is >> ch;    // whitespace skipped, first non-white char in ch
 115 
 116   if (!is.fail())
 117   {
 118     do
 119     {
 120       string += ch;
 121       is.get(ch);
 122     } while (!is.fail() && !isspace(ch));
 123 
 124     if (isspace(ch))    // put whitespace back on the stream
 125     {
 126       is.putback(ch);
 127     }
 128   }
 129   return is;
 130 }

 143 /*
 144  * @brief Split a string by a delimiter
 145  * @param delim The delimiter to use
 146  * @param limit Optional param which specified the max number of elements to return
 147  * @note Excess whitespace is trimmed similar to Ruby's split.
 148  */
 149 Array<String> String::split(const String& delim, size_t limit) const {
 150   if (!length()) return Array<String>();
 151   if (limit == 1) {
 152     String list[] = {*this};
 153     return Array<String>(list, 1);
 154   }
 155 
 156   const String space(' ');
 157   Array<String> array;
 158   // Use a temporary - is a fast reference and never modified, so no COW ever used.
 159   String str(*this);
 160   size_t pos = 0;
 161 
 162   while (array.size() < limit - 1) {
 163     // Trim out left whitespace
 164     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 165 
 166     // All done when there's nothing left
 167     if (!str) break;
 168 
 169     if ((pos = str.find(delim)) == npos)
 170       pos = str.length();
 171     array << str(0, pos);
 172     str += pos + delim.length();
 173   }
 174 
 175   // Add on extra
 176   if (limit != npos) {
 177     // Trim out left whitespace
 178     if (delim == space) while (str.length() && str[0] == ' ') ++str;
 179     array << str;
 180   }
 181 
 182   return array;
 183 }

 203 String& String::chomp() {
 204   if (length() && (*this)[length() - 1] == '\n') --(*this);
 205   if (length() && (*this)[length() - 1] == '\r') --(*this);
 206   return *this;

 209 String& String::trim() {
 210   // ltrim
 211   while (length() && isspace((*this)[0])) { ++offset; subLength(1); }
 212   // rtrim
 213   while (length() && isspace((*this)[length() - 1])) subLength(1);
 214   return *this;

 217 size_t String::find(const String& str) const {
 218   if (length() >= str.length()) {
 219     const size_t last_pos = length() - str.length();
 220     for (size_t pos = 0; pos <= last_pos; ++pos)
 221       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 222         return pos;
 223   }
 224   return npos;
 225 }

 227 size_t String::ifind(const String& str) const {
 228   String str_upper(str), this_upper(*this);
 229   std::transform(str.begin(), str.end(), str_upper.begin(), ::toupper);
 230   std::transform(this->begin(), this->end(), this_upper.begin(), ::toupper);
 231   return this_upper.find(str_upper);
 232 }

 241 size_t String::rfind(const String& str, const size_t lpos) const {
 242   if (length() >= str.length()) {
 243     const size_t last_pos = length() - str.length();
 244     for (size_t pos = last_pos; pos + 1 > lpos; --pos)
 245       if (str[0] == (*this)[pos] && !memcmp(begin() + pos, str.begin(), std::min(str.length(), length() - pos)))
 246         return pos;
 247   }
 248   return npos;

 251 String String::sub(const String& search, const String& replacement, int limit) const {
 252   String newStr(length()), search_str(*this);
 253   size_t pos = 0;
 254   int cnt = 0;
 255   // Search in our const string so that stuff like \->\\ doesnt become \\\, \\\\ etc.
 256   while ((pos = search_str.find(search)) != npos) {
 257     newStr += search_str(0, pos);
 258     newStr += replacement;
 259     search_str += pos + search.length();
 260     if (limit != -1 && ++cnt == limit) break;
 261   }
 262   // Left over
 263   if (search_str.length())
 264     newStr += search_str;
 265   return newStr;
 266 }

 268 String String::subst(HashTable<String, String> hashes) const {
 269   String newStr(*this);
 270   Array<String> keys(hashes.keys());
 271   Array<String> values(hashes.values());
 272   for (size_t i = 0; i < keys.length(); ++i) {
 273     const String key(keys[i]), value(values[i]);
 274     //::printf("%s -> %s\n", key.c_str(), value.c_str());
 275     newStr = newStr.sub(key, value);
 276   }
 277   return newStr;

 280 /**
 281  * @brief Splits off tokens
 282  * @param str The string to split a token off of
 283  * @param delim The delimiter to tokenize with
 284  * This will modify the given string to remove the first token, which is returned.
 285  * If no more tokens exist, the remaining string is returned and the string becomes "". Future calls will return a blank String("")
 286  */
 287 String newsplit(String& str, char delim)
 288 {
 289   if (!str.length()) return "";
 290   size_t pos = str.find(delim);
 291   if (pos == String::npos)
 292     pos = str.length();
 293 
 294   String first(str(0, pos));
 295 
 296   /* Trim out runs of whitespaces */
 297   if (delim == ' ') {
 298     while ((str.begin() + pos + 1) < str.end() && str[pos + 1] == ' ')
 299       ++pos;
 300   }
 301 
 302 //  str = str(pos + 1, str.length() - pos + 1);
 303   str += pos + 1;
 304   return first;
 305 }
