  57 String ScriptInterpTCL::eval(const String& script) {
  58   if (Tcl_EvalEx(interp, script.c_str(), script.length(), TCL_EVAL_GLOBAL) == TCL_OK) {
  59     Tcl_Obj* value = Tcl_GetObjResult(interp);
  60     return tcl_to_c_cast<String>::from(value, this);

  66 ScriptInterp::LoadError ScriptInterpTCL::loadScript(const String& fileName, String& resultStr) {
  67   if (fileName.rfind(".tcl", fileName.length() - 4) == String::npos)
  68     return SCRIPT_LOAD_WRONG_INTERP;
  69   if (Tcl_EvalFile(interp, *fileName) != TCL_OK) {
  70     Tcl_Obj* value = Tcl_GetObjResult(interp);
  71     resultStr = tcl_to_c_cast<String>::from(value, this);
  72     return SCRIPT_LOAD_ERROR;
  73   }
  74   return SCRIPT_LOAD_OK;
  75 }

  80 int ScriptInterpTCL::_createCommand_callback(ClientData clientData, Tcl_Interp *interp, int objc, Tcl_Obj *CONST objv[]) {
  81   String cmdName(tcl_to_c_cast<String>::from(objv[0], nullptr));
  82   script_cmd_handler_clientdata* ccd = CmdHandlerData[cmdName];
  83 
  84   if ((size_t(objc) - 1) < ccd->callbackParamMin || size_t(objc) - 1 > ccd->callbackParamMax) {
  85     String errorResult(String::printf("Wrong # args. Expected %zu, got %d.", ccd->callbackParamMin, objc - 1));
  86     if (ccd->usage)
  87       errorResult += String::printf(" Should be \"%s %s\"", cmdName.c_str(), ccd->usage);
  88     Tcl_SetObjResult(interp, c_to_tcl_cast<String>::from(errorResult, interp));
  89     return TCL_ERROR;
  90   }
  91 
  92   try {
  93     ccd->callback_proxy->call(objc, reinterpret_cast<void* CONST*>(objv), ccd->si, interp);
  94   } catch (bd::String& e) {
  95     Tcl_SetObjResult(interp, c_to_tcl_cast<String>::from(e, interp));
  96     return TCL_ERROR;
  97   } catch (...) {
  98     Tcl_SetObjResult(interp, c_to_tcl_cast<const char*>::from("Unhandled exception.", interp));
  99     return TCL_ERROR;
 100   }
 101   return TCL_OK;
 102 }

 104 void ScriptInterpTCL::setupTraces(const String& varName, ClientData var, Tcl_VarTraceProc* get, Tcl_VarTraceProc* set, link_var_hook hook_func) {
 105   Tcl_SetVar(interp, *varName, "", TCL_GLOBAL_ONLY);
