  45 template <class T, class Allocator = std::allocator<T> >
  46 /**
  47  * @class ArrayRef
  48  * @brief Helps the String and Array classes with reference counting
  49  * @todo look into something like a string object made of string pointers, and << displays all of the pointed to objects. (this would save copying to construct immuatable concatenated strings, but would only work with the << stream
  50  */
  51 class ArrayRef {
  52   public:
  53     typedef typename Allocator::pointer iterator;
  54     mutable Allocator alloc;
  55     mutable size_t size; //Capacity of buffer
  56     union {
  57       mutable iterator buf;
  58       mutable T buf_local[16];
  59     };
  60     mutable std::atomic<int> refs; //References
  61 
  62     ArrayRef(const Allocator& allocator = Allocator()) : alloc(allocator), size(0), buf(nullptr), refs(1) {};
  63     ~ArrayRef() {
  64       if (buf) {
  65         FreeBuf(buf);
  66       }
  67     };
  68     /**
  69      * @brief Ensure that the buffer capacity() is >= newSize; else grow/copy into larger buffer.
  70      * @param newSize A size that we need to Allocate the buffer to.
  71      * @param scaling_factor How much to multiply the size by to help avoid later resizing
  72      * @param offset The offset of the old buffer so we know where to start
  73      * @param sublen The length of the subarray in use
  74      * @pre newSize is > 0 (assumed as size_t is unsigned)
  75      * @post The buffer is at least nsize bytes long.
  76      * @post If the buffer had to grow, the old data was deep copied into the new buffer and the old deleted.
  77      */
  78     void Reserve(size_t newSize, double scaling_factor, size_t& offset, size_t sublen) const
  79     {
  80       /* Don't new if we already have enough room! */
  81       if (size < newSize) {
  82         newSize = std::max(size_t(size * scaling_factor), newSize);
  83 
  84         iterator newbuf = alloc.allocate(newSize, buf);
  85 
  86         if (newbuf != buf) {
  87           // Initialize new memory
  88           std::uninitialized_fill(newbuf, newbuf + newSize, T());
  89           if (buf) {
  90             /* Copy old buffer into new - only copy the subarray */
  91             std::copy(buf + offset, buf + offset + sublen, newbuf);
  92             FreeBuf(buf);
  93           }
  94           buf = newbuf;
  95           size = newSize;
  96           offset = 0;
  97         }
  98       } else if ((size - offset) < newSize) {
  99         // There's enough room in the current buffer, but we're offsetted/shifted to a point where there's no room left
 100         // Shift everything to the beginning and reset the offset
 101         /* Only copy the subarray */
 102         std::memmove(static_cast<void*>(buf), static_cast<void*>(buf + offset), sublen);
 103         offset = 0;
 104       }
 105     }
 106 
 107     /**
 108      * @brief Free's up the allocated buffer.
 109      * @param p The buffer to be free'd
 110      * @post The buffer is deleted.
 111      * @todo Implement mempool here.
 112      */
 113     inline void FreeBuf(iterator p) const {
 114       for (iterator i = p; i != p + size; ++i) {
 115         alloc.destroy(i);
 116       }
 117       alloc.deallocate(p, size);
 118     }
 119 
 120     /**
 121      * @brief Is this ReferenceCountedArray shared?
 122      */
 123     inline bool isShared() const { return refs > 1; };
 124   private:
 125     // No copying allowed
 126     ArrayRef(const ArrayRef&) = delete;
 127     ArrayRef& operator=(const ArrayRef&) = delete;
 128 };

 131 /**
 132  * @class Slice
 133  * @brief Safe subarray reading and writing.
 134  * This class should be optimized away and fully inlined such that:
 135  * ReferenceCountedArray s("look over there");
 136  * s(0, 4) = "blah"';
 137  * Should be rewritten as:
 138  * s.replace(0, "blah", 4);
 139  */
 140 class Slice {
 141   private:
 142     T& rca;
 143     int start;
 144     int len;
 145 
 146     Slice() = delete;
 147 
 148   public:
 149     Slice(T& _rca, int _start, int _len) : rca(_rca), start(_start), len(_len) {};
 150     Slice(const Slice& slice) : rca(slice.rca), start(slice.start), len(slice.len) {};
 151 
 152     /**
 153      * @brief return a new (const) slice
 154      */
 155     inline operator T() const {
 156       T newArray(rca);
 157       newArray.slice(start, len);
 158       return newArray;
 159     };
 160 
 161     friend void swap(Slice& a, Slice& b) {
 162       using std::swap;
 163 
 164       swap(a.alloc, b.alloc);
 165       swap(a.rca, b.rca);
 166       swap(a.start, b.start);
 167       swap(a.len, b.len);
 168     }
 169 
 170     /**
 171      * @brief Assign a Slice to a Slice
 172      */
 173     inline Slice& operator=(const Slice& slice) {
 174       (*this) = T(slice);
 175       return (*this);
 176     }
 177 
 178     /**
 179      * @brief Assign to a Slice
 180      * @todo This needs to account for negative start/len
 181      */
 182     inline Slice& operator=(const T& array) {
 183       rca.replace(start, array, len);
 184       return (*this);
 185     }

 197 template <class T, class Allocator = std::allocator<T> >
 198 /**
 199  * @class ReferenceCountedArray
 200  * @brief Common template base class for String and Array
 201  */
 202 class ReferenceCountedArray : public ReferenceCountedArrayBase {
 203   public:
 204     typedef T value_type;
 205 
 206     typedef typename Allocator::size_type             size_type;
 207     typedef typename Allocator::difference_type       difference_type;
 208     typedef typename Allocator::pointer               pointer;
 209     typedef typename Allocator::const_pointer         const_pointer;
 210     typedef typename Allocator::reference             reference;
 211     typedef typename Allocator::const_reference       const_reference;
 212     typedef typename Allocator::pointer               iterator;
 213     typedef typename Allocator::const_pointer         const_iterator;
 214     typedef std::random_access_iterator_tag           iterator_category;
 215     typedef std::reverse_iterator<const_iterator>     const_reverse_iterator;
 216     typedef std::reverse_iterator<iterator>           reverse_iterator;
 217 
 218   private:
 219     /**
 220      * @brief Detach from the shared reference.
 221      * This is only called when losing the old buffer or when modifying the buffer (and copy-on-write is used)
 222      * @note This does not free the old reference, as it is still in use
 223      */
 224     void doDetach() const {
 225       decRef();
 226       Ref = nullptr;
 227       sublen = 0;
 228       offset = 0;
 229       my_hash = 0;
 230     }
 231     /**
 232      * @brief Increment our reference counter.
 233      */
 234     inline int incRef() const { return Ref ? ++Ref->refs : 0; };
 235 
 236     /**
 237      * @brief Decrement our reference counter.
 238      */
 239     inline int decRef() const { return Ref ? --Ref->refs : 0; };
 240 
 241   protected:
 242     /**
 243      * @brief Set the lengths to the specified length
 244      * @param newLen the new length to set to
 245      */
 246     inline void setLength(size_t newLen) const { sublen = newLen; my_hash = 0; };
 247 
 248     /**
 249      * @sa setLength()
 250      */
 251     inline void addLength(size_t diff) const { sublen += diff; my_hash = 0; };
 252 
 253     /**
 254      * @sa setLength()
 255      */
 256     inline void subLength(size_t diff) const { sublen -= diff; my_hash = 0; };
 257 
 258     /**
 259      * @brief Mutable Ref->buf+offset reference for use internally
 260      */
 261     //pointer mdata() const { return Buf() + offset; };
 262 
 263     /**
 264      * @brief Mutable Ref->buf reference for use internally
 265      */
 266     inline pointer Buf(size_t pos = 0) const { return Ref ? &Ref->buf[offset + pos] : nullptr; };
 267 
 268     /**
 269      * @brief Ref->buf reference for use internally
 270      */
 271     inline const_pointer constBuf(size_t pos = 0) const { return Buf(pos); };
 272   private:
 273     Allocator alloc;
 274     /**
 275      * @brief The array reference for reference counting
 276      * This is mutable so that Ref->refs can be modified, which really is mutable
 277      */
 278     mutable ArrayRef<value_type, Allocator> *Ref;
 279   protected:
 280     /**
 281      * Return the real buffer's start point, without accounting for offset. This is used for cleaning the buffer when needed.
 282      */
 283     inline const_pointer real_begin() const { return Ref ? Ref->buf : nullptr; };
 284 
 285     /**
 286      * This is for subarrays: so we know where the subarray starts.
 287      */
 288     mutable size_t offset;
 289     /**
 290      * This is for subarrays: so we know where the subarray ends.
 291      */
 292     mutable size_t sublen;
 293 
 294     /**
 295      * Cache of current hash() result. 0 if stale
 296      */
 297     mutable size_t my_hash;
 298 
 299   private:
 300     /**
 301      * @brief Detach from the reference
 302      * This is called when the old buffer is no longer needed.
 303      * ie, operator=() was called.
 304      */
 305     void Detach() {
 306       if (isShared()) {
 307         doDetach();
 308       } else {
 309         setLength(0);
 310         offset = 0;
 311         my_hash = 0;
 312       }
 313     }
 314 
 315     /**
 316      * @brief Free up our reference if we have the last one.
 317      * @post The reference counter is decremented.
 318      * @post If this was the last Reference, it is free'd
 319      * This is only called in ~Array() and operator=(Array&).
 320      * It checks whether of not this Array was the last reference to the buffer, and if it was, it removes it.
 321      */
 322     inline void CheckDeallocRef() {
 323       if (Ref && decRef() < 1) {
 324         delete Ref;
 325         Ref = nullptr;
 326       }
 327     }
 328 
 329     //    void COW(size_t) const;
 330     /**
 331      * @brief Ensure that our internal buffer is unshared.
 332      * @param n Create/Grow the buffer to this size.
 333      * @pre n is > 0; this is assumed due to size_t being unsigned though.
 334      * @post The internal buffer/data is unshared
 335      * @post The buffer is at least size n.
 336      * @post The buffer is deep copied to a new buffer.
 337      *
 338      * Ensure that our internal buffer is unshared.
 339      * If needed, performs a deep copy into a new buffer (COW).
 340      * Also take a hint size n of the new ReferenceCountedArray's size as to avoid needless copying/allocing.
 341      */
 342     void COW(size_t n) const {
 343       const_pointer oldBuf = constBuf();
 344       const size_t oldLength = length();
 345 
 346       doDetach(); //Detach from the shared reference
 347       Reserve(std::max(oldLength, n), _rca_cow_scaling_factor); //Will set capacity()/size
 348       std::copy(oldBuf, oldBuf + oldLength, Buf());
 349       setLength(oldLength);
 350     }
 351 
 352   protected:
 353     /**
 354      * @brief Force COW if needed
 355      * @post The array is no longer shared, if it was.
 356      */
 357     inline void getOwnCopy() const { AboutToModify(length()); };
 358 
 359     /**
 360      * @brief Warn the reference counting that it may need to COW
 361      * @post The buffer is detached/COW, and possibly larger
 362      */
 363     inline void AboutToModify(size_t n) const {
 364       my_hash = 0;
 365       if (isShared())
 366         COW(n); // Clears the offset
 367       else {
 368         Reserve(n, _rca_cow_scaling_factor);
 369         /* Shift the offset away */
 370       }
 371     }
 372   public:
 373     ReferenceCountedArray(const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(nullptr), offset(0), sublen(0), my_hash(0) {
 374     };
 375     ReferenceCountedArray(const ReferenceCountedArray& rca) : ReferenceCountedArrayBase(), alloc(rca.alloc), Ref(rca.Ref), offset(rca.offset), sublen(rca.sublen), my_hash(rca.my_hash) { incRef(); };
 376     ReferenceCountedArray(ReferenceCountedArray&& rca) : ReferenceCountedArrayBase(), alloc(std::move(rca.alloc)), Ref(std::move(rca.Ref)), offset(std::move(rca.offset)), sublen(std::move(rca.sublen)), my_hash(std::move(rca.my_hash)) {
 377       rca.alloc = Allocator();
 378       rca.Ref = nullptr;
 379       rca.offset = 0;
 380       rca.sublen = 0;
 381       rca.my_hash = 0;
 382     };

 384     /**
 385      * @brief Create an empty container with at least the specified bytes in size.
 386      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 387      * @post This ReferenceCountedArray's memory will also never be shrunk.
 388      * @post A buffer has been created.
 389      *
 390      * The idea behind this is that if a specific size was asked for, the buffer is like
 391      * a char buf[N];
 392      */
 393     explicit ReferenceCountedArray(const size_t newSize, const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(nullptr), offset(0), sublen(0), my_hash(0) {
 394       if (newSize) {

 396       }
 397     };
 398     /**
 399      * @brief Create a container filled with n copies of the given value.
 400      * @param newSize Reserve at least this many bytes for this ReferenceCountedArray.
 401      * @param value The value to populate the array with
 402      * @post This ReferenceCountedArray's memory will also never be shrunk.
 403      * @post A buffer has been created.
 404      *
 405      */
 406     ReferenceCountedArray(const size_t newSize, const value_type value, const Allocator& allocator = Allocator()) : ReferenceCountedArrayBase(), alloc(allocator), Ref(nullptr), offset(0), sublen(0), my_hash(0) {
 407       if (newSize) {
 408         Reserve(newSize);
 409 
 410         for (size_t i = 0; i < newSize; ++i) {
 411           *(Buf(i)) = value;
 412         }
 413         this->setLength(newSize);
 414       }
 415     }

 531     void resize(size_t len, value_type value = value_type()) {
 532       if (len < this->length()) {
 533         this->subLength(this->length() - len);
 534       } else {

 536         for (size_t i = 0; i < (len - this->length()); ++i)

 538         this->addLength(len - this->length());

 624    /**
 625      * @brief Return a hash of every element in the array. Cache result as well.
 626      * @note DJB's hash function
 627      */
 628     virtual size_t hash() const {
 629       if (my_hash != 0) return my_hash;
 630       HashType hasher;
 631       size_t _hash = 5381;
 632 
 633       for(size_t i = 0; i < this->length(); ++i)
 634         _hash = ((_hash << 5) + _hash) + hasher(this->data()[i]);
 635       return (my_hash = (_hash & 0x7FFFFFFF));
 636     }

 662     /**
 663      * @brief Checks if the buffer has the given index or not.
 664      * @return Boolean true/false as to whether or not index exists.
 665      * @param pos Index to check.
 666      */
 667     inline bool hasIndex(size_t pos) const {
 668       if (pos >= (offset + length())) {
 669 #ifdef DEBUG
 670         ::printf("ATTEMPT TO ACCESS INDEX %zu/%zu\n", pos, offset + length());
 671 #endif
 672         return 0;
 673       }
 674       return (pos < length());
 675     };

 677     /**
 678      * @sa at()
 679      * Unlinke at() this is unchecked.
 680      */
 681     inline value_type read(size_t pos) const { return *(constBuf(pos)); };
 682 
 683     /**
 684      * @brief Write an item to the given index
 685      */
 686     inline void write(size_t pos, value_type item) {
 687       getOwnCopy();
 688       *(Buf(pos)) = item;
 689     };

 691     /**
 692      * @brief Safe element access operator
 693      * @todo This is only called on a (const) ReferenceCountedArray, but should for a ReferenceCountedArray as well.
 694      */
 695     inline value_type operator[](size_t pos) const { return read(pos); };

 697     /**
 698      * @class Cref
 699      * @brief Safe element reading and writing.
 700      * This class should be optimized away and fully inlined such that:
 701      * ReferenceCountedArray s;
 702      * s[0] = 'a';
 703      * Should be rewritten as:
 704      * s.write(0, 'a');
 705      */
 706     class Cref {
 707       private:
 708         friend class ReferenceCountedArray;
 709         ReferenceCountedArray& rca;
 710         size_t k;
 711 
 712         /**
 713          * @brief Used by Cref operator[]
 714          */
 715         Cref(ReferenceCountedArray& _rca, size_t pos) : rca(_rca), k(pos) {};
 716         Cref() = delete;
 717 
 718       public:
 719         Cref(const Cref& cref) : rca(cref.rca), k(cref.k) {};
 720         inline Cref& operator=(const Cref& cref) {
 721           (*this) = value_type(cref);
 722           return (*this);
 723         }
 724 
 725         public:
 726         /**
 727          * @sa ReferenceCountedArray::operator[]
 728          */
 729         inline operator value_type() const { return rca.read(k); };
 730 
 731         /**
 732          * Stroustrup shows using this as void with no return value, but that breaks chaining a[n] = b[n] = 'b';
 733          */
 734         inline Cref& operator=(value_type c) {
 735           rca.write(k, c);
 736           return (*this);
 737         };
 738     };

 740     /**
 741      * @brief Returns 'Cref' class for safe (cow) writing.
 742      * @sa Cref
 743      */
 744     inline Cref operator[](size_t pos) { return Cref(*this, pos); };

 746     /**
 747      * @brief Returns the character at the given index.
 748      * @return The character at the given index.
 749      * @param pos Index to return.
 750      * @pre The index must exist.
 751      * @sa operator[]()
 752      * @todo Perhaps this should throw an exception if out of range?
 753      */
 754     inline value_type at(size_t pos) const { return hasIndex(pos) ? (*this)[pos] : 0; };
 755 
 756     /**
 757      * @param start The offset to begin the subarray from (indexed from 0)
 758      * @param len The length of the subarray to return
 759      * The returned slice is a reference to the original array until modified.
 760      */
 761     void slice(int start, int len = -1) {
 762       if (len == -1) len = int(length()) - start;
 763       // Start is after the end, set us to an empty array
 764       if (start >= static_cast<signed>(length())) {
 765         offset = length();
 766         setLength(0);
 767         return;
 768       }
 769 
 770       //Count backwards from the end
 771       if (start < 0)
 772         start = length() + start;
 773       // Start was before the beginning, just reset to the beginning
 774       if (start < 0)
 775         start = 0;
 776 
 777       offset += start;
 778       //If the length of the subarray exceeds the end of the array, truncate to the end of the array
 779       if (start + len >= static_cast<signed>(length()))
 780         len = length() - start;
 781       // If the length is negative, stop from counting backwards from the end
 782       else if (len < 0)
 783         len = length() - start + len;
 784       setLength(len);
 785     }

 787     /**
 788      * @brief Appends one item to end of buffer.
 789      * @param item The item to be appended.
 790      * @post The buffer is allocated.
 791      * @post The item is appended at the end of the buffer.
 792      * This is the same as inserting the item at the end of the buffer.
 793      */
 794     inline void append(const_reference item) { insert(length(), item); };
 795     /**
 796      * @brief Appends given rca to the end of buffer
 797      * @param rca The rca to be appended.
 798      * @param n How many characters to copy from the ReferenceCountedArray object.
 799      * @post The buffer is allocated.
 800      * This is the same as inserting the rca at the end of the buffer.
 801      */
 802     inline void append(const ReferenceCountedArray& rca, size_t n = npos) { insert(length(), rca, n); };
 803

 805     /**
 806      * @brief Inserts a ReferenceCountedArray object into our buffer
 807      * @param pos The index to insert at.
 808      * @param rca The rca to insert.
 809      * @param n The length to insert.
 810      * @post The buffer contains n items from rca inserted at index pos.
 811      */
 812     void insert(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 813       if (n == 0) return;
 814       if (pos && !hasIndex(pos-1)) return;
 815 
 816       size_t slen = rca.length();
 817 
 818       /* New rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 819       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 820         *this = rca;
 821         return;
 822       }
 823 
 824       if (n == npos || n > slen)
 825         n = slen;
 826       slen -= slen - n;
 827       AboutToModify(length() + slen);
 828       std::memmove(static_cast<void*>(Buf() + pos + slen), static_cast<void*>(Buf() + pos), length() - pos);
 829       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 830       addLength(slen);
 831     }
 832 
 833     /**
 834      * @brief Insert an item at the given index.
 835      * @param pos The index to insert at.
 836      * @param item The item to be inserted.
 837      * @post A buffer is allocated.
 838      * @post If the old buffer was too small, it is enlarged.
 839      * @post The item is inserted at the given index.
 840      */
 841     void insert(size_t pos, const_reference item)
 842     {
 843       if (pos && !hasIndex(pos-1)) return;
 844 
 845       AboutToModify(length() + 1);
 846       std::memmove(static_cast<void*>(Buf() + pos + 1), static_cast<void*>(Buf() + pos), length() - pos);
 847       *(Buf(pos)) = item;
 848       addLength(1);
 849     }

 851     /**
 852      * @brief Replace the given index with the given item.
 853      * @param pos The index to replace.
 854      * @param item The item to replace with.
 855      * @post The given index has been replaced.
 856      * @post COW is done if needed.
 857      */
 858     void replace(size_t pos, const_reference item) {
 859       if (pos && !hasIndex(pos-1)) return;
 860 
 861       getOwnCopy();
 862       *(Buf(pos)) = item;
 863     }
 864 
 865     /**
 866      * @brief Replaces n elements in our buffer at index pos with the given ReferenceCountedArray object
 867      * @param pos The index to replace at.
 868      * @param rca The ReferenceCountedArray object to replace with.
 869      * @param n The number of characters to use for the replace.
 870      */
 871     void replace(size_t pos, const ReferenceCountedArray& rca, size_t n = npos) {
 872       if (n == 0) return;
 873       if (pos && !hasIndex(pos-1)) return;
 874 
 875       size_t slen = rca.length();
 876 
 877       /* Replace rca is longer than ours, and inserting at 0, just replace ours with a reference of theirs */
 878       if (pos == 0 && slen > length() && (n == npos || n == slen)) {
 879         *this = rca;
 880         return;
 881       }
 882 
 883       if (n == npos || n > slen)
 884         n = slen;
 885       slen -= slen - n;
 886 
 887       size_t newlen = pos + slen;
 888 
 889       if (newlen >= length()) {
 890         AboutToModify(newlen);
 891       } else {
 892         newlen = length();
 893         getOwnCopy();
 894       }
 895       std::copy(rca.begin(), rca.begin() + slen, Buf() + pos);
 896       setLength(newlen);
 897     }
