  36 template <class Key, class Value>
  37 /**
  38   * @class BinaryTree
  39   * @brief Binary tree data structure
  40   */
  41 class BinaryTree {
  42   class BinaryTreeIterator;
  43 
  44   public:
  45     typedef BinaryTreeIterator iterator;
  46     typedef KeyValue<Key, Value> iterator_type;
  47 
  48   private:
  49     class Node {
  50       public:
  51         Node *left;
  52         Node *right;
  53         iterator_type kv;
  54 
  55         Node(const Key k, const Value v) : left(nullptr), right(nullptr), kv(k, v) { };
  56         Node(const Node& n) : left(n.left ? new Node(*(n.left)) : nullptr),
  57                               right(n.right ? new Node(*(n.right)) : nullptr),
  58                               kv(n.kv) {};
  59     };
  60 
  61     size_t my_size;
  62     Node *root;
  63 
  64     /** 
  65       * @brief Returns a reference to the node with a matching key
  66       * @param search Where to start the search in the tree
  67       * @param key The key to search for
  68       * @return The node searched for, if found, otherwise NULL
  69       */
  70     Node*& fetchNode(Node* const* search, const Key& key) const {
  71       while ((*search) != nullptr) {
  72         if (key < (*search)->kv.key())
  73           search = &(*search)->left;
  74         else if (key > (*search)->kv.key())
  75           search = &(*search)->right;
  76         else {
  77           return const_cast<Node*&>(*search);
  78         }
  79       }
  80       return const_cast<Node*&>(*search);
  81     }
  82 
  83     /** 
  84       * @brief Insert a node into the given tree
  85       * @param search The tree to insert the node into
  86       * @param node The node to insert
  87       * @sa fetchNode
  88       * This first looks for the where the node SHOULD be in the tree.
  89       * It will of course fail, but this is the location to place the node.
  90       */
  91     inline void insertNode(Node** search, Node* const node) {
  92       Node*& insertAt = fetchNode(search, node->kv.key());
  93       insertNode(insertAt);
  94     }
  95 
  96     inline void insertNode(Node*& insertAt, Node* const node) {
  97       insertAt = node;
  98       ++my_size;
  99     }
 100 
 101     /**
 102       * @brief Delete a node from the tree. (possibly recursive)
 103       * @param node A reference to the node to delete.
 104       */
 105     void deleteNode(Node*& node) {
 106       if (node == nullptr)
 107         return;
 108       if (node->left == nullptr) {
 109         Node* temp = node->right;
 110         delete node;
 111         node = temp;
 112         --my_size;
 113       } else if (node->right == nullptr) {
 114         Node* temp = node->left;
 115         delete node;
 116         node = temp;
 117         --my_size;
 118       } else {
 119         //Two children, find max of left subtree and swap
 120         Node** temp = &(node->left);
 121 
 122         while ((*temp)->right != nullptr)
 123           temp = &(*temp)->right;
 124         node->kv = (*temp)->kv;
 125         deleteNode(*temp);
 126       }
 127     }
 128   public:
 129     /** 
 130       * @brief Checks whether the given key is in the tree
 131       * @param key The key to search for
 132       * @return true/false depending on found/not-found
 133       */
 134     bool contains(const Key& key) const {
 135       if (isEmpty())
 136         return false;
 137       if (fetchNode(&root, key))
 138         return true;
 139       return false;
 140     }
 141 
 142     /**
 143       * @brief Returns the current size of the tree
 144       * @return The current size of the tree
 145       */
 146     inline size_t size() const { return my_size; };
 147     inline bool isEmpty() const { return size() == 0; };
 148     inline explicit operator bool() const { return !isEmpty(); };
 149 
 150   public:
 151     BinaryTree() : my_size(0), root(nullptr) {};
 152     BinaryTree(const BinaryTree& tree) : my_size(tree.my_size), root(new Node(*(tree.root))) {};
 153     BinaryTree(BinaryTree&& tree) : my_size(std::move(tree.my_size)), root(std::move(tree.root)) {
 154       tree.root = nullptr;
 155       tree.my_size = 0;
 156     };
 157 
 158     friend void swap(BinaryTree& a, BinaryTree& b) {
 159       using std::swap;
 160 
 161       swap(a.my_size, b.my_size);
 162       swap(a.root, b.root);
 163     }
 164 
 165     BinaryTree& operator=(BinaryTree tree) {
 166       swap(*this, tree);
 167       return *this;
 168     }
 169 
 170     /**
 171       * @brief Destructor
 172       * @sa deleteNode
 173       * This simply deletes the root node over and over until the root is NULL.
 174       * There may possibly be a quicker way that does not use recursion (deleteNode does) but this seems the simplest.
 175       */
 176     virtual ~BinaryTree() { clear(); };
 177 
 178     /**
 179       * @brief insert Key/Value pair into tree
 180       * @param key The key to be inserted (or replaced)
 181       * @param value The value to be inserted (or replaced)
 182       * @post The tree's size is increased by 1 if the element was not already in the tree
 183       */
 184     bool insert(const Key& key, const Value& value) {
 185       Node*& rNode = fetchNode(&root, key);
 186       if (rNode)
 187         rNode->kv = iterator_type(key, value);
 188       else {
 189         Node *newNode = new Node(key, value);
 190         insertNode(rNode, newNode);
 191       }
 192       return true;
 193     }
 194 
 195     /**
 196       * @brief Associate array type accessor (rvalue)
 197       * @sa getValue
 198       * @param key The key to search for
 199       */
 200     inline const Value operator[](const Key& key) const { return getValue(key); }
 201 
 202     /**
 203       * @brief Associate array type accessor (lvalue)
 204       * @param key The key to search for
 205       * @sa insert
 206       * If the key is not in the tree, it is inserted, and the value set to the rvalue given.
 207       */
 208     Value& operator[](const Key& key) {
 209       Node*& node = fetchNode(&root, key);
 210       if (!node)
 211         insertNode(node, new Node(key, Value()));
 212       //The key was inserted at the return node's address! node is now guranteed to be NON-NULL ! */
 213       return node->kv.v;
 214     }
 215 
 216     /** 
 217       * @brief Remove the given key (and its value) from the tree
 218       * @param key The key to be searched/removed
 219       * @return Whether or not the key was removed
 220       */
 221     bool remove(const Key& key) {
 222       Node*& node = fetchNode(&root, key);
 223 
 224       if (node != nullptr) {
 225         deleteNode(node);
 226         return true;
 227       }
 228       return false;
 229     }
 230 
 231     inline void clear() {
 232       while (root != nullptr)
 233         deleteNode(root);
 234     }
 235 
 236     /**
 237       * @brief Find a value in the tree from the key given
 238       * @param key The key to search for
 239       * @return The value of the key searched for, or NULL if not found
 240       */
 241     Value getValue(const Key& key) const {
 242       const Value empty;
 243       if (isEmpty()) return empty;
 244 
 245       Node* node = fetchNode(&root, key);
 246       if (node)
 247         return node->kv.value();
 248       return empty;
 249     }

 269   private:
 270     class BinaryTreeIterator : public Iterator<iterator_type> {
 271       friend class BinaryTree;
 272       typedef BinaryTree<Key, Value> Tree;
 273       private:
 274         Tree* tree;
 275         size_t index;
 276         size_t my_size;
 277         bool reverse;
 278         iterator_type* storage;
 279 
 280         void fillArray(int& i, const Node* node) {
 281           if (node == nullptr) 
 282             return;
 283           fillArray(i, node->left);
 284           storage[i++] = iterator_type(node->kv);
 285           fillArray(i, node->right);
 286         }
 287 
 288         BinaryTreeIterator(Tree* t, Node* node, size_t size, bool _reverse = 0) : Iterator<iterator_type>(),
 289                                                                           tree(t),
 290                                                                           index(_reverse ? size - 1 : 0),
 291                                                                           my_size(size), 
 292                                                                           reverse(_reverse),
 293                                                                           storage(new iterator_type[size]) {
 294           int i = 0;
 295           fillArray(i, node);
 296         }
 297         //BinaryTreeIterator& operator=(const BinaryTreeIterator&); ///<Block implicit copy constructor
 298       public:
 299         BinaryTreeIterator(const iterator& iter) : Iterator<iterator_type>(), 
 300                                                    tree(iter.tree),
 301                                                    index(iter.index), 
 302                                                    my_size(iter.my_size), 
 303                                                    reverse(iter.reverse),
 304                                                    storage(new iterator_type[iter.my_size]) {
 305           for (size_t i = 0; i < my_size; ++i)
 306             storage[i] = iter.storage[i];
 307         }
 308 
 309         BinaryTreeIterator() : Iterator<iterator_type>(),
 310                                tree(nullptr),
 311                                index(0),
 312                                my_size(0),
 313                                reverse(0),
 314                                storage(nullptr) {
 315         };
 316 
 317         iterator& operator=(const iterator& iter) {
 318           if (&iter != this) {
 319             if (my_size != iter.my_size) {
 320               delete[] storage;
 321               storage = new iterator_type[iter.my_size];
 322             }
 323 
 324             tree = iter.tree;
 325             index = iter.index;
 326             my_size = iter.my_size;
 327 
 328             for (size_t i = 0; i < my_size; ++i)
 329               storage[i] = iter.storage[i];
 330           }
 331           return *this;
 332         }

 334         virtual ~BinaryTreeIterator() { delete[] storage; }

 336         virtual void remove() {
 337           tree->remove(storage[index].key());

 340           for (int i = index; i < my_size; ++i)
 341             storage[i] = storage[i + 1];
 342           //The iterator is 'invalid' now, but the index is shifter down in case iter++ is called
 343           --index;
 344 */
 345         }
 346 
 347         virtual explicit operator bool() const {
 348           return ((reverse && index > 0) || (!reverse && index < (my_size)));
 349         };
 350 
 351         virtual operator iterator_type() { return operator*(); };
 352         virtual iterator_type& operator*(){ return storage[index]; }
 353 
 354         /**
 355          * @brief Postfix increment
 356          */
 357         virtual iterator operator++(int) {
 358           iterator tmp(*this);
 359           ++index;
 360           return tmp;
 361         }
 362 
 363         /**
 364          * @brief Prefix increment
 365          */
 366         virtual iterator& operator++() {
 367           ++index;
 368           return *this;
 369         }

 373          */
 374         virtual iterator operator--(int) {
 375           iterator tmp(*this);
 376           --index;

 378         }

 381          * @brief Prefix decrement
 382          */
 383         virtual iterator& operator--() {
 384           --index;
 385           return *this;
 386         }
 387     };
 388   public:
 389     inline iterator begin() { return iterator(this, root, my_size); };
 390     inline iterator end() { return iterator(this, root, my_size, 1); };
