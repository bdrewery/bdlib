  28 #ifndef _BD_STACK_H
  29 #define _BD_STACK_H 1
  30 
  31 #include "bdlib.h"
  32 #include "List.h"
  33 
  34 BDLIB_NS_BEGIN
  35 
  36 template <class T>
  37 /**
  38  * @class Stack
  39  * @brief A FILO Stack, what else!
  40  */
  41 class Stack : public List<T> {
  42   typedef T iterator_type;
  43   typedef typename List<iterator_type>::Node Node;
  44 
  45   public:
  46     Stack() : List<iterator_type>() {};
  47     virtual ~Stack() {};
  48 
  49     /**
  50      * @brief Pop an element off the top of the stack
  51      */
  52     iterator_type pop() {
  53       iterator_type item = this->head->item;
  54       this->remove(item);
  55       return item;
  56     };
  57 
  58     /**
  59      * @sa pop
  60      */
  61     inline friend Stack<iterator_type>& operator>> (Stack<iterator_type>& stack, iterator_type& item) {
  62       item = stack.pop();
  63       return stack;
  64     }
  65 
  66     /**
  67      * @brief Peek at the first element, but don't pop it off
  68      * @todo Throw an exception?
  69      */
  70     inline const iterator_type peek() const {
  71       return this->isEmpty() ? iterator_type() : this->head->item;
  72     }
  73 
  74     /**
  75      * @brief Push an element on to the top of stack
  76      * @param item The element to push back
  77      */
  78     inline void push(const iterator_type &item) {
  79       this->insert(item);
  80     };
  81 
  82     /**
  83      * @sa push
  84      */
  85     inline friend Stack<iterator_type>& operator<< (Stack<iterator_type>& stack, const iterator_type& item) {
  86       stack.insert(item);
  87       return stack;
  88     }
  89 };
  90 
  91 BDLIB_NS_END
  92 #endif /* _BD_STACK_H */
