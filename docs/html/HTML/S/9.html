  23 #ifndef _BD_STACK_H
  24 #define _BD_STACK_H 1

  26 #include "bdlib.h"
  27 #include "List.h"

  29 BDLIB_NS_BEGIN

  31 template <class T>
  32 /**
  33  * @class Stack
  34  * @brief A FILO Stack, what else!
  35  */
  36 class Stack : public List<T> {
  37   typedef T iterator_type;
  38   typedef typename List<iterator_type>::Node Node;
  39 
  40   public:
  41     Stack() : List<iterator_type>() {};
  42     virtual ~Stack() {};
  43 
  44     /**
  45      * @brief Pop an element off the top of the stack
  46      */
  47     iterator_type pop() {
  48       iterator_type item = this->head->ptr;
  49       this->remove(item);
  50       return item;
  51     };
  52 
  53     /**
  54      * @sa pop
  55      */
  56     inline friend Stack<iterator_type>& operator>> (Stack<iterator_type>& stack, iterator_type& item) {
  57       item = stack.pop();
  58       return stack;
  59     }
  60 
  61     /**
  62      * @brief Peek at the first element, but don't pop it off
  63      * @todo Throw an exception?
  64      */
  65     inline const iterator_type peek() const {
  66       return this->isEmpty() ? iterator_type() : this->head->ptr;
  67     }
  68 
  69     /**
  70      * @brief Push an element on to the top of stack
  71      * @param item The element to push back
  72      */
  73     inline void push(const iterator_type &item) {
  74       this->insert(item);
  75     };
  76 
  77     /**
  78      * @sa push
  79      */
  80     inline friend Stack<iterator_type>& operator<< (Stack<iterator_type>& stack, const iterator_type& item) {
  81       stack.insert(item);
  82       return stack;
  83     }
  84 };
  85 
  86 BDLIB_NS_END
  87 #endif /* _BD_STACK_H */
