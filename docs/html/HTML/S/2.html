  35 bool Socket::create() {
  36   int ptype = 0;
  37 
  38   if (flags & SOCKET_UDP)
  39     ptype = SOCK_DGRAM;
  40   else if (flags & SOCKET_TCP)
  41     ptype = SOCK_STREAM;
  42 
  43   sock = socket(pfamily, ptype, 0);
  44   if (sock == -1)
  45     return false;
  46 }
