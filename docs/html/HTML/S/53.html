  46 void Stream::Reserve(const size_t newSize) const {
  47   if (newSize < capacity())
  48     return;
  49   str.Reserve(STREAM_BLOCKSIZE * ((newSize + STREAM_BLOCKSIZE -1) / STREAM_BLOCKSIZE));
  50 }

  52 int Stream::seek(int offset, int whence) {
  53   int newpos;
  54 
  55   switch (whence) {
  56     case SEEK_SET:
  57       newpos = offset;
  58       break;
  59     case SEEK_CUR:
  60       newpos = tell() + offset;
  61       break;
  62     case SEEK_END:
  63       newpos = capacity() - offset;
  64       break;
  65     default:
  66       newpos = tell();
  67   }
  68   if (newpos < 0)
  69     newpos = 0;
  70   else if (static_cast<size_t>(newpos) > capacity())
  71     newpos = capacity();
  72   pos = newpos;
  73 
  74   return newpos;
  75 }

  77 String Stream::read(size_t maxSize, char delim) {
  78   size_t toRead = std::min(maxSize, length() - pos);
  79 
  80   /* No need to split the string, return a substring */
  81   if (!delim) {
  82     String ret(str(pos, toRead));
  83     pos += toRead;
  84     return ret;
  85   }
  86 
  87   /* Must reconstruct a new string */
  88   String ret(toRead);
  89 
  90   while ((ret.length() < toRead) && (str[pos] != delim)) {
  91     ret += str[pos++];
  92   }
  93   if (str.hasIndex(pos) && str[pos] == delim) {
  94     ret += delim;
  95     ++pos;
  96   }
  97 
  98   return ret;
  99 }

 101 int Stream::loadFile(const String& file)
 102 {
 103   int fd = open(file.c_str(), O_RDONLY);
 104   if (fd == -1)
 105     return 1;
 106   int ret = loadFile(fd);
 107   close(fd);
 108   return ret;

 111 int Stream::loadFile(const int fd)
 112 {
 113   clear();
 114 #ifdef HAVE_MMAP
 115   size_t size = lseek(fd, 0, SEEK_END);
 116   Reserve(size);
 117   void* map = mmap(0, size, PROT_READ, MAP_SHARED, fd, 0);
 118   if (map == MAP_FAILED) {
 119     close(fd);
 120     return 1;
 121   }
 122   loading = 1;
 123 
 124   puts(String(static_cast<const char*>(map), size));
 125 
 126   munmap(static_cast<void*>(map), size);
 127 #else
 128   FILE *f = fdopen(fd, "rb");
 129   if (f == NULL)
 130     return 1;
 131 
 132   loading = 1;
 133   fseek(f, 0, SEEK_END);
 134   size_t size = ftell(f);
 135   fseek(f, 0, SEEK_SET);
 136   Reserve(size);
 137 
 138   size_t len = 0;
 139   char buf[STREAM_BLOCKSIZE + 1];
 140 
 141   while ((len = fread(buf, 1, sizeof(buf) - 1, f))) {
 142     puts(String(buf, len));
 143   }
 144   fclose(f);
 145 #endif
 146   seek(0, SEEK_SET);
 147   loading = 0;
 148   return 0;
 149 }

 151 int Stream::writeFile(const String& file, mode_t mode) const
 152 {
 153   int fd = open(file.c_str(), O_CREAT|O_RDWR|O_TRUNC, mode);
 154   if (fd == -1)
 155     return 1;
 156   int ret = writeFile(fd);
 157   close(fd);
 158   return ret;
 159 }

 161 int Stream::writeFile(const int fd) const
 162 {
 163 #ifdef HAVE_MMAP
 164   /* Write to end of file to make its size match */
 165   /* Seek past wherever it is at now plus our length */
 166   if (ftruncate(fd, length())) return 1;
 167   if (!length())
 168     return 0;
 169 
 170   void* map = mmap(0, length(), PROT_READ|PROT_WRITE, MAP_SHARED, fd, 0);
 171 
 172   if (map == MAP_FAILED) return 1;
 173 
 174   std::memcpy(map, str.data(), length());
 175 
 176   if (munmap(map, length()) == -1) return 1;
 177 #else
 178   FILE *f = fdopen(fd, "wb");
 179   if (f == NULL)
 180     return 1;
 181 
 182   if ((fwrite(str.data(), 1, length(), f) != length()) || (fflush(f))) {
 183     fclose(f);
 184     return 1;
 185   }
 186   fclose(f);
 187 #endif
 188   return 0;
 189 }
