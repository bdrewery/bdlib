  29 int Stream::seek (int offset, int whence) {
  30   int newpos;
  31 
  32   switch (whence) {
  33     case SEEK_SET:
  34       newpos = offset;
  35       break;
  36     case SEEK_CUR:
  37       newpos = tell() + offset;
  38       break;
  39     case SEEK_END:
  40       newpos = capacity() - offset;
  41       break;
  42     default:
  43       newpos = tell();
  44   }
  45   if (newpos < 0)
  46     newpos = 0;
  47   else if (static_cast<size_t>(newpos) > capacity())
  48     newpos = capacity();
  49   pos = newpos;
  50 
  51   return newpos;
  52 }

  54 String Stream::read (size_t maxSize, char delim) {
  55   size_t toRead = std::min(maxSize, length() - pos);
  56 
  57   /* No need to split the string, return a substring */
  58   if (!delim) {
  59     String ret(str(pos, toRead));
  60     pos += toRead;
  61     return ret;
  62   }
  63 
  64   /* Must reconstruct a new string */
  65   String ret(toRead);
  66 
  67   while ((ret.length() < toRead) && (str[pos] != delim)) {
  68     ret += str[pos++];
  69   }
  70   if (str.hasIndex(pos) && str[pos] == delim) {
  71     ret += delim;
  72     ++pos;
  73   }

 137 int Stream::writeFile(const int fd) const
 138 {
 139 #ifdef HAVE_MMAP
 140   /* Write to end of file to make its size match */
 141   /* Seek past wherever it is at now plus our length */
 142   if (ftruncate(fd, length())) return 1;
 143   if (!length())
 144     return 0;
 145 
 146   void* map = mmap(0, length(), PROT_READ|PROT_WRITE, MAP_SHARED, fd, 0);
 147 
 148   if (map == MAP_FAILED) return 1;
 149 
 150   std::memcpy(map, str.data(), length());
 151

 156     return 1;
