  36 bool AtomicFile::open(const String& fname, mode_t mode) {
  37   size_t slash_pos;
  38   String dir;
  39 
  40   this->_fname = fname;
  41   this->_mode = mode;
  42 
  43   /*
  44    * Get a tempfile in the same directory as the target so that rename(2)
  45    * is not cross-device.
  46    */
  47   if ((slash_pos = this->_fname.rfind("/")) == String::npos) {
  48     slash_pos = 0;
  49     dir = "./";
  50   } else {
  51     dir = this->_fname(0, slash_pos + 1);       /* include the '/' */
  52   }
  53 
  54   this->_tmpname = String::printf("%s.%s.XXXXXX", dir.c_str(),
  55       static_cast<String>(this->_fname(slash_pos)).c_str());
  56 
  57   /* XXX: Move nul-writing to String per C++11 */
  58   this->_tmpname.resize(this->_tmpname.length() + 1);
  59   this->_fd = mkstemp(this->_tmpname.begin());
  60 
  61   return this->is_open();
  62 }
  63 
  64 bool AtomicFile::abort() {
  65   if (!this->is_open()) {
  66     return false;
  67   }
  68   ::unlink(this->_tmpname.c_str());
  69   ::close(this->_fd);
  70   this->_fd = -1;
  71   return true;
  72 }
  73 
  74 bool AtomicFile::commit() {
  75   bool result, remove_temp;
  76 
  77   result = false;
  78   remove_temp = true;
  79   if (this->_mode != mode_t(-1) && fchmod(this->_fd, this->_mode) != 0) {
  80     goto cleanup;
  81   }
  82   if (::fsync(this->_fd) != 0) {
  83     goto cleanup;
  84   }
  85   if (::close(this->_fd) != 0) {
  86     goto cleanup;
  87   }
  88   if (::rename(this->_tmpname.c_str(), this->_fname.c_str()) != 0) {
  89     goto cleanup;
  90   }
  91 
  92   remove_temp = false;
  93   result = true;
  94 
  95 cleanup:
  96   if (remove_temp) {
  97     ::unlink(this->_tmpname.c_str());
  98   }
  99   this->_fd = -1;
 100   return result;
 101 }
