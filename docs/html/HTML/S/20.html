 322 void ArrayTest :: hashTest(void)
 323 {
 324   str_c = new Array<String>();
 325 
 326   (*str_a).push("test");
 327   (*str_a).push("test2");
 328   (*str_b).push("test");

 351   *str_b = *str_a;

 364

 375 void ArrayTest :: operatorsTest(void)
 376 {
 377   str_a->push("1");
 378   str_a->push("2");
 379   str_b->push("3");
 380   str_b->push("4");
 381 
 382   str_c = new Array<String>();
 383   (*str_c) = (*str_a) + (*str_b);
 384 
 385   CPPUNIT_ASSERT_STRING_EQUAL("1 2 3 4", str_c->join(' '));
 386   CPPUNIT_ASSERT_EQUAL(size_t(4), str_c->size());
 387 
 388   (*str_c) += 1;
 389   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4", str_c->join(' '));
 390   CPPUNIT_ASSERT_EQUAL(size_t(3), str_c->size());
 391 
 392   (*str_b)[0] = "5";
 393   (*str_b)[1] = "6";
 394   (*str_c) += (*str_b);
 395 
 396   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4 5 6", str_c->join(' '));
 397   CPPUNIT_ASSERT_EQUAL(size_t(5), str_c->size());
 398 
 399   (*str_b) = (*str_c)--;
 400 
 401   CPPUNIT_ASSERT_STRING_EQUAL("6", str_b->join(' '));
 402   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4 5", str_c->join(' '));
 403   CPPUNIT_ASSERT_EQUAL(size_t(1), str_b->size());
 404   CPPUNIT_ASSERT_EQUAL(size_t(4), str_c->size());
 405 
 406   --(*str_c);
 407 
 408   CPPUNIT_ASSERT_STRING_EQUAL("2 3 4", str_c->join(' '));
 409   CPPUNIT_ASSERT_EQUAL(size_t(3), str_c->size());
 410 
 411   ++(*str_c);
 412   CPPUNIT_ASSERT_STRING_EQUAL("3 4", str_c->join(' '));
 413   CPPUNIT_ASSERT_EQUAL(size_t(2), str_c->size());
 414 
 415   (*str_b) = (*str_c)++;
 416   CPPUNIT_ASSERT_EQUAL(size_t(1), str_c->size());
 417   CPPUNIT_ASSERT_STRING_EQUAL("3", str_b->join(' '));
 418   CPPUNIT_ASSERT_STRING_EQUAL("4", str_c->join(' '));
 419 }
