  63 {

  71   a->push("Blah");
  72   CPPUNIT_ASSERT_EQUAL((size_t)4, a->size());
  73   a->push("Blah");
  74   CPPUNIT_ASSERT_EQUAL((size_t)5, a->size());
  75 
  76   (*a) << "Blah1" << "Blah2";
  77   CPPUNIT_ASSERT_EQUAL((size_t)7, a->size());

  80 void StackTest :: popTest (void)
  81 {
  82   a->push("Blahstart");
  83   a->push("Bleck");
  84   a->push("Blah");
  85   a->push("Blah");
  86   a->push("Blahend");
  87 
  88   CPPUNIT_ASSERT_EQUAL((size_t)5, a->size());
  89   CPPUNIT_ASSERT_STRING_EQUAL("Blahend", a->pop());
  90 
  91   CPPUNIT_ASSERT_EQUAL((size_t)4, a->size());
  92 
  93   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->peek());
  94   CPPUNIT_ASSERT_EQUAL((size_t)4, a->size());
  95 
  96   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->peek());
  97   CPPUNIT_ASSERT_EQUAL((size_t)4, a->size());
  98 
  99   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->pop());
 100   
 101   CPPUNIT_ASSERT_EQUAL((size_t)3, a->size());
 102   CPPUNIT_ASSERT_STRING_EQUAL("Blah", a->pop());
 103   
 104   CPPUNIT_ASSERT_EQUAL((size_t)2, a->size());
 105   CPPUNIT_ASSERT_STRING_EQUAL("Bleck", a->pop());
 106   
 107   CPPUNIT_ASSERT_EQUAL((size_t)1, a->size());
 108   CPPUNIT_ASSERT_STRING_EQUAL("Blahstart", a->pop());
 109   
 110   CPPUNIT_ASSERT_EQUAL((size_t)0, a->size());
 111 
 112   (*a) << "One" << "Two" << "Three";
 113   String one, two, three;
 114   (*a) >> three >> two;
 115   CPPUNIT_ASSERT_EQUAL((size_t)1, a->size());
 116   CPPUNIT_ASSERT_STRING_EQUAL("Three", three);
 117   CPPUNIT_ASSERT_STRING_EQUAL("Two", two);
 118   (*a) >> one;
 119   CPPUNIT_ASSERT_EQUAL((size_t)0, a->size());
 120   CPPUNIT_ASSERT_STRING_EQUAL("One", one);
 121 }

 123 void StackTest :: iterateTest (void)
 124 {
 125   (*a) << "1" << "2" << "3" << "4";
 126   Stack<String>::iterator iter(a->begin());
 127   int count = 4;
 128   for (; iter; ++iter) {
 129     char buf[2] = "";
 130     sprintf(buf, "%d", count);
 131     --count;
 132     CPPUNIT_ASSERT_STRING_EQUAL(buf, a->pop());
