 370 void ScriptInterpTCLTest :: createCommandEventTest (void)
 371 {
 372   ScriptInterpTCL tcl_script;
 373   ScriptCallbacker* scb;
 374   String result, input;
 375   Array<String> params;
 376 
 377   tcl_script.createCommand("param_test", param_test);
 378 
 379   /* Try a TCL Callback from C++ (event binding from C)
 380    * Execute script_bind from TCL, which will record the bound event callback in Events
 381    * Then call the Event and verify that it returns our expected result
 382    */
 383   // Create script_bind which will add the passed proc to our event handling procedures (Just an Array for now)
 384   tcl_script.createCommand("bind", script_bind);
 385   // Bind an event
 386   tcl_script.eval("proc echo {args} { return $args }");
 387   tcl_script.eval("bind test echo");
 388 
 389   // Call the event from C++ and then verify it echos back the input
 390   scb = Events["test"];
 391   input = "my params";
 392   params << input;
 393   // FIXME: This call line needs help to accept arbitrary params
 394   result = scb->call(params);
 395   CPPUNIT_ASSERT_STRING_EQUAL(bd::String::printf("{%s}", input.c_str()), result);
 396   params.clear();
 397 
 398   tcl_script.eval("bind \"complex test\" param_test");
 399   scb = Events["complex test"];
 400 
 401   // Now trigger the callback and verify that it calls the passed event
 402   params << "test";
 403   params << "42";
 404   result = scb->call(params);
 405   CPPUNIT_ASSERT_STRING_EQUAL("I got 2 args, arg1: test arg2: 42", result);
 406   params.clear();
 407 
 408   params << "some argument";
 409   params << "45";
 410   result = scb->call(params);
 411   CPPUNIT_ASSERT_STRING_EQUAL("I got 2 args, arg1: some argument arg2: 45", result);
 412 }
