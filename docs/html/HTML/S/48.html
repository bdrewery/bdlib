  43 static threadMainRet_t threadMain(void* object) {
  44   Thread* thread = (Thread*) object;
  45   return (threadMainRet_t) thread->run(thread->param);

  48 int Thread::start(void* _param) {
  49   if (!isStarted()) {
  50     param = _param;
  51 #if defined(USE_PTHREAD)
  52     /* Initialize the pthread_attr needed for pthread .. */
  53     pthread_attr_t attr;
  54 
  55     pthread_attr_init(&attr);
  56     /* ... and set the detached state if we are to be detached */
  57     if (isDetached())
  58       pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_DETACHED);  
  59 
  60     /* Attempt to create the thread, and set to started if successfull */
  61     if (pthread_create(&(this->handle), &attr, threadMain, this) == 0)
  62       setStarted();
  63     pthread_attr_destroy(&attr);
  64 /* endif USE_PTHREAD */
  65 #elif defined(WIN32)  
  66     /* Attempt to create the thread */
  67     if (this->handle = (HANDLE)_beginthreadex(NULL, 0, threadMain, this, 0, &(this->id)) {
  68       if (isDetached())
  69         CloseHandle(this->handle);
  70       setStarted();
  71     }
  72 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  73     /* No thread support, pretend the thread started */
  74     setStarted();
  75 #endif 
  76   }
  77   return isStarted();
  78 }

  80 void Thread::detach(void) {
  81   if (shouldDetach()) {
  82 #if defined(USE_PTHREAD)
  83     pthread_detach(this->handle);
  84 #elif defined(WIN32)
  85     CloseHandle(this->handle);
  86 #elif defined(NO_THREADS) /* NO THREAD SUPPORT */
  87     ; /* nop */
  88 #endif
  89   }
  90   setDetached();
  91 }

  93 void* Thread::wait(void) {
  94   void* _status = NULL;
  95 
  96   if (shouldDetach()) {
  97 #if defined(USE_PTHREAD)
  98     pthread_join(this->handle, &_status);
  99 #elif defined(WIN32)
 100     WaitForSingleObject(this->handle, INFINITE);
 101     GetExitCodeThread(this->handle, (DWORD*) &_status);
 102     CloseHandle(this->handle);
 103 #endif /* WIN32 */
 104     setDetached();
 105   }
 106   return _status;
 107 }

 109 void Thread::stop(void) {
 110   if (shouldDetach()) {
 111 #if defined(USE_PTHREAD)
 112     pthread_cancel(this->handle);
 113     pthread_detach(this->handle);
 114 #elif defined(WIN32)
 115     TerminateThread(this->handle, 0);
 116     CloseHandle(this->handle);
 117 #endif
 118     setDetached();
 119   }
 120 }

 134 void Thread::sleep(int delay) {
 135 #if defined(USE_PTHREAD) || defined(NO_THREADS)
 136   timeval timeout;
 137 
 138   timeout.tv_sec = delay / 1000;
 139   timeout.tv_usec = (delay * 1000) % 1000000;
 140   select(0, (fd_set*) NULL, (fd_set*) NULL, (fd_set*) NULL, &timeout);
 141 #elif defined(WIN32)
 142   ::Sleep(delay);
 143 #endif
 144 }
