  35 template <class Key, class Value>
  36 /**
  37   * @class HashTable
  38   * @brief HashTable data structure
  39   * @todo replace()
  40   * @todo iterators
  41   * @todo resizing/growing
  42   *
  43   */
  44 class HashTable {
  45   private:
  46     static const size_t default_list_size = 100;
  47     typedef KeyValue<Key, Value> iterator_type;
  48     typedef Hash<Key> hasher;
  49     typedef void (*hash_table_block)(const Key, Value, void *param);
  50 
  51     List<iterator_type> *list;
  52     size_t _size;
  53     size_t _capacity;
  54     hasher _hash;
  55 
  56     inline size_t getIndex(const Key &key) const {
  57       return _hash(key) % _capacity;
  58     }
  59   public:
  60     HashTable() : list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {};
  61     explicit HashTable(size_t capacity_in) : list(new List<iterator_type>[capacity_in]), _size(0), _capacity(capacity_in), _hash() {};
  62     HashTable(const HashTable<Key, Value> &table) : list(new List<iterator_type>[table._capacity]), _size(table._size), _capacity(table._capacity), _hash(table._hash) {
  63       for (size_t i = 0; i < _capacity; ++i)
  64           list[i] = table.list[i];
  65     };
  66 
  67     virtual ~HashTable() {
  68       delete[] list;
  69     }
  70 
  71     void clear() {
  72       for (size_t i = 0; i < _capacity; ++i)
  73         list[i].clear();
  74       _size = 0;
  75     }
  76 
  77     /**
  78      * @brief A ruby stule block which will yield to the passed callback for each Key/Value pair.
  79      * @param block The block to execute for each element
  80      * @param param An optional parameter to pass to the block.
  81      */
  82     void each(hash_table_block block, void* param = NULL) {
  83       if (!size()) return;
  84 
  85       typename List<iterator_type>::iterator iter;
  86       // Make a list of KeyValues to yield from.
  87       // Don't yield in this loop as the block may actually modify (this), thus making this iterator stale
  88       List<iterator_type> items;
  89       for (size_t i = 0; i < _capacity; ++i) {
  90         if (list[i].size()) {
  91           for (iter = list[i].begin(); iter; (++iter)) {
  92             items << *iter;
  93           }
  94         }
  95       }
  96 
  97       // Now yield on our temporary, so (this) isn't a factor.
  98       for (iter = items.begin(); iter; (++iter)) {
  99         iterator_type kv = *iter;
 100         block(kv.key(), kv.value(), param);
 101       }
 102     }
 103 
 104     HashTable &operator = (const HashTable<Key, Value> &table) { 
 105       if (&table != this) {
 106         delete[] list;
 107         list = new List<iterator_type>[_capacity];
 108         for (size_t i = 0; i < _capacity; ++i) 
 109           list[i] = table.list[i];
 110         _size = table._size;
 111         _capacity = table._capacity;
 112         _hash = table._hash;
 113       }
 114       return *this; 
 115     }
 116 
 117     inline size_t size() const { return _size; };
 118     inline size_t capacity() const { return _capacity; };
 119     inline bool isEmpty() const { return size() == 0; };
 120     //inline operator bool() const { return !isEmpty(); };
 121 
 122     bool insert(const Key &key, const Value &value) { 
 123       if (contains(key)) return false;
 124       list[getIndex(key)] << iterator_type(key, value);
 125       ++_size;
 126       return true;
 127     }
 128 
 129     inline bool contains(const Key &key) const {
 130       if (isEmpty()) return false;
 131       return list[getIndex(key)].contains(iterator_type(key, Value()));
 132     };
 133 
 134     bool remove(const Key &key) {
 135       if (isEmpty()) return false;
 136       if (list[getIndex(key)].remove(iterator_type(key, Value()))) {
 137         --_size;
 138         return true;
 139       }
 140       return false;
 141     };
 142 
 143     inline Value getValue(const Key &key) const {
 144       Value empty;
 145       if (isEmpty()) return empty;
 146       return list[getIndex(key)].find(iterator_type(key, Value())).value();
 147     };
 148 
 149     /**
 150       * @brief Associate array type accessor (rvalue)
 151       * @param key The key to search for
 152       */
 153     inline const Value operator [](const Key& key) const { return getValue(key); }
 154 
 155     /**
 156      * @brief Return an array of all the keys
 157      */
 158     Array<Key> keys() const {
 159       Array<Key> tmp(size());
 160 
 161       typename List<iterator_type>::iterator iter;
 162       for (size_t i = 0; i < capacity(); ++i) {
 163         if (list[i].size()) {
 164           for (iter = list[i].begin(); iter; (++iter)) {
 165             iterator_type kv = *iter;
 166             tmp << kv.key();
 167           }
 168         }
 169       }
 170       return tmp;
 171     }
 172 
 173     /**
 174      * @brief Return an array of all the values
 175      */
 176     Array<Value> values() const {
 177       Array<Value> tmp(size());
 178 
 179       typename List<iterator_type>::iterator iter;
 180       for (size_t i = 0; i < capacity(); ++i) {
 181         if (list[i].size()) {
 182           for (iter = list[i].begin(); iter; (++iter)) {
 183             iterator_type kv = *iter;
 184             tmp << kv.value();
 185           }
 186         }
 187       }
 188       return tmp;
 189     }
 190     
 191     /**
 192       * @brief Associate array type accessor (lvalue)
 193       * @param key The key to search for
 194       * @sa find_or_insert_key 
 195       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 196       */
 197     inline Value& operator [](const Key& key) {
 198       return find_or_insert_key(key);
 199     }
 200  
 201     /**
 202      * @brief Find a key in the list or insert it.
 203      */
 204     inline Value& find_or_insert_key(const Key& key) {
 205       if (!contains(key))
 206         insert(key, Value());
 207       return list[getIndex(key)].findRef(iterator_type(key, Value())).v;
 208     }
 209 };
