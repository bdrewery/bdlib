  34 template <class Key, class Value>
  35 /**
  36   * @class HashTable
  37   * @brief HashTable data structure
  38   * @todo replace()
  39   * @todo iterators
  40   * @todo resizing/growing
  41   *
  42   */
  43 class HashTable {
  44   private:
  45     static const size_t default_list_size = 100;
  46     typedef KeyValue<Key, Value> iterator_type;
  47     typedef Hash<Key> hasher;
  48     typedef void (*hash_table_block)(const Key, Value, void *param);
  49 
  50     List<iterator_type> *_list;
  51     size_t _size;
  52     size_t _capacity;
  53     hasher _hash;
  54 
  55     inline size_t getIndex(const Key& key) const {
  56       return _hash(key) % _capacity;
  57     }
  58   public:
  59     HashTable() : _list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {};
  60     explicit HashTable(size_t capacity_in) : _list(new List<iterator_type>[capacity_in]), _size(0), _capacity(capacity_in), _hash() {};
  61     HashTable(const HashTable<Key, Value>& table) : _list(new List<iterator_type>[table._capacity]), _size(table._size), _capacity(table._capacity), _hash(table._hash) {
  62       for (size_t i = 0; i < _capacity; ++i)
  63           _list[i] = table._list[i];
  64     };
  65 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  66     HashTable(HashTable<Key, Value>&& table) : _list(NULL), _size(0), _capacity(default_list_size), _hash() {
  67       swap(*this, table);
  68     }
  69 #endif
  70 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  71     HashTable(std::initializer_list<iterator_type> list) : _list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {
  72       *this = list;
  73     }
  74 #endif
  75 
  76     virtual ~HashTable() {
  77       delete[] _list;
  78     }
  79 
  80     void clear() {
  81       for (size_t i = 0; i < _capacity; ++i)
  82         _list[i].clear();
  83       _size = 0;
  84     }
  85 
  86     /**
  87      * @brief A ruby stule block which will yield to the passed callback for each Key/Value pair.
  88      * @param block The block to execute for each element
  89      * @param param An optional parameter to pass to the block.
  90      */
  91     void each(hash_table_block block, void* param = NULL) {
  92       if (!size()) return;
  93 
  94       // Make a list of KeyValues to yield from.
  95       // Don't yield in this loop as the block may actually modify (this), thus making this iterator stale
  96       List<iterator_type> items;
  97       for (size_t i = 0; i < _capacity; ++i) {
  98         if (_list[i].size()) {
  99           for (auto iter = _list[i].begin(); iter; (++iter)) {
 100             items << *iter;
 101           }
 102         }
 103       }
 104 
 105       // Now yield on our temporary, so (this) isn't a factor.
 106       for (auto iter = items.begin(); iter; (++iter)) {
 107         iterator_type kv = *iter;
 108         block(kv.key(), kv.value(), param);
 109       }
 110     }
 111 
 112     friend void swap(HashTable<Key, Value>& a, HashTable<Key, Value>& b) {
 113       using std::swap;
 114 
 115       swap(a._list, b._list);
 116       swap(a._size, b._size);
 117       swap(a._capacity, b._capacity);
 118       swap(a._hash, b._hash);
 119     }
 120 
 121     HashTable& operator=(HashTable<Key, Value> table) {
 122       swap(*this, table);
 123       return *this; 
 124     }
 125 
 126 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 127     /**
 128      * @brief Create an array from an initializer list
 129      * @param list An initializer_list
 130      */
 131     HashTable& operator=(std::initializer_list<iterator_type> list) {
 132       for (iterator_type item : list) {
 133         (*this)[item.key()] = item.value();
 134       }
 135       return *this;
 136     }
 137 #endif
 138 
 139     inline size_t size() const { return _size; };
 140     inline size_t capacity() const { return _capacity; };
 141     inline bool isEmpty() const { return size() == 0; };
 142     inline explicit operator bool() const { return !isEmpty(); };
 143 
 144     bool insert(const Key& key, const Value& value) {
 145       if (contains(key)) return false;
 146       _list[getIndex(key)] << iterator_type(key, value);
 147       ++_size;
 148       return true;
 149     }
 150 
 151     inline bool contains(const Key& key) const {
 152       if (isEmpty()) return false;
 153       return _list[getIndex(key)].contains(iterator_type(key, Value()));
 154     };
 155 
 156     bool remove(const Key& key) {
 157       if (isEmpty()) return false;
 158       if (_list[getIndex(key)].remove(iterator_type(key, Value()))) {
 159         --_size;
 160         return true;
 161       }
 162       return false;
 163     };
 164 
 165     inline Value getValue(const Key& key) const {
 166       Value empty;
 167       if (isEmpty()) return empty;
 168       return _list[getIndex(key)].find(iterator_type(key, Value())).value();
 169     };
 170 
 171     /**
 172       * @brief Associate array type accessor (rvalue)
 173       * @param key The key to search for
 174       */
 175     inline const Value operator[](const Key& key) const { return getValue(key); }
 176 
 177     /**
 178      * @brief Return an array of all the keys
 179      */
 180     Array<Key> keys() const {
 181       Array<Key> tmp(size());
 182 
 183       for (size_t i = 0; i < capacity(); ++i) {
 184         if (_list[i].size()) {
 185           for (auto iter = _list[i].begin(); iter; (++iter)) {
 186             iterator_type kv = *iter;
 187             tmp << kv.key();
 188           }
 189         }
 190       }
 191       return tmp;
 192     }
 193 
 194     /**
 195      * @brief Return an array of all the values
 196      */
 197     Array<Value> values() const {
 198       Array<Value> tmp(size());
 199 
 200       for (size_t i = 0; i < capacity(); ++i) {
 201         if (_list[i].size()) {
 202           for (auto iter = _list[i].begin(); iter; (++iter)) {
 203             iterator_type kv = *iter;
 204             tmp << kv.value();
 205           }
 206         }
 207       }
 208       return tmp;
 209     }
 210     
 211     /**
 212       * @brief Associate array type accessor (lvalue)
 213       * @param key The key to search for
 214       * @sa find_or_insert_key 
 215       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 216       */
 217     inline Value& operator[](const Key& key) {
 218       return find_or_insert_key(key);
 219     }
 220  
 221     /**
 222      * @brief Find a key in the list or insert it.
 223      */
 224     inline Value& find_or_insert_key(const Key& key) {
 225       if (!contains(key))
 226         insert(key, Value());
 227       return _list[getIndex(key)].findRef(iterator_type(key, Value())).v;
 228     }
 229 };
