  38 template <class Key, class Value>
  39 /**
  40   * @class HashTable
  41   * @brief HashTable data structure
  42   * @todo replace()
  43   * @todo iterators
  44   * @todo resizing/growing
  45   *
  46   */
  47 class HashTable {
  48   private:
  49     static const size_t default_list_size = 100;
  50     typedef KeyValue<Key, Value> iterator_type;
  51     typedef Hash<Key> hasher;
  52     typedef void (*hash_table_block)(const Key, Value, void *param);
  53 
  54     List<iterator_type> *_list;
  55     size_t _size;
  56     size_t _capacity;
  57     hasher _hash;
  58 
  59     inline size_t getIndex(const Key& key) const {
  60       return _hash(key) % _capacity;
  61     }
  62   public:
  63     HashTable() : _list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {};
  64     explicit HashTable(size_t capacity_in) : _list(new List<iterator_type>[capacity_in]), _size(0), _capacity(capacity_in), _hash() {};
  65     HashTable(const HashTable<Key, Value>& table) : _list(new List<iterator_type>[table._capacity]), _size(table._size), _capacity(table._capacity), _hash(table._hash) {
  66       for (size_t i = 0; i < _capacity; ++i)
  67           _list[i] = table._list[i];
  68     };
  69 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  70     HashTable(HashTable<Key, Value>&& table) : _list(std::move(table._list)), _size(std::move(table._Size)), _capacity(std::move(table._capacity)), _hash(std::move(table._hash)) {
  71       _list = NULL;
  72       _size = 0;
  73       _capacity = default_list_size;
  74       _hash = _hash();
  75     }
  76 #endif
  77 #ifdef __GXX_EXPERIMENTAL_CXX0X__
  78     HashTable(std::initializer_list<iterator_type> list) : _list(new List<iterator_type>[default_list_size]), _size(0), _capacity(default_list_size), _hash() {
  79       *this = list;
  80     }
  81 #endif
  82 
  83     virtual ~HashTable() {
  84       delete[] _list;
  85     }
  86 
  87     void clear() {
  88       for (size_t i = 0; i < _capacity; ++i)
  89         _list[i].clear();
  90       _size = 0;
  91     }
  92 
  93     /**
  94      * @brief A ruby stule block which will yield to the passed callback for each Key/Value pair.
  95      * @param block The block to execute for each element
  96      * @param param An optional parameter to pass to the block.
  97      */
  98     void each(hash_table_block block, void* param = NULL) {
  99       if (!size()) return;
 100 
 101       // Make a list of KeyValues to yield from.
 102       // Don't yield in this loop as the block may actually modify (this), thus making this iterator stale
 103       typename List<iterator_type>::iterator iter;
 104       List<iterator_type> items;
 105       for (size_t i = 0; i < _capacity; ++i) {
 106         if (_list[i].size()) {
 107           for (iter = _list[i].begin(); iter; (++iter)) {
 108             items << *iter;
 109           }
 110         }
 111       }
 112 
 113       // Now yield on our temporary, so (this) isn't a factor.
 114       for (iter = items.begin(); iter; (++iter)) {
 115         iterator_type kv = *iter;
 116         block(kv.key(), kv.value(), param);
 117       }
 118     }
 119 
 120     friend void swap(HashTable<Key, Value>& a, HashTable<Key, Value>& b) {
 121       using std::swap;
 122 
 123       swap(a._list, b._list);
 124       swap(a._size, b._size);
 125       swap(a._capacity, b._capacity);
 126       swap(a._hash, b._hash);
 127     }
 128 
 129     HashTable& operator=(HashTable<Key, Value> table) {
 130       swap(*this, table);
 131       return *this; 
 132     }
 133 
 134 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 135     /**
 136      * @brief Create an array from an initializer list
 137      * @param list An initializer_list
 138      */
 139     HashTable& operator=(std::initializer_list<iterator_type> list) {
 140       for (iterator_type item : list) {
 141         (*this)[item.key()] = item.value();
 142       }
 143       return *this;
 144     }
 145 #endif
 146 
 147     inline size_t size() const { return _size; };
 148     inline size_t capacity() const { return _capacity; };
 149     inline bool isEmpty() const { return size() == 0; };
 150 #ifdef __GXX_EXPERIMENTAL_CXX0X__
 151     inline explicit operator bool() const { return !isEmpty(); };
 152 #endif
 153 
 154     bool insert(const Key& key, const Value& value) {
 155       if (contains(key)) return false;
 156       _list[getIndex(key)] << iterator_type(key, value);
 157       ++_size;
 158       return true;
 159     }
 160 
 161     inline bool contains(const Key& key) const {
 162       if (isEmpty()) return false;
 163       return _list[getIndex(key)].contains(iterator_type(key, Value()));
 164     };
 165 
 166     bool remove(const Key& key) {
 167       if (isEmpty()) return false;
 168       if (_list[getIndex(key)].remove(iterator_type(key, Value()))) {
 169         --_size;
 170         return true;
 171       }
 172       return false;
 173     };
 174 
 175     inline Value getValue(const Key& key) const {
 176       Value empty;
 177       if (isEmpty()) return empty;
 178       return _list[getIndex(key)].find(iterator_type(key, Value())).value();
 179     };
 180 
 181     /**
 182       * @brief Associate array type accessor (rvalue)
 183       * @param key The key to search for
 184       */
 185     inline const Value operator[](const Key& key) const { return getValue(key); }
 186 
 187     /**
 188      * @brief Return an array of all the keys
 189      */
 190     Array<Key> keys() const {
 191       Array<Key> tmp(size());
 192 
 193       typename List<iterator_type>::iterator iter;
 194       for (size_t i = 0; i < capacity(); ++i) {
 195         if (_list[i].size()) {
 196           for (iter = _list[i].begin(); iter; (++iter)) {
 197             iterator_type kv = *iter;
 198             tmp << kv.key();
 199           }
 200         }
 201       }
 202       return tmp;
 203     }
 204 
 205     /**
 206      * @brief Return an array of all the values
 207      */
 208     Array<Value> values() const {
 209       Array<Value> tmp(size());
 210 
 211       typename List<iterator_type>::iterator iter;
 212       for (size_t i = 0; i < capacity(); ++i) {
 213         if (_list[i].size()) {
 214           for (iter = _list[i].begin(); iter; (++iter)) {
 215             iterator_type kv = *iter;
 216             tmp << kv.value();
 217           }
 218         }
 219       }
 220       return tmp;
 221     }
 222     
 223     /**
 224       * @brief Associate array type accessor (lvalue)
 225       * @param key The key to search for
 226       * @sa find_or_insert_key 
 227       * If the key is not in the table, it is inserted, and the value set to the rvalue given.
 228       */
 229     inline Value& operator[](const Key& key) {
 230       return find_or_insert_key(key);
 231     }
 232  
 233     /**
 234      * @brief Find a key in the list or insert it.
 235      */
 236     inline Value& find_or_insert_key(const Key& key) {
 237       if (!contains(key))
 238         insert(key, Value());
 239       return _list[getIndex(key)].findRef(iterator_type(key, Value())).v;
 240     }
 241 };
